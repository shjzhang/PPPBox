41  /* Returns the effect of ocean tides loading (meters) at the given
42  * station and epoch, in the Up-East-North (UEN) reference frame.
43  *
44  * @param name Station name (case is NOT relevant).
45  * @param time Epoch to look up
46  *
47  * @return a Triple with the ocean tidas loading effect, in meters and
48  * in the UEN reference frame.
49  *
50  * @throw InvalidRequest If the request can not be completed for any
51  * reason, this is thrown. The text may have additional information
52  * about the reason the request failed.
53  */
54  Triple OceanLoading::getOceanLoading( const string& name,
55  const CommonTime& t )
56  throw(InvalidRequest)
57  {
58 
59  const int NUM_COMPONENTS = 3;
60  const int NUM_HARMONICS = 11;
61 
62  Matrix<double> harmonics(6,11,0.0);
63 
64  // Get harmonics data from file
65  harmonics = blqData.getTideHarmonics(name);
66 
67  Vector<double> arguments(11,0.0);
68 
69  // Compute arguments
70  arguments = getArg(t);
71 
72  Triple oLoading;
73 
74  for(int i=0; i<NUM_COMPONENTS; i++)
75  {
76 
77  double temp(0.0);
78  for(int k=0; k<NUM_HARMONICS; k++)
79  {
80 
81  temp += harmonics(i,k) *
82  std::cos( arguments(k) - harmonics( (i+3),k)*DEG_TO_RAD );
83 
84  }
85 
86  // This Triple is in Up, West, South reference frame
87  oLoading[i] = temp;
88 
89  } // End of 'for(int i=0; i<NUM_COMPONENTS; i++)'
90 
91  // Let's change Triple to Up, East, North [UEN] reference frame
92  oLoading[1] = -oLoading[1];
93  oLoading[2] = -oLoading[2];
94 
95  return oLoading;
96 
97  } // End of method 'OceanLoading::getOceanLoading()'

118  /* Compute the value of the corresponding astronomical arguments,
119  * in radians. This routine is based on IERS routine ARG.f.
120  *
121  * @param time Epoch of interest
122  *
123  * @return A Vector<double> of 11 elements with the corresponding
124  * astronomical arguments to be used in ocean loading model.
125  */
126  Vector<double> OceanLoading::getArg(const CommonTime& time)
127  {

129  const int NUM_HARMONICS = 11;
130 
131  // Let's store some important values
132  Vector<double> sig(NUM_HARMONICS,0.0);
133  sig(0) = 1.40519e-4;
134  sig(1) = 1.45444e-4;
135  sig(2) = 1.37880e-4;
136  sig(3) = 1.45842e-4;
137  sig(4) = 0.72921e-4;
138  sig(5) = 0.67598e-4;
139  sig(6) = 0.72523e-4;
140  sig(7) = 0.64959e-4;
141  sig(8) = 0.053234e-4;
142  sig(9) = 0.026392e-4;
143  sig(10)= 0.003982e-4;
144 
145  Matrix<double> angfac(4,NUM_HARMONICS,0.0);
146  angfac(0,0) = 2.0; angfac(1,0) = -2.0;
147  angfac(2,0) = 0.0; angfac(3,0) = 0.0;
148  angfac(0,1) = 0.0; angfac(1,1) = 0.0;
149  angfac(2,1) = 0.0; angfac(3,1) = 0.0;

154  angfac(0,4) = 1.0; angfac(1,4) = 0.0;
155  angfac(2,4) = 0.0; angfac(3,4) = 0.25;
156  angfac(0,5) = 1.0; angfac(1,5) = -2.0;
157  angfac(2,5) = 0.0; angfac(3,5) = -0.25;
158  angfac(0,6) = -1.0; angfac(1,6) = 0.0;
159  angfac(2,6) = 0.0; angfac(3,6) = -0.25;
160  angfac(0,7) = 1.0; angfac(1,7) = -3.0;

162  angfac(0,8) = 0.0; angfac(1,8) = 2.0;

169

171 
172  // Get day of year
173  short year(static_cast<YDSTime>(time).year);
174 
175  // Fractional part of day, in seconds
176  double fday(static_cast<YDSTime>(time).sod);
177 
178  // Compute time
179  double d(static_cast<YDSTime>(time).doy+365.0*(year-1975.0)+floor((year-1973.0)/4.0));
180  double t((27392.500528+1.000000035*d)/36525.0);
181 
182  // Mean longitude of Sun at beginning of day
183  double H0((279.69668+(36000.768930485+3.03e-4*t)*t)*DEG_TO_RAD);
184 
185  // Mean longitude of Moon at beginning of day
186  double S0( (((1.9e-6*t - 0.001133)*t +
187  481267.88314137)*t + 270.434358)*DEG_TO_RAD );
188 
189  // Mean longitude of lunar perigee at beginning of day
190  double P0( (((-1.2e-5*t - 0.010325)*t +
191  4069.0340329577)*t + 334.329653)*DEG_TO_RAD );
192 
193  for(int k=0; k<NUM_HARMONICS; k++)
194  {
195 
196  double temp( sig(k)*fday + angfac(0,k)*H0 +
197  angfac(1,k)*S0 + angfac(2,k)*P0 + angfac(3,k)*TWO_PI );
198 
199  arguments(k) = fmod(temp,TWO_PI);
200 
201  if (arguments(k) < 0.0)
202  {
203  arguments(k) = arguments(k) + TWO_PI;
204  }
205 
206  } // End of 'for(int k=0; k<NUM_HARMONICS; k++)'
207 
208  return arguments;
