87  void OrbElemCNAV::loadData( const ObsID& obsIDArg,
88  const SatID& satIDArg,
89  const PackedNavBits& message10,
90  const PackedNavBits& message11,
91  const PackedNavBits& messageClk)
92  throw( InvalidParameter )
93  {
94 
95  obsID = obsIDArg;
96  satID = satIDArg;
97 
98  // Message Type 10 data
99  unsigned long TOWCount10 = message10.asUnsignedLong(20, 17, 6);
100  short TOWWeek = message10.asUnsignedLong(38, 13, 1);
101  L1Health = message10.asUnsignedLong(51, 1, 1);
102  L2Health = message10.asUnsignedLong(52, 1, 1);
103  L5Health = message10.asUnsignedLong(53, 1, 1);
104  double Top = message10.asUnsignedLong(54, 11, 300);
105  URAed = message10.asLong(65, 5, 1);
106  double Toe = message10.asUnsignedLong(70, 11, 300);
107  double deltaA = message10.asSignedDouble(81, 26, -9);
108  Adot = message10.asSignedDouble(107, 25, -21);
109  dn = message10.asDoubleSemiCircles(132, 17, -44);
110  dndot = message10.asDoubleSemiCircles(149, 23, -57);
111  M0 = message10.asDoubleSemiCircles(172, 33, -32);
112  ecc = message10.asUnsignedDouble(205, 33, -34);
113  w = message10.asDoubleSemiCircles(238, 33, -32);
114  short ISFVal = message10.asUnsignedLong(271, 1, 1);
115  if (ISFVal==1) IntegrityStatusFlag = true;
116  L2CPhasing = message10.asUnsignedLong(272, 1, 1);
117 
118  // Message Type 11 data
119  unsigned long TOWCount11 = message11.asUnsignedLong(20, 17, 6);
120  double Toe11 = message11.asUnsignedLong(38,11,300);
121  OMEGA0 = message11.asDoubleSemiCircles(49, 33, -32);
122  i0 = message11.asDoubleSemiCircles(82, 33, -32);
123  double deltaOMEGAdot = message11.asDoubleSemiCircles(115, 17, -44);
124  idot = message11.asDoubleSemiCircles(132, 15, -44);
125  Cis = message11.asSignedDouble(147, 16, -30);
126  Cic = message11.asSignedDouble(163, 16, -30);
127  Crs = message11.asSignedDouble(179, 24, -8);
128  Crc = message11.asSignedDouble(203, 24, -8);
129  Cus = message11.asSignedDouble(227, 21, -30);
130  Cuc = message11.asSignedDouble(248, 21, -30);
131 
132  // Message Type Clock data
133  unsigned long TOWCountClk = messageClk.asUnsignedLong(20, 17, 6);
134  URAned0 = messageClk.asLong(49, 5, 1);
135  URAned1 = messageClk.asUnsignedLong(54, 3, 1);
136  URAned2 = messageClk.asUnsignedLong(57, 3, 1);
137  double Toc = messageClk.asUnsignedLong(60, 11, 300);
138  af0 = messageClk.asSignedDouble(71, 26, -35);
139  af1 = messageClk.asSignedDouble(97, 20, -48);
140  af2 = messageClk.asSignedDouble(117, 10, -60);
141 
142  A = A_REF_GPS + deltaA;
143  OMEGAdot = OMEGADOT_REF_GPS + deltaOMEGAdot;
144 
145  if(Toe!=Toc ||
146  Toe!=Toe11)
147  {
148  InvalidParameter exc("Toe(s) and Toc are not consistent.");
149  GPSTK_THROW(exc);
150  }
151 
152  healthy = false;
153  if (obsIDArg.band == ObsID::cbL2 && L2Health == 0) healthy = true;
154  if (obsIDArg.band == ObsID::cbL5 && L5Health == 0) healthy = true;
155 
156  double timeDiff = Toe - TOWCount10;
157  short epochWeek = TOWWeek;
158  if (timeDiff < -HALFWEEK) epochWeek++;
159  else if (timeDiff > HALFWEEK) epochWeek--;
160 
161  // Note that TOW times are referenced to the beginning of
162  // the next message. Therefore, the transmit time is actually
163  // PRIOR to the TOW count in SOW. However, the amount prior
164  // is dependent on the signal. 12 sec for L2C, 6 sec for L5
165  unsigned long SOWOffset = 6;
166  if (obsID.code==ObsID::tcC2M ||
167  obsID.code==ObsID::tcC2L ||
168  obsID.code==ObsID::tcC2LM) SOWOffset=12;
169 
170  // Message Type 10 CONTAINS the week number. Therefore, there
171  // should not be any SOW rollover to deal with for Message Type 10.
172  // However, we don't know for certain that the 11 and the clock (31-37)
173  // messages were collected in the same week (probably, but not certainly).
174  // Therefore, we need to look for week rollovers.
175  long SOW10 = TOWCount10 - SOWOffset;
176  ctMsg10 = GPSWeekSecond(TOWWeek, SOW10, TimeSystem::GPS);
177 
178  // Set up temp variables for MsgType 11 SOW and week.
179  short Week11 = TOWWeek;
180  long SOW11 = TOWCount11-SOWOffset;
181 
182  // First, check to see that the TOW -> transmit time adjust did not
183  // cross the week boundary.
184  if (SOW11<0) SOW11 += gpstk::FULLWEEK;
185 
186  // Then check against the MsgType 10 XMit time/week
187  long check = (SOW10-SOW11);
188  if (check>gpstk::HALFWEEK) Week11++;
189  else if (check<-gpstk::HALFWEEK) Week11--;
190  ctMsg11 = GPSWeekSecond(Week11, SOW11, TimeSystem::GPS);
191 
192  // Now follow the same process for the clock message times.
193  short WeekClk = TOWWeek;
194  long SOWClk = TOWCountClk - SOWOffset;
195  if (SOWClk<0) SOWClk += gpstk::FULLWEEK;
196  check = (SOW10-SOW11);
197  if (check>gpstk::HALFWEEK) WeekClk++;
198  else if (check<-gpstk::HALFWEEK) WeekClk--;
199  ctMsgClk = GPSWeekSecond(WeekClk, SOWClk, TimeSystem::GPS);
200 
201  // The fit interval is nominally the middle of the trasmission interval.
202  // However, interval may "start late" for an upload. Therefore, start
203  // from the Toe (rounded to the middle) and work forward/backward from
204  // there. The "beginning" is nominally the even two hour interval prior
205  // to the Toe; however, if the Toe is offset from the middle, the
206  // beginning is the later of the beginning of fit and the transmission
207  // time.
208  long testSOW1 = (static_cast<GPSWeekSecond>(ctMsg10)).sow;
209  long testSOW2 = (static_cast<GPSWeekSecond>(ctMsg11)).sow;
210  long testSOW3 = (static_cast<GPSWeekSecond>(ctMsgClk)).sow;
211 
212  long midPointSOW = (long) Toe;
213  long Xmit = 0;
214  long offsetFromEven2Hours = midPointSOW % TWO_HOURS;
215  bool smallOffset = false;
216  if (offsetFromEven2Hours!=NINTY_MINUTES)
217  {
218  smallOffset = true;
219 
220  // Compute the "small offset". Negative value means the
221  // midPoint is early.
222  unsigned long sizeOfOffset = offsetFromEven2Hours - NINTY_MINUTES;
223 
224  // Apply the offset such that the midPointSOW is now the
225  // nominal midpoint of the tranmission interval.
226  midPointSOW -= sizeOfOffset;
227  }
228 
229  if (smallOffset) // May not have been an even two hour change
230  {
231  long leastSOW = testSOW1;
232  if(testSOW2<leastSOW) leastSOW = testSOW2;
233  if(testSOW3<leastSOW) leastSOW = testSOW3;
234  Xmit = leastSOW - (leastSOW % (4*SOWOffset)); // See -705B Table 20-XII
235  // Still need a test to assure that the begin of fit doesn't'
236  // exceed half the interval from the mid-point. (During live
237  // CNAV test of June 2013, Msg 11 was cutting over early.)
238  long defBegFit = midPointSOW - NINTY_MINUTES;
239  if (Xmit<defBegFit) Xmit = defBegFit;
240  }
241  else
242  {
243  long leastSOW = testSOW1;
244  if(testSOW2<leastSOW) leastSOW = testSOW2;
245  if(testSOW3<leastSOW) leastSOW = testSOW3;
246  Xmit = leastSOW - (leastSOW % TWO_HOURS);
247  long defBegFit = midPointSOW - NINTY_MINUTES;
248  if (Xmit<defBegFit) Xmit = defBegFit;
249  }
250  beginValid = GPSWeekSecond(TOWWeek, Xmit, TimeSystem::GPS);
251 
252  // Top must be before transmission.
253  // Check for week rollover between Top and TOWCount.
254  short TopWeek = TOWWeek;
255  double testSOW = (static_cast<GPSWeekSecond>(beginValid)).sow;
256  if(Top>testSOW) TopWeek--;
257 
258  ctTop = GPSWeekSecond(TopWeek, Top, TimeSystem::GPS);
259  ctToe = GPSWeekSecond(epochWeek, Toe, TimeSystem::GPS);
260  ctToc = GPSWeekSecond(epochWeek, Toc, TimeSystem::GPS);
261 
262  // The end of validity is 1.5 hours after the nominal midpoint
263  double endSOW = midPointSOW + NINTY_MINUTES;
264  short endWeek = TOWWeek;
265  if (endSOW> gpstk::FULLWEEK)
266  {
267  endSOW -= gpstk::FULLWEEK;
268  endWeek++;
269  }
270  endValid = GPSWeekSecond(endWeek, endSOW, TimeSystem::GPS);
271 
272  dataLoadedFlag = true;
273  } // end of loadData()
