74  void FICData::reallyPutRecord(FFStream& ffs) const
75  throw(std::exception, gpstk::StringUtils::StringException,
76  gpstk::FFStreamError)
77  {
78  if (!isValid())
79  {
80  FFStreamError e("Tried to write an invalid FIC block");
81  GPSTK_THROW(e);
82  }
83 
84  // is this a FIC or FICA stream?
85  // remember that a FICAStream derives from FICStream, so check
86  // for FICA first
87  if (dynamic_cast<FICAStream*>(&ffs))
88  {
89  FICAStream& strm = dynamic_cast<FICAStream&>(ffs);
90 
91  strm << '\n' << blockString;
92  strm << setw(6) << blockNum
93  << setw(5) << f.size()
94  << setw(5) << i.size()
95  << setw(5) << c.size();
96  unsigned long index;
97  for (index = 0; index < f.size(); index++)
98  {
99  if ((index % 4) == 0)
100  strm << '\n';
101  double l10 = ::log10(::fabs(f[index]));
102  if ( ((l10 >= 100) || (l10 <= -100)) && (f[index] != 0))
103  strm << doub2for(f[index], 20, 3);
104  else
105  strm << doub2for(f[index], 20, 2);
106  }
107  for (index = 0; index < i.size(); index++)
108  {
109  if ((index % 6) == 0)
110  strm << '\n';
111  strm << setw(12) << i[index];
112  }
113  for (index = 0; index < c.size(); index++)
114  {
115  if ((index % 8) == 0)
116  strm << '\n';
117  strm << setw(8) << c[index];
118  }
119  }
120  else
121  {
122  if (dynamic_cast<FICStream*>(&ffs))
123  {
124  FICStream& strm = dynamic_cast<FICStream&>(ffs);
125 
126  strm << blockString;
127  strm.writeData<int32_t>(hostToIntel(blockNum));
128  strm.writeData<uint32_t>(hostToIntel(static_cast<uint32_t>(f.size())));
129  strm.writeData<uint32_t>(hostToIntel(static_cast<uint32_t>(i.size())));
130  strm.writeData<uint32_t>(hostToIntel(static_cast<uint32_t>(c.size())));
131  unsigned long index;
132  for (index = 0; index < f.size(); index++)
133  strm.writeData<double>(hostToIntel(f[index]));
134  for (index = 0; index < i.size(); index++)
135  strm.writeData<int32_t>(hostToIntel(i[index]));
136  for (index = 0; index < c.size(); index++)
137  strm.writeData<char>(hostToIntel(c[index]));
138  }
139  // bad cast!!!
140  else
141  {
142  FFStreamError e("FICData tried to read from a non-FIC file");
143  GPSTK_THROW(e);
144  }
145  }

148  bool FICData::isValid() const
149  {
150  switch(blockNum) {
151  case 109:
152  if((f.size() == 0) && (i.size() == 32) && (c.size() == 0))
153  return true;
154  break;
155  case 9:
156  if((f.size() == 60) && (i.size() == 0) && (c.size() == 0))
157  return true;
158  break;
159  case 162:
160  if((f.size() == 0) && (i.size() == 15) && (c.size() == 0))
161  return true;
162  break;
163  case 62:
164  if(( (f.size() == 20) || (f.size() == 32) || (f.size() == 14) ||
165  (f.size() == 23) || (f.size() == 47) || (f.size() == 29) )
166  && (i.size() == 6) && (c.size() == 0))
167  return true;
168  break;
169  default:
170  return false;
171  break;
172  }
173  return false;
