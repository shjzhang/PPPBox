107  inline
108  short getLegacyFitInterval(const short iodc, const short fiti)
109  throw(gpstk::InvalidRequest )
110  {
111  /* check the IODC */
112  if (iodc < 0 || iodc > 1023)
113  {
114  /* error in iodc, return minimum fit */
115  return 4;
116  }
117 
118  if ( ( ( (fiti == 0) && (iodc & 0xFF) < 240 )
119  || (iodc & 0xFF) > 255 ) )
120  {
121  /* fit interval of 4 hours */
122  return 4;
123  }
124  else if (fiti == 1)
125  {
126  if( ((iodc & 0xFF) < 240 || (iodc & 0xFF) > 255))
127  {
128  /* fit interval of 6 hours */
129  return 6;
130  }
131  else if(iodc >=240 && iodc <=247)
132  {
133  /* fit interval of 8 hours */
134  return 8;
135  }
136  else if( ( (iodc >= 248) && (iodc <= 255) ) || iodc == 496 )
137  {
138  /* fit interval of 14 hours */
139  return 14;
140  }
141  else if((iodc >= 497 && iodc <=503) || (iodc >= 1021 && iodc <= 1023))
142  {
143  /* fit interval of 26 hours */
144  return 26;
145  }
146  else if(iodc >= 504 && iodc <=510)
147  {
148  /* fit interval of 50 hours */
149  return 50;
150  }
151  else if( iodc == 511 || ( (iodc >= 752) && (iodc <= 756) ) )
152  {
153  /* fit interval of 74 hours */
154  return 74;
155  }
156  else if(iodc == 757)
157  {
158  /* fit interval of 98 hours */
159  return 98;
160  }
161  else
162  {
163  InvalidRequest exc("Invalid IODC Value For sv Block");
164  GPSTK_THROW(exc);
165  }
166  }
167  else
168  {
169  /* error in ephemeris/iodc, return minimum fit */
170  return 4;
171  }
172 
173  return 0; // never reached
174  }

291  inline double getWavelength(const SatID& sat, const int& n, const int N=0)
292  throw()
293  {
294  switch(sat.system) {
295  case SatID::systemGPS:
296  if(n == 1) return L1_WAVELENGTH_GPS;
297  else if(n == 2) return L2_WAVELENGTH_GPS;
298  else if(n == 5) return L5_WAVELENGTH_GPS;
299  break;
300  case SatID::systemGlonass:
301  if(n == 1) return (C_MPS/(L1_FREQ_GLO + N*L1_FREQ_STEP_GLO));
302  else if(n == 2) return (C_MPS/(L2_FREQ_GLO + N*L2_FREQ_STEP_GLO));
303  break;
304  case SatID::systemGalileo:
305  if(n == 1) return L1_WAVELENGTH_GAL;
306  else if(n == 5) return L5_WAVELENGTH_GAL;
307  else if(n == 6) return L6_WAVELENGTH_GAL;
308  else if(n == 7) return L7_WAVELENGTH_GAL;
309  else if(n == 8) return L8_WAVELENGTH_GAL;
310  break;
311  case SatID::systemGeosync:
312  if(n == 1) return L1_WAVELENGTH_GEO;
313  else if(n == 5) return L5_WAVELENGTH_GEO;
314  break;
315  case SatID::systemBeiDou:
316  //MGEX data uses 2 if(n == 1) return L1_WAVELENGTH_BDS;
317  if(n == 2) return L1_WAVELENGTH_BDS;
318  else if(n == 7) return L2_WAVELENGTH_BDS;
319  else if(n == 6) return L3_WAVELENGTH_BDS;
320  break;
321  case SatID::systemQZSS:
322  if(n == 1) return L1_WAVELENGTH_QZS;
323  else if(n == 2) return L2_WAVELENGTH_QZS;
324  else if(n == 5) return L5_WAVELENGTH_QZS;
325  else if(n == 6) return L6_WAVELENGTH_QZS;
326  break;
327  default:
328  break;
329  }
330 
331  return 0.0;
332  }

337  inline double getBeta(const SatID& sat, const int& na, const int& nb) throw()
338  {
339  double wla = getWavelength(sat,na);
340  double wlb = getWavelength(sat,nb);
341  if(wla == 0.0 || wlb == 0.0) return 0.0;
342  return wlb/wla;
343  }

349  inline double getAlpha(const SatID& sat, const int& na, const int& nb) throw()
350  {
351  double beta(getBeta(sat,na,nb));
352  if(beta == 0.0) return 0.0;
353  return (beta*beta-1.0);
