61  // This routine uses the Householder algorithm to update the SRI
62  // state and covariance.
63  // Input:
64  // R a priori SRI matrix (upper triangular, dimension N)
65  // Z a priori SRI data vector (length N)
66  // A concatentation of H and D : A = H || D, where
67  // H Measurement partials, an M by N matrix.
68  // D Data vector, of length M
69  // H and D may have row dimension > M; then pass M:
70  // M (optional) Row dimension of H and D
71  // Output:
72  // Updated R and Z. H is trashed, but the data vector D
73  // contains the residuals of fit (D - A*state).
74  // Return values:
75  // SrifMU returns void, but throws exceptions if the input matrices
76  // or vectors have incompatible dimensions.
77  //
78  // Measurment noise associated with H and D must be white
79  // with unit covariance. If necessary, the data can be 'whitened'
80  // before calling this routine in order to satisfy this requirement.
81  // This is done as follows. Compute the lower triangular square root
82  // of the covariance matrix, L, and replace H with inverse(L)*H and
83  // D with inverse(L)*D.
84  //
85  // The Householder transformation is simply an orthogonal
86  // transformation designed to make the elements below the diagonal
87  // zero. It works by explicitly performing the transformation, one
88  // column at a time, without actually constructing the transformation
89  // matrix. The matrix is transformed as follows
90  // [ A(m,n) ] => [ sum a ]
91  // [ ] => [ 0 A'(m-1,n-1) ]
92  // after which the same transformation is applied to A' matrix, until A'
93  // has only one row or column. The transformation that zeros the diagonal
94  // below the (k,k) element also replaces the (k,k) element and modifies
95  // the matrix elements for columns >= k and rows >=k, but does not affect
96  // the matrix for columns < k or rows < k.
97  // Column k (=0..min(m,n)-1) of the input matrix A(m,n) can be zeroed
98  // below the diagonal (columns < k have already been so zeroed) as follows:
99  // let y be the vector equal to column k at the diagonal and below,
100  // ( so y(j)==A(k+j,k), y(0)==A(k,k), y.size = m-k )
101  // let sum = -sign(y(0))*|y|,
102  // define vector u by u(0) = y(0)-sum, u(j)=y(j) for j>0 (j=1..m-k)
103  // and finally define b = 1/(sum*u(0)).
104  // Redefine column k with A(k,k)=sum and A(k+j,k)=0, j=1..m, and
105  // then for each column j > k, (j=k+1..n)
106  // compute g = b*sum[u(i)*A(k+i,j)], i=0..m-k-1,
107  // replace A(k+i,j) with A(k+i,j)+g*u(i), for i=0..m-k-1
108  // Most algorithms don't handle special cases:
109  // 1. If column k is already zero below the diagonal, but A(k,k)!=0, then
110  // y=[A(k,k),0,0,...0], sum=-A(k,k), u(0)=2A(k,k), u=[2A(k,k),0,0,...0]
111  // and b = -1/(2*A(k,k)^2). Then, zeroing column k only changes the sign
112  // of A(k,k), and for the other columns j>k, g = -A(k,j)/A(k,k) and only
113  // row k is changed.
114  // 2. If column k is already zero below the diagonal, AND A(k,k) is zero,
115  // then y=0,sum=0,u=0 and b is infinite...the transformation is undefined.
116  // However this column should be skipped (Biermann Appendix VII.B).
117  //
118  // Ref: Bierman, G.J. "Factorization Methods for Discrete Sequential
119  // Estimation," Academic Press, 1977.
120 
124  template <class T>
125  void SrifMU(Matrix<T>& R, Vector<T>& Z, Matrix<T>& A, unsigned int M=0)
126  throw(MatrixException)
127  {
128  if(A.cols() <= 1 || A.cols() != R.cols()+1 || Z.size() < R.rows()) {
129  if(A.cols() > 1 && R.rows() == 0 && Z.size() == 0) {
130  // create R and Z
131  R = Matrix<double>(A.cols()-1,A.cols()-1,0.0);
132  Z = Vector<double>(A.cols()-1,0.0);
133  }
134  else {
135  MatrixException me("Invalid input dimensions:\n R has dimension "
136  + StringUtils::asString<int>(R.rows()) + "x"
137  + StringUtils::asString<int>(R.cols()) + ",\n Z has length "
138  + StringUtils::asString<int>(Z.size()) + ",\n and A has dimension "
139  + StringUtils::asString<int>(A.rows()) + "x"
140  + StringUtils::asString<int>(A.cols()));
141  GPSTK_THROW(me);
142  }
143  }
144 
145  const T EPS=-T(1.e-200);
146  unsigned int m=M, n=R.rows();
147  if(m==0 || m > A.rows()) m=A.rows();
148  unsigned int np1=n+1; // if np1 = n, state vector Z is not updated
149  unsigned int i,j,k;
150  T dum, delta, beta;
151 
152  for(j=0; j<n; j++) { // loop over columns
153  T sum = T(0);
154  for(i=0; i<m; i++)
155  sum += A(i,j)*A(i,j); // sum squares of elements in this column below d
156  if(sum <= T(0)) continue;
157 
158  dum = R(j,j);
159  sum += dum * dum; // add diagonal element
160  sum = (dum > T(0) ? -T(1) : T(1)) * ::sqrt(sum);
161  delta = dum - sum;
162  R(j,j) = sum;
163 
164  if(j+1 > np1) break;
165 
166  beta = sum*delta; // beta must be negative
167  if(beta > EPS) continue;
168  beta = T(1)/beta;
169 
170  for(k=j+1; k<np1; k++) { // columns to right of diagonal
171  sum = delta * (k==n ? Z(j) : R(j,k));
172  for(i=0; i<m; i++)
173  sum += A(i,j) * A(i,k);
174  if(sum == T(0)) continue;
175 
176  sum *= beta;
177  if(k==n) Z(j) += sum*delta;
178  else R(j,k) += sum*delta;
179 
180  for(i=0; i<m; i++)
181  A(i,k) += sum * A(i,j);
182  }
183  }
184  } // end SrifMU

227  // Compute Cholesky decomposition of symmetric positive definite matrix using Crout
228  // algorithm. A = L*L^T where A and L are (nxn) and L is lower triangular reads:
229  // [ A00 A01 A02 ... A0n ] = [ L00 0 0 0 ... 0 ][ L00 L10 L20 ... L0n ]
230  // [ A10 A11 A12 ... A1n ] = [ L10 L11 0 0 ... 0 ][ 0 L11 L21 ... L1n ]
231  // [ A20 A21 A22 ... A2n ] = [ L20 L21 L22 0 ... 0 ][ 0 0 L22 ... L2n ]
232  // ... ... ...
233  // [ An0 An1 An2 ... Ann ] = [ Ln0 Ln1 Ln2 0 ... Lnn][ 0 0 0 ... Lnn ]
234  // but multiplying out gives
235  // A = [ L00^2
236  // [ L00*L10 L11^2+L10^2
237  // [ L00*L20 L11*L21+L10*L20 L22^2+L21^2+L20^2
238  // ...
239  // Aii = Lii^2 + sum(k=0,i-1) Lik^2
240  // Aij = Lij*Ljj + sum(k=0,j-1) Lik*Ljk
241  // These can be inverted by looping over columns, and filling L from diagonal down.

249  template <class T>
250  Matrix<T> lowerCholesky(const Matrix<T>& A) throw(MatrixException)
251  {
252  if(A.rows() != A.cols() || A.rows() == 0) {
253  MatrixException me("Invalid input dimensions: "
254  + StringUtils::asString<int>(A.rows()) + "x"
255  + StringUtils::asString<int>(A.cols()));
256  GPSTK_THROW(me);
257  }
258 
259  const unsigned int n=A.rows();
260  unsigned int i,j,k;
261  Matrix<T> L(n,n,0.0);
262 
263  for(j=0; j<n; j++) { // loop over cols
264  T d(A(j,j));
265  for(k=0; k<j; k++) d -= L(j,k)*L(j,k);
266  if(d <= 0.0) {
267  MatrixException e("Non-positive eigenvalue; "
268  "Cholesky requires positive-definite input");
269  GPSTK_THROW(e);
270  }
271  L(j,j) = ::sqrt(d);
272  for(i=j+1; i<n; i++) { // loop over rows below diagonal
273  d = A(i,j);
274  for(k=0; k<j; k++) d -= L(i,k)*L(j,k);
275  L(i,j) = d/L(j,j);
276  }
277  }
278 
279  return L;
280  }

322  // Invert the upper triangular matrix stored in the square matrix UT, using a very
323  // efficient algorithm. Throw MatrixException if the matrix is singular.
324  // If the pointers are defined, on exit (but not if an exception is thrown),
325  // they return the smallest and largest eigenvalues of the matrix.
326 
335  template <class T>
336  Matrix<T> inverseUT(const Matrix<T>& UT, T *ptrSmall=NULL, T *ptrBig=NULL)
337  throw(MatrixException)
338  {
339  if(UT.rows() != UT.cols() || UT.rows() == 0) {
340  MatrixException me("Invalid input dimensions: "
341  + StringUtils::asString<int>(UT.rows()) + "x"
342  + StringUtils::asString<int>(UT.cols()));
343  GPSTK_THROW(me);
344  }
345 
346  unsigned int i,j,k,n=UT.rows();
347  T big(0),small(0),sum,dum;
348  Matrix<T> Inv(UT);
349 
350  // start at the last row,col
351  dum = UT(n-1,n-1);
352  if(dum == T(0)) {
353  SingularMatrixException e("Singular matrix");
354  GPSTK_THROW(e);
355  }
356 
357  big = small = fabs(dum);
358  Inv(n-1,n-1) = T(1)/dum;
359  if(n == 1) return Inv; // 1x1 matrix
360  for(i=0; i<n-1; i++) Inv(n-1,i)=0;
361 
362  // now move to rows i = n-2 to 0
363  for(i=n-2;; i--) {
364  if(UT(i,i) == T(0))
365  throw SingularMatrixException("Singular matrix");
366 
367  if(fabs(UT(i,i)) > big) big = fabs(UT(i,i));
368  if(fabs(UT(i,i)) < small) small = fabs(UT(i,i));
369  dum = T(1)/UT(i,i);
370  Inv(i,i) = dum; // diagonal element first
371 
372  // now do off-diagonal elements (i,i+1) to (i,n-1)
373  for(j=i+1; j<n; j++) {
374  sum = T(0);
375  for(k=i+1; k<=j; k++)
376  sum += Inv(k,j) * UT(i,k);
377  Inv(i,j) = - sum * dum;
378  }
379  for(j=0; j<i; j++) Inv(i,j)=0;
380 
381  if(i==0) break; // NB i is unsigned, hence 0-1 = 4294967295!
382  }
383 
384  if(ptrSmall) *ptrSmall=small;
385  if(ptrBig) *ptrBig=big;
386 
387  return Inv;
388  }

391  //---------------------------------------------------------------------------------
392  // Given an upper triangular matrix UT, compute the symmetric matrix
393  // UT * transpose(UT) using a very efficient algorithm.
398  template <class T>
399  Matrix<T> UTtimesTranspose(const Matrix<T>& UT)
400  throw(MatrixException)
401  {
402  const unsigned int n=UT.rows();
403  if(n == 0 || UT.cols() != n) {
404  MatrixException me("Invalid input dimensions: "
405  + StringUtils::asString<int>(UT.rows()) + "x"
406  + StringUtils::asString<int>(UT.cols()));
407  GPSTK_THROW(me);
408  }
409 
410  unsigned int i,j,k;
411  T sum;
412  Matrix<T> S(n,n);
413 
414  for(i=0; i<n-1; i++) { // loop over rows of UT, except the last
415  sum = T(0); // diagonal element (i,i)
416  for(j=i; j<n; j++)
417  sum += UT(i,j)*UT(i,j);
418  S(i,i) = sum;
419  for(j=i+1; j<n; j++) { // loop over columns to right of (i,i)
420  sum = T(0);
421  for(k=j; k<n; k++)
422  sum += UT(i,k) * UT(j,k);
423  S(i,j) = S(j,i) = sum;
424  }
425  }
426  S(n-1,n-1) = UT(n-1,n-1)*UT(n-1,n-1); // the last diagonal element
427 
428  return S;
429  }
