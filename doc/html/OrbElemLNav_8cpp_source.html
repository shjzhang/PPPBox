83  void OrbElemLNav::loadData( const long SF1[10],
84  const long SF2[10],
85  const long SF3[10],
86  const SatID satIDArg,
87  const short XmitGPSWeek )
88  throw( InvalidParameter )
89  {
90  double ficOut[60];
91  double ficTemp[60];
92  EngNav engNavDummy; // Need this to initialize static data
93  if(!EngNav::subframeConvert(SF1, XmitGPSWeek, ficTemp))
94  {
95  InvalidParameter exc("Invalid SF1 Data");
96  GPSTK_THROW(exc);
97  }
98  for(int i = 0; i < 20; i++)
99  {
100  ficOut[i] = ficTemp[i];
101  }
102  if(!EngNav::subframeConvert(SF2, XmitGPSWeek, ficTemp))
103  {
104  InvalidParameter exc("Invalid SF2 Data");
105  GPSTK_THROW(exc);
106  }
107  for(int i = 0; i < 20; i++)
108  {
109  ficOut[i+20] = ficTemp[i];
110  }
111  if(!EngNav::subframeConvert(SF3, XmitGPSWeek, ficTemp))
112  {
113  InvalidParameter exc("Invalid SF3 Data");
114  GPSTK_THROW(exc);
115  }
116  for(int i = 0; i < 20; i++)
117  {
118  ficOut[i+40] = ficTemp[i];
119  }
120  //ficOut[19] = PRNID;
121 
122  // Fill in the variables unique to OrbElemLNav
123  HOWtime[0] = static_cast<long>( ficOut[2] );
124  ASalert[0] = static_cast<short>( ficOut[3] );
125  codeflags = static_cast<short>( ficOut[6] );
126  accFlag = static_cast<short>( ficOut[7] );
127  health = static_cast<short>( ficOut[8] );
128  IODC = static_cast<short>( ldexp( ficOut[9], -11 ) );
129  L2Pdata = static_cast<short>( ficOut[10] );
130  Tgd = ficOut[11];
131 
132  HOWtime[1] = static_cast<long>( ficOut[22] );
133  ASalert[1] = static_cast<short>( ficOut[23] );
134  IODE = static_cast<short>( ldexp( ficOut[25], -11 ) );
135  fitint = static_cast<short>( ficOut[34] );
136 
137  HOWtime[2] = static_cast<long>( ficOut[42] );
138  ASalert[2] = static_cast<short>( ficOut[43] );
139  AODO = static_cast<long>( ficOut[35] );
140 
141  short fullXmitWeekNum = static_cast<short>( ficOut[5] );
142 
143  // Fill in the variables in the OrbElem parent
144  // - - - First the simple copies - - -
145  double Toc = ficOut[12]; // OrbElem only stores fully qualified times
146  // see notes below.
147  af2 = ficOut[13];
148  af1 = ficOut[14];
149  af0 = ficOut[15];
150 
151  Crs = ficOut[26];
152  dn = ficOut[27];
153  M0 = ficOut[28];
154  Cuc = ficOut[29];
155  ecc = ficOut[30];
156  Cus = ficOut[31];
157  double AHalf = ficOut[32]; // Not a member of OrbElem. See notes below.
158  double Toe = ficOut[33]; // OrbElem only stores fully qualified times
159  // see notes below.
160 
161  Cic = ficOut[45];
162  OMEGA0 = ficOut[46];
163  Cis = ficOut[47];
164  i0 = ficOut[48];
165  Crc = ficOut[49];
166  w = ficOut[50];
167  OMEGAdot = ficOut[51];
168  idot = ficOut[53];
169 
170  // Store the satellite ID
171  satID = satIDArg;
172 
173  // - - - Now work on the things that need to be calculated - - -
174  // The observation ID has a type of navigation, but the
175  // carrier and code types are undefined. They could be
176  // L1/L2 C/A, P, Y,.....
177  obsID.type = ObsID::otNavMsg;
178  obsID.band = ObsID::cbUndefined;
179  obsID.code = ObsID::tcUndefined;
180 
181  // Beginning of Validity
182  // New concept. Admit the following.
183  // (a.) The collection system may not capture the data at earliest transmit.
184  // (b.) The collection system may not capture the three SFs consecutively.
185  // Consider a couple of IS-GPS-200 promises,
186  // (c.) By definition, beginning of validity == beginning of transmission.
187  // (d.) Except for uploads, cutovers will only happen on hour boundaries
188  // (e.) Cutovers can be detected by non-even Toc.
189  // (f.) Even uploads will cutover on a frame (30s) boundary.
190  // Therefore,
191  // 1.) If Toc is NOT even hour interval, pick lowest HOW time,
192  // round back to even 30s. That's the earliest Xmit time we can prove.
193  // NOTE: For the case where this is the SECOND SF 1/2/3 after an upload,
194  // this may yield a later time as such a SF 1/2/3 will be on a even
195  // hour boundary. Unfortunately, we have no way of knowing whether
196  // this item is first or second after upload without additional information.
197  // 2.) If Toc IS even two hour interval, pick time from SF 1,
198  // round back to nearest EVEN two hour boundary. This assumes collection
199  // SOMETIME in first hour of transmission. Could be more
200  // complete by looking at fit interval and IODC to more accurately
201  // determine earliest transmission time.
202  long longToc = (long) Toc;
203  double XmitSOW = 0.0;
204  if ( (longToc % 3600) != 0) // NOT an even two hour change
205  {
206  long leastHOW = HOWtime[0];
207  if (HOWtime[1]<leastHOW) leastHOW = HOWtime[1];
208  if (HOWtime[2]<leastHOW) leastHOW = HOWtime[2];
209  long Xmit = leastHOW - (leastHOW % 30);
210  XmitSOW = (double) Xmit;
211  }
212  else
213  {
214  // This assumes the data are collected sometime in the
215  // first two hours of transmission. For extended ops
216  // that might not be true. Think about this.
217  long Xmit = HOWtime[0] - HOWtime[0] % 7200;
218  XmitSOW = (double) Xmit;
219  }
220  beginValid = GPSWeekSecond( fullXmitWeekNum, XmitSOW, TimeSystem::GPS );
221 
222  // Determine Transmit Time
223  // Transmit time is the actual time this
224  // SF 1/2/3 sample was collected
225  long leastHOW = HOWtime[0];
226  if (HOWtime[1]<leastHOW) leastHOW = HOWtime[1];
227  if (HOWtime[2]<leastHOW) leastHOW = HOWtime[2];
228  long Xmit = leastHOW - (leastHOW % 30);
229  transmitTime = GPSWeekSecond( fullXmitWeekNum, (double)Xmit, TimeSystem::GPS );
230 
231  // Fully qualified Toe and Toc
232  // As broadcast, Toe and Toc are in GPS SOW and do not include
233  // the GPS week number. OrbElem (rightly) insists on having a
234  // Toe and Toc in CommonTime objects which implies determining
235  // the week number.
236  double timeDiff = Toe - XmitSOW;
237  short epochWeek = fullXmitWeekNum;
238  if (timeDiff < -HALFWEEK) epochWeek++;
239  else if (timeDiff > HALFWEEK) epochWeek--;
240 
241  ctToc = GPSWeekSecond(epochWeek, Toc, TimeSystem::GPS);
242  ctToe = GPSWeekSecond(epochWeek, Toe, TimeSystem::GPS);
243 
244  // End of Validity.
245  // The end of validity is calculated from the fit interval
246  // and the Toe. The fit interval is either trivial
247  // (if fit interval flag==0, fit interval is 4 hours)
248  // or a look-up table based on the IODC.
249  // Round the Toe value to the hour to elminate confusion
250  // due to possible "small offsets" indicating uploads
251  short fitHours = getLegacyFitInterval(IODC, fitint);
252  long ToeOffset = (long) Toe % 3600;
253  double adjToe = Toe; // Default case
254  if (ToeOffset)
255  {
256  adjToe += 3600.0 - (double)ToeOffset; // If offset, then adjust to remove it
257  }
258  long endFitSOW = adjToe + (fitHours/2)*3600;
259  short endFitWk = epochWeek;
260  if (endFitSOW >= FULLWEEK)
261  {
262  endFitSOW -= FULLWEEK;
263  endFitWk++;
264  }
265  endValid = GPSWeekSecond(endFitWk, endFitSOW, TimeSystem::GPS);
266 
267  // Semi-major axis and time-rate-of-change of semi-major axis
268  // Note: Legacy navigation message (SF 1/2/3) used SQRT(A).
269  // The CNAV and CNAV-2 formats use deltaA and Adot. As a
270  // result, OrbElem uses A and Adot and SQRT(A) and deltaA
271  // are converted to A at runtime.
272  A = AHalf * AHalf;
273  Adot = 0.0;
274  // Legacy nav doesn't have Rate of Chagen to corerction to mean motion,
275  // so set it to zero.
276  dndot = 0.0;
277 
278  // Health
279  // OrbElemLNav stores the full 8 bits health from the legacy
280  // navigation message. OrElem only stores the true/false,
281  // use/don't use based on whether the 8 bit health is 0 or non-zero
282  healthy = (health==0);
283 
284  // After all this is done, declare that data has been loaded
285  // into this object (so it may be used).
286  dataLoadedFlag = true;
287 
288  return;

303  void OrbElemLNav::adjustBeginningValidity()
304  {
305  if (!dataLoaded()) return;
306 
307  // The nominal beginning of validity is calculated from
308  // the fit interval and the Toe. The fit interval is
309  // either trivial (if fit interval flag==0, fit interval is 4 hours)
310  // or a look-up table based on the IODC.
311  short fitHours = getLegacyFitInterval(IODC, fitint);
312  long oneHalfInterval = ((long)fitHours/2) * 3600;
313 
314  // If we assume this is the SECOND set of elements in a set
315  // (which is an assumption of this function - see the .hpp) then
316  // the "small offset in Toe" will actually push the Toe-oneHalfInterval
317  // too early. For example, consider the following case.
318  // Toe : 19:59:44 (really near 20:00:00)
319  // first xMit : 18:00:00 (nominal)
320  // Blindly setting beginValid top Toe - 1/2 fit interval will
321  // result in 17:59:44. But 18:00:00 actually is the right answer
322  // because the -16 second offset is an artifact.
323  //
324  // Therefore, we are FIRST going to remove that offset,
325  // THEN determine beginValid.
326  long sow = (long) (static_cast<GPSWeekSecond>(ctToe)).sow;
327  short week = (static_cast<GPSWeekSecond>(ctToe)).week;
328  sow = sow + (3600 - (sow%3600));
329  CommonTime adjustedToe = GPSWeekSecond(week, (double) sow);
330  adjustedToe.setTimeSystem(TimeSystem::GPS);
331 
332  beginValid = adjustedToe - oneHalfInterval;
333  return;
334  }
