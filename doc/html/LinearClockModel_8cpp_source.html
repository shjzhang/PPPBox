66  void LinearClockModel::addEpoch(const ORDEpoch& oe)
67  throw(gpstk::InvalidValue)
68  {
69  ORDEpoch::ORDMap::const_iterator itr;
70  const gpstk::CommonTime t=oe.time;
71 
72  // Start off by getting an estimate of this epoch's clock
73  // note that this also sets the prn status map
74  gpstk::Stats<double> stat = simpleOrdClock(oe);
75  SvStatusMap& statusMap = prnStatus[t];
76  statusMap = status;
77 
78  double mean;
79  if (clockModel.N()==0)
80  {
81  double clkc = stat.Average();
82 #pragma unused(clkc)
83  startTime = endTime = baseTime = t;
84  tossCount = 0;
85  }
86 
87  const double deltaT = t-baseTime;
88 
89  if (t<startTime)
90  startTime=t;
91  if (t>endTime)
92  endTime=t;
93 
94  if (clockModel.N()>24)
95  mean = clockModel.Slope()*deltaT + clockModel.Intercept();
96  else
97  mean = stat.Average();
98 
99  if (std::abs(stat.Average() - mean) > 20)
100  {
101  cout << t
102  << " slope=" << setw(12) << clockModel.Slope()
103  << ", intercept=" << setw(8) << clockModel.Intercept()
104  << ", est=" << setw(8) << clockModel.Slope()*deltaT + clockModel.Intercept()
105  << ", N=" << setw(6) << clockModel.N()
106  << ", stdev=" << setw(6) << clockModel.StdDevY()
107  << endl;
108  tossCount++;
109  if (tossCount>5)
110  {
111  reset();
112  cout << "Reseting model" << endl;
113  }
114  }
115  else
116  {
117  tossCount=0;
118  for (itr = oe.ords.begin(); itr != oe.ords.end(); itr++)
119  if (statusMap[itr->second.getSvID()] == USED)
120  {
121  const double ord = itr->second.getORD();
122  clockModel.Add(deltaT, ord);
123  std::pair<const double,double> o(deltaT, ord);
124  clockObs.insert(o);
125  }
126  }
127 
128  std::multimap<double,double>::iterator i1,i2;
129  i1 = clockObs.begin();
130  while (i1!=clockObs.end())
131  {
132  i2=i1;
133  i1++;
134  double dt = i2->first;
135  double ord = i2->second;
136  if ((deltaT - dt)>1800)
137  {
138  clockObs.erase(i2);
139  clockModel.Subtract(dt, ord);
140  }
141  else
142  break;
143  }
144  }
