188  template<class T>
189  void BivarStats<T>::add(const T& x, const T& y)
190  {
191  if (ns == 0)
192  {
193  sumX = sumY = sumX2 = sumY2 = sumXY = T(0);
194  xMin = xMax = x;
195  yMin = yMax = y;
196  scaleX = scaleY = T(1);
197  }
198 
199  if (scaled)
200  {
201  if (scaleX==T(1) && x!=T()) scaleX=ABS(x);
202  if (scaleY==T(1) && y!=T()) scaleY=ABS(y);
203  T tx(x/scaleX);
204  T ty(y/scaleY);
205  sumX += tx;
206  sumY += ty;
207  sumX2 += tx*tx;
208  sumY2 += ty*ty;
209  sumXY += tx*ty;
210  }
211  else
212  {
213  sumX += x;
214  sumY += y;
215  sumX2 += x*x;
216  sumY2 += y*y;
217  sumXY += x*y;
218  }
219 
220  if(x < xMin) xMin=x;
221  if(x > xMax) xMax=x;
222  if(y < yMin) yMin=y;
223  if(y > yMax) yMax=y;
224  ns++;
225  }

405  BivarStats<T>& BivarStats<T>::operator+=(BivarStats<T>& S)
406  {
407  if(ns + S.ns == 0) return *this;
408  xMin = std::min(xMin, S.xMin);
409  xMax = std::max(xMax, S.xMax);
410  yMin = std::min(yMin, S.yMin);
411  yMax = std::max(yMax, S.yMax);
412  T xscaler( S.scaleX/scaleX ), yscaler( S.scaleY/scaleY );
413  sumX += xscaler * S.sumX;
414  sumY += yscaler * S.sumY;
415  sumX2 += xscaler * xscaler * S.sumX2;
416  sumY2 += yscaler * yscaler * S.sumY2;
417  sumXY += xscaler * yscaler * S.sumXY;
418  ns += S.ns;
419  return *this;
