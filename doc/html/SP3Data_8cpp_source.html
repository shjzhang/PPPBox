52  void SP3Data::reallyGetRecord(FFStream& ffs)
53  throw(exception, FFStreamError, StringException)
54  {
55  // cast the stream to be an SP3Stream
56  SP3Stream& strm = dynamic_cast<SP3Stream&>(ffs);
57 
58  // status says which records have been written already this call
59  // status = 1 if an epoch record was read
60  // status = 2 if an P or V record was read
61  // status = 3 if an EP or EV record was read
62  int status = 0;
63 
64  // version to be written out is determined by written header (stored in strm)
65  //bool isVerA = (strm.header.getVersion() == SP3Header::SP3a);
66  bool isVerC = (strm.header.getVersion() == SP3Header::SP3c);
67 
68  // correlation flag will be set if there is an EP/EV record
69  correlationFlag = false;
70 
71  // default for the 'c' arrays is all zeros
72  sdev[0] = sdev[1] = sdev[2] = sdev[3] = 0;
73  sig[0] = sig[1] = sig[2] = sig[3] = 0;
74  correlation[0] = correlation[1] = correlation[2]
75  = correlation[3] = correlation[4] = correlation[5] = 0;
76 
77  // TimeSystem for this stream
78  TimeSystem timeSystem;
79  timeSystem.fromString(strm.header.timeSystemString());
80 
81  // loop until an error occurs, or until the entire record (which may consist
82  // of two lines) is read.
83  bool unexpectedEOF=false;
84  while(1) {
85 
86  // set the time to be the last epoch read by the stream
87  time = strm.currentEpoch;
88 
89  // process the lastLine string stored in strm - contains the last line read
90  // empty record
91  if(strm.lastLine.size() < 3) { // 3 b/c it may contain "EOF"
92  // nothing in lastLine - do nothing here, get another line
93  ;
94  }
95 
96  // EOF has been read
97  else if(strm.lastLine.substr(0,3) == string("EOF")) {
98  // if a data record has already been processed during this call,
99  // then return here, and let the next call process this EOF.
100  // This gives the caller a chance to process the data before hitting EOF.
101 
102  // if an epoch record was processed this call, that's an error
103  if(status == 1) {
104  FFStreamError ffse("EOF was found immediately after epoch line");
105  GPSTK_THROW(ffse);
106  }
107 
108  // some other record was processed this call, so quit
109  if(status > 1) break;
110 
111  // this next read had better fail - if it does not, then there is
112  // an "EOF" that is followed by something other than the file EOF.
113  try {
114  strm.formattedGetLine(strm.lastLine, true); // true is 'expect EOF'
115  }
116  catch(EndOfFile& err) {
117  break; // normal exit; the stream will now be at eof()
118  }
119 
120  // the GetLine succeeded, so this is an error
121  FFStreamError err("EOF text found, followed by: " + strm.lastLine);
122  GPSTK_THROW(err);
123  }
124 
125  // Epoch line read
126  else if(strm.lastLine[0] == '*') {
127 
128  // if an epoch record was already processed this call, that's an error
129  // TD consider simply removing this if(status==1) entirely. Some SP3 files
130  // particularly those generated from a realtime stream, have consecutive
131  // epoch lines. Why would we not just ignore the epoch with no data?
132  if(status == 1) {
133  FFStreamError ffse("Consecutive epoch records found");
134  GPSTK_THROW(ffse);
135  }
136 
137  // if another record has been process during this call, quit now
138  if(status > 1) break;
139 
140  // process an epoch record
141  status = 1;
142 
143  // mark this record as non-data, in case another P|V record is not read
144  sat = SatID();
145 
146  // warn if the line is short but not too short
147  if(strm.lastLine.size() <= 30 && strm.lastLine.size() > 26)
148  strm.warnings.push_back(string("Warning (SP3 std): short epoch line: ")
149  + strm.lastLine);
150 
151  // throw if the line is short
152  if(strm.lastLine.size() <= 26) { // some igs files cut seconds short 30)
153  FFStreamError err("Invalid line length "
154  + asString(strm.lastLine.size()));
155  GPSTK_THROW(err);
156  }
157 
158  // parse the epoch line
159  RecType = strm.lastLine[0];
160  int year = asInt(strm.lastLine.substr(3,4));
161  int month = asInt(strm.lastLine.substr(8,2));
162  int dom = asInt(strm.lastLine.substr(11,2));
163  int hour = asInt(strm.lastLine.substr(14,2));
164  int minute = asInt(strm.lastLine.substr(17,2));
165  double second = asInt(strm.lastLine.substr(20,10));
166  CivilTime t;
167  try {
168  t = CivilTime(year, month, dom, hour, minute, second, timeSystem);
169  }
170  catch (gpstk::Exception& e) {
171  FFStreamError fe("Invalid time in:" + strm.lastLine);
172  GPSTK_THROW(fe);
173  }
174  time = strm.currentEpoch = static_cast<CommonTime>(t);
175  }
176 
177  // P or V record read
178  else if(strm.lastLine[0] == 'P' || strm.lastLine[0] == 'V') {
179 
180  // if another record has been process during this call, quit now
181  if(status > 0) break;
182 
183  // process this P|V
184  status = 2;
185  RecType = strm.lastLine[0]; // P or V
186 
187  // if its version c and the line is short (<80) but valid(>59),
188  // then add blanks at the end to make it 80 character.
189  if (isVerC && strm.lastLine.size() < 80 && strm.lastLine.size() > 59)
190  leftJustify(strm.lastLine,80);
191 
192  // throw if the line is short
193  if ((!isVerC && strm.lastLine.size() < 60) ||
194  (isVerC && strm.lastLine.size() < 80) ) {
195  FFStreamError err("Invalid line length ("
196  + asString(strm.lastLine.size())
197  + ") for line:\n" + strm.lastLine);
198  GPSTK_THROW(err);
199  }
200 
201  // parse the line
202  sat = static_cast<SatID>(SP3SatID(strm.lastLine.substr(1,3)));
203 
204  x[0] = asDouble(strm.lastLine.substr(4,14)); // XYZ
205  x[1] = asDouble(strm.lastLine.substr(18,14));
206  x[2] = asDouble(strm.lastLine.substr(32,14));
207  clk = asDouble(strm.lastLine.substr(46,14)); // Clock
208 
209  // the rest is version c only
210  if(isVerC) {
211  sig[0] = asInt(strm.lastLine.substr(61,2)); // sigma XYZ
212  sig[1] = asInt(strm.lastLine.substr(64,2));
213  sig[2] = asInt(strm.lastLine.substr(67,2));
214  sig[3] = asInt(strm.lastLine.substr(70,3)); // sigma clock
215 
216  if(RecType == 'P') { // P flags
217  clockEventFlag = clockPredFlag
218  = orbitManeuverFlag = orbitPredFlag = false;
219  if(strm.lastLine[74] == 'E') clockEventFlag = true;
220  if(strm.lastLine[75] == 'P') clockPredFlag = true;
221  if(strm.lastLine[78] == 'M') orbitManeuverFlag = true;
222  if(strm.lastLine[79] == 'P') orbitPredFlag = true;
223  }
224  }
225  }
226 
227  // EP or EV correlation record read
228  else if(strm.lastLine[0] == 'E' &&
229  (strm.lastLine[1] == 'P' || strm.lastLine[1] == 'V'))
230  {
231  // throw if correlation record did not follow corresponding P|V record
232  if(status != 2 || strm.lastLine[1] != RecType) {
233  Exception e("correlation EP|V record mismatched with previous record");
234  GPSTK_THROW(e);
235  }
236 
237  // process EP|V record
238  status = 3;
239 
240  // throw if line is short
241  if(strm.lastLine.size()<80) {
242  FFStreamError err("Invalid SP3c correlation line length ("
243  + asString(strm.lastLine.size())
244  + ") for line:\n" + strm.lastLine);
245  GPSTK_THROW(err);
246  }
247 
248  // parse the line
249  sdev[0] = abs(asInt(strm.lastLine.substr(4,4)));
250  sdev[1] = abs(asInt(strm.lastLine.substr(9,4)));
251  sdev[2] = abs(asInt(strm.lastLine.substr(14,4)));
252  sdev[3] = abs(asInt(strm.lastLine.substr(19,7)));
253  correlation[0] = asInt(strm.lastLine.substr(27,8));
254  correlation[1] = asInt(strm.lastLine.substr(36,8));
255  correlation[2] = asInt(strm.lastLine.substr(45,8));
256  correlation[3] = asInt(strm.lastLine.substr(54,8));
257  correlation[4] = asInt(strm.lastLine.substr(63,8));
258  correlation[5] = asInt(strm.lastLine.substr(72,8));
259 
260  // tell the caller that correlation data is now present
261  correlationFlag = true;
262  }
263 
264  else { // Unknown record
265  FFStreamError err("Unknown record label " + strm.lastLine.substr(0,2));
266  GPSTK_THROW(err);
267  }
268 
269  // be tolerant of files without EOF -- IGS!
270  // if previous iteration of the loop found unexpected EOF, then quit here.
271  if(unexpectedEOF) {
272  // add a warning
273  strm.warnings.push_back(string("Warning (SP3 std): Unexpected EOF"));
274 
275  // clear the buffer so it won't be (re)processed next call
276  strm.lastLine = string("EOF");
277 
278  // clear the eof bit and return, so the user will process this SP3Data
279  strm.clear(ios::eofbit);
280  status = 3;
281  }
282  else { // normal flow
283  // read next line into the lastLine
284  try {
285  strm.formattedGetLine(strm.lastLine);
286  }
287  catch(FFStreamError& err) {
288  string what = err.what().substr(0,21);
289  //cout << "Found unexpected EOF" << endl;
290  //cout << "what is " << what << endl;
291  //cout << "Here is the buffer:\n" << strm.lastLine << endl;
292  if(what == string("text 0:Unexpected EOF")) {
293  unexpectedEOF = true;
294 
295  // there could still be unprocessed data in the buffer:
296  if(strm.lastLine.size() < 3) status = 3; // nothing there, quit
297  else status = 0; // go back and process it
298  }
299  else
300  GPSTK_RETHROW(err);
301  }
302  }
303 
304  if(status == 3) break; // quit if found EOF or EP|EV was processed, but
305  // go back if lastLine was empty (0)
306  // or if epoch was processed (1)
307  // or if P|V was processed (2)
308 
309  } // end while loop processing records
310 
311  } // end reallyGetRecord()

313  void SP3Data::reallyPutRecord(FFStream& ffs) const
314  throw(exception, FFStreamError, StringException)
315  {
316  string line;
317 
318  // cast the stream to be an SP3Stream
319  SP3Stream& strm = dynamic_cast<SP3Stream&>(ffs);
320 
321  // version to be written out is determined by written header (stored in strm)
322  bool isVerA = (strm.header.getVersion() == SP3Header::SP3a);
323  bool isVerC = (strm.header.getVersion() == SP3Header::SP3c);
324 
325  // output Epoch Header Record
326  if(RecType == '*') {
327  CivilTime civTime(time);
328  line = "* ";
329  line += civTime.printf(" %4Y %2m %2d %2H %2M");
330  line += " " + rightJustify(civTime.printf("%.8f"),11);
331  }
332 
333  // output Position and Clock OR Velocity and Clock Rate Record
334  else {
335  line = RecType; // P or V
336  if(isVerA) {
337  if(sat.system != SatID::systemGPS) {
338  FFStreamError fse("Cannot output non-GPS to SP3a");
339  GPSTK_THROW(fse);
340  }
341  line += rightJustify(asString(sat.id),3);
342  }
343  else
344  line += static_cast<SP3SatID>(sat).toString(); // sat ID
345 
346  line += rightJustify(asString(x[0],6),14); // XYZ
347  line += rightJustify(asString(x[1],6),14);
348  line += rightJustify(asString(x[2],6),14);
349  line += rightJustify(asString(clk,6),14); // Clock
350 
351  if(isVerC) {
352  line += rightJustify(asString(sig[0]),3); // sigma XYZ
353  line += rightJustify(asString(sig[1]),3);
354  line += rightJustify(asString(sig[2]),3);
355  line += rightJustify(asString(sig[3]),4); // sigma Clock
356 
357  if(RecType == 'P') { // flags or blanks
358  line += string(" ");
359  line += (clockEventFlag ? string("E") : string(" "));
360  line += (clockPredFlag ? string("P") : string(" "));
361  line += string(" ");
362  line += (orbitManeuverFlag ? string("M") : string(" "));
363  line += (orbitPredFlag ? string("P") : string(" "));
364  }
365  }
366 
367  // if version is 'c' and correlation flag is set,
368  // then output the P|V Correlation Record
369  if(isVerC && correlationFlag) {
370 
371  // first output the P|V record you just built
372  strm << line << endl;
373  strm.lineNumber++;
374 
375  // now build and output the correlation record
376  if(RecType == 'P') // P or V
377  line = "EP ";
378  else
379  line = "EV ";
380  line += rightJustify(asString(sdev[0]),5); // stddev X
381  line += rightJustify(asString(sdev[1]),5); // stddev Y
382  line += rightJustify(asString(sdev[2]),5); // stddev Z
383  line += rightJustify(asString(sdev[3]),8); // stddev Clk
384  for(int i=0; i<6; i++) // correlations
385  line += rightJustify(asString(correlation[i]),9);
386  }
387  }
388 
389  // write the line just built
390  strm << line << endl;
391  strm.lineNumber++;
392 
393  } // end reallyPutRecord()

395  void SP3Data::dump(ostream& s, bool includeC) const throw()
396  {
397  // dump record type (PV*), sat id, and current epoch
398  s << RecType << " " << static_cast<SP3SatID>(sat).toString() << " "
399  << (static_cast<CivilTime>(time)).printf("%Y/%02m/%02d %2H:%02M:%06.3f")
400  << " = "
401  << (static_cast<GPSWeekSecond>(time)).printf("%F/%10.3g");
402 
403  if(RecType != '*') { // not epoch line
404  s << fixed << setprecision(6)
405  << " X=" << setw(14) << x[0] // XYZ
406  << " Y=" << setw(14) << x[1]
407  << " Z=" << setw(14) << x[2]
408  << " C=" << setw(14) << clk; // clk
409 
410  if(includeC) {
411  s << " sX=" << setw(2) << sig[0] // sigma XYZ
412  << " sY=" << setw(2) << sig[1]
413  << " sZ=" << setw(2) << sig[2]
414  << " sC=" << setw(3) << sig[3]; // sigma clock
415 
416  if(RecType == 'P') // flags
417  s << " " << (clockEventFlag ? "clockEvent" : "-")
418  << " " << (clockPredFlag ? "clockPrediction" : "-")
419  << " " << (orbitManeuverFlag ? "orbitManeuver" : "-")
420  << " " << (orbitPredFlag ? "orbitPrediction" : "-");
421 
422  if(correlationFlag) // stddevs and correlations
423  s << endl
424  << " and E" << RecType
425  << " cXX=" << setw(4) << sdev[0]
426  << " cYY=" << setw(4) << sdev[1]
427  << " cZZ=" << setw(4) << sdev[2]
428  << " cCC=" << setw(7) << sdev[3]
429  << " cXY=" << setw(8) << correlation[0]
430  << " cXZ=" << setw(8) << correlation[1]
431  << " cXC=" << setw(8) << correlation[2]
432  << " cYZ=" << setw(8) << correlation[3]
433  << " cYC=" << setw(8) << correlation[4]
434  << " cZC=" << setw(8) << correlation[5];
435  }
436  }
437 
438  s << endl;
439 
440  } // end dump()
