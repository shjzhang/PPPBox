52 using namespace std;

55  // Constructor
56  GenXSequence::GenXSequence( const unsigned int initialState,
57  const unsigned int tapRegister,
58  const int initialLengthOfSequence,
59  const int maxDelay )
60  {
61 
62  /*
63  * NOTE: the x register functions assume a LSB-to-MSB shift. Therefore,
64  * the tap register definition and the initialization definition for the
65  * X register correspond to the diagrams in ICD-GPS-200, assuming
66  * "stage 0" is the LSB. That means the diagrams have to be read from
67  * LSB (LEFT) -> MSB (RIGHT). The corresponding text doesn't have this
68  * problem.
69  */
70  unsigned int mask12bits[12] = { 0x0001, 0x0002, 0x0004, 0x0008,
71  0x0010, 0x0020, 0x0040, 0x0080,
72  0x0100, 0x0200, 0x0400, 0x0800 };
73 
74  unsigned int reg = initialState;
75  lengthOfSequence = initialLengthOfSequence;
76  maxOfSequence = lengthOfSequence + maxDelay;
77 
78  uint32_t output;
79  word_num = 0;
80  bit_num = 0;
81  int andBits;
82  int i;
83 
84  // Clear the output array
85  for ( i=0; i<MAX_WORD; ++i ) bits[i] = 0x00000000;
86  debugPrint = false;
87 
88  for ( i=0; i<lengthOfSequence ; ++i)
89  {
90  // Get current output and store it away
91  if ( (reg & 0x0800) != 0 ) output = 0x00000001;
92  else output = 0x00000000;
93  addBitToSequence( output );
94 
95  // Calculate next input bit
96  andBits = reg & tapRegister;
97  reg <<= 1;
98  reg &= 0x0FFF;
99  int cnt = 0;
100  for ( int bit12cnt=0; bit12cnt<12; ++bit12cnt)
101  {
102  if (( andBits & mask12bits[bit12cnt] ) != 0) ++cnt;
103  }
104  int newBit = cnt % 2;
105  reg |= newBit;
106  }
107 
108  // Fill delay bits with copies of the last ouptut bit
109  for ( i=0; i<maxDelay; ++i) addBitToSequence( output );
110 
111  // When finished, make sure the last word (which is probably
112  // a partially-filled word) is left-justified.
113  if (bit_num>0) bits[word_num] <<= (MAX_BIT-bit_num);
114  }
115 
116  // Private helper method to avoid duplicate code.
117  void GenXSequence::addBitToSequence( uint32_t newBit )
118  {
119  // Left shift any pre-existing data, then OR on the new
120  // data (assumed to be right-justified).
121  bits[word_num] <<= 1;
122  bits[word_num] |= newBit;
123 
124  /*
125  Increment bit pointer and check for word overflow.
126  NOTE: Overflow of the WORD pointer is a "programming problem"
127  that is unrecoverable and should NEVER happen in production.
128  */
129  ++bit_num;
130  if (bit_num>=MAX_BIT)
131  {
132  ++word_num;
133  bit_num=0;
134  }
135  }
136 
137  /*
138  Given a bit position within the X sequence (numbered starting at 0),
139  return the next 32 bits. Note: if there are insufficient bits left
140  to fill the request, wrap around to the beginning of the sequence.
141  Note that the location of the wrap around can be modified using the
142  function GenXSequence::setEndOfSequence( int los );
143  */
144  uint32_t GenXSequence::operator[] ( int ia )
145  {
146  uint32_t retArg = 0x00000000;
147  int i = ia;
148  if (i >= lengthOfSequence) i = i % lengthOfSequence;
149  int ndx1 = i / MAX_BIT;
150  int offset = i % MAX_BIT;
151  if ( (i+MAX_BIT) <= lengthOfSequence )
152  {
153  if (offset==0) retArg = bits[ndx1];
154  else retArg = merge( bits[ndx1], bits[ndx1+1], offset );
155  }
156  /*
157  Complicated case when coming up to end of sequence. May have to
158  put together parts of up to three words to get 32 bits. Two words
159  at the end of sequence plus "wrap around" bits from beginning of
160  sequence.
161  First: If end of sequence doesn't fall in current word, use up
162  remaining bits in the current word.
163  Second: use bits up to the end of sequence.
164  Third: fill remaining bits from the beginning of the sequence.
165  */
166  else
167  {
168  int numRemainingInSequence = lengthOfSequence - i;
169  int numRemainingInWord = MAX_BIT - offset;
170  int numFilled = 0;
171 
172  /*
173  Get bits (if any) from next-to-last word.
174  */
175  if (numRemainingInWord<numRemainingInSequence)
176  {
177  retArg = bits[ndx1++] << offset;
178  numRemainingInSequence -= numRemainingInWord;
179  numFilled = numRemainingInWord;
180  }
181 
182  uint32_t temp = bits[ndx1];
183  /*
184  Get bits from last word
185  Case 1: No bits from previous word, need only "middle" section
186  from last word.
187  Case 2: Need all bits available from final word (may or
188  may not have bits from preceding word)
189  */
190  if (numFilled==0 && offset!=0)
191  {
192  temp <<= offset; // Move to left to clear excess msb
193  temp >>= (MAX_BIT-numRemainingInSequence); // Shift right to clear excess lsb
194  temp <<= (MAX_BIT - (numRemainingInSequence+numFilled) );
195  retArg |= temp;
196  }
197  else
198  {
199  temp >>= (MAX_BIT-numRemainingInSequence);
200  temp <<= (MAX_BIT-(numRemainingInSequence+numFilled));
201  retArg |= temp;
202  }
203 
204  // Finally, add bits from the "wraparound" word at the
205  // beginning of the array.
206  retArg |= bits[0] >> (numRemainingInSequence+numFilled);
207  }
208  return(retArg);
209  }
