136  * don't have azimuth dependent phase center patterns.
137  *
138  * This method returns a Triple, in UEN system, with the

187  /* Get antenna phase center variation.
188  *
189  * This method returns a Triple, in UEN system, with the
190  * elevation and azimuth-dependent phase center variation.
191  *
192  * @param[in] freq Frequency
193  * @param[in] elevation Elevation (degrees)
194  * @param[in] azimuth Azimuth (degrees)
195  *
196  * @warning The phase center variation Triple is in UEN system.
197  */
198  Triple Antenna::getAntennaPCVariation( frequencyType freq,
199  double elevation,
200  double azimuth ) const
201  throw(InvalidRequest)
202  {
203 
204  // The angle should be measured respect to zenith
205  double angle( 90.0 - elevation );
206 
207  // Check that angle is within limits
208  if( ( angle < zen1 ) ||
209  ( angle > zen2 ) )
210  {
211  InvalidRequest e("Elevation is out of allowed range.");
212  GPSTK_THROW(e);
213  }
214 
215  // Reduce azimuth to 0 <= azimuth < 360 interval
216  while( azimuth < 0.0 )
217  {
218  azimuth += 360.0;
219  }
220  while( azimuth >= 360.0 )
221  {
222  azimuth -= 360.0;
223  }
224 
225  // Look for this frequency in pcMap
226  // Define iterator
227  PCDataMap::const_iterator it( pcMap.find(freq) );
228  if( it != pcMap.end() )
229  {
230 
231  // Get the right azimuth interval
232  const double lowerAzimuth( std::floor(azimuth/dazi) * dazi );
233  const double upperAzimuth( lowerAzimuth + dazi );
234 
235  // Look for data vectors
236  AzimuthDataMap::const_iterator it2( (*it).second.find(lowerAzimuth) );
237  AzimuthDataMap::const_iterator it3( (*it).second.find(upperAzimuth) );
238 
239  // Find the fraction from 'lowerAzimuth'
240  const double fractionalAzimuth( ( azimuth - lowerAzimuth ) /
241  ( upperAzimuth - lowerAzimuth ) );
242 
243  // Check if 'azimuth' exactly corresponds to a value in the map
244  if( fractionalAzimuth == 0.0 )
245  {
246  // Check if there is data for 'lowerAzimuth'
247  if( it2 != (*it).second.end() )
248  {
249  // Get the normalized angle
250  const double normalizedAngle( (angle-zen1)/dzen );
251 
252  // Return result. Only the "Up" component is important.
253  Triple result( linearInterpol( (*it2).second, normalizedAngle ),
254  0.0,
255  0.0 );
256 
257  return result;
258 
259  }
260  else
261  {
262  InvalidRequest e("No data was found for this azimuth.");
263  GPSTK_THROW(e);
264  }
265  }
266  else
267  {
268  // We have to interpolate
269  // Check if there is data for 'lowerAzimuth' and 'upperAzimuth'
270  if( it2 != (*it).second.end() &&
271  it3 != (*it).second.end() )
272  {
273  // Get the normalized angle
274  const double normalizedAngle( (angle-zen1)/dzen );
275 
276  // Find values corresponding to both azimuths
277  double val1( linearInterpol( (*it2).second, normalizedAngle ) );
278  double val2( linearInterpol( (*it3).second, normalizedAngle ) );
279 
280  // Return result. Only the "Up" component is important.
281  Triple result( ( val1 + (val2-val1) * fractionalAzimuth ),
282  0.0,
283  0.0 );
284 
285  return result;
286 
287  }
288  else
289  {
290  InvalidRequest e("Not enough data was found for this azimuth.");
291  GPSTK_THROW(e);
292  }
293  }
294 
295  }
296  else
297  {
298 
299  InvalidRequest e("No data was found for this frequency.");
300  GPSTK_THROW(e);
301 
302  } // End of 'if( it != pcMap.end() )...'
303 
304  } // End of method 'Antenna::getAntennaPCVariation()'

379  /* Linear interpolation as function of normalized angle
380  *
381  * @param[in] dataVector std::vector holding data.
382  * @param[in] normalizedAngle Normalized angle.
383  *
384  * 'normalizedAngle' is a value corresponding to the original angle
385  * divided by the angle interval.
386  */
387  double Antenna::linearInterpol( const std::vector<double>& dataVector,
388  double normalizedAngle ) const
389  {
390 
391  // Get the index value 'normalizedAngle' is equivalent to
392  int index( static_cast<int>( std::floor(normalizedAngle) ) );
393 
394  // Find the fraction from 'index'
395  double fraction( normalizedAngle - std::floor(normalizedAngle) );
396 
397  // Check if 'normalizedAngle' is exactly a value in the map
398  if( fraction == 0.0 )
399  {
400  // Return result
401  return dataVector[index];
402  }
403  else
404  {
405  // In this case, we have to interpolate
406  double val1( dataVector[index] );
407  double val2( dataVector[index+1] );
408 
409  // Return result
410  return ( val1 + (val2-val1) * fraction );
411  }
412 
413  } // End of method 'Antenna::elevationInterpol()'
