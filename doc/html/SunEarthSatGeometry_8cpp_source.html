59 // Given a Position, compute unit (ECEF) vectors in the Up, East and North directions
60 // at that position. Use geodetic coordinates, i.e. 'up' is perpendicular to the
61 // geoid. Return the vectors in the form of a 3x3 Matrix<double>, this is in fact the
62 // rotation matrix that will take an ECEF vector into an 'up-east-north' vector.
63 Matrix<double> UpEastNorth(Position& P, bool geocentric) throw(Exception)
64 {
65 try {
66  Matrix<double> R = NorthEastUp(P,geocentric);
67  for(int i=0; i<3; i++) { double r=R(0,i); R(0,i)=R(2,i); R(2,i)=r; }
68  return R;
69 }
70 catch(Exception& e) { GPSTK_RETHROW(e); }
71 catch(exception& e) { Exception E("std except: "+string(e.what())); GPSTK_THROW(E); }
72 catch(...) { Exception e("Unknown exception"); GPSTK_THROW(e); }
73 }

85 // Same as UpEastNorth(), but with rows re-ordered.
86 Matrix<double> NorthEastUp(Position& P, bool geocentric) throw(Exception)
87 {
88 try {
89  Matrix<double> R(3,3);
90  P.transformTo(geocentric ? Position::Geodetic : Position::Geocentric);
91 
92  double lat = (geocentric ? P.getGeodeticLatitude() : P.getGeocentricLatitude())
93  * DEG_TO_RAD; // rad N
94  double lon = P.getLongitude() * DEG_TO_RAD; // rad E
95  double ca = ::cos(lat);
96  double sa = ::sin(lat);
97  double co = ::cos(lon);
98  double so = ::sin(lon);
99 
100  // This is the rotation matrix which will
101  // transform X=(x,y,z) into (R*X)(north,east,up)
102  R(0,0) = -sa*co; R(0,1) = -sa*so; R(0,2) = ca;
103  R(1,0) = -so; R(1,1) = co; R(1,2) = 0.;
104  R(2,0) = ca*co; R(2,1) = ca*so; R(2,2) = sa;
105 
106  // The rows of R are also the unit vectors, in ECEF, of north,east,up;
107  // R = (N && E && U) = transpose(N || E || U).
108  //Vector<double> N = R.rowCopy(0);
109  //Vector<double> E = R.rowCopy(1);
110  //Vector<double> U = R.rowCopy(2);
111 
112  return R;
113 }
114 catch(Exception& e) { GPSTK_RETHROW(e); }
115 catch(exception& e) { Exception E("std except: "+string(e.what())); GPSTK_THROW(E); }
116 catch(...) { Exception e("Unknown exception"); GPSTK_THROW(e); }
117 }

127 // -----------------------------------------------------------------------------------
128 // Generate a 3x3 rotation Matrix, for direct rotations about one axis
129 // (for XYZ, axis=123), given the rotation angle in radians;
130 // @param angle in radians.
131 // @param axis 1,2,3 as rotation about X,Y,Z.
132 // @return Rotation matrix (3x3).
133 // @throw InvalidInput if axis is anything other than 1, 2 or 3.
134 Matrix<double> SingleAxisRotation(double angle, const int axis)
135  throw(Exception)
136 {
137 try {
138  if(axis < 1 || axis > 3) {
139  Exception e(string("Invalid axis (1,2,3 <=> X,Y,Z): ")
140  + StringUtils::asString(axis));
141  GPSTK_THROW(e);
142  }
143  Matrix<double> R(3,3,0.0);
144 
145  int i1=axis-1; // axis = 1 : 0,1,2
146  int i2=i1+1; if(i2 == 3) i2=0; // axis = 2 : 1,2,0
147  int i3=i2+1; if(i3 == 3) i3=0; // axis = 3 : 2,0,1
148 
149  R(i1,i1) = 1.0;
150  R(i2,i2) = R(i3,i3) = ::cos(angle);
151  R(i3,i2) = -(R(i2,i3) = ::sin(angle));
152 
153  return R;
154 }
155 catch(Exception& e) { GPSTK_RETHROW(e); }
156 catch(exception& e) { Exception E("std except: "+string(e.what())); GPSTK_THROW(E); }
157 catch(...) { Exception e("Unknown exception"); GPSTK_THROW(e); }
158 }

161 // Consider the sun and the earth as seen from the satellite. Let the sun be a circle
162 // of angular radius r, center in direction s, and the earth be a (larger) circle
163 // of angular radius R, center in direction e. The circles overlap if |e-s| < R+r;
164 // complete overlap if |e-s| < R-r. The satellite is in penumbra if R-r < |e-s| < R+r,// it is in umbra if |e-s| < R-r.
165 // Let L == |e-s|. What is the area of overlap in penumbra : R-r < L < R+r ?
166 // Call the two points where the circles intersect p1 and p2. Draw a line from e to s;
167 // call the points where this line intersects the two circles r1 and R1, respectively.
168 // Draw lines from e to s, e to p1, e to p2, s to p1 and s to p2. Call the angle
169 // between e-s and e-p1 alpha, and that between s-e and s-p1, beta.
170 // Draw a rectangle with top and bottom parallel to e-s passing through p1 and p2,
171 // and with sides passing through s and r1. Similarly for e and R1. Note that the
172 // area of intersection lies within the intersection of these two rectangles.
173 // Call the area of the rectangle outside the circles A and B. The height H of the
174 // rectangles is
175 // H = 2Rsin(alpha) = 2rsin(beta)
176 // also L = rcos(beta)+Rcos(alpha)
177 // The area A will be the area of the rectangle
178 // minus the area of the wedge formed by the angle 2*alpha
179 // minus the area of the two triangles which meet at s :
180 // A = RH - (2alpha/2pi)*pi*R*R - 2*(1/2)*(H/2)Rcos(alpha)
181 // Similarly
182 // B = rH - (2beta/2pi)*pi*r*r - 2*(1/2)*(H/2)rcos(beta)
183 // The area of intersection will be the area of the rectangular intersection
184 // minus the area A
185 // minus the area B
186 // Intersection = H(R+r-L) - A - B
187 // = HR+Hr-HL -HR+alpha*R*R+(H/2)Rcos(alpha) -Hr+beta*r*r+(H/2)rcos(beta)
188 // Cancel terms, and substitute for L using above equation L = ..
189 // = -(H/2)rcos(beta)-(H/2)Rcos(alpha)+alpha*R*R+beta*r*r
190 // substitute for H/2
191 // = -R*R*sin(alpha)cos(alpha)-r*r*sin(beta)cos(beta)+alpha*R*R+beta*r*r
192 // Intersection = R*R*[alpha-sin(alpha)cos(alpha)]+r*r*[beta-sin(beta)cos(beta)]
193 // Solve for alpha and beta in terms of R, r and L using the H and L relations above
194 // (r/R)cos(beta)=(L/R)-cos(alpha)
195 // (r/R)sin(beta)=sin(alpha)
196 // so
197 // (r/R)^2 = (L/R)^2 - (2L/R)cos(alpha) + 1
198 // cos(alpha) = (R/2L)(1+(L/R)^2-(r/R)^2)
199 // cos(beta) = (L/r) - (R/r)cos(alpha)
200 // and 0 <= alpha or beta <= pi
201 //
202 // AngRadEarth angular radius of the earth as seen at the satellite
203 // AngRadSun angular radius of the sun as seen at the satellite
204 // AngSeparation angular distance of the sun from the earth
205 // return fraction (0 <= f <= 1) of area of sun covered by earth
206 // units only need be consistent
207 double ShadowFactor(double AngRadEarth, double AngRadSun, double AngSeparation)
208 {
209 try {
210  if(AngSeparation >= AngRadEarth+AngRadSun) return 0.0;
211  if(AngSeparation <= fabs(AngRadEarth-AngRadSun)) return 1.0;
212  double r=AngRadSun, R=AngRadEarth, L=AngSeparation;
213  if(AngRadSun > AngRadEarth) { r=AngRadEarth; R=AngRadSun; }
214  double cosalpha = (R/L)*(1.0+(L/R)*(L/R)-(r/R)*(r/R))/2.0;
215  double cosbeta = (L/r) - (R/r)*cosalpha;
216  double sinalpha = ::sqrt(1-cosalpha*cosalpha);
217  double sinbeta = ::sqrt(1-cosbeta*cosbeta);
218  double alpha = ::asin(sinalpha);
219  double beta = ::asin(sinbeta);
220  double shadow = r*r*(beta-sinbeta*cosbeta)+R*R*(alpha-sinalpha*cosalpha);
221  shadow /= ::acos(-1.0)*AngRadSun*AngRadSun;
222  return shadow;
223 }
224 catch(Exception& e) { GPSTK_RETHROW(e); }
225 catch(exception& e) { Exception E("std except: "+string(e.what())); GPSTK_THROW(E); }
226 catch(...) { Exception e("Unknown exception"); GPSTK_THROW(e); }
227 }

229 // -----------------------------------------------------------------------------------
230 // Compute the satellite attitude, given the time and the satellite position SV.
231 // If the SolarSystem is valid, use it; otherwise use SolarPosition.
232 // See two versions of SatelliteAttitude() for the user interface.
233 // Return a 3x3 Matrix which contains, as rows, the unit (ECEF) vectors X,Y,Z in the
234 // body frame of the satellite, namely
235 // Z = along the boresight (i.e. towards Earth center),
236 // Y = perpendicular to both Z and the satellite-sun direction, and
237 // X = completing the orthonormal triad. X will generally point toward the sun.
238 // Thus this rotation matrix R * (ECEF XYZ vector) = components in body frame, and
239 // R.transpose() * (sat. body. frame vector) = ECEF XYZ components.
240 // Also return the shadow factor = fraction of sun's area not visible to satellite.
241 Matrix<double> doSatAtt(const CommonTime& tt, const Position& SV,
242  const SolarSystem& SSEph, const EarthOrientation& EO,
243  double& sf)
244  throw(Exception)
245 {
246  try {
247  int i;
248  double d,svrange,DistSun,AngRadSun,AngRadEarth,AngSeparation;
249  Position X,Y,Z,T,S,Sun;
250  Matrix<double> R(3,3);
251 
252  // Z points from satellite to Earth center - along the antenna boresight
253  Z = SV;
254  Z.transformTo(Position::Cartesian);
255  svrange = Z.mag();
256  d = -1.0/Z.mag();
257  Z = d * Z; // reverse and normalize Z
258 
259  // get the Sun's position
260  if(SSEph.JPLNumber() > -1) {
261  //SolarSystem& mySSEph=const_cast<SolarSystem&>(SSEph);
262  Sun = const_cast<SolarSystem&>(SSEph).WGS84Position(SolarSystem::Sun,tt,EO);
263  }
264  else {
265  double AR;
266  Sun = SolarPosition(tt, AR);
267  }
268  DistSun = Sun.radius();
269 
270  // apparent angular radius of sun = 0.2666/distance in AU (deg)
271  AngRadSun = 0.2666/(DistSun/149598.0e6);
272  AngRadSun *= DEG_TO_RAD;
273 
274  // angular radius of earth at sat
275  AngRadEarth = ::asin(6378137.0/svrange);
276 
277  // T points from satellite to sun
278  T = Sun; // vector earth to sun
279  T.transformTo(Position::Cartesian);
280  S = SV;
281  S.transformTo(Position::Cartesian);
282  T = T - S; // sat to sun=(E to sun)-(E to sat)
283  d = 1.0/T.mag();
284  T = d * T; // normalize T
285 
286  AngSeparation = ::acos(Z.dot(T)); // apparent angular distance, earth
287  // to sun, as seen at satellite
288  // is satellite in eclipse?
289  try { sf = ShadowFactor(AngRadEarth, AngRadSun, AngSeparation); }
290  catch(Exception& e) { GPSTK_RETHROW(e); }
291 
292  // Y is perpendicular to Z and T, such that ...
293  Y = Z.cross(T);
294  d = 1.0/Y.mag();
295  Y = d * Y; // normalize Y
296 
297  // ... X points generally in the direction of the sun
298  X = Y.cross(Z); // X will be unit vector
299  if(X.dot(T) < 0) { // need to reverse X, hence Y also
300  X = -1.0 * X;
301  Y = -1.0 * Y;
302  }
303 
304  // fill the matrix and return it
305  for(i=0; i<3; i++) {
306  R(0,i) = X[i];
307  R(1,i) = Y[i];
308  R(2,i) = Z[i];
309  }
310 
311  return R;
312  }
313  catch(Exception& e) { GPSTK_RETHROW(e); }
314  catch(exception& e) {Exception E("std except: "+string(e.what())); GPSTK_THROW(E);}
315  catch(...) { Exception e("Unknown exception"); GPSTK_THROW(e); }
316 }

345 //------------------------------------------------------------------------------------
346 // Compute the azimuth and nadir angle, in the satellite body frame,
347 // of receiver Position RX as seen at the satellite Position SV. The nadir angle
348 // is measured from the Z axis, which points to Earth center, and azimuth is
349 // measured from the X axis.
350 // @param Position SV Satellite position
351 // @param Position RX Receiver position
352 // @param Matrix<double> Rot Rotation matrix (3,3), output of SatelliteAttitude
353 // @param double nadir Output nadir angle in degrees
354 // @param double azimuth Output azimuth angle in degrees
355 void SatelliteNadirAzimuthAngles(const Position& SV,
356  const Position& RX,
357  const Matrix<double>& Rot,
358  double& nadir,
359  double& azimuth)
360  throw(Exception)
361 {
362  try {
363  if(Rot.rows() != 3 || Rot.cols() != 3) {
364  Exception e("Rotation matrix invalid");
365  GPSTK_THROW(e);
366  }
367 
368  double d;
369  Position RmS;
370  // RmS points from satellite to receiver
371  RmS = RX - SV;
372  RmS.transformTo(Position::Cartesian);
373  d = RmS.mag();
374  if(d == 0.0) {
375  Exception e("Satellite and Receiver Positions identical");
376  GPSTK_THROW(e);
377  }
378  RmS = (1.0/d) * RmS;
379 
380  Vector<double> XYZ(3),Body(3);
381  XYZ(0) = RmS.X();
382  XYZ(1) = RmS.Y();
383  XYZ(2) = RmS.Z();
384  Body = Rot * XYZ;
385 
386  nadir = ::acos(Body(2)) * RAD_TO_DEG;
387  azimuth = ::atan2(Body(1),Body(0)) * RAD_TO_DEG;
388  if(azimuth < 0.0) azimuth += 360.;
389  }
390  catch(Exception& e) { GPSTK_RETHROW(e); }
391  catch(exception& e) {Exception E("std except: "+string(e.what())); GPSTK_THROW(E);}
392  catch(...) { Exception e("Unknown exception"); GPSTK_THROW(e); }
393 }
