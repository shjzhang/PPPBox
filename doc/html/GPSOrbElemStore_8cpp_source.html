220  bool GPSOrbElemStore::addOrbElem(const OrbElem& eph)
221  throw(InvalidParameter,Exception)
222  {
223  bool rc = false;
224  try
225  {
226 
227  SatID sid = eph.satID;
228  OrbElemMap& oem = ube[sid];
229  string ts = "%02m/%02d/%02y %02H:%02M:%02S";
230  int PRN = sid.id;
231  int testPRN = 0;
232 #pragma unused(PRN,testPRN,rc)
233 
234  // if map is empty, load object and return
235  if (oem.size()==0)
236  {
237  oem[eph.beginValid] = eph.clone();
238  updateInitialFinal(eph);
239  return (true);
240  }
241  // Search for beginValid in current keys.
242  // If found candidate, should be same data
243  // as already in table. Test this by comparing
244  // Toe values.
245  OrbElemMap::iterator it = oem.find(eph.beginValid);
246  if(it!=oem.end())
247  {
248  const OrbElem* oe = it->second;
249  // Found duplicate already in table
250  if(oe->ctToe==eph.ctToe)
251  {
252  return (false);
253  }
254  // Found matching beginValid but different Toe - This shouldn't happen
255  else
256  {
257  string str = "Unexpectedly found matching beginValid times";
258  stringstream os;
259  os << eph.satID.id;
260  str += " but different Toe. PRN= " + os.str();
261  str += ", beginValid= " + printTime(eph.beginValid,ts);
262  str += ", Toe(map)= " + printTime(eph.ctToe,ts);
263  str += ", Toe(candidate)= "+ printTime(oe->ctToe," %6.0g");
264  str += ". ";
265  InvalidParameter exc( str );
266  GPSTK_THROW(exc);
267  }
268  }
269  // Did not already find match to
270  // beginValid in map
271  // N.B:: lower_bound will reutrn element beyond key since there is no match
272  it = oem.lower_bound(eph.beginValid);
273  // Case where candidate is before beginning of map
274  if(it==oem.begin())
275  {
276  const OrbElem* oe = it->second;
277  if(oe->ctToe==eph.ctToe)
278  {
279  oem.erase(it);
280  oem[eph.beginValid] = eph.clone();
281  updateInitialFinal(eph);
282  return (true);
283  }
284  oem[eph.beginValid] = eph.clone();
285  updateInitialFinal(eph);
286  return (true);
287  }
288  // Case where candidate is after end of current map
289  if(it==oem.end())
290  {
291  // Get last item in map and check Toe
292  OrbElemMap::reverse_iterator rit = oem.rbegin();
293  const OrbElem* oe = rit->second;
294  if(oe->ctToe!=eph.ctToe)
295  {
296  oem[eph.beginValid] = eph.clone();
297  updateInitialFinal(eph);
298  return (true);
299  }
300  return (false);
301  }
302  // case where candidate is "In the middle"
303  // Check if iterator points to late transmission of
304  // same OrbElem as candidate
305  const OrbElem* oe = it->second;
306  if(oe->ctToe==eph.ctToe)
307  {
308  oem.erase(it);
309  oem[eph.beginValid] = eph.clone();
310  updateInitialFinal(eph);
311  return (true);
312  }
313  // Two cases:
314  // (a.) Candidate is late transmit copy of
315  // previous OrbElem in table - discard (do nothing)
316  // (b.) Candidate OrbElem is not in table
317 
318  // Already checked for it==oem.beginValid() earlier
319  it--;
320  const OrbElem* oe2 = it->second;
321  if(oe2->ctToe!=eph.ctToe)
322  {
323  oem[eph.beginValid] = eph.clone();
324  updateInitialFinal(eph);
325  return (true);
326  }
327  return (false);
328 
329  }
330  catch(Exception& e)
331  {
332  GPSTK_RETHROW(e)

338  void GPSOrbElemStore::edit(const CommonTime& tmin, const CommonTime& tmax)
339  throw()
340  {
341  for(UBEMap::iterator i = ube.begin(); i != ube.end(); i++)
342  {
343  OrbElemMap& eMap = i->second;
344 
345  OrbElemMap::iterator lower = eMap.lower_bound(tmin);
346  if (lower != eMap.begin())
347  {
348  for (OrbElemMap::iterator emi = eMap.begin(); emi != lower; emi++)
349  delete emi->second;
350  eMap.erase(eMap.begin(), lower);
351  }
352 
353  OrbElemMap::iterator upper = eMap.upper_bound(tmax);
354  if (upper != eMap.end())
355  {
356  for (OrbElemMap::iterator emi = upper; emi != eMap.end(); emi++)
357  delete emi->second;
358  eMap.erase(upper, eMap.end());
359  }
360  }
361 
362  initialTime = tmin;
363  finalTime = tmax;
364  }

380 // Goal is to find the set of orbital elements that would have been
381 // used by a receiver in real-time. That is to say, the most recently
382 // broadcast elements (assuming the receiver has visibility to the SV
383 // in question).

386  const OrbElem*
387  GPSOrbElemStore::findOrbElem(const SatID& sat, const CommonTime& t) const
388  throw( InvalidRequest )
389  {
390  // Check to see that there exists a map of orbital elements
391  // relevant to this SV.
392  UBEMap::const_iterator prn_i = ube.find(sat);
393  if (prn_i == ube.end())
394  {
395  InvalidRequest e("No orbital elements for satellite " + asString(sat));
396  GPSTK_THROW(e);
397  }
398 
399  // Define reference to the relevant map of orbital elements
400  const OrbElemMap& em = prn_i->second;
401 
402  // The map is ordered by beginning times of validity, which
403  // is another way of saying "earliest transmit time". A call
404  // to em.lower_bound( t ) will return the element of the map
405  // with a key "one beyond the key" assuming the t is NOT a direct
406  // match for any key.
407 
408  // First, check for the "direct match" case
409  OrbElemMap::const_iterator it = em.find(t);
410  // If that fails, then use lower_bound( )
411  if (it == em.end( ))
412  {
413  it = em.lower_bound(t);
414 
415  // Tricky case here. If the key is beyond the last key in the table,
416  // lower_bound( ) will return em.end( ). However, this doesn't entirely
417  // settle the matter. It is theoretically possible that the final
418  // item in the table may have an effectivity that "stretches" far enough
419  // to cover time t. Therefore, if it==em.end( ) we need to check
420  // the period of validity of the final element in the table against
421  // time t.
422  //
423  if (it==em.end())
424  {
425  OrbElemMap::const_reverse_iterator rit = em.rbegin();
426  if (rit->second->isValid(t)) return(rit->second); // Last element in map works
427 
428  // We reached the end of the map, checked the end of the map,
429  // and we still have nothing.
430  string mess = "All orbital elements found for satellite " + asString(sat) + " are too early for time "
431  + (static_cast<CivilTime>(t)).printf("%02m/%02d/%04Y %02H:%02M:%02S %P");
432  InvalidRequest e(mess);
433  GPSTK_THROW(e);
434  }
435  }
436 
437  // If the algorithm found a direct match, then we should
438  // probably use the PRIOR set since it takes ~30 seconds
439  // from beginning of transmission to complete reception.
440  // If lower_bound( ) was called, it points to the element
441  // after the time of the key.
442  // So either way, it points ONE BEYOND the element we want.
443  //
444  // The exception is if it is pointing to em.begin( ). If that is the case,
445  // then all of the elements in the map are too late.
446  if (it==em.begin())
447  {
448  string mess = "All orbital elements found for satellite " + asString(sat) + " are too late for time "
449  + (static_cast<CivilTime>(t)).printf("%02m/%02d/%04Y %02H:%02M:%02S %P");
450  InvalidRequest e(mess);
451  GPSTK_THROW(e);
452  }
453 
454  // The iterator should be a valid iterator and set one beyond
455  // the item of interest. However, there may be gaps in the
456  // middle of the map and cases where periods of effectivity do
457  // not overlap. That's OK, the key represents the EARLIEST
458  // time the elements should be used. Therefore, we can
459  // decrement the counter and test to see if the element is
460  // valid.
461  it--;
462  if (!(it->second->isValid(t)))
463  {
464  // If we reach this throw, the cause is a "hole" in the middle of a map.
465  string mess = "No orbital elements found for satellite " + asString(sat) + " at "
466  + (static_cast<CivilTime>(t)).printf("%02m/%02d/%04Y %02H:%02M:%02S %P");
467  InvalidRequest e(mess);
468  GPSTK_THROW(e);
469  }
470  return(it->second);
471  }

476 //-----------------------------------------------------------------------------
477 
478  const OrbElem*
479  GPSOrbElemStore::findNearOrbElem(const SatID& sat, const CommonTime& t) const
480  throw(InvalidRequest)
481  {
482  // Check for any OrbElem for this SV
483  UBEMap::const_iterator prn_i = ube.find(sat);
484  if (prn_i == ube.end())
485  {
486  InvalidRequest e("No OrbElem for satellite " + asString(sat));
487  GPSTK_THROW(e);
488  }
489 
490  // FIRST, try to find the elements that were
491  // actually being broadcast at the time of
492  // interest. That will ALWAYS be the most
493  // correct response. IF YOU REALLY THINK
494  // OTHERWISE CALL ME AND LET'S TALK ABOUT
495  // IT - Brent Renfro
496  try
497  {
498  const OrbElem* oep = findOrbElem(sat, t);
499  return(oep);
500  }
501  // No OrbElem in store for requested sat time
502  catch(InvalidRequest)
503  {
504  // Create references to map for this satellite
505  const OrbElemMap& em = prn_i->second;
506  /*
507  Three Cases:
508  1. t is within a gap within the store
509  2. t is before all OrbElem in the store
510  3. t is after all OrbElem in the store
511  */
512 
513  // Attempt to find next in store after t
514  OrbElemMap::const_iterator itNext = em.lower_bound(t);
515  // Test for case 2
516  if(itNext==em.begin())
517  {
518  return(itNext->second);
519  }
520  // Test for case 3
521  if(itNext==em.end())
522  {
523  OrbElemMap::const_reverse_iterator rit = em.rbegin();
524  return(rit->second);
525  }
526  // Handle case 1
527  // Know that itNext is not the beginning, so safe to decrement
528  CommonTime nextBeginValid = itNext->first;
529  OrbElemMap::const_iterator itPrior = itNext;
530  itPrior--;
531  CommonTime lastEndValid = itPrior->second->endValid;
532  double diffToNext = nextBeginValid-t;
533  double diffFromLast = t - lastEndValid;
534  if(diffToNext>diffFromLast)
535  {
536  return(itPrior->second);
537  }
538  return(itNext->second);
539  }
540  }

542 // See notes in the .hpp. This function is designed to be called
543 // AFTER all elements are loaded. It can then make adjustments to
544 // time relationships based on inter-comparisons between sets of
545 // elements that cannot be performed until the ordering has been
546 // determined.
547 //-----------------------------------------------------------------------------
548  void GPSOrbElemStore::rationalize( )
549  {
550  UBEMap::iterator it;
551  for (it = ube.begin(); it != ube.end(); it++)
552  {
553  OrbElemMap& em = it->second;
554  OrbElemMap::iterator ei;
555  OrbElemMap::iterator eiPrev;
556  bool begin = true;
557  double previousOffset = 0.0;
558  bool previousIsOffset = false;
559  bool currentIsOffset = false;
560 
561  //string tForm = "%02H:%02M:%02S";
562 
563  // Scan the map for this SV looking for
564  // uploads. Uploads are identifed by
565  // Toe values that are offset from
566  // an even hour.
567  OrbElem* oePrev = 0;
568  for (ei = em.begin(); ei != em.end(); ei++)
569  {
570  currentIsOffset = false; // start with this assumption
571  OrbElem* oe = ei->second;
572  long Toe = (long) (static_cast<GPSWeekSecond> (oe->ctToe)).sow;
573  double currentOffset = Toe % 3600;
574 
575  //cout << "Top of For Loop. oe->beginValid = " << printTime(oe->beginValid,tForm);
576  //cout << ", currentOffset =" << currentOffset << endl;
577 
578  if ( (currentOffset)!=0)
579  {
580  currentIsOffset = true;
581 
582  //cout << "*** Found an offset" << endl;
583  //cout << " currentIsOffest: " << currentIsOffset;
584  //cout << " previousIsOffest: " << previousIsOffset;
585  //cout << " current, previous Offset = " << currentOffset << ", " << previousOffset << endl;
586 
587  // If this set is offset AND the previous set is offset AND
588  // the two offsets are the same, then this is the SECOND
589  // set of elements in an upload. In that case the OrbElem
590  // load routines have conservatively set the beginning time
591  // of validity to the transmit time because there was no
592  // way to prove this was the second data set. Since we can
593  // now prove its second by observing the ordering, we can
594  // adjust the beginning of validity as needed.
595  // Since the algorithm is dependent on the message
596  // format, this must be done in OrbElem.
597  //
598  // IMPORTANT NOTE: We also need to adjust the
599  // key in the map, which is based on the beginning
600  // of validity. However, we can't do it in this
601  // loop without destroying the integrity of the
602  // iterator. This is handled later in a second
603  // loop. See notes on the second loop, below.
604  if (previousIsOffset &&
605  currentIsOffset &&
606  currentOffset==previousOffset)
607  {
608  //cout << "*** Adjusting beginValid" << endl;
609  oe->adjustBeginningValidity();
610  }
611 
612  // If the previous set is not offset, then
613  // we've found an upload
614  // For that matter, if previous IS offset, but
615  // the current offset is different than the
616  // previous, then it is an upload.
617  if (!previousIsOffset ||
618  (previousIsOffset && (currentOffset!=previousOffset) ) )
619  {
620  // Record the offset for later reference
621  previousOffset = currentOffset;
622 
623  // Adjust the ending time of validity of any elements
624  // broadcast BEFORE the new upload such that they
625  // end at the beginning validity of the upload.
626  // That beginning validity value should already be
627  // set to the transmit time (or earliest transmit
628  // time found) by OrbElem and GPSOrbElemStore.addOrbElem( )
629  // This adjustment is consistent with the IS-GPS-XXX
630  // rules that a new upload invalidates previous elements.
631  // Note that this may be necessary for more than one
632  // preceding set of elements.
633  if (!begin)
634  {
635  //cout << "*** Adjusting endValid Times" << endl;
636  OrbElemMap::iterator ri;
637  ri = em.find(oePrev->beginValid); // We KNOW it exists in the map
638  bool done = false;
639  while (!done)
640  {
641  OrbElem* oeRev = ri->second;
642  //cout << "Testing Toe of " << printTime(oeRev->ctToe,"%02H:%02M:%02S");
643  //cout << " with endValid of " << printTime(oeRev->endValid,"%02H:%02M:%02S") << endl;
644 
645  // If the current set of elements has an ending
646  // validity prior to the upload, then do NOT
647  // adjust the ending and set done to true.
648  if (oeRev->endValid <= oe->beginValid) done = true;
649 
650  // Otherwise, adjust the ending validity to
651  // match the begin of the upload.
652  else oeRev->endValid = oe->beginValid;
653 
654  // If we've reached the beginning of the map, stop.
655  // Otherwise, decrement and test again.
656  if (ri!=em.begin()) ri--;
657  else done = true;
658  }
659  }
660  }
661  }
662 
663  // Update condition flags for next loop
664  previousIsOffset = currentIsOffset;
665  oePrev = oe; // May need this for next loop.
666  begin = false;
667  //cout << "Bottom of For loop. currentIsOffset: " << currentIsOffset <<
668  // ", previousIsOffset: " << previousIsOffset << endl;
669  } //end inner for-loop
670 
671  // The preceding process has left some elements in a condition
672  // when the beginValid value != the key in the map. This
673  // must be addressed, but the map key is a const (by definition).
674  // We have to search the map for these disagreements. When found,
675  // the offending item need to be copied out of the map, the
676  // existing entry deleted, the item re-entered with the new key.
677  // Since it is unsafe to modify a map while traversing the map,
678  // each time this happens, we have to reset the loop process.
679  //
680  // NOTE: Simplisitically, we could restart the loop at the
681  // beginning each time. HOWEVER, we sometimes load a long
682  // period in a map (e.g. a year). At one upload/day, that
683  // would mean ~365 times, each time one day deeper into the map.
684  // As an alternate, we use the variable loopStart to note how
685  // far we scanned prior to finding a problem and manipulating
686  // the map. Then we can restart at that point.
687  bool done = false;
688  CommonTime loopStart = CommonTime::BEGINNING_OF_TIME;
689  while (!done)
690  {
691  ei = em.lower_bound(loopStart);
692  while (ei!=em.end())
693  {
694  OrbElem* oe = ei->second;
695  if (ei->first!=oe->beginValid)
696  {
697  OrbElem* oeAdj= oe->clone(); // Adjustment was done in
698  // first loop above.
699  delete ei->second; // oe becomes invalid.
700  em.erase(ei); // Remove the map entry.
701  em[oeAdj->beginValid] = oeAdj->clone(); // Add back to map
702  break; // exit this while loop without finishing
703  }
704  loopStart = ei->first; // Scanned this far successfully, so
705  // save this as a potential restart point.
706  ei++;
707  if (ei==em.end()) done = true; // Successfully completed
708  // the loop w/o finding any
709  // mismatches. We're done.
710  }
711  }
712 
713  // Well, not quite done. We need to update the initial/final
714  // times of the map.
715  const OrbElemMap::iterator Cei = em.begin( );
716  initialTime = Cei->second->beginValid;
717  const OrbElemMap::reverse_iterator rCei = em.rbegin();
718  finalTime = rCei->second->endValid;
719 
720  } // end outer for-loop
721  }
