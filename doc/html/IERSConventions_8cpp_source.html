244  {
245  if(from==to) return time;

248  if(mapTSName.size()==0)

253  mapTSName[TS_TAI] = "TAI";
254  mapTSName[TS_TT] = "TT";
255  }
256 
257  std::map<TimeSystemEnum,std::string>::const_iterator itf,itt,ite;
258  itf = mapTSName.find(from);
259  itt = mapTSName.find(to);

262  if( (itf==ite) || (itt==ite) )

264  Exception e("Can't convert the Time System from "
265  + ((itf==ite)?"Unknown":itf->second) + std::string(" to")

268  GPSTK_THROW(e);

273  ConvertFunPtr funPtr1(0);
