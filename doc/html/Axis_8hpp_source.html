87  Axis(double ix, double iy, double ex, double ey, double length, double imin, double imax, const AxisStyle &style=AxisStyle())
88  {
89  //Calculate direction
90  /*
91  * tan(dir) = opposite / adjacent;
92  * arctan(opposite/adjacent) = dir;
93  * opposite = (ey-iy);
94  * adjacent = (ex-ix);
95  */
96  double direction;
97  if(ex==ix)
98  {
99  if(ey < iy) direction=vdraw::PI;
100  else direction=0;
101  }
102  else if(ey==iy)
103  {
104  if(ex < ix) direction=vdraw::PI+vdraw::HALF_PI;
105  else direction=vdraw::HALF_PI;
106  }
107  else
108  {
109  direction=atan((ey-iy)/(ex-ix));
110  if(direction < 0) direction=vdraw::TWO_PI+direction;
111  }
112  init(ix,iy,direction,length,imin,imax,style);
113  }
