38  void SeriesList::findMinMax(double& minX, double &maxX, double& minY, double& maxY)
39  {
40  minX = DBL_MAX;
41  maxX = DBL_MIN;
42  minY = DBL_MAX;
43  maxY = DBL_MIN;
44 
45  for(unsigned int i=0; i<pointlists.size(); i++)
46  {
47  vector<pair<double, double> >& series = pointlists[i];
48  vector<pair<double, double> >::iterator it;
49  for (it = series.begin(); it!=series.end(); it++)
50  {
51  double x = it->first;
52  double y = it->second;
53  maxX = (x>maxX ? x : maxX);
54  minX = (x<minX ? x : minX);
55  maxY = (y>maxY ? y : maxY);
56  minY = (y<minY ? y : minY);
57  }
58  }

61  void SeriesList::drawInFrame(Frame& innerFrame, double minX, double maxX, double minY, double maxY)
62  {
63  double multX = innerFrame.getWidth()/(maxX-minX);
64  double multY = innerFrame.getHeight()/(maxY-minY);
65 
66  // Draw lines
67  for(int i=0;i<getNumSeries();i++)
68  {
69  innerFrame.push_state();
70 
71  StrokeStyle s = getStyle(i);
72  Marker m = getMarker(i);
73 
74  if(m.getColor().isClear() && s.getColor().isClear())
75  {
76  innerFrame << Comment("Plot contained data with clear stroke and marker. Skipping.");
77  continue;
78  }
79 
80  vector< pair<double,double> >& vec = getPointList(i);
81  Path curve(vec,innerFrame.lx(), innerFrame.ly());
82 
83  // What I'd give for a line of haskell...
84  // map (\(x,y) -> (multX*(x-minX), multY*(y-minY))) vector
85  map_object map_instance(multX,minX,multY,minY);
86 
87  innerFrame.setMarker(m);
88  innerFrame.setLineStyle(s);
89 
90  if(s.getColor().isClear())
91  {
92  // crop
93  auto_ptr< Path > cropX = Splitter::cropToBox(minX,maxX,minY,maxY,curve);
94 
95  // Fit it to the box.
96  std::for_each(cropX->begin(), cropX->end(), map_instance);
97 
98  // Draw the line
99  innerFrame.line(*cropX);
100  }
101  else
102  {
103  // interpolate
104  auto_ptr< std::list<Path> > interpX = Splitter::interpToBox(minX,maxX,minY,maxY,curve);
105 
106  for(std::list<Path>::iterator i=interpX->begin();i!=interpX->end();i++)
107  {
108  // Fit it to the box
109  std::for_each(i->begin(), i->end(), map_instance);
110 
111  // Draw the line
112  innerFrame.line(*i);
113  }
114  }
115  innerFrame.pop_state();
116  }
117  }

119  void SeriesList::drawLegend(Frame& frame, double pointsize, unsigned int columns)
120  {
121  if(columns <= 1)
122  drawLegendSegment(frame,pointsize,0,titles.size());
123  else
124  {
125  // Make a grid with one row and the correct number of columns
126  GridLayout gl(frame,1,columns);
127  // The number of elements in each column will be the same for all
128  // columns but the last.
129  unsigned int n = (titles.size()/columns) + (titles.size()%columns?1:0);
130  // Add the legend segment for each column
131  for(unsigned int i=0; i<columns; i++)
132  {
133  Frame t(gl.getFrame(0,i));
134  drawLegendSegment(t,pointsize,i*n,min(n,(unsigned int)(titles.size()-i*n)));
135  }

139  void SeriesList::drawLegendSegment(Frame& frame, double pointsize,
140  unsigned int begin, unsigned int n)
141  {
142  // If we aren't drawing anything, don't bother with all the effort :)
143  if(n == 0)
144  return;
145  // Spacer is the number of points between the drawn segment and the text
146  // of the label as well as the added number of spacing between each
147  // series in the list
148  double spacer = 5;
149  // Test to see if we are dealing with a scatter plot or not
150  // as this determines what is drawn on the left
151  bool lines = false;
152  double mwidth = 0;
153  double height = pointsize;
154  for(unsigned int i=0;i<styles.size();i++)
155  {
156  if(!markers[i].getColor().isClear())
157  {
158  mwidth = max(mwidth,markers[i].getRange()*2);
159  height = max(height,mwidth);
160  }
161 
162  if(!styles[i].getColor().isClear())
163  lines = true;
164  }
165  // Add spacer to the height
166  height += spacer;
167  // width = width needed to draw this
168  // lbegin = x offset where line will begin
169  // lwidth = length of the lines to draw the sample
170  double width = 30;
171  double lbegin = 0;
172  double lwidth = 30;
173  if(mwidth) // If we have markers...
174  {
175  lbegin = mwidth/2;
176  if(lines)
177  {
178  width = mwidth*3;
179  lwidth = mwidth*2;
180  }
181  else
182  {
183  width = mwidth;
184  lwidth = 0;
185  }
186  }
187  // TextStyle
188  TextStyle style;
189  style.setPointSize(pointsize);
190 
191  // Begin drawing?
192  for(unsigned int i=begin;i<begin+n;i++)
193  {
194  // Draw the sample:
195  double y = frame.getHeight() - height/2.0 - height*i;
196  Line l(lbegin+spacer,y,lbegin+lwidth+spacer,y);
197  l.setStrokeStyle(styles[i]);
198  l.setMarker(markers[i]);
199  frame << l;
200 
201  // Add the series title
202  frame << Text(titles[i].c_str(),width+spacer*2,(y-(pointsize/2.0)),style,Text::LEFT);
203  }
204  }
205 }
