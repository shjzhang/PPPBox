39  /* Get a combination of observations from a Rinex3ObsData object
40  *
41  * @param rinexData The Rinex data set holding the observations
42  * @param indexObs1 Index representing the observation type #1.
43  * @param indexObs2 Index representing the observation type #2.
44  *
45  * @return
46  * Number of SVs with this combination of observables available
47  *
48  * @note
49  * The indexes are obtained from the RINEX Observation Header
50  * using method 'Rinex3ObsHeader::getObsIndex()'.
51  */
52  int ExtractCombinationData::getData( const Rinex3ObsData& rinexData,
53  int indexObs1,
54  int indexObs2 )
55  throw(InvalidRequest)
56  {
57 
58  try
59  {
60 
61  // Let's make sure each time we start with clean Vectors
62  availableSV.resize(0);
63  obsData.resize(0);
64 
65  // Create a CheckPRData object with the given limits
66  CheckPRData checker(minPRange, maxPRange);
67 
68  // Let's define the "it" iterator to visit the observations PRN map
69  // RinexSatMap is a map from SatID to RinexObsTypeMap:
70  // std::map<SatID, RinexObsTypeMap>
71  Rinex3ObsData::DataMap::const_iterator it;
72  for ( it = rinexData.obs.begin();
73  it != rinexData.obs.end();
74  it++ )
75  {
76 
77  // The satellites are stored in the first elements of the map...
78  SatID sat(it->first);
79  // .. and vectors of available obs are in the second elements
80  vector<RinexDatum> vecData(it->second);
81 
82  // Extract observation values
83  double obsValue1( (vecData[indexObs1]).data );
84  double obsValue2( (vecData[indexObs2]).data );
85 
86  // Compute the combination
87  double combinationValue( getCombination( obsValue1, obsValue2 ) );
88 
89  // Let's check if the combination is between the limits
90  if ( !(checkData) || (checker.check(combinationValue)) )
91  {
92 
93  // Store all relevant data of this epoch
94  availableSV = availableSV && sat;
95  obsData = obsData && combinationValue;
96 
97  }
98 
99  } // End of data combination extraction from this epoch
100 
101  }
102  catch(...)
103  {
104  InvalidRequest e("Unable to compute combination from Rinex3ObsData object");
105  GPSTK_THROW(e);
106  }
107 
108  // Let's record the number of SV with this type of data available
109  numSV = (int)obsData.size();
110 
111  // If everything is fine so far, then the results should be valid
112  valid = true;
113 
114  return numSV;
