41 int ReadMatrix(gpstk::Matrix<double>& M, string& file)
42 {
43  ifstream infile(file.c_str());
44  if(!infile) return -1;
45 
46  enum Type{GEN,LOW,UPT,DIA,SYM,SQU};
47  string labels[6]={"GEN","LOW","UPT","DIA","SYM","SQU"};
48  bool dimmed=false;
49  char ch;
50  int r=0,c=0,i=0,j=0;
51  string buffer;
52  Type type=GEN;
53  while(infile >> buffer) {
54  if(buffer[0] == '#') { // skip to end of line
55  while(infile.get(ch)) { if(ch=='\n') break; }
56  }
57  else {
58  string::size_type in;
59  if((in=buffer.find("r=")) != string::npos) {
60  r = strtol(buffer.substr(in+2).c_str(),0,10);
61  //cout << "Found r = " << r << endl;
62  if(type != GEN) c=r;
63  }
64  else if((in=buffer.find("c=")) != string::npos) {
65  c = strtol(buffer.substr(in+2).c_str(),0,10);
66  //cout << "Found c = " << c << endl;
67  if(type != GEN) r=c;
68  }
69  else if((in=buffer.find("t=")) != string::npos) {
70  if((in=buffer.find("LOW")) != string::npos) type=LOW;
71  if((in=buffer.find("UPT")) != string::npos) type=UPT;
72  if((in=buffer.find("DIA")) != string::npos) type=DIA;
73  if((in=buffer.find("SYM")) != string::npos) type=SYM;
74  if((in=buffer.find("SQU")) != string::npos) type=SQU;
75  //cout << "Found type " << labels[type] << endl;
76  }
77  else if(buffer == string(":::")) {
78  if(r*c == 0) return 0;
79  M = gpstk::Matrix<double>(r,c,double(0));
80  dimmed = true;
81  }
82  else {
83  if(!dimmed) continue;
84  M(i,j) = strtod(buffer.c_str(),0);
85  j++;
86  switch(type) {
87  case LOW: if(j>i) { i++; j=0; } break;
88  case UPT: if(j>=c) { i++; j=i; } break;
89  case DIA: i=j; break;
90  case SYM: M(j-1,i) = M(i,j-1); if(j>i) { i++; j=0; } break;
91  case SQU: case GEN: default: if(j>=c) { i++; j=0; } break;
92  }
93  }
94  }
95  }
96  infile.close();
97  if(!dimmed) return -2;
98  return 0;
99 }
