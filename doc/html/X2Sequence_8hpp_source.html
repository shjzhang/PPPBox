200  /*
201  Given a bit position within the X2 sequence (numbered starting at -37),
202  return the next 32 bits. Note: if there are insufficient bits left
203  to fill the request, wrap around to the beginning of the sequence.
204  */
205  inline uint32_t X2Sequence::operator[] ( long i )
206  {
207  long adjustedCount = i + X2A_EPOCH_DELAY;
208 
209  uint32_t retArg = 0L;
210  int ndx1 = adjustedCount / MAX_BIT;
211  int offset = adjustedCount - (ndx1 * MAX_BIT);
212  if ( (adjustedCount+MAX_BIT) <= MAX_X2_COUNT )
213  {
214  if (offset==0) retArg = bitsP[ndx1];
215  else retArg = merge( bitsP[ndx1], bitsP[ndx1+1], offset );
216  }
217  /*
218  Complicated case when coming up to end of sequence. May have to
219  put together parts of up to three words to get 32 bits. The problem
220  is complicated because the word at the end of the array is partial
221  AND the beginning of sequence (BOW) occurs in mid-word due to the
222  PRN offset. Some numbers:
223 
224  Number of bits available in word N : 25
225  Number of bits available in word at BOS : 27
226 
227  Possible cases:
228  1.) Combine bits from [n-1], [n], and [BOS] - word n will provide
229  25 bits. Therefore, some combination of (n-1,BOS) from the
230  choice of (1,4), (2,3), (3,2), (4,1).
231 
232  2.) Combine bits from [n-1] and [BOS] - word n-1 will provide 25-5
233  bits. Therefore, BOS will provide 7-27 bits.
234 
235  3.) Combine bits from [n-1], [BOS], and [BOS+1] - word n-1 provides
236  4-1 bits. BOS provides 27 bits (running total to 31-28 bits).
237  BOS+1 provides 1-4 bits.
238  */
239  else
240  {
241  int numRemainingInSequence = MAX_X2_COUNT - adjustedCount;
242  int numRemainingInWord;
243  int numFilled = 0;
244 
245  // Handle word n-1
246  if (ndx1==NUM_X2_WORDS-2)
247  {
248  numRemainingInWord = MAX_BIT - offset;
249  retArg = bitsP[ndx1++] << offset;
250  numFilled = numRemainingInWord;
251  numRemainingInSequence -= numRemainingInWord;
252  }
253 
254  // Handle word n
255  uint32_t temp = bitsP[ndx1];
256  numRemainingInWord = numRemainingInSequence;
257  temp >>= (MAX_BIT-numRemainingInWord);
258  temp <<= (MAX_BIT-(numRemainingInWord+numFilled));
259  retArg |= temp;
260  numFilled += numRemainingInSequence;
261 
262  // Wrap to front. Recall that "front" is actually bit
263  // 37 in sequence due to "beginning of week" delay
264  numRemainingInWord = (2 * MAX_BIT) - X2A_EPOCH_DELAY;
265  int numNeeded = MAX_BIT - numFilled;
266 
267  // Case where all bits needed are in word 1 of sequence array
268  //(which only has 27 "useful" bits)
269  if (numNeeded <= numRemainingInWord)
270  {
271  temp = bitsP[1] << (MAX_BIT - numRemainingInWord);
272  temp >>= (MAX_BIT - numNeeded);
273  retArg |= temp;
274  }
275  // Case where all bits in word 1 are needed plus some bits in word 2.
276  else
277  {
278  // Clearing high-order bits
279  temp = bitsP[1] << (MAX_BIT - numRemainingInWord);
280  temp >>= (MAX_BIT - numRemainingInWord);
281  temp <<= (MAX_BIT - (numRemainingInWord+numFilled));
282  retArg |= temp;
283 
284  // Fetch remaining bits from next word
285  numFilled += numRemainingInWord;
286  numNeeded = MAX_BIT - numFilled;
287  temp = bitsP[2] >> (MAX_BIT - numNeeded);
288  retArg |= temp;
289  }
290  }
291  return(retArg);
292  }
