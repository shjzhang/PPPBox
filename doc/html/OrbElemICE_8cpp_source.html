95  // See IS-GPS-800 3.5.3.8
96  double OrbElemICE::getIAURAned(const CommonTime& t) const
97  throw( InvalidRequest )
98  {
99  if (!dataLoaded())
100  {
101  InvalidRequest exc("Required data not stored.");
102  GPSTK_THROW(exc);
103  }
104 
105  double ned0 = ura2CNAVaccuracy( URAned0 );
106  double ned1 = 1.0/std::pow(2.0, (14.0 + URAned1));
107  double ned2 = 1.0/std::pow(2.0, (28.0 + URAned2));
108  double deltaT = t - ctTop;
109  if(deltaT <= 93600)
110  return ( ned0 + ned1 * deltaT);
111  else
112  return ned0 + ned1 * deltaT + ned2 * std::pow((deltaT - 93600), 2.0);
113  }

144  void OrbElemICE::adjustBeginningValidity()
145  {
146  if (!dataLoaded()) return;
147 
148  // If we assume this is the SECOND set of elements in a set
149  // (which is an assumption of this method - see the .hpp) then
150  // the "small offset in Toe" will actually push the Toe-oneHalfInterval
151  // too early. For example, consider the following case.
152  // Toe : 19:29:44 (really near 19:30:00)
153  // first xMit : 18:00:00 (nominal)
154  // Blindly setting beginValid to Toe - 1/2 fit interval will
155  // result in 17:59:44. But 18:00:00 actually is the right answer
156  // because the -16 second offset is an artifact.
157  //
158  // Therefore, we are FIRST going to remove that offset,
159  // THEN (re-)determine beginValid.
160  long sow = (long) (static_cast<GPSWeekSecond>(ctToe)).sow;
161  short week = (static_cast<GPSWeekSecond>(ctToe)).week;
162  long offsetFrom2Hours = sow % TWO_HOURS;
163  if (offsetFrom2Hours!=NINTY_MINUTES)
164  {
165  sow = sow - (offsetFrom2Hours-NINTY_MINUTES);
166  }
167  CommonTime adjustedToe = GPSWeekSecond(week, (double) sow);
168  adjustedToe.setTimeSystem(TimeSystem::GPS);
169 
170  // For CNAV the nominal beginning of validity is ninty minutes
171  // prior to the Toe.
172  beginValid = adjustedToe - NINTY_MINUTES;
173 
174  return;
175  }
