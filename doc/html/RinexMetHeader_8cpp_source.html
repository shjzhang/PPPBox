69  std::string RinexMetHeader::bitString(unsigned long vb, char quote,
70  std::string sep)
71  {
72  unsigned long b = 1;
73  std::string rv;
74  while (b)
75  {
76  if (vb & b)
77  {
78  if (rv.length())
79  rv += sep;
80  if (quote)
81  rv += quote + bitsAsString((validBits)b) + quote;
82  else
83  rv += bitsAsString((validBits)b);
84  }
85  b <<= 1;
86  }
87  return rv;
88  }

90  void RinexMetHeader::reallyPutRecord(FFStream& ffs) const
91  throw(std::exception, FFStreamError,
92  gpstk::StringUtils::StringException)
93  {
94  RinexMetStream& strm = dynamic_cast<RinexMetStream&>(ffs);
95 
96  // Since they want to output this header, let's store it
97  // internally for use by the data.
98  strm.header = (*this);
99 
100  unsigned long allValid;
101  if (version == 2.0 ) allValid = allValid20;
102  else if (version == 2.1 ) allValid = allValid21;
103  else if (version == 2.11) allValid = allValid211;
104  else if (version == 3.0 ) allValid = allValid211;
105  else
106  {
107  FFStreamError err("Unknown RINEX version: " + asString(version,2));
108  err.addText("Make sure to set the version correctly.");
109  GPSTK_THROW(err);
110  }
111 
112  if ((valid & allValid) != allValid)
113  {
114  string errstr("Incomplete or invalid header: missing: ");
115  errstr += bitString(allValid & ~valid);
116  FFStreamError err(errstr);
117  err.addText("Make sure you set all header valid bits for all of the available data.");
118  GPSTK_THROW(err);
119  }
120 
121  string line;
122 
123  if (valid & validVersion)
124  {
125  line = rightJustify(asString(version,2), 9);
126  line += string(11, ' ');
127  line += leftJustify(fileType, 40);
128  line += stringVersion;
129  strm << line << endl;
130  strm.lineNumber++;
131  }
132  if (valid & validRunBy)
133  {
134  line = leftJustify(fileProgram,20);
135  line += leftJustify(fileAgency,20);
136  SystemTime sysTime;
137  string curDate = (static_cast<CivilTime>(sysTime)).printf("%04Y%02m%02d %02H%02M%02S %P");
138  line += leftJustify(curDate, 20);
139  line += stringRunBy;
140  strm << line << endl;
141  strm.lineNumber++;
142  }
143  if (valid & validComment)
144  {
145  vector<string>::const_iterator itr = commentList.begin();
146  while (itr != commentList.end())
147  {
148  line = leftJustify((*itr), 60);
149  line += stringComment;
150  strm << line << endl;
151  strm.lineNumber++;
152  itr++;
153  }
154  }
155  if (valid & validMarkerName)
156  {
157  line = leftJustify(markerName, 60);
158  line += stringMarkerName;
159  strm << line << endl;
160  strm.lineNumber++;
161  }
162  if (valid & validMarkerNumber)
163  {
164  line = leftJustify(markerNumber, 60);
165  line += stringMarkerNumber;
166  strm << line << endl;
167  strm.lineNumber++;
168  }
169  if (valid & validObsType)
170  {
171  line = rightJustify(asString(obsTypeList.size()),6);
172  vector<RinexMetType>::const_iterator itr = obsTypeList.begin();
173  size_t numWritten = 0;
174  while (itr != obsTypeList.end())
175  {
176  numWritten++;
177  // continuation lines
178  if ((numWritten % (maxObsPerLine+1)) == 0)
179  {
180  line += stringObsType;
181  strm << line << endl;
182  strm.lineNumber++;
183  line = string(6,' ');
184  }
185  line += rightJustify(convertObsType(*itr), 6);
186  itr++;
187  }
188  // pad the line out to 60 chrs and add label
189  line += string(60 - line.size(), ' ');
190  line += stringObsType;
191  strm << line << endl;
192  strm.lineNumber++;
193  }
194  if (valid & validSensorType)
195  {
196  // Write out only the sensor types that are in the obsTypeList.
197  vector<sensorType>::const_iterator itr = sensorTypeList.begin();
198  while (itr != sensorTypeList.end())
199  {
200  if (std::find(obsTypeList.begin(), obsTypeList.end(),
201  (*itr).obsType) != obsTypeList.end())
202  {
203  line = leftJustify((*itr).model, 20);
204  line += leftJustify((*itr).type, 20);
205  line += string(6, ' ');
206  line += rightJustify(asString((*itr).accuracy,1),7);
207  line += string(4, ' ');
208  line += convertObsType((*itr).obsType);
209  line += string(1, ' ');
210  line += stringSensorType;
211  strm << line << endl;
212  strm.lineNumber++;
213  }
214  itr++;
215  }
216  }
217  if (valid & validSensorPos)
218  {
219  // Write out only the sensor positions that are in the obsTypeList.
220  vector<sensorPosType>::const_iterator itr = sensorPosList.begin();
221  while (itr != sensorPosList.end())
222  {
223  if (std::find(obsTypeList.begin(), obsTypeList.end(),
224  (*itr).obsType) != obsTypeList.end())
225  {
226  line = rightJustify(asString((*itr).position[0],4),14);
227  line += rightJustify(asString((*itr).position[1],4),14);
228  line += rightJustify(asString((*itr).position[2],4),14);
229  line += rightJustify(asString((*itr).height,4),14);
230  line += string(1, ' ');
231  line += convertObsType((*itr).obsType);
232  line += string(1, ' ');
233  line += stringSensorPos;
234  strm << line << endl;
235  strm.lineNumber++;
236  }
237  itr++;
238  }
239  }
240  if (valid & validEoH)
241  {
242  line = string(60, ' ');
243  line += stringEoH;
244  strm << line << endl;
245  strm.lineNumber++;
246  }
247  }

249  void RinexMetHeader::reallyGetRecord(FFStream& ffs)
250  throw(std::exception, FFStreamError,
251  gpstk::StringUtils::StringException)
252  {
253  RinexMetStream& strm = dynamic_cast<RinexMetStream&>(ffs);
254 
255  // if already read, just return
256  if (strm.headerRead == true)
257  return;
258 
259  valid = 0;
260 
261  // Clear out structures in case the last read was a partial header.
262  commentList.clear();
263  obsTypeList.clear();
264  sensorTypeList.clear();
265  sensorPosList.clear();
266 
267  int numObs;
268 
269  while (!(valid & validEoH))
270  {
271  string line;
272  strm.formattedGetLine(line);
273 
274  if (line.length()<60 || line.length()>81)
275  {
276  FFStreamError e("Bad line length");
277  GPSTK_THROW(e);
278  }
279 
280  string thisLabel = strip(line.substr(60,20));
281 
282  if (thisLabel == stringVersion)
283  {
284  version = asDouble(line.substr(0,20));
285  fileType = strip(line.substr(20,20));
286  if ( (fileType[0] != 'M') && (fileType[0] != 'm'))
287  {
288  FFStreamError e("This isn't a Rinex Met file");
289  GPSTK_THROW(e);
290  }
291  valid |= validVersion;
292  }
293  else if (thisLabel == stringRunBy)
294  {
295  fileProgram = strip(line.substr(0,20));
296  fileAgency = strip(line.substr(20,20));
297  date = strip(line.substr(40,20));
298  valid |= validRunBy;
299  }
300  else if (thisLabel == stringComment)
301  {
302  commentList.push_back(strip(line.substr(0,60)));
303  valid |= validComment;
304  }
305  else if (thisLabel == stringMarkerName)
306  {
307  markerName = strip(line.substr(0,60));
308  valid |= validMarkerName;
309  }
310  else if (thisLabel == stringMarkerNumber)
311  {
312  markerNumber = strip(line.substr(0,20));
313  valid |= validMarkerNumber;
314  }
315  else if (thisLabel == stringObsType)
316  {
317  // read the first line
318  if (!(valid & validObsType))
319  {
320  numObs = gpstk::StringUtils::asInt(line.substr(0,6));
321  for (int i = 0; (i < numObs) && (i < maxObsPerLine); i++)
322  {
323  int currPos = i * 6 + 6;
324  if (line.substr(currPos, 4) != string(4, ' '))
325  {
326  FFStreamError e("Format error for line type " + stringObsType);
327  GPSTK_THROW(e);
328  }
329  obsTypeList.push_back(convertObsType(line.substr(currPos + 4, 2)));
330  }
331  valid |= validObsType;
332  }
333  // read continuation lines
334  else
335  {
336  int currentObsTypes = obsTypeList.size();
337  for (int i = currentObsTypes;
338  (i < numObs) && (i < (maxObsPerLine + currentObsTypes));
339  i++)
340  {
341  int currPos = (i % maxObsPerLine) * 6 + 6;
342  if (line.substr(currPos, 4) != string(4,' '))
343  {
344  FFStreamError e("Format error for line type " + stringObsType);
345  GPSTK_THROW(e);
346  }
347  obsTypeList.push_back(convertObsType(line.substr(currPos + 4, 2)));
348  }
349  }
350  }
351  else if (thisLabel == stringSensorType)
352  {
353  if (line.substr(40,6) != string(6, ' '))
354  {
355  FFStreamError e("Format error for line type " + stringSensorType);
356  GPSTK_THROW(e);
357  }
358  sensorType st;
359  st.model = strip(line.substr(0,20));
360  st.type = strip(line.substr(20,20));
361  st.accuracy = asDouble(line.substr(46,9));
362  st.obsType = convertObsType(line.substr(57,2));
363 
364  sensorTypeList.push_back(st);
365 
366  // only set this valid if there are exactly
367  // the same number in both lists
368  if (sensorTypeList.size() == obsTypeList.size())
369  {
370  valid |= validSensorType;
371  }
372  else
373  {
374  valid &= ~(long)validSensorType;
375  }
376  }
377  else if (thisLabel == stringSensorPos)
378  {
379  // read XYZ and H and obs type
380  sensorPosType sp;
381  sp.position[0] = asDouble(line.substr(0,14));
382  sp.position[1] = asDouble(line.substr(14,14));
383  sp.position[2] = asDouble(line.substr(28,14));
384  sp.height = asDouble(line.substr(42,14));
385 
386  sp.obsType = convertObsType(line.substr(57,2));
387 
388  sensorPosList.push_back(sp);
389 
390  // Only barometer is required, so set it valid only if you see that record.
391  if (sp.obsType == PR)
392  {
393  valid |= validSensorPos;
394  }
395  }
396  else if (thisLabel == stringEoH)
397  {
398  valid |= validEoH;
399  }
400  else
401  {
402  FFStreamError e("Unknown header label " + thisLabel);
403  GPSTK_THROW(e);
404  }
405  }
406 
407  unsigned long allValid;
408  if (version == 2.0 ) allValid = allValid20;
409  else if (version == 2.1 ) allValid = allValid21;
410  else if (version == 2.11) allValid = allValid21;
411  else if (version == 3.0 ) allValid = allValid21;
412  else
413  {
414  FFStreamError e("Unknown or unsupported RINEX version " +
415  asString(version));
416  GPSTK_THROW(e);
417  }
418 
419  if ( (allValid & valid) != allValid)
420  {
421  string errstr("Incomplete or invalid header: missing: ");
422  errstr += bitString(allValid & ~valid);
423  FFStreamError err(errstr);
424  GPSTK_THROW(err);
425  }
426 
427  strm.header = *this;
428  strm.headerRead = true;
429  }
