85  // add to statistics, and to weighted average solution and covariance
86  void add(const Vector<double>& Sol, const Matrix<double>& Cov)
87  throw(Exception)
88  {
89  try {
90  // add to the statistics
91  for(int i=0; i<3; i++) S[i].Add(Sol(i));
92 
93  // NB do NOT include clock(s); this can ruin the position average
94  Vector<double> Sol3(Sol);
95  Sol3.resize(3); // assumes position states come first
96  Matrix<double> Cov3(Cov,0,0,3,3);
97 
98  // information matrix (position only)
99  Matrix<double> Info(inverseSVD(Cov3));
100  if(N == 0) { // first call: dimension and set to zero
101  sumInfo = Matrix<double>(3,3,0.0);
102  sumInfoState = Vector<double>(3,0.0);
103  }
104 
105  // add to the total information
106  sumInfo += Info;
107  sumInfoState += Info * Sol3;
108  ++N;
109  }
110  catch(Exception& e) { GPSTK_RETHROW(e); }
111  }

246  void add(const Vector<double>& Sol, const Matrix<double>& Cov,
247  const Vector<double>& PreFitResid, const Matrix<double>& Partials,
248  const Matrix<double>& invMeasCov)
249  throw(Exception)
250  {
251  was.add(Sol, Cov);
252 
253  // first solution: apriori solution has no clock, so PFR bad
254  if(was.getN() == 1) return;
255 
256  try {
257  Matrix<double> Part(Partials,0,0,Partials.rows(),3);
258  Matrix<double> invMC(invMeasCov);
259  if(invMC.rows() == 0) {
260  invMC=Matrix<double>(Part.rows(),Part.rows());
261  ident(invMC);
262  }
263  Matrix<double> sumInfo(was.getInfo());
264  Matrix<double> Ginv(Part*sumInfo*transpose(Part) + invMC);
265  Matrix<double> G(inverseSVD(Ginv));
266  Vector<double> Gpfr(G*PreFitResid);
267  APV += dot(PreFitResid,Gpfr);
268  ndata += PreFitResid.size();
269  ndof = ndata-sumInfo.rows();
270  }
271  catch(Exception& e) {
272  e.addText("APV failed.");
273  GPSTK_RETHROW(e);
274  }
275  }
