36  pair<list<Path*>*, list<Path*>*> Splitter::splith(double splitter, Path* p, bool top, bool bottom, bool continuous)
37  {
38  if(!(top || bottom) || p->empty())
39  return pair<list<Path* >*, list<Path*>*>((list<Path* >*)0,(list<Path* >*)0);
40 
41  list<Path*> *tl, *bl;
42  if(top)
43  tl = new list<Path*> ();
44  if(bottom)
45  bl = new list<Path*> ();
46 
47  Path* current = new Path(0,0);
48 
49  Path::iterator i=p->begin();
50  bool above = i->second > splitter;
51 
52  if(above && top) current->addPointAbsolute(i->first, i->second);
53  else if(!above && bottom) current->addPointAbsolute(i->first, i->second);
54 
55  i++;
56 
57  double tdouble = 0;
58  for (;i != p->end(); i++)
59  {
60  if (above && (i->second < splitter))
61  {
62  if(top)
63  {
64  if(continuous)
65  {
66  Path::iterator j = i;
67  j--;
68  tdouble = intersecth(splitter,*i,*j);
69  current->addPointAbsolute(tdouble,splitter);
70  }
71 
72  if(!current->empty()) tl->push_back(current);
73  current = new Path(0,0);
74 
75  if(continuous)
76  current->addPointAbsolute(tdouble,splitter);
77  }
78  above = false;
79  }
80  else if(!above && (i->second > splitter))
81  {
82  if(bottom)
83  {
84  if(continuous)
85  {
86  Path::iterator j = i;
87  j--;
88  tdouble = intersecth(splitter,*i,*j);
89  current->addPointAbsolute(tdouble,splitter);
90  }
91 
92  if(!current->empty()) bl->push_back(current);
93  current = new Path(0,0);
94 
95  if(continuous)
96  current->addPointAbsolute(tdouble,splitter);
97  }
98  above = true;
99  }
100  else if(i->second == splitter)
101  {
102  Path::iterator j=i;
103  j++;
104 
105  /*
106  * Only add, push, and make a new path if it doesn't "bounce" off of the
107  * splitter.
108  * So we do a lookahead of one with j.
109  * We will _skip_ this area iff:
110  * - j is at the end of p
111  * - (i-1) is above and j is below
112  * - (i-1) is below and j is above
113  */
114  if( (j != p->end()) &&
115  ((above && (j->second < splitter)) ||
116  (!above && (j->second > splitter))) )
117  {
118  current->addPointAbsolute(i->first, i->second);
119  if (above && top)
120  tl->push_back(current);
121  else if(!above && bottom)
122  bl->push_back(current);
123  current = new Path(0,0);
124  above = !above;
125  }
126  }
127  if ((above && top) || (!above && bottom))
128  current->addPointAbsolute(i->first, i->second);
129  }
130 
131  if(!current->empty())
132  {
133  if(above && top) tl->push_back(current);
134  else if(!above && bottom) bl->push_back(current);
135  }
136 
137  return pair<list<Path*>*, list<Path*>*>(tl, bl);
138  }

140  list<Path*>* Splitter::splitvgap(double gap, Path* p)
141  {
142  if((p==0)||(p->empty()))
143  return 0;
144 
145  list<Path*> *paths = new list<Path*> ();
146 
147  Path* current = new Path(0,0);
148 
149  Path::iterator i=p->begin();
150  Path::iterator last=i;
151  current->addPointAbsolute(i->first, i->second);
152  i++;
153 
154  gap = (gap<0?-gap:gap);
155 
156  for (;i != p->end(); i++, last++)
157  {
158  double dist = i->first - last->first;
159  dist = (dist<0?-dist:dist);
160  if (dist >= gap)
161  {
162  paths->push_back(current);
163  current = new Path(0,0);
164  }
165  current->addPointAbsolute(i->first, i->second);
166  }
167 
168  if(!current->empty())
169  paths->push_back(current);
170 
171  return paths;
172  }

174  std::pair<double,double> Splitter::intersectBox(
175  const std::pair<double,double> inside, const std::pair<double,double> outside,
176  double minX, double maxX, double minY, double maxY)
177  {
178  double x,y;
179  if(outside.first < minX)
180  {
181  x = minX;
182  y = intersectv(x, inside, outside);
183  if(y>minY && y<maxY) return std::pair<double,double>(x,y);
184  }
185  else if(outside.first > maxX)
186  {
187  x = maxX;
188  y = intersectv(x, inside, outside);
189  if(y>minY && y<maxY) return std::pair<double,double>(x,y);
190  }
191 
192  if(outside.second < minY)
193  {
194  y = minY;
195  x = intersecth(y, inside, outside);
196  return std::pair<double,double>(x,y);
197  }
198  else if(outside.second > maxY)
199  {
200  y = maxY;
201  x = intersecth(y, inside, outside);
202  return std::pair<double,double>(x,y);
203  }
204 
205  // outside isn't out of the box, maybe inside is...
206  if(!inBox(inside,minX,maxX,minY,maxY))
207  return intersectBox(outside,inside,minX,maxX,minY,maxY);
208 
209  // Neither outside nor inside are out of the box. Just give back one of
210  // the points (the first)
211  return inside;
212  }

214  std::auto_ptr< std::list< vdraw::Path > > Splitter::interpToBox(double minX, double maxX, double minY, double maxY, const vdraw::Path& p)
215  {
216  using namespace vdraw;
217  Path::const_iterator i=p.begin(),j=p.begin();
218  bool inside = inBox(*i,minX,maxX,minY,maxY);
219  bool lastinside = inside;
220 
221  double cx, cy;
222  p.getOrigin(cx,cy);
223 
224  std::auto_ptr< std::list< Path > > thelist(new std::list<Path>());
225  Path current(cx,cy);
226  if(inside)
227  current.push_back(*i);
228  i++;
229 
230  for(;i!=p.end();i++,j++,lastinside=inside)
231  {
232  inside = inBox(*i,minX,maxX,minY,maxY);
233  if(!inside && lastinside)
234  {
235  current.push_back(intersectBox(*j,*i,minX,maxX,minY,maxY));
236  thelist->push_back(current);
237  current = Path(cx,cy);
238  }
239  else if(inside && !lastinside)
240  {
241  current.push_back(intersectBox(*i,*j,minX,maxX,minY,maxY));
242  current.push_back(*i);
243  }
244  else if(inside)
245  {
246  current.push_back(*i);
247  }
248  }
249 
250  if(current.size() != 0)
251  thelist->push_back(current);
252 
253  return thelist;
254  }
