65 // position for one or more sites, and the resulting ocean loading file will be

206 //------------------------------------------------------------------------------------
207 // Compute the site displacement vector at the given time for the given site.
208 // The site must have been successfully initialized; if not an exception is
209 // thrown.
210 // @param site string Input name of the site; must be the same as previously
211 // successfully passed to initializeSites().
212 // @param t DayTime Input time of interest.
213 // @return Triple containing the North, East and Up components of the site
214 // displacement in meters.
215 // @throw if the site has not been initialized.
216 Triple OceanLoading::computeDisplacement(string site, DayTime t) throw(Exception)
217 {
218 try {
219  if(!isValid(site)) {
220  Exception e("Site " + site + " has not been initialized.");
221  GPSTK_THROW(e);
222  }
223 
224  // get the coefficients for this site
225  vector<double> coeff = coefficientMap[site];
226 
227  // get the astronomical arguments in radians
228  // NB DayTime::year() returns short
229  double angles[11];
230  SchwiderskiArg(int(t.year())-1900, t.DOY(), t.secOfDay(), angles);
231 
232  // compute the radial, west and south components
233  // coefficients are stored by rows: radial, west, south; first amp, then phase
234  // column order same as in SchwiderskiArg() [ as in the file ]
235  Triple dc;
236  for(int i=0; i<3; i++) { // components
237  dc[i] = 0.0;
238  for(int j=0; j<11; j++) // tidal modes
239  dc[i] += coeff[i*11+j]*cos(angles[j]-coeff[33+i*11+j]*DEG_TO_RAD);
240  }
241 
242  // now convert radial,west,south to north,east,up
243  double temp=dc[0];
244  dc[0] = -dc[2]; // N = -S
245  dc[1] = -dc[1]; // E = -W
246  dc[2] = temp; // U = rad
247 
248  return dc;
249 }
250 catch(Exception& e) { GPSTK_RETHROW(e); }
251 catch(exception& e) { Exception E("std except: "+string(e.what())); GPSTK_THROW(E); }
252 catch(...) { Exception e("Unknown exception"); GPSTK_THROW(e); }
253 }

257 // return angles in radians
258 void OceanLoading::SchwiderskiArg(int iyear, int iday, double fday, double* angles)
259  throw()
260 {
261  // ordering is: M2, S2, N2, K2, K1, O1, P1, Q1, Mf, Mm, Ssa
262  // which are : { semi-diurnal }{ diurnal }{long-period}
263  static const double speed[11] = { 1.40519E-4, 1.45444E-4, 1.37880E-4, 1.45842E-4,
264  0.72921E-4, 0.67598E-4, 0.72523E-4, 0.64959E-4,
265  0.053234E-4, 0.026392E-4, 0.003982E-4 };
266  // sun
267  static const double angfac[44] = { 2.0, 0.0, 2.0, 2.0, // 4 : M2, S2, N2, K2
268  1.0, 1.0, -1.0, 1.0, // 8 : K1, O1, P1, Q1
269  0.0, 0.0, 2.0, // 11 : Mf, Mm, Ssa
270  // moon
271  -2.0, 0.0, -3.0, 0.0, // 15 : M2, S2, N2, K2
272  0.0, -2.0, 0.0, -3.0, // 19 : K1, O1, P1, Q1
273  2.0, 1.0, 0.0, // 22 : Mf, Mm, Ssa
274  // lunar perigee
275  0.0, 0.0, 1.0, 0.0, // 26 : M2, S2, N2, K2
276  0.0, 0.0, 0.0, 1.0, // 30 : K1, O1, P1, Q1
277  0.0, -1.0, 0.0, // 33 : Mf, Mm, Ssa
278  // two pi
279  0.0, 0.0, 0.0, 0.0, // 37 : M2, S2, N2, K2
280  0.25,-0.25,-0.25,-0.25, // 41 : K1, O1, P1, Q1
281  0.0, 0.0, 0.0}; // 44 : Mf, Mm, Ssa
282  static const double twopi = 6.28318530718;
283  static const double dtr = 0.0174532925199;
284 
285  // fractional part of the day in seconds
286  //int iday = int(day);
287  //double fday = (day-double(iday))*86400.0;
288  int icapd = iday + 365*(iyear-75)+((iyear-73)/4);
289  //double capt = (27392.500528+1.000000035*double(icapd))/36525.0;
290  double capt = 0.74996579132101300 + 2.73785088295687885e-5 * double(icapd);
291  // mean longitude of sun at beginning of day
292  double H0 = (279.69668+(36000.768930485+0.000303*capt)*capt)*dtr;
293  // mean longitude of moon at beginning of day
294  double S0 = (((0.0000019*capt-0.001133)*capt+481267.88314137)*capt+270.434358)*dtr;
295  // mean longitude of lunar perigee at beginning of day
296  double P0 = (((-0.000012*capt-0.010325)*capt+4069.0340329577)*capt+334.329653)*dtr;
297 
298  //LOG(INFO) << "Schwiderski " << iday << " " << fixed << setprecision(5)
299  //<< setw(11) << fday << " " << icapd << " " << capt
300  //<< " " << H0 << " " << S0 << " " << P0;
301 
302  for(int k=0; k<11; k++) {
303  angles[k] = speed[k]*fday + angfac[k]*H0
304  + angfac[11+k]*S0
305  + angfac[22+k]*P0
306  + angfac[33+k]*twopi;
307  angles[k] = fmod(angles[k],twopi);
308  if(angles[k] < 0.0) angles[k] += twopi;
309  }
310 
311 }
