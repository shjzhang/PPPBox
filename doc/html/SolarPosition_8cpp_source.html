33 // Compute Greenwich Mean Sidereal Time in degrees
34 static double GMST(gpstk::CommonTime t)
35 {
36 
37  static const long JulianEpoch=2451545;
38  // days since epoch
39  double days =( static_cast<JulianDate>(t).jd - JulianEpoch); // days
40  if(days <= 0.0) days -= 1.0;
41  double Tp = days/36525.0; // dim-less
42 
43  // Compute GMST in radians
44  double G;
45  // seconds (24060s = 6h 41min)
46  //G = 24110.54841 + (8640184.812866 + (0.093104 - 6.2e-6*Tp)*Tp)*Tp; // sec
47  //G /= 86400.0; // instead, divide the numbers above manually
48  G = 0.279057273264 + 100.0021390378009*Tp // seconds/86400 = days
49  + (0.093104 - 6.2e-6*Tp)*Tp*Tp/86400.0;
50  G += static_cast<YDSTime>(t).sod/86400.0; // days
51 
52  // put answer between 0 and 360 deg
53  G = fmod(G,1.0);
54  while(G < 0.0) G += 1.0;
55  G *= 360.0; // degrees
56 
57  return G;
58 }

62 //------------------------------------------------------------------------------------
63 // accuracy is about 1 arcminute, when t is within 2 centuries of 2000.
64 // Ref. Astronomical Almanac pg C24, as presented on USNO web site.
65 // input
66 // t epoch of interest
67 // output
68 // lat,lon,R latitude, longitude and distance (deg,deg,m in ECEF) of sun at t.
69 // AR apparent angular radius of sun as seen at Earth (deg) at t.
70 Position SolarPosition(CommonTime t, double& AR) throw()
71 {
72  //const double mPerAU = 149598.0e6;
73  double D; // days since J2000
74  double g,q; // q is mean longitude of sun, corrected for aberration
75  double L; // sun's geocentric apparent ecliptic longitude (deg)
76  //double b=0; // sun's geocentric apparent ecliptic latitude (deg)
77  double e; // mean obliquity of the ecliptic (deg)
78  //double R; // sun's distance from Earth (m)
79  double RA; // sun's right ascension (deg)
80  double DEC; // sun's declination (deg)
81  //double AR; // sun's apparent angular radius as seen at Earth (deg)
82 
83  D = static_cast<JulianDate>(t).jd - 2451545.0;
84  g = (357.529 + 0.98560028 * D) * DEG_TO_RAD;
85  // AA 1990 has g = (357.528 + 0.9856003 * D) * DEG_TO_RAD;
86  q = 280.459 + 0.98564736 * D;
87  // AA 1990 has q = 280.460 + 0.9856474 * D;
88  L = (q + 1.915 * std::sin(g) + 0.020 * std::sin(2*g)) * DEG_TO_RAD;
89 
90  e = (23.439 - 0.00000036 * D) * DEG_TO_RAD;
91  // AA 1990 has e = (23.439 - 0.0000004 * D) * DEG_TO_RAD;
92  RA = std::atan2(std::cos(e)*std::sin(L),std::cos(L)) * RAD_TO_DEG;
93  DEC = std::asin(std::sin(e)*std::sin(L)) * RAD_TO_DEG;
94 
95  // equation of time = apparent solar time minus mean solar time
96  // = [q-RA (deg)]/(15deg/hr)
97 
98  // compute the hour angle of the vernal equinox = GMST and convert RA to lon
99  double lon = fmod(RA-GMST(t),360.0);
100  if(lon < -180.0) lon += 360.0;
101  if(lon > 180.0) lon -= 360.0;
102 
103  double lat = DEC;
104 
105  // ECEF unit vector in direction Earth to sun
106  double xhat = std::cos(lat*DEG_TO_RAD)*std::cos(lon*DEG_TO_RAD);
107  double yhat = std::cos(lat*DEG_TO_RAD)*std::sin(lon*DEG_TO_RAD);
108  double zhat = std::sin(lat*DEG_TO_RAD);
109 
110  // R in AU
111  double R = 1.00014 - 0.01671 * std::cos(g) - 0.00014 * std::cos(2*g);
112  // apparent angular radius in degrees
113  AR = 0.2666/R;
114  // convert to meters
115  R *= 149598.0e6;
116 
117  Position ES;
118  ES.setECEF(R*xhat,R*yhat,R*zhat);
119  return ES;
120 }

123 // Compute the position (latitude and longitude, in degrees) of the sun
124 // given the day of year and the hour of the day.
125 // Adapted from sunpos by D. Coco 12/15/94
126 void CrudeSolarPosition(CommonTime t, double& lat, double& lon) throw()
127 {
128  int doy = static_cast<YDSTime>(t).doy;
129  int hod = int(static_cast<YDSTime>(t).sod/3600.0 + 0.5);
130  lat = std::sin(23.5*DEG_TO_RAD)*std::sin(TWO_PI*double(doy-83)/365.25);
131  lat = lat / std::sqrt(1.0-lat*lat);
132  lat = RAD_TO_DEG*std::atan(lat);
133  lon = 180.0 - hod*15.0;
134 }

136 //------------------------------------------------------------------------------------
137 // Consider the sun and the earth as seen from the satellite. Let the sun be a circle
138 // of angular radius r, center in direction s, and the earth be a (larger) circle
139 // of angular radius R, center in direction e. The circles overlap if |e-s| < R+r;
140 // complete overlap if |e-s| < R. Let L == |e-s|.
141 // What is the area of overlap if R-r < L < R+r ?
142 // Call the two points where the circles intersect p1 and p2. Draw a line from e to s;
143 // call the points where this line intersects the two circles r1 and R1, respectively.
144 // Draw lines from e to s, e to p1, e to p2, s to p1 and s to p2. Call the angle
145 // between e-s and e-p1 alpha, and that between s-e and s-p1, beta.
146 // Draw a rectangle with top and bottom parallel to e-s passing through p1 and p2,
147 // and with sides passing through s and r1. Similarly for e and R1. Note that the
148 // area of intersection lies within the intersection of these two rectangles.
149 // Call the area of the rectangle outside the circles A and B. The height H of the
150 // rectangles is
151 // H = 2Rsin(alpha) = 2rsin(beta)
152 // also L = rcos(beta)+Rcos(alpha)
153 // The area A will be the area of the rectangle
154 // minus the area of the wedge formed by the angle 2*alpha
155 // minus the area of the two triangles which meet at s :
156 // A = RH - (2alpha/2pi)*pi*R*R - 2*(1/2)*(H/2)Rcos(alpha)
157 // Similarly
158 // B = rH - (2beta/2pi)*pi*r*r - 2*(1/2)*(H/2)rcos(beta)
159 // The area of intersection will be the area of the rectangular intersection
160 // minus the area A
161 // minus the area B
162 // Intersection = H(R+r-L) - A - B
163 // = HR+Hr-HL -HR+alpha*R*R+(H/2)Rcos(alpha) -Hr+beta*r*r+(H/2)rcos(beta)
164 // Cancel terms, and substitute for L using above equation L = ..
165 // = -(H/2)rcos(beta)-(H/2)Rcos(alpha)+alpha*R*R+beta*r*r
166 // substitute for H/2
167 // = -R*R*sin(alpha)cos(alpha)-r*r*sin(beta)cos(beta)+alpha*R*R+beta*r*r
168 // Intersection = R*R*[alpha-sin(alpha)cos(alpha)]+r*r*[beta-sin(beta)cos(beta)]
169 // Solve for alpha and beta in terms of R, r and L using the H and L relations above
170 // (r/R)cos(beta)=(L/R)-cos(alpha)
171 // (r/R)sin(beta)=sin(alpha)
172 // so
173 // (r/R)^2 = (L/R)^2 - (2L/R)cos(alpha) + 1
174 // cos(alpha) = (R/2L)(1+(L/R)^2-(r/R)^2)
175 // cos(beta) = (L/r) - (R/r)cos(alpha)
176 // and 0 <= alpha or beta <= pi
177 //
178 // Rearth angular radius of the earth as seen at the satellite
179 // Rsun angular radius of the sun as seen at the satellite
180 // dES angular distance of the sun from the earth
181 // return fraction (0 <= f <= 1) of area of sun covered by earth
182 // units only need be consistent
183 double shadowFactor(double Rearth, double Rsun, double dES) throw()
184 {
185  if(dES >= Rearth+Rsun) return 0.0;
186  if(dES <= std::fabs(Rearth-Rsun)) return 1.0;
187  double r=Rsun, R=Rearth, L=dES;
188  if(Rsun > Rearth) { r=Rearth; R=Rsun; }
189  double cosalpha = (R/L)*(1.0+(L/R)*(L/R)-(r/R)*(r/R))/2.0;
190  double cosbeta = (L/r) - (R/r)*cosalpha;
191  double sinalpha = ::sqrt(1-cosalpha*cosalpha);
192  double sinbeta = ::sqrt(1-cosbeta*cosbeta);
193  double alpha = ::asin(sinalpha);
194  double beta = ::asin(sinbeta);
195  double shadow = r*r*(beta-sinbeta*cosbeta)+R*R*(alpha-sinalpha*cosalpha);
196  shadow /= ::acos(-1.0)*Rsun*Rsun;
197  return shadow;
198 }

201 // From AA 1990 D46
202 Position LunarPosition(CommonTime t, double& AR) throw()
203 {
204  // days since J2000
205  double N = static_cast<JulianDate>(t).jd-2451545.0;
206  // centuries since J2000
207  double T = N/36525.0;
208  // ecliptic longitude
209  double lam = DEG_TO_RAD*(218.32 + 481267.883*T
210  + 6.29 * ::sin(DEG_TO_RAD*(134.9+477198.85*T))
211  - 1.27 * ::sin(DEG_TO_RAD*(259.2-413335.38*T))
212  + 0.66 * ::sin(DEG_TO_RAD*(235.7+890534.23*T))
213  + 0.21 * ::sin(DEG_TO_RAD*(269.9+954397.70*T))
214  - 0.19 * ::sin(DEG_TO_RAD*(357.5+ 35999.05*T))
215  - 0.11 * ::sin(DEG_TO_RAD*(259.2+966404.05*T)));
216  // ecliptic latitude
217  double bet = DEG_TO_RAD*(5.13 * ::sin(DEG_TO_RAD*( 93.3+483202.03*T))
218  + 0.28 * ::sin(DEG_TO_RAD*(228.2+960400.87*T))
219  - 0.28 * ::sin(DEG_TO_RAD*(318.3+ 6003.18*T))
220  - 0.17 * ::sin(DEG_TO_RAD*(217.6-407332.20*T)));
221  // horizontal parallax
222  double par = DEG_TO_RAD*(0.9508
223  + 0.0518 * ::cos(DEG_TO_RAD*(134.9+477198.85*T))
224  + 0.0095 * ::cos(DEG_TO_RAD*(259.2-413335.38*T))
225  + 0.0078 * ::cos(DEG_TO_RAD*(235.7+890534.23*T))
226  + 0.0028 * ::cos(DEG_TO_RAD*(269.9+954397.70*T)));
227 
228  // obliquity of the ecliptic
229  double eps = (23.439 - 0.00000036 * N) * DEG_TO_RAD;
230 
231  // convert ecliptic lon,lat to geocentric lon,lat
232  double l = ::cos(bet)*::cos(lam);
233  double m = ::cos(eps)*::cos(bet)*::sin(lam) - ::sin(eps)*::sin(bet);
234  double n = ::sin(eps)*::cos(bet)*::sin(lam) + ::cos(eps)*::sin(bet);
235 
236  // convert to right ascension and declination,
237  // (referred to mean equator and equinox of date)
238  double RA = ::atan2(m,l) * RAD_TO_DEG;
239  double DEC = ::asin(n) * RAD_TO_DEG;
240 
241  // compute the hour angle of the vernal equinox = GMST and convert RA to lon
242  double lon = ::fmod(RA-GMST(t),360.0);
243  if(lon < -180.0) lon += 360.0;
244  if(lon > 180.0) lon -= 360.0;
245 
246  double lat = DEC;
247 
248  // apparent semidiameter of moon (in radians)
249  AR = 0.2725 * par;
250  // moon distance in meters
251  double R = 1.0 / ::sin(par);
252  R *= 6378137.0;
253 
254  // ECEF vector in direction Earth to moon
255  double x = R*std::cos(lat*DEG_TO_RAD)*std::cos(lon*DEG_TO_RAD);
256  double y = R*std::cos(lat*DEG_TO_RAD)*std::sin(lon*DEG_TO_RAD);
257  double z = R*std::sin(lat*DEG_TO_RAD);
258 
259  Position EM;
260  EM.setECEF(x,y,z);
261 
262  return EM;
263 }
