34  /* Compute satellite position & velocity at the given time
35  * using this ephemeris.
36  *
37  * @throw InvalidRequest if required data has not been stored.
38  */
39  Xvt GloEphemeris::svXvt(const CommonTime& epoch) const
40  throw( gpstk::InvalidRequest )
41  {
42 
43  // Check that the given epoch is within the available time limits.
44  // We have to add a margin of 15 minutes (900 seconds).
45  if ( epoch < (ephTime - 900.0) ||
46  epoch >= (ephTime + 900.0) )
47  {
48  InvalidRequest e( "Requested time is out of ephemeris data" );
49  GPSTK_THROW(e);
50  }
51 
52  // Values to be returned will be stored here
53  Xvt sv;
54 
55  // If the exact epoch is found, let's return the values
56  if ( epoch == ephTime ) // exact match for epoch
57  {
58 
59  sv.x[0] = x[0]*1.e3; // m
60  sv.x[1] = x[1]*1.e3; // m
61  sv.x[2] = x[2]*1.e3; // m
62  sv.v[0] = v[0]*1.e3; // m/sec
63  sv.v[1] = v[1]*1.e3; // m/sec
64  sv.v[2] = v[2]*1.e3; // m/sec
65 
66  // In the GLONASS system, 'clkbias' already includes the
67  // relativistic correction, therefore we must substract the late
68  // from the former.
69  sv.relcorr = sv.computeRelativityCorrection();
70  sv.clkbias = clkbias + clkdrift * (epoch - ephTime) - sv.relcorr;
71  sv.clkdrift = clkdrift;
72  sv.frame = ReferenceFrame::PZ90;
73 
74  // We are done, let's return
75  return sv;
76 
77  }
78 
79  // Get the data out of the GloRecord structure
80  double px( x[0] ); // X coordinate (km)
81  double vx( v[0] ); // X velocity (km/s)
82  double ax( a[0] ); // X acceleration (km/s^2)
83  double py( x[1] ); // Y coordinate
84  double vy( v[1] ); // Y velocity
85  double ay( a[1] ); // Y acceleration
86  double pz( x[2] ); // Z coordinate
87  double vz( v[2] ); // Z velocity
88  double az( a[2] ); // Z acceleration
89 
90  // We will need some PZ-90 ellipsoid parameters
91  PZ90Ellipsoid pz90;
92  double we( pz90.angVelocity() );
93 
94  // Get sidereal time at Greenwich at 0 hours UT
95  double gst( getSidTime( ephTime ) );
96  double s0( gst*PI/12.0 );
97  YDSTime ytime( ephTime );
98  double numSeconds( ytime.sod );
99  double s( s0 + we*numSeconds );
100  double cs( std::cos(s) );
101  double ss( std::sin(s) );
102 
103  // Initial state matrix
104  Vector<double> initialState(6), accel(3), dxt1(6), dxt2(6), dxt3(6),
105  dxt4(6), tempRes(6);
106 
107  // Get the reference state out of GloEphemeris object data. Values
108  // must be rotated from PZ-90 to an absolute coordinate system
109  // Initial x coordinate (m)
110  initialState(0) = (px*cs - py*ss);
111  // Initial y coordinate
112  initialState(2) = (px*ss + py*cs);
113  // Initial z coordinate
114  initialState(4) = pz;
115 
116  // Initial x velocity (m/s)
117  initialState(1) = (vx*cs - vy*ss - we*initialState(2) );
118  // Initial y velocity
119  initialState(3) = (vx*ss + vy*cs + we*initialState(0) );
120  // Initial z velocity
121  initialState(5) = vz;
122 
123 
124  // Integrate satellite state to desired epoch using the given step
125  double rkStep( step );
126 
127  if ( (epoch - ephTime) < 0.0 ) rkStep = step*(-1.0);
128  CommonTime workEpoch( ephTime );
129 
130  double tolerance( 1e-9 );
131  bool done( false );
132  while (!done)
133  {
134 
135  // If we are about to overstep, change the stepsize appropriately
136  // to hit our target final time.
137  if( rkStep > 0.0 )
138  {
139  if( (workEpoch + rkStep) > epoch )
140  rkStep = (epoch - workEpoch);
141  }
142  else
143  {
144  if ( (workEpoch + rkStep) < epoch )
145  rkStep = (epoch - workEpoch);
146  }
147 
148  numSeconds += rkStep;
149  s = s0 + we*( numSeconds );
150  cs = std::cos(s);
151  ss = std::sin(s);
152 
153  // Accelerations are computed once per iteration
154  accel(0) = ax*cs - ay*ss;
155  accel(1) = ax*ss + ay*cs;
156  accel(2) = az;
157 
158  dxt1 = derivative( initialState, accel );
159  for( int j = 0; j < 6; ++j )
160  tempRes(j) = initialState(j) + rkStep*dxt1(j)/2.0;
161 
162  dxt2 = derivative( tempRes, accel );
163  for( int j = 0; j < 6; ++j )
164  tempRes(j) = initialState(j) + rkStep*dxt2(j)/2.0;
165 
166  dxt3 = derivative( tempRes, accel );
167  for( int j = 0; j < 6; ++j )
168  tempRes(j) = initialState(j) + rkStep*dxt3(j);
169 
170  dxt4 = derivative( tempRes, accel );
171  for( int j = 0; j < 6; ++j )
172  initialState(j) = initialState(j) + rkStep * ( dxt1(j)
173  + 2.0 * ( dxt2(j) + dxt3(j) ) + dxt4(j) ) / 6.0;
174 
175 
176  // If we are within tolerance of the target time, we are done.
177  workEpoch += rkStep;
178  if ( std::fabs(epoch - workEpoch ) < tolerance )
179  done = true;
180 
181  } // End of 'while (!done)...'
182 
183 
184  px = initialState(0);
185  py = initialState(2);
186  pz = initialState(4);
187  vx = initialState(1);
188  vy = initialState(3);
189  vz = initialState(5);
190 
191  sv.x[0] = 1000.0*( px*cs + py*ss ); // X coordinate
192  sv.x[1] = 1000.0*(-px*ss + py*cs); // Y coordinate
193  sv.x[2] = 1000.0*pz; // Z coordinate
194  sv.v[0] = 1000.0*( vx*cs + vy*ss + we*(sv.x[1]/1000.0) ); // X velocity
195  sv.v[1] = 1000.0*(-vx*ss + vy*cs - we*(sv.x[0]/1000.0) ); // Y velocity
196  sv.v[2] = 1000.0*vz; // Z velocity
197 
198  // In the GLONASS system, 'clkbias' already includes the relativistic
199  // correction, therefore we must substract the late from the former.
200  sv.relcorr = sv.computeRelativityCorrection();
201  sv.clkbias = clkbias + clkdrift * (epoch - ephTime) - sv.relcorr;
202  sv.clkdrift = clkdrift;
203  sv.frame = ReferenceFrame::PZ90;
204 
205  // We are done, let's return
206  return sv;
207 
208 
209  } // End of method 'GloEphemeris::svXvt(const CommonTime& t)'

387  // Compute true sidereal time (in hours) at Greenwich at 0 hours UT.
388  double GloEphemeris::getSidTime( const CommonTime& time ) const
389  {
390 
391  // The following algorithm is based on the paper:
392  // Aoki, S., Guinot,B., Kaplan, G. H., Kinoshita, H., McCarthy, D. D.
393  // and P.K. Seidelmann. 'The New Definition of Universal Time'.
394  // Astronomy and Astrophysics, 105, 359-361, 1982.
395 
396  // Get the Julian Day at 0 hours UT (jd)
397  YDSTime ytime( time );
398  double year( ytime.year );
399  int doy( ytime.doy );
400  int temp( floor(365.25 * (year - 1.0)) + doy );
401 
402  double jd( static_cast<double>(temp)+ 1721409.5 );
403 
404  // Compute the Julian centuries (36525 days)
405  double jc( (jd - 2451545.0)/36525.0 );
406 
407  // Compute the sidereal time, in seconds
408  double sid( 24110.54841 + jc * ( 8640184.812866
409  + jc * ( 0.093104 - jc * 0.0000062 ) ) );
410 
411  sid = sid / 3600.0;
412  sid = fmod(sid, 24.0);
413  if( sid < 0.0 ) sid = sid + 24.0;
414 
415  return sid;
416 
417  }; // End of method 'GloEphemeris::getSidTime()'

420  // Function implementing the derivative of GLONASS orbital model.
421  Vector<double> GloEphemeris::derivative( const Vector<double>& inState,
422  const Vector<double>& accel )
423  const
424  {
425 
426  // We will need some important PZ90 ellipsoid values
427  PZ90Ellipsoid pz90;
428  const double j20( pz90.j20() );
429  const double mu( pz90.gm_km() );
430  const double ae( pz90.a_km() );
431 
432  // Let's start getting the current satellite position and velocity
433  double x( inState(0) ); // X coordinate
434  double vx( inState(1) ); // X velocity
435  double y( inState(2) ); // Y coordinate
436  double vy( inState(3) ); // Y velocity
437  double z( inState(4) ); // Z coordinate
438  double vz( inState(5) ); // Z velocity
439 #pragma unused(vx,vy,vz)
440 
441  double r2( x*x + y*y + z*z );
442  double r( std::sqrt(r2) );
443  double xmu( mu/r2 );
444  double rho( ae/r );
445  double xr( x/r );
446  double yr( y/r );
447  double zr( z/r );
448  double zr2( zr*zr );
449  double k1(j20*xmu*1.5*rho*rho);
450  double cm( k1*(1.0-5.0*zr2) );
451  double cmz( k1*(3.0-5.0*zr2) );
452  double k2(cm-xmu);
453 
454  double gloAx( k2*xr + accel(0) );
455  double gloAy( k2*yr + accel(1) );
456  double gloAz( (cmz-xmu)*zr + accel(2) );
457 
458  Vector<double> dxt(6, 0.0);
459 
460  // Let's insert data related to X coordinates
461  dxt(0) = inState(1); // Set X' = Vx
462  dxt(1) = gloAx; // Set Vx' = gloAx
463 
464  // Let's insert data related to Y coordinates
465  dxt(2) = inState(3); // Set Y' = Vy
466  dxt(3) = gloAy; // Set Vy' = gloAy
467 
468  // Let's insert data related to Z coordinates
469  dxt(4) = inState(5); // Set Z' = Vz
470  dxt(5) = gloAz; // Set Vz' = gloAz
471 
472  return dxt;
473 
474  } // End of method 'GloEphemeris::derivative()'
