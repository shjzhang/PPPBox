155  // function that returns the cosine of angle between this and right
156  double Triple :: cosVector(const Triple& right) const
157  throw(GeometryException)
158  {
159  double rx, ry, cosvects;
160 
161  rx = dot(*this);
162  ry = right.dot(right);
163 
164  if (rx <= 1e-14 || ry <= 1e-14)
165  {
166  GPSTK_THROW(GeometryException("Divide by Zero Error"));
167  }
168  cosvects = dot(right) / ::sqrt(rx * ry);
169 
170  /* this if checks for and corrects round off error */
171  if (fabs(cosvects) > 1.0e0)
172  {
173  cosvects = fabs(cosvects) / cosvects;
174  }
175 
176  return cosvects;

203  // Calculates a satellites azimuth from a station
204  double Triple :: azAngle(const Triple& right) const
205  throw(GeometryException)
206  {
207  double xy, xyz, cosl, sinl, sint, xn1, xn2, xn3, xe1, xe2;
208  double z1, z2, z3, p1, p2, test, alpha;
209 
210  xy = (*this)[0] * (*this)[0] + (*this)[1] * (*this)[1] ;
211  xyz = xy + (*this)[2] * (*this)[2] ;
212  xy = ::sqrt(xy);
213  xyz = ::sqrt(xyz);
214 
215  if (xy <= 1e-14 || xyz <=1e-14)
216  GPSTK_THROW(GeometryException("Divide by Zero Error"))
217 
218  cosl = (*this)[0] / xy;
219  sinl = (*this)[1] / xy;
220  sint = (*this)[2] / xyz;
221 
222  xn1 = -sint * cosl;
223  xn2 = -sint * sinl;
224  xn3 = xy/xyz;
225 
226  xe1 = -sinl;
227  xe2 = cosl;
228 
229  z1 = right[0] - (*this)[0];
230  z2 = right[1] - (*this)[1];
231  z3 = right[2] - (*this)[2];
232 
233  p1 = (xn1 * z1) + (xn2 * z2) + (xn3 * z3) ;
234  p2 = (xe1 * z1) + (xe2 * z2) ;
235 
236  test = fabs(p1) + fabs(p2);
237 
238  if (test < 1.0e-16)
239  {
240  GPSTK_THROW(GeometryException("azAngle(), failed p1+p2 test."));
241  }
242 
243  alpha = 90 - ::atan2(p1, p2) * RAD_TO_DEG;
244  if (alpha < 0)
245  {
246  return alpha + 360;
247  }
248  else
249  {
250  return alpha;
251  }
252  }
