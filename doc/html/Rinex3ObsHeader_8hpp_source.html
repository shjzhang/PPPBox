311  bool isValid() const
312  {
313  if(!validEoH) return false;
314  unsigned long allValid;
315  if( version < 3.00) allValid = allValid2;
316  else if(version < 3.01) allValid = allValid30;
317  else if(version < 3.02) allValid = allValid301;
318  else allValid = allValid302;
319  return ((valid & allValid) == allValid);
320  }
