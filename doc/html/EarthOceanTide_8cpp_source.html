52  void EarthOceanTide::loadTideFile(std::string fileName,int NMAX,double XMIN)
53  {
54  if(isLoaded) return;
55 
56  // open the file
57  std::ifstream fin(fileName.c_str());
58  if(!fin.good())
59  {
60  Exception e("Can not Open the CSR Ocean Tide File:"+fileName);
61  GPSTK_THROW(e);
62  }
63 
64  // read the file
65  std::string buf;
66 
67  // line 1 ,skip it
68  getline(fin,buf);
69 
70  // line 2
71  getline(fin,buf);
72  NWAV = asInt(buf.substr(0,4));
73  NTOT = asInt(buf.substr(4,4));
74  NMX = asInt(buf.substr(8,4));
75  MMX = asInt(buf.substr(12,4));
76 
77  // line 3 ,skip it
78  getline(fin,buf);
79 
80  // line 4
81  getline(fin,buf);
82  RRE = asDouble(buf.substr(0,21));
83  RHOW = asDouble(buf.substr(21,21));
84  XME = asDouble(buf.substr(42,21));
85  PFCN = asDouble(buf.substr(63,21));
86  XXX = asDouble(buf.substr(84,21));
87 
88  // line 5 ~ 8
89  for(int i=0;i<4;i++)
90  {
91  getline(fin,buf);
92  for(int j=0;j<6;j++ )
93  {
94  if((i*6+j) > 19) break;
95  tideData.KNMP[i*6+j] = asDouble(buf.substr(j*21,21));
96  }
97  }
98 
99  //IGNORE THE NWAV LINE
100  for(int i=0;i<NWAV;i++) getline(fin,buf);
101 
102  // CEXTRACT REQUIRED INFORMATION FROM NEXT NTOT LINES
103  int id = 0;
104  for(int i=0;i<NTOT;i++)
105  {
106  getline(fin,buf);
107 
108  tideData.NDOD[id][0] = asInt(buf.substr(13,1));
109  tideData.NDOD[id][1] = asInt(buf.substr(14,1));
110  tideData.NDOD[id][2] = asInt(buf.substr(15,1));
111 
112  tideData.NDOD[id][3] = asInt(buf.substr(17,1));
113  tideData.NDOD[id][4] = asInt(buf.substr(18,1));
114  tideData.NDOD[id][5] = asInt(buf.substr(19,1));
115 
116  tideData.NM[id][0] = asInt(buf.substr(24,2));
117  tideData.NM[id][1] = asInt(buf.substr(26,2));
118 
119  tideData.CSPM[id][0] = asDouble(buf.substr(30,22));
120  tideData.CSPM[id][1] = asDouble(buf.substr(52,22));
121  tideData.CSPM[id][2] = asDouble(buf.substr(74,22));
122  tideData.CSPM[id][3] = asDouble(buf.substr(96,22));
123 
124 
125  if((tideData.NM[id][0]<=NMAX) &&
126  (
127  (fabs(tideData.CSPM[id][0])>XMIN) ||
128  (fabs(tideData.CSPM[id][1])>XMIN) ||
129  (fabs(tideData.CSPM[id][2])>XMIN) ||
130  (fabs(tideData.CSPM[id][3])>XMIN))
131  )
132  {
133  for(int j=1;j<6;j++) tideData.NDOD[id][j]-=5;
134 
135  id ++;
136  tideData.NTACT = id;
137  }

139  /*
140  if(m_csr_otide.NTACT>=MAXTRM)

142  cerr << "ERROR: SR OTIDES: NOT ALL TERMS AVAILABLE IN FILE" << endl;
143  exit(1);
144  }*/
145  }

147  // close the file
148  fin.close();

150  isLoaded = true;
151 
152  FAC[0] = 1.0;
153  for(int i=1;i<=30;i++)
154  {
155  FAC[i]=FAC[i-1]*i;
156  }

160  /* Ocean pole tide to normalized earth potential coefficients
161  *
162  * @param mjdUtc UTC in MJD
163  * @param dC Correction to normalized coefficients dC
164  * @param dS Correction to normalized coefficients dS
165  * C20 C21 C22 C30 C31 C32 C33 C40 C41 C42 C43 C44
166  */
167  void EarthOceanTide::getOceanTide(double mjdUtc, double dC[], double dS[] )
168  {
169  try
170  {
171  loadTideFile(fileName,maxN,minX);
172  }
173  catch (...)
174  {
175  // faild to get the ocean tide model
176  // return zeros
177  const int n = (maxN-1)*(maxN+4)/2;
178  for(int i=0;i<n;i++)
179  {
180  dC[i] = 0.0;
181  dS[i] = 0.0;
182  }
183 
184  return;
185  }
186 
187  UTCTime utc;
188  utc=mjdUtc;
189 
190  // CC PURPOSE : COMPUTE DOODSON'S FUNDAMENTAL ARGUMENTS (BETA)
191  // CC AND FUNDAMENTAL ARGUMENTS FOR NUTATION (FNUT)
192  double BETA[6]={0.0};
193  double FNUT[5] ={0.0};
194  ReferenceFrames::doodsonArguments(utc.asUT1(), utc.asTT(), BETA,FNUT);
195 
196  for(int i=0;i<tideData.NTACT;i++)
197  {
198  int N= tideData.NM[i][0];
199  int M= tideData.NM[i][1];
200 
201  if(tideData.NM[i][0]>maxN) continue;
202 
203  double delta = M ? 0 : 1;
204 
205  double FNM = 4.0*ASConstant::PI*G*RHOW/GE* std::sqrt(FAC[N+M]/FAC[N-M]/(2.0*N+1.0)/(2.0-delta))
206  *(1.0+tideData.KNMP[N-1])/(2.0*N+1)/100.0;
207 
208  double ARG =0;
209  for(int j=0;j<6;j++)
210  {
211  ARG +=tideData.NDOD[i][j]*BETA[j];
212  }
213 
214  double CARG = std::cos(ARG);
215  double SARG = std::sin(ARG);
216 
217  int index=N*(N+1)/2-3+M;
218 
219  dC[index]=dC[index]
220  +FNM*(
221  (tideData.CSPM[i][0]+tideData.CSPM[i][2])*CARG
222  +(tideData.CSPM[i][1]+tideData.CSPM[i][3])*SARG);
223 
224  dS[index]=dS[index]
225  +FNM*(
226  (tideData.CSPM[i][1]-tideData.CSPM[i][3])*CARG
227  -(tideData.CSPM[i][0]-tideData.CSPM[i][2])*SARG);
228  }
229 
230  } // End of method 'EarthOceanTide::getOceanTide()'
