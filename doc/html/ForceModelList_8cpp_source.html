46  Vector<double> ForceModelList::getDerivatives(UTCTime utc, EarthBody& bref, Spacecraft& sc)
47  {
48  const int np = setFMT.size(); //getNP();
49 
50  a.resize(3,0.0);
51  da_dr.resize(3,3,0.0);
52  da_dv.resize(3,3,0.0);
53  da_dp.resize(3,np,0.0);
54 
55  da_dcd.resize(3,1,0.0);
56  da_dcr.resize(3,1,0.0);
57 
58  for(std::list<ForceModel*>::iterator it = forceList.begin();
59  it != forceList.end();
60  ++it)
61  {
62  (*it)->doCompute(utc,bref,sc);
63 
64  a += (*it)->getAccel();
65  da_dr += (*it)->partialR();
66  da_dv += (*it)->partialV();
67  //da_dp += (*it)->partialP();
68 
69  //cout<<(*it)->modelName()<<endl;
70 
71  da_dcd += (*it)->partialCd();
72  da_dcr += (*it)->partialCr();
73 
74  }
75 
76  // declare a counter
77  int i = 0;
78 
79  for(std::set<ForceModel::ForceModelType>::iterator it = setFMT.begin();
80  it!=setFMT.end();
81  it++)
82  {
83  if((*it)==ForceModel::Cd)
84  {
85  da_dp(0,i) = da_dcd(0,0);
86  da_dp(1,i) = da_dcd(1,0);
87  da_dp(2,i) = da_dcd(2,0);
88  }
89  else if((*it)==ForceModel::Cr)
90  {
91  da_dp(0,i) = da_dcr(0,0);
92  da_dp(1,i) = da_dcr(1,0);
93  da_dp(2,i) = da_dcr(2,0);
94  }
95  else
96  {
97  Exception e("Error in ForceModelList::getDerivatives():Unexpect ForceModelType");
98  GPSTK_THROW(e);
99  }
100 
101  i++;
102  }
103 
104  /* Transition Matrix (6+np)*(6+np)
105  | |
106  | dr_dr0 dr_dv0 dr_dp0 |
107  | |
108  phi= | dv_dr0 dv_dv0 dv_dp0 |
109  | |
110  | 0 0 I |
111  | |
112  */
113  Matrix<double> phi = sc.getTransitionMatrix();
114 
115  /* A Matrix (6+np)*(6+np)
116  | |
117  | 0 I 0 |
118  | |
119  A = | da_dr da_dv da_dp |
120  | |
121  | 0 0 0 |
122  | |
123  */
124  Matrix<double> A = getAMatrix();
125 
126  /* dphi Matrix
127  | |
128  | dv_dr0 dv_dv0 dv_dp0 |
129  | |
130  dphi = | da_dr0 da_dv0 da_dp0 |
131  | |
132  | 0 0 0 |
133  | |
134 
135  da_dr0 = da_dr*dr_dr0 + da_dv*dv_dr0
136 
137  da_dv0 = da_dr*dr_dv0 + da_dv*dv_dv0
138 
139  da_dp0 = da_dr*dr_dp0 + da_dv*dv_dp0 + da_dp0;
140  */
141  Matrix<double> dphi = A * phi;
142 
143 
144  Vector<double> r = sc.R();
145  Vector<double> v = sc.V();
146 
147  gpstk::Vector<double> dy(42+6*np,0.0);
148 
149  dy(0) = v(0); // v
150  dy(1) = v(1);
151  dy(2) = v(2);
152  dy(3) = a(0); // a
153  dy(4) = a(1);
154  dy(5) = a(2);
155 
156  for(int i=0;i<3;i++)
157  {
158  for(int j=0;j<3;j++)
159  {
160  dy(6+i*3+j) = dphi(i,j); // dv_dr0
161  dy(15+i*3+j) = dphi(i,j+3); // dv_dv0
162  dy(24+3*np+i*3+j) = dphi(i+3,j); // da_dr0
163  dy(33+3*np+i*3+j) = dphi(i+3,j+3); // da_dv0
164  }
165  for(int k=0;k<np;k++)
166  {
167  dy(24+i*np+k) = dphi(i,i*np+k); // dv_dp0
168  dy(42+3*np+i*np+k) = dphi(i+3,i*np+k); // da_dp0
169  }
170  }
171 
172  return dy;
