60  void Axis::setAngle(double angle)
61  {
62  // Save the angle, normalized to be between 0 and 360
63  line_direction = fmod(angle,vdraw::TWO_PI);
64  if(line_direction < 0) line_direction += 360.0;
65 
66  // Direction of the line, lets keep cos/sin to a minimum
67  cosdir = cos(angle);
68  sindir = sin(angle);
69 
70  // Ticks are 90 degrees rotated
71  costic = cos(angle+vdraw::HALF_PI);
72  sintic = sin(angle+vdraw::HALF_PI);
73  }

124  void Axis::axisTickRecursive(bool draw, int depth, double ratio, double dist, double length)
125  {
126  // Base case:
127  if(depth < 0) return;
128  // Bounds checking
129  if(ratio > 1.0001) return;
130  if(ratio < 0.0)
131  {
132  if(ratio+dist < 0.0) return;
133  draw = false;
134  }
135  // Draw
136  if(draw)
137  {
138  drawTick(ratio,length);
139  }
140  // Recurse
141  if(axis_style.subticks_per_tick < 1)
142  return;
143  double t = dist / (1.0+axis_style.subticks_per_tick);
144  axisTickRecursive(false,depth-1,ratio ,t,length*axis_style.tick_depth_multiplier);
145  for(int i=1;i<=axis_style.subticks_per_tick;i++)
146  axisTickRecursive(true ,depth-1,ratio+i*t,t,length*axis_style.tick_depth_multiplier);
147  }

156  void Axis::fromTic(double &x1, double &y1, double &x2, double &y2,
157  double distratio, double length)
158  {
159  double tx,ty;
160  fromRatio(tx,ty,distratio);
161  bool ta = axis_style.tick_position!=AxisStyle::BELOW;
162  bool tb = axis_style.tick_position!=AxisStyle::ABOVE;
163  x1 = tx; x2 = tx; y1 = ty; y2 = ty;
164  if(ta&&tb) length=length/2;
165  if(ta)
166  {
167  x1 = tx+costic*length;
168  y1 = ty+sintic*length;
169  }
170  if(tb)
171  {
172  x2 = tx-costic*length;
173  y2 = ty-sintic*length;
174  }

177  void Axis::guessTickInfo(double &startpos, double &startval, double &distpos, double &distval)
178  {
179  double tickdx;
180 
181  if(gap<=0.0)
182  tickdx = pow(10.0,floor(log10(fabs(max-min))));
183  else
184  tickdx = gap;
185 
186  if (fmod(max,tickdx)!= 0)
187  {
188  if (max>0)
189  axesMax = max-fmod(max,tickdx)+tickdx;
190  else
191  axesMax = max-fmod(max,tickdx);
192  }
193  else
194  axesMax = max;
195  if (fmod(min,tickdx) != 0)
196  {
197  if (min<0)
198  axesMin = min-fmod(min,tickdx)-tickdx;
199  else
200  axesMin = min-fmod(min,tickdx);
201  }
202  else
203  axesMin = min;
204 
205  double w = max-min;
206  double aw = axesMax-axesMin;
207  // See function header comments about these values.
208  startpos = 0.0;
209  startval = axesMin;
210  distpos = tickdx/aw;
211  distval = tickdx;
212  //printf("--><%f,%f> (%f,%f) [%f,%f,%f,%f] %f\n",
213  // min,max,axesMin,axesMax,startpos,startval,distpos,distval,tickdx);
214  if(axis_style.tight_bounds)
215  {
216  startpos = (axesMin-min)/w;
217  distpos = tickdx/w;
218  }
219  else
220  {
221  max = axesMax;
222  min = axesMin;
223  w = aw;
224  }
225  if((w / tickdx) < 4 )
226  {
227  distpos /= 4;
228  distval /= 4;
229  }
230  }

233  {
234  using namespace vdraw;

238  int n=-1;

242  // This returns -1 if it fails, i.e. label_format was bad.

244  const char * c = NULL;

249  else if(pval >= 100000) c="%4g";
250  else if(pval >= 1) c="%g"; //c="%4.f";
251  else if(pval >= 0.1) c="%g"; //c="%.1f";
252  else if(pval >= 0.01) c="%g"; //c="%.2f";
253  else c="%4.e";
254  // FIXME Fix labels for values with small later numbers, for example if
255  // value is like 1.3003003003 and want it displayed as 1.3
256  }
257  std::sprintf(buffer,c,value);
258  std::string textString = std::string(buffer);
259  Text t(textString.c_str(),x,y,style);

272  bool Axis::label(vdraw::Text &t, double value, int direction, const vdraw::TextStyle &style)
273  {
274  using namespace vdraw;
275  // Quadrant
276  // 2 | 1
277  // ---|---
278  // 3 | 4
279  short quad = 0;
280  if(sindir>0)
281  {
282  if(cosdir>0) quad = 1;
283  else quad = 2;
284  }
285  else
286  {
287  if(cosdir>0) quad = 4;
288  else quad = 3;
289  }
290  bool above = direction==AxisStyle::ABOVE;
291 
292  if(line_direction == UP || line_direction == DOWN)
293  {
294  // Simply draw it to the left or right, horizontal
295  // Set alignment
296  if(cos(line_direction-(direction*vdraw::HALF_PI)) > 0)
297  t.setAlignment(Text::LEFT);
298  else
299  t.setAlignment(Text::RIGHT);
300  // Bump down some based on text height
301  t.setPosition(t.x,t.y-f.up()*style.getPointSize()/2);
302  }
303  else if(line_direction == LEFT || line_direction == RIGHT)
304  {
305  // Simply draw it above or below, horizontal, centered
306  // Set alignment
307  t.setAlignment(Text::CENTER);
308  // Bump down some based on text height
309  if(sindir < 0.0 || (sindir == 0.0 && direction == AxisStyle::BELOW))
310  t.setPosition(t.x,t.y-f.up()*style.getPointSize());
311  }
312  else
313  {
314  // Either make the text horizontal regardless of the line's directions or
315  // angle the text to be 90 degrees off the direction of the line, in the
316  // direction specified. Make sure text is right-side-up.
317  // FIXME If above is made to mean "Above on the page" then...
318  // if(cosdir<0) above = !above;
319  if(false) // axis_style.label_horizontal
320  {
321  bool bumpdown = false;
322  // Horizontal labels
323  switch(quad)
324  {
325  case 1:
326  if(above) { t.setAlignment(Text::RIGHT); }
327  else { t.setAlignment(Text::LEFT); bumpdown=true; }
328  break;
329  case 2:
330  if(above) { t.setAlignment(Text::RIGHT); bumpdown=true;}
331  else { t.setAlignment(Text::LEFT); }
332  break;
333  case 3:
334  if(above) { t.setAlignment(Text::LEFT); bumpdown=true; }
335  else { t.setAlignment(Text::RIGHT); }
336  break;
337  case 4:
338  if(above) { t.setAlignment(Text::LEFT); }
339  else { t.setAlignment(Text::RIGHT); bumpdown=true; }
340  break;
341  }
342  if(bumpdown) t.setPosition(t.x,t.y-f.up()*style.getPointSize());
343  }
344  else
345  {
346  // TODO Angled labels
347  }
348  }
349  f.text(t);
350  return true;
351  }
