38  // Explicit constructor, from the 7 parameters.
39  HelmertTransform::HelmertTransform(
40  const ReferenceFrame& from, const ReferenceFrame& to,
41  const double& Rx, const double& Ry, const double& Rz,
42  const double& Tx, const double& Ty, const double& Tz,
43  const double& Sc, const string& Desc, CommonTime epoch)
44  throw(InvalidRequest)
45  {
46  // copy input
47  // NB input is in degrees, members in radians
48  rx = Rx*DEG_TO_RAD; ry = Ry*DEG_TO_RAD; rz = Rz*DEG_TO_RAD;
49  tx = Tx; ty = Ty; tz = Tz;
50  Scale = Sc;
51  description = Desc;
52  Epoch = epoch;
53  fromFrame = from;
54  toFrame = to;
55  if(from == ReferenceFrame::Unknown || to == ReferenceFrame::Unknown) {
56  InvalidRequest e("Invalid Helmert transformation with Unknown frame");
57  GPSTK_THROW(e);
58  }
59 
60  // check that rotation angles are small; sin x ~ x at 0.244 radians = 13.9 deg
61  if(::fabs(rx) > 1.e-3 || ::fabs(ry) > 1.e-3 || ::fabs(rz) > 1.e-3) {
62  InvalidRequest e("Invalid Helmert transformation : "
63  "small angle approximation.");
64  GPSTK_THROW(e);
65  }
66 
67  // rotation matrix. NB. small angle approximation is used. NB. by construction
68  // transpose(Rotation) == inverse(Rotation) (given small angle approximation).
69  Rotation = Matrix<double>(3,3,0.0);
70  Rotation(0,0) = 1.0;
71  Rotation(0,1) = -rz;
72  Rotation(0,2) = ry;
73 
74  Rotation(1,0) = rz;
75  Rotation(1,1) = 1.0;
76  Rotation(1,2) = -rx;
77 
78  Rotation(2,0) = -ry;
79  Rotation(2,1) = rx;
80  Rotation(2,2) = 1.0;
81 
82  // translation vector
83  Translation = Vector<double>(3);
84  Translation(0) = tx;
85  Translation(1) = ty;
86  Translation(2) = tz;
87  }

92  {

117  << " Description: " << description;
118  return (oss.str());
119  }

121  // Transform Position to another frame using this transform or its inverse.
122  // @param Position& pos position to be transformed; unchanged on output.
123  // @param Position& result position after transformation.
124  // @throw if transformation, or inverse, cannot act on ReferenceFrame of input.
125  void HelmertTransform::transform(const Position& pos, Position& result)
126  throw(InvalidRequest)
127  {
128  if(pos.getReferenceFrame() == fromFrame) { // transform
129  result = pos;
130  result.transformTo(Position::Cartesian);
131  Vector<double> vec(3),res(3);
132  vec(0) = result[0];
133  vec(1) = result[1];
134  vec(2) = result[2];
135  res = Rotation*vec + Scale*vec + Translation;
136  result[0] = res(0);
137  result[1] = res(1);
138  result[2] = res(2);
139  result.setReferenceFrame(toFrame);
140  }
141  else if(pos.getReferenceFrame() == toFrame) { // inverse transform
142  result = pos;
143  result.transformTo(Position::Cartesian);
144  Vector<double> vec(3),res(3);
145  vec(0) = result[0];
146  vec(1) = result[1];
147  vec(2) = result[2];
148  res = transpose(Rotation) * (vec - Scale*vec - Translation);
149  result[0] = res(0);
150  result[1] = res(1);
151  result[2] = res(2);
152  result.setReferenceFrame(fromFrame);
153  }
154  else {
155  InvalidRequest e("Helmert tranformation cannot act on frame "
156  + pos.getReferenceFrame().asString());
157  GPSTK_THROW(e);
158  }
159  }
