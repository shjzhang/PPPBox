67  void RinexNavHeader::reallyPutRecord(FFStream& ffs) const
68  throw(std::exception, FFStreamError, StringException)
69  {
70  RinexNavStream& strm = dynamic_cast<RinexNavStream&>(ffs);
71 
72  strm.header = (*this);
73 
74  unsigned long allValid;
75  if (version == 2.0) allValid = allValid20;
76  else if (version == 2.1) allValid = allValid21;
77  else if (version == 2.11) allValid = allValid211;
78  else
79  {
80  FFStreamError err("Unknown RINEX version: " + asString(version,3));
81  err.addText("Make sure to set the version correctly.");
82  GPSTK_THROW(err);
83  }
84 
85  if ((valid & allValid) != allValid)
86  {
87  FFStreamError err("Incomplete or invalid header.");
88  err.addText("Make sure you set all header valid bits for all of the available data.");
89  GPSTK_THROW(err);
90  }
91 
92  string line;
93 
94  if (valid & versionValid)
95  {
96  line = rightJustify(asString(version,3), 10);
97  line += string(10, ' ');
98  line += string("NAVIGATION"); //leftJustify(fileType, 20);
99  line += string(30, ' ');
100  line += versionString;
101  strm << line << endl;
102  strm.lineNumber++;
103  }
104  if (valid & runByValid)
105  {
106  line = leftJustify(fileProgram,20);
107  line += leftJustify(fileAgency,20);
108  SystemTime dt;
109  string dat = (static_cast<CivilTime>(dt)).printf("%02m/%02d/%04Y %02H:%02M:%02S");
110  line += leftJustify(dat, 20);
111  line += runByString;
112  strm << line << endl;
113  strm.lineNumber++;
114  }
115  if (valid & commentValid)
116  {
117  vector<string>::const_iterator itr = commentList.begin();
118  while (itr != commentList.end())
119  {
120  line = leftJustify((*itr), 60);
121  line += commentString;
122  strm << line << endl;
123  strm.lineNumber++;
124  itr++;
125  }
126  }
127  if (valid & ionAlphaValid)
128  {
129  line = string(2, ' ');
130  for (int i = 0; i < 4; i++)
131  {
132  line += rightJustify(doub2for(ionAlpha[i], 12, 2),12); // should be 12.4
133  }
134  line += string(10, ' ');
135  line += ionAlphaString;
136  strm << line << endl;
137  strm.lineNumber++;
138  }
139  if (valid & ionBetaValid)
140  {
141  line = string(2, ' ');
142  for (int i = 0; i < 4; i++)
143  {
144  line += rightJustify(doub2for(ionBeta[i], 12, 2),12);
145  }
146  line += string(10, ' ');
147  line += ionBetaString;
148  strm << line << endl;
149  strm.lineNumber++;
150  }
151  if (valid & deltaUTCValid)
152  {
153  line = string(3, ' ');
154  //line += string(2, ' ');
155  line += doub2for(A0, 19, 2);
156  line += doub2for(A1, 19, 2);
157  line += rightJustify(asString(UTCRefTime),9);
158  line += rightJustify(asString(UTCRefWeek),9);
159  line += string(1, ' ');
160  line += deltaUTCString;
161  strm << line << endl;
162  strm.lineNumber++;
163  }
164  if (valid & leapSecondsValid)
165  {
166  line = rightJustify(asString(leapSeconds), 6);
167  line += string(54, ' ');
168  line += leapSecondsString;
169  strm << line << endl;
170  strm.lineNumber++;
171  }
172  if (valid & endValid)
173  {
174  line = string(60,' ');
175  line += endOfHeader;
176  strm << line << endl;
177  strm.lineNumber++;
178  }

182  void RinexNavHeader::reallyGetRecord(FFStream& ffs)
183  throw(std::exception, FFStreamError, StringException)
184  {
185  RinexNavStream& strm = dynamic_cast<RinexNavStream&>(ffs);
186 
187  // if already read, just return
188  if (strm.headerRead == true)
189  return;
190 
191  valid = 0;
192 
193  // clear out anything that was unsuccessfully read the first
194  commentList.clear();
195 
196  while (! (valid & endValid))
197  {
198  string line;
199  strm.formattedGetLine(line);
200  StringUtils::stripTrailing(line);
201 
202  if (line.length()==0) continue;
203  else if (line.length()<60 || line.length()>80)
204  {
205  FFStreamError e("Invalid line length");
206  GPSTK_THROW(e);
207  }
208 
209  string thisLabel(line, 60, 20);
210 
211  if (thisLabel == versionString)
212  {
213  version = asDouble(line.substr(0,20));
214  fileType = strip(line.substr(20,20));
215  if ( (fileType[0] != 'N') &&
216  (fileType[0] != 'n'))
217  {
218  FFStreamError e("This isn't a Rinex Nav file");
219  GPSTK_THROW(e);
220  }
221  valid |= versionValid;
222  }
223  else if (thisLabel == runByString)
224  {
225  fileProgram = strip(line.substr(0,20));
226  fileAgency = strip(line.substr(20,20));
227  date = strip(line.substr(40,20));
228  valid |= runByValid;
229  }
230  else if (thisLabel == commentString)
231  {
232  commentList.push_back(strip(line.substr(0,60)));
233  valid |= commentValid;
234  }
235  else if (thisLabel == ionAlphaString)
236  {
237  for(int i = 0; i < 4; i++)
238  ionAlpha[i] = gpstk::StringUtils::for2doub(line.substr(2 + 12 * i,12));
239  valid |= ionAlphaValid;
240  }
241  else if (thisLabel == ionBetaString)
242  {
243  for(int i = 0; i < 4; i++)
244  ionBeta[i] = gpstk::StringUtils::for2doub(line.substr(2 + 12 * i,12));
245  valid |= ionBetaValid;
246  }
247  else if (thisLabel == deltaUTCString)
248  {
249  A0 = gpstk::StringUtils::for2doub(line.substr(3,19));
250  A1 = gpstk::StringUtils::for2doub(line.substr(22,19));
251  UTCRefTime = asInt(line.substr(41,9));
252  UTCRefWeek = asInt(line.substr(50,9));
253  valid |= deltaUTCValid;
254  }
255  else if (thisLabel == leapSecondsString)
256  {
257  leapSeconds = asInt(line.substr(0,6));
258  valid |= leapSecondsValid;
259  }
260  else if (thisLabel == endOfHeader)
261  {
262  valid |= endValid;
263  }
264  else
265  {
266  throw(FFStreamError("Unknown header label at line " +
267  asString<size_t>(strm.lineNumber)));
268  }
269  }
270 
271  unsigned long allValid;
272  if (version == 2.0) allValid = allValid20;
273  else if (version == 2.1) allValid = allValid21;
274  else if (version == 2.11) allValid = allValid211;
275  else
276  {
277  FFStreamError e("Unknown or unsupported RINEX version " +
278  asString(version));
279  GPSTK_THROW(e);
280  }
281 
282  if ( (allValid & valid) != allValid)
283  {
284  FFStreamError e("Incomplete or invalid header");
285  GPSTK_THROW(e);
286  }
287 
288  // we got here, so something must be right...
289  strm.header = *this;
290  strm.headerRead = true;
291  }

293  void RinexNavHeader::dump(ostream& s) const
294  {
295  int i;
296  s << "---------------------------------- REQUIRED ----------------------------------\n";
297  s << "Rinex Version " << fixed << setw(5) << setprecision(2) << version
298  << ", File type " << fileType << ".\n";
299  s << "Prgm: " << fileProgram << ", Run: " << date << ", By: " << fileAgency << endl;
300 
301  s << "(This header is ";
302  if((valid & allValid211) == allValid211) s << "VALID 2.11";
303  else if((valid & allValid21) == allValid21) s << "VALID 2.1";
304  else if((valid & allValid20) == allValid20) s << "VALID 2.0";
305  else s << "NOT VALID";
306  s << " Rinex.)\n";
307 
308  if(!(valid & versionValid)) s << " Version is NOT valid\n";
309  if(!(valid & runByValid)) s << " Run by is NOT valid\n";
310  if(!(valid & endValid)) s << " End is NOT valid\n";
311 
312  s << "---------------------------------- OPTIONAL ----------------------------------\n";
313  if(valid & ionAlphaValid) { s << "Ion alpha:";
314  for(i=0; i<4; i++) s << " " << scientific << setprecision(4) << ionAlpha[i];
315  s << endl; }
316  else s << " Ion alpha is NOT valid\n";
317  if(valid & ionBetaValid) { s << "Ion beta:";
318  for(i=0; i<4; i++) s << " " << scientific << setprecision(4) << ionBeta[i];
319  s << endl; }
320  else s << " Ion beta is NOT valid\n";
321  if(valid & deltaUTCValid) s << "Delta UTC: A0="
322  << scientific << setprecision(12) << A0 << ", A1="
323  << scientific << setprecision(12) << A1 << ", UTC ref = ("
324  << UTCRefWeek << "," << UTCRefTime << ")\n";
325  else s << " Delta UTC is NOT valid\n";
326  if(valid & leapSecondsValid) s << "Leap seconds: " << leapSeconds << endl;
327  else s << " Leap seconds is NOT valid\n";
328  if(commentList.size() > 0) {
329  s << "Comments (" << commentList.size() << ") :\n";
330  for(size_t i=0; i<commentList.size(); i++)
331  s << commentList[i] << endl;
332  }
333  s << "-------------------------------- END OF HEADER -------------------------------\n";
334  }
