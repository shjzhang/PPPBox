143  void EarthSolidTide::getSolidTide(double mjdUtc, double dC[], double dS[] )
144  {
145  UTCTime utc(mjdUtc);
146 
147  Matrix<double> E = ReferenceFrames::J2kToECEFMatrix(utc);
148 
149  Vector<double> moonReci = ReferenceFrames::getJ2kPosition(utc.asTDB(),SolarSystem::Moon)*1000.0;
150  Vector<double> sunReci = ReferenceFrames::getJ2kPosition(utc.asTDB(),SolarSystem::Sun)*1000.0;
151 
152  Vector<double> moonR = E * moonReci; // in ecef m
153  Vector<double> sunR = E * sunReci; // in ecef m
154 
155  Position moonP(moonR(0),moonR(1),moonR(2));
156  Position sunP(sunR(0),sunR(1),sunR(2));
157 
158  double r_sun, phi_sun, lamda_sun;
159  r_sun = norm(sunR);
160  phi_sun = sunP.getGeocentricLatitude()*ASConstant::PI/180.0;
161  lamda_sun = sunP.getLongitude()*ASConstant::PI/180.0;
162 
163  double r_lunar, phi_lunar, lamda_lunar;
164  r_lunar = norm(moonR);
165  phi_lunar = moonP.getGeocentricLatitude()*ASConstant::PI/180.0;
166  lamda_lunar = moonP.getLongitude()*ASConstant::PI/180.0;
167 
168 
169  // reference bern 5 TIDPT2.f
170  /*
171  PERTURBING ACCELERATION DUE TO TIDES CORRESPONDING TO IERS STANDARDS 2003.
172  STEP 1 CORRECTIONS OF SOLID EARTH TIDES INCLUDED,
173  STEP 2 ONLY TERM DUE TO K1. SOLID EARTH POLE TIDES INCLUDED
174  OCEAN TIDE TERMS UP TO N=M=4 INCLUDED
175  */
176 
177  /* IERS2003, P60
178  Elastic Earth Anelastic Earth
179  n m knm k+nm Reknm Imknm k+nm
180  2 0 0.29525 .0.00087 0.30190 .0.00000 .0.00089
181  2 1 0.29470 .0.00079 0.29830 .0.00144 .0.00080
182  2 2 0.29801 .0.00057 0.30102 .0.00130 .0.00057
183  3 0 0.093 ?ก่ ?ก่ ?ก่
184  3 1 0.093 ?ก่ ?ก่ ?ก่
185  3 2 0.093 ?ก่ ?ก่ ?ก่
186  3 3 0.094 ?ก่ ?ก่ ?ก่
187  */
188  complex<double> k[10] = // Anelastic Earth
189  {
190  complex<double >(0.30190, 0.0), // 20
191  complex<double >(0.29830,-0.00144), // 21
192  complex<double >(0.30102,-0.00130), // 22
193  complex<double >(0.093, 0.0), // 30
194  complex<double >(0.093, 0.0), // 31
195  complex<double >(0.093, 0.0), // 32
196  complex<double >(0.094, 0.0), // 33
197  complex<double >(-0.00089, 0.0), // k+ 20
198  complex<double >(-0.00080, 0.0), // k+ 21
199  complex<double >(-0.00057, 0.0) // k+ 22
200  };
201 
202  complex<double> res[7];
203 
204  //----------------------------------------------------------------------
205  // The first step of the computation ,refer to "IERS conventions 2003" P59
206  // Each iteration for dC[n,m] and dS[n,m]
207  for(int n=2;n<=3;n++)
208  {
209  for(int m=0;m<=n;m++)
210  {
211  int index = n * n - 2 * n + m; //index in the returning value array
212 
213  double Nnm = normFactor( n, m ); //normalization coefficents of degree n and order m
214 
215  // Pnm: normalized Legendre polynomials of degress n and order m
216  // 0 for sun and 1 for lunar each
217  double sunPnm = Nnm * legendrePoly( n, m, std::sin( phi_sun) );
218  double moonPnm = Nnm * legendrePoly( n, m, std::sin( phi_lunar) );
219 
220  double sunTemp = (ASConstant::GM_Sun/ASConstant::GM_Earth)*std::pow(ASConstant::R_Earth/r_sun,n+1) * sunPnm;
221  double moonTemp = (ASConstant::GM_Moon/ASConstant::GM_Earth)*std::pow(ASConstant::R_Earth/r_lunar,n+1)*moonPnm;
222 
223  // Exp(-m*lamda*i) for sun and lunar each
224  complex<double> c_sun = complex<double>( std::cos( - m * lamda_sun ), std::sin( - m * lamda_sun ) );
225  complex<double> c_lunar = complex<double>( std::cos( - m * lamda_lunar ), std::sin( - m * lamda_lunar ) );
226 
227  res[index] = sunTemp * c_sun + moonTemp * c_lunar;
228 
229  dC[index] = (k[index]*res[index]).real()/(2.0*n+1.0);
230  dS[index] = -(k[index]*res[index]).imag()/(2.0*n+1.0);
231 
232  } // 'for(int m=0;m<=n;m++)'
233 
234  } // 'for(int n=2;n<=3;n++)'
235 
236  // The correction of dC[4,i] and dS[4,i](i=0,1,2) produced by degree 2 tide
237  // The only difference from the above dC[2,i] and dS[2,i] is value of k replaced by k+
238  for(int n = 0; n <= 2; n ++ )
239  {
240  int index = 2 * 2 - 2 * 2 + n; // liuwk
241  complex<double> c_temp = k[n+7 ] * res[ index ]; // liuwk
242  dC[7+n] = c_temp.real() / 5.0;
243  dS[7+n] =-c_temp.imag() / 5.0;
244  }
245 
246 
247  //-------------------------------------------------------------
248  // The second step
249 
250  // COMPUTE DOODSON'S FUNDAMENTAL ARGUMENTS (BETA)
251  double BETA[6] = {0.0};
252  double Dela[5] = {0.0};
253  ReferenceFrames::doodsonArguments(utc.asUT1(),utc.asTT(),BETA,Dela);
254  double GMST = ReferenceFrames::iauGmst00(utc.asUT1(), utc.asTT());
255 
256 
257  for(int i=0;i<48;i++)
258  {
259  // Computation of thet_f
260  double thet_f = (GMST+ASConstant::PI)-(Argu_C21[i][2]*Dela[0]+Argu_C21[i][3]*Dela[1]+Argu_C21[i][4]*Dela[2]
261  + Argu_C21[i][5]*Dela[3]+Argu_C21[i][6]*Dela[4]);
262 
263  double t_s = std::sin(thet_f);
264  double t_c = std::cos(thet_f);
265 
266  // Resulted from formula 5b in chapter 6.1
267  dC[1] += ((Argu_C21[i][0]*t_s+Argu_C21[i][1]*t_c )*1e-12);
268  dS[1] += ((Argu_C21[i][0]*t_c-Argu_C21[i][1]*t_s )*1e-12);
269  }
270 
271  for(int i=0;i<2;i++)
272  {
273  // Input the computation of thet_f
274  double thet_f = 2*(GMST+ASConstant::PI)-(Argu_C22[i][1]*Dela[0]+Argu_C22[i][2]*Dela[1]+Argu_C22[i][3]*Dela[2]
275  + Argu_C22[i][4]*Dela[3]+Argu_C22[i][5]*Dela[4]);
276 
277  double t_s = std::sin(thet_f);
278  double t_c = std::cos(thet_f);
279 
280  // Resulted from formula 5b in chapter 6.1
281  // The corrections are only to the real part.
282  dC[2] += ((Argu_C22[i][0]*t_c)*1e-12 );
283  dS[2] += ((-Argu_C22[i][0]*t_s)*1e-12 );
284  }
285 
286 
287  for(int i=0;i<21;i++)
288  {
289  // Input the computation of thet_f
290  double thet_f = -(Argu_C20[i][2]*Dela[0]+Argu_C20[i][3]*Dela[1]+Argu_C20[i][4]*Dela[2]
291  + Argu_C20[i][5]*Dela[3]+Argu_C20[i][6]*Dela[4]);
292 
293  double t_s = std::sin(thet_f);
294  double t_c = std::cos(thet_f);
295 
296  // Resulted from formula 5a in chapter 6.1
297  // Modified, 05.12.2009
298  // dC[0] += ( ( Argu_C20[i][0] * t_c - Argu_C20[i][1] * t_s ) * 1e-12 );
299  dC[0]+=((Argu_C20[i][0]*t_c+Argu_C20[i][1]*t_s)*1e-12);
300  }
301 
302  //--------------------------------------------------------------------------------
303  // the third step
304  //
305  //C REMOVE PREMANENT TIDE FROM C02 (FOR JGM-3 NOT FOR GEM-T3)
306  //C ---------------------------------------------------------
307  //IF (IZTID.EQ.1) CPOT(4)=CPOT(4)+1.3914129D-8*K20
308 
309  }

312  double EarthSolidTide::normFactor(int n, int m)
313  {
314  // The input should be n >= m >= 0
315 
316  double fac(1.0);
317  for(int i = (n-m+1); i <= (n+m); i++)
318  {
319  fac = fac * double(i);
320  }
321 
322  double delta = (m == 0) ? 1.0 : 0.0;
323 
324  double num = (2.0 * n + 1.0) * (2.0 - delta);
325 
326  // We should make sure fac!=0, but it won't happen on the case,
327  // so we just skip handling it
328  double out = std::sqrt(num/fac);
329 
330  return out;
331 
332  } // End of method 'EarthSolidTide::normFactor'

335  // Legendre polynomial
336  double EarthSolidTide::legendrePoly(int n, int m, double u)
337  {
338  // reference:Satellite Orbits Montenbruck. P66
339  if(0==n && 0==m)
340  {
341  return 1.0;
342  }
343  else if(m==n)
344  {
345  return (2.0*m-1.0)*std::sqrt(1.0-u*u)*legendrePoly(n-1,m-1,u);
346  }
347  else if(n==m+1)
348  {
349  return (2.0*m+1)*u*legendrePoly(m,m,u);
350  }
351  else
352  {
353  return ((2.0*n-1.0)*u*legendrePoly(n-1,m,u)-(n+m-1.0)*legendrePoly(n-2,m,u))/(n-m);
354  }
355 
356  } // End of method 'EarthSolidTide::legendrePoly()'
