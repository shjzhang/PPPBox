118  void Add(T d, T t, T w=T(1))
119  {
120  // n_-1
121  // Equation is sum[t^i * X(i)] = d OR P * X = d
122  // i=0 1xn_ n_ 1
123  Vector<T> P(n_);
124  T tt=T(1);
125  for(size_t i=0; i<n_; i++) { P(i)=tt; tt *= t; }
126  Npts++;
127  Matrix<T> PP;
128  PP = outer(P,P);
129  PP *= w;
130  InfMatrix += PP;
131  P *= d*w;
132  InfData += P;
133  Inverted = false;
134  }

184  T Evaluate(T t)
185  {
186  if(n_ <= 0) { Singular=true; return T(0); }
187  Solve();
188  if(Singular) return T(0);
189 
190  T d,tn;
191  d = X(0);
192  tn = t;
193  for(size_t i=1; i<X.size(); i++) {
194  d += X(i)*tn;
195  tn *= t;
196  }
197  return d;

202  Vector<T> Evaluate(const Vector<T>& Vt)
203  {
204  Vector<T> R;
205  if(n_ <= 0) { Singular=true; return R; }
206  Solve();
207  if(Singular) return R;
208 
209  T tn;
210  R = Vector<T>(Vt.size());
211  for(size_t j=0; j<Vt.size(); j++) {
212  R(j) = X(0);
213  tn = Vt(j);
214  for(size_t i=1; i<X.size(); i++) {
215  R(j) += X(i)*tn;
216  tn *= Vt(j);
217  }
218  }
219  return R;
