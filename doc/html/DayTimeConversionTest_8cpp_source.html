120 bool testRandomAccessors(DayTime &dtb, DayTime &dte, long ndates)
121 {
122  bool cumulativeResult=true;
123 
124  // Seed the random number generator
125  gpstk::DayTime dt;
126  unsigned int seed= (unsigned int) dt.GPSsow();
127  srand(seed);
128 
129  unsigned long dayDiff = (unsigned long)
130  ceil(dte.MJD() - dtb.MJD());
131 
132  for (int j=0;j<ndates;++j) // Loop through set of random days
133  {
134  double dayDelta = floor( rand()*1./ RAND_MAX * dayDiff);
135  double sodDelta = rand()*1./RAND_MAX;
136  gpstk::DayTime testDate;
137 
138  testDate.setMJD(dtb.MJD()+dayDelta+sodDelta);
139 
140  short year = testDate.year();
141  short month = testDate.month();
142  short dom = testDate.day();
143  short hour = testDate.hour();
144  short minute = testDate.minute();
145  double seconds = testDate.second();
146  short doy = testDate.DOY();
147  double sod = testDate.DOYsecond();
148  short week = testDate.GPSfullweek()%1024;
149  double sow = testDate.GPSsow();
150  long zcount = testDate.GPSzcount();
151  short hintYear = testDate.year();
152  double MJD = testDate.MJD();
153  cumulativeResult = cumulativeResult &&
154  testMutators( year, month, dom, hour, minute, seconds,
155  doy, sod,
156  week, sow, zcount, hintYear,
157  MJD );
158  } // End loop over random dates
159 
160  return cumulativeResult;
161 }
