69  // parses the command line input
70  void
71  CommandOptionParser::parseOptions(int argc,
72  char* argv[])
73  {
74  // this maps the index of optionVec to the command line options
75  CommandOptionMap com;
76 
77  // keep track of the order of command options
78  unsigned int order = 0;
79 
80  // get the program name. If there's a / in it, take the part
81  // after the last / as the program name
82  progName = string(argv[0]);
83  string::size_type slashPos = progName.rfind('/');
84  if (slashPos != string::npos)
85  progName = progName.substr(slashPos + 1);
86 
87  string shortOptString;
88  struct option* optArray = NULL;
89  unsigned long optArraySize = 0;
90 
91  CommandOption *trailing = NULL;
92 
93  // build the getopt and getopt_long inputs
94  CommandOptionVec::size_type index;
95  for(index = 0; index < optionVec.size(); index++)
96  {
97  // add short options
98  switch (optionVec[index]->optType)
99  {
100  case CommandOption::trailingType:
101  if (trailing)
102  errorStrings.push_back("More than one trailing argument"
103  " object used (programming error");
104  else
105  trailing = optionVec[index];
106  break;
107  case CommandOption::stdType:
108  if (optionVec[index]->shortOpt != 0)
109  {
110  shortOptString += optionVec[index]->toGetoptShortOption();
111  com[string(1,optionVec[index]->shortOpt)] = optionVec[index];
112  }
113 
114  // add long options
115  if (!optionVec[index]->longOpt.empty())
116  {
117  resizeOptionArray(optArray, optArraySize);
118  optArray[optArraySize - 1] =
119  optionVec[index]->toGetoptLongOption();
120  com[optionVec[index]->longOpt] = optionVec[index];
121  }
122 
123  // keep track of whether or not there are required or
124  // optional arguments
125  if (optionVec[index]->required)
126  hasRequiredArguments = true;
127  else
128  hasOptionalArguments = true;
129  break;
130  default:
131  // do nothing
132  break;
133  }
134  }
135 
136  // add the getopt_long terminator value
137  resizeOptionArray(optArray, optArraySize);
138  struct option lastOption = {0,0,0,0};
139  optArray[optArraySize - 1] = lastOption;
140 
141  // use '+' to make getopt not mangle the inputs (if i remember right)
142  shortOptString.insert((string::size_type)0, (string::size_type)1, '+');
143 
144  int cha;
145  int optionIndex;
146 
147  // disable internal error messages
148  opterr = 0;
149 
150  while (optind < argc)
151  {
152  if ((cha = getopt_long(argc, argv, shortOptString.c_str(),
153  optArray, &optionIndex)) == -1)
154  {
155  if (!trailing)
156  errorStrings.push_back("Excess arguments");
157  break;
158  }
159 
160  order++;
161 
162  // Solaris uses '?' for all getopt errors. Linux uses '?'
163  // for unknown options and ':' for options that require
164  // arguments but don't have then. That's why the error
165  // message is "option error" cause we can't differentiate
166  // what the REAL error is...
167  if ((cha == '?') || (cha == ':'))
168  {
169  // get the argument that had the error and write an
170  // error string
171  string errorArg;
172  // for a character option error
173  if (optopt != 0)
174  errorArg = string(1, (char)optopt);
175  // for a getopt_long error
176  else
177  errorArg = argv[optind - 1];
178  errorStrings.push_back(string("Option error: " + errorArg));
179  }
180  // otherwise this is probably a found option
181  else
182  {
183  string thisOption;
184 
185  // determine whether it found the short or long version
186  if (cha != 0)
187  thisOption = string(1,(char)cha);
188  else
189  thisOption = string(optArray[optionIndex].name);
190 
191  // try to find the option in our option map
192  map<string, CommandOption*>::iterator itr = com.find(thisOption);
193 
194  if (itr != com.end())
195  {
196  CommandOption* pickedOption = (*itr).second;
197  // if there is an value for this option...
198  if (optarg)
199  {
200  if (pickedOption->optFlag == CommandOption::noArgument)
201  {
202  errorStrings.push_back(string("Option ") +
203  thisOption +
204  string(" has an argument but it shouldn't."));
205  }
206  // add this argument to the picked option and
207  // increment the count
208  else if (pickedOption->optFlag == CommandOption::hasArgument)
209  {
210  pickedOption->value.push_back(string(optarg));
211  pickedOption->count++;
212  pickedOption->order = order;
213  }
214 
215  }
216  // no value for option...
217  else
218  {
219  if (pickedOption->optFlag == CommandOption::hasArgument)
220  {
221  errorStrings.push_back(string("Option ") +
222  thisOption +
223  string(" has no argument when it should."));
224  }
225  // increment the picked option's count
226  else if (pickedOption->optFlag == CommandOption::noArgument)
227  {
228  pickedOption->count++;
229  pickedOption->order = order;
230  }
231  }
232  } // itr != end()
233  else
234  {
235  errorStrings.push_back("Unknown option error");
236  }
237  } // else cha ==
238  } // getopt_long
239 
240  // check for remaining arguments
241  if (optind < argc)
242  {
243  if (trailing)
244  {
245  int i;
246  for(i = optind; i < argc; i++)
247  {
248  trailing->value.push_back(string(argv[i]));
249  trailing->count++;
250  }
251  }
252  // the case where trailing==null is handled above
253  }
254 
255  for(index = 0; index < optionVec.size(); index++)
256  {
257  string retVal = optionVec[index]->checkArguments();
258  if (!retVal.empty())
259  errorStrings.push_back(retVal);
260 
261  // check max count
262  if (optionVec[index]->maxCount != 0)
263  {
264  if (optionVec[index]->count > optionVec[index]->maxCount)
265  {
266  string errstr("Option ");
267  errstr += optionVec[index]->getOptionString();
268  errstr += string(" appeared more times than allowed.");
269  errorStrings.push_back(errstr);
270  }
271  }
272  }
273 
274  delete [] optArray;
275  }

285  // prints the required arguments first (if any) then the optional
286  // ones (if any)
287  ostream& CommandOptionParser::displayUsage(ostream& out, bool doPretty)
288  {
289  CommandOptionVec::size_type index;
290  CommandOption *trailing = NULL;
291 
292  char *colch = getenv("COLUMNS");
293  int columns = 80;
294  unsigned maxlen = 0;
295  if (colch)
296  {
297  string colStr(colch);
298  columns = asInt(colStr);
299  }
300 
301  // find the trailing argument if any, and max option string length
302  for (index = 0; index < optionVec.size(); index++)
303  {
304  if (optionVec[index]->optType == CommandOption::trailingType)
305  trailing = optionVec[index];
306  else if (optionVec[index]->optType == CommandOption::stdType)
307  maxlen = std::max(maxlen,
308  unsigned(optionVec[index]->getFullOptionString().length()));
309  }
310 
311  out << "Usage: " << progName;
312  if (hasRequiredArguments || hasOptionalArguments)
313  out << " [OPTION] ...";
314  if (trailing)
315  out << " " << trailing->description;
316  out << endl
317  << text ;
318 // << (doPretty ? prettyPrint(text,"\n","","",columns) : text);
319 // << endl
320 // << endl
321 // << "Command options:" << endl;
322 
323  for(int required = 1; required >= 0; required--)
324  {
325  if (required==1 && hasRequiredArguments)
326  out << endl << "Required arguments:" << endl;
327  else if (required==0 && hasOptionalArguments)
328  out << endl << "Optional arguments:" << endl;
329 
330  for(index = 0; index < optionVec.size(); index++)
331  {
332  if ((optionVec[index]->required == (required==1)) &&
333  (optionVec[index]->optType == CommandOption::stdType))
334  {
335  string optstr(optionVec[index]->getFullOptionString());
336  string desc(optionVec[index]->description);
337  string indent(maxlen, ' ');
338 
339  if(doPretty) {
340  leftJustify(optstr, maxlen);
341  prettyPrint(desc, "\n", indent, optstr, columns);
342  }
343  out << desc;
344  if(!doPretty) out << endl;
345  }
346  }
347  }
348 
349  return out;
350  }

354  unsigned long& oldSize)
355  {
356  struct option* newArray = new struct option[1 + oldSize];
