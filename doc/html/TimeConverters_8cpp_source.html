34  // These two routines convert 'integer JD' and calendar time; they were
35  // derived from Sinnott, R. W. "Bits and Bytes" Sky & Telescope Magazine,
36  // Vol 82, p. 183, August 1991, and The Astronomical Almanac, published
37  // by the U.S. Naval Observatory.
38  // NB range of applicability of this routine is from 0JD (4713BC)
39  // to approx 3442448JD (4713AD).
40  void convertJDtoCalendar( long jd,
41  int& iyear,
42  int& imonth,
43  int& iday )
44  {
45  long L, M, N, P, Q;
46  if(jd > 2299160) // after Oct 4, 1582
47  {
48  L = jd + 68569;
49  M = (4 * L) / 146097;
50  L = L - ((146097 * M + 3) / 4);
51  N = (4000 * (L + 1)) / 1461001;
52  L = L - ((1461 * N) / 4) + 31;
53  P = (80 * L) / 2447;
54  iday = int(L - (2447 * P) / 80);
55  L = P / 11;
56  imonth = int(P + 2 - 12 * L);
57  iyear = int(100 * (M - 49) + N + L);
58  }
59  else
60  {
61  P = jd + 1402;
62  Q = (P - 1) / 1461;
63  L = P - 1461 * Q;
64  M = (L - 1) / 365 - L / 1461;
65  N = L - 365 * M + 30;
66  P = (80 * N) / 2447;
67  iday = int(N - (2447 * P) / 80);
68  N = P / 11;
69  imonth = int(P + 2 - 12 * N);
70  iyear = int(4 * Q + M + N - 4716);
71  if(iyear <= 0)
72  {
73  --iyear;
74  }
75  }
76  // catch century/non-400 non-leap years
77  if(iyear > 1599 &&
78  !(iyear % 100) &&
79  (iyear % 400) &&
80  imonth == 2 &&
81  iday == 29)
82  {
83  imonth = 3;
84  iday = 1;
85  }
86  }

88  long convertCalendarToJD( int yy,
89  int mm,
90  int dd )
91  {
92  if(yy == 0)
93  --yy; // there is no year 0
94 
95  if(yy < 0)
96  ++yy;
97 
98  long jd;
99  double y = static_cast<double>( yy );
100  double m = static_cast<double>( mm );
101  //double d = static_cast<double>( dd );
102 
103  // In the conversion from the Julian Calendar to the Gregorian
104  // Calendar the day after October 4, 1582 was October 15, 1582.
105  //
106  // if the date is before October 15, 1582
107  if(yy < 1582 || (yy == 1582 && (mm < 10 || (mm == 10 && dd < 15))))
108  {
109  jd = 1729777 + dd + 367 * yy
110  - static_cast<long>(7 * ( y + 5001 +
111  static_cast<long>((m - 9) / 7)) / 4)
112  + static_cast<long>(275 * m / 9);
113  }
114  else // after Oct 4, 1582
115  {
116  jd = 1721029 + dd + 367 * yy
117  - static_cast<long>(7 * (y + static_cast<long>((m + 9) / 12)) / 4)
118  - static_cast<long>(3 * (static_cast<long>((y + (m - 9) / 7) / 100)
119  + 1) / 4)
120  + static_cast<long>(275 * m / 9);
121 
122  // catch century/non-400 non-leap years
123  if( (! (yy % 100) &&
124  (yy % 400) &&
125  mm > 2 &&
126  mm < 9) ||
127  (!((yy - 1) % 100) &&
128  ((yy - 1) % 400) &&
129  mm == 1))
130  {
131  --jd;
132  }
133  }
134  return jd;
135  }

137  void convertSODtoTime( double sod,
138  int& hh,
139  int& mm,
140  double& sec )
141  {
142  // Get us to within one day.
143  if (sod < 0)
144  {
145  sod += (1 +
146  static_cast<unsigned long>(sod / SEC_PER_DAY)) * SEC_PER_DAY ;
147  }
148  else if (sod >= SEC_PER_DAY)
149  {
150  sod -= static_cast<unsigned long>(sod / SEC_PER_DAY) * SEC_PER_DAY ;
151  }
152 
153  double temp; // variable to hold the integer part of sod
154  sod = modf(sod, &temp); // sod holds the fraction, temp the integer
155  long seconds = static_cast<long>(temp); // get temp into a real integer
156 
157  hh = seconds / 3600 ;
158  mm = (seconds % 3600) / 60 ;
159  sec = double(seconds % 60) + sod ;
160 
161  }
