85  void OrbElemFIC9::loadData( const FICData& fic9 )
86  throw( InvalidParameter )
87  {
88  if (fic9.blockNum!=9)
89  {
90  InvalidParameter exc("Invalid FIC Block: "+StringUtils::asString(fic9.blockNum));
91  GPSTK_THROW(exc);
92  }
93 
94  // Fill in the variables unique to OrbElemFIC9
95  HOWtime[0] = static_cast<long>( fic9.f[2] );
96  ASalert[0] = static_cast<short>( fic9.f[3] );
97  codeflags = static_cast<short>( fic9.f[6] );
98  accFlag = static_cast<short>( fic9.f[7] );
99  health = static_cast<short>( fic9.f[8] );
100  IODC = static_cast<short>( ldexp( fic9.f[9], -11 ) );
101  L2Pdata = static_cast<short>( fic9.f[10] );
102  Tgd = fic9.f[11];
103 
104  HOWtime[1] = static_cast<long>( fic9.f[22] );
105  ASalert[1] = static_cast<short>( fic9.f[23] );
106  IODE = static_cast<short>( ldexp( fic9.f[25], -11 ) );
107  fitint = static_cast<short>( fic9.f[34] );
108 
109  HOWtime[2] = static_cast<long>( fic9.f[42] );
110  ASalert[2] = static_cast<short>( fic9.f[43] );
111 
112  short fullXmitWeekNum = static_cast<short>( fic9.f[5] );
113 
114  // Fill in the variables in the OrbElem parent
115  // - - - First the simple copies - - -
116  double Toc = fic9.f[12]; // OrbElem only stores fully qualified times
117  // see notes below.
118  af2 = fic9.f[13];
119  af1 = fic9.f[14];
120  af0 = fic9.f[15];
121 
122  Crs = fic9.f[26];
123  dn = fic9.f[27];
124  M0 = fic9.f[28];
125  Cuc = fic9.f[29];
126  ecc = fic9.f[30];
127  Cus = fic9.f[31];
128  double AHalf = fic9.f[32]; // Not a member of OrbElem. See notes below.
129  double Toe = fic9.f[33]; // OrbElem only stores fully qualified times
130  // see notes below.
131 
132  Cic = fic9.f[45];
133  OMEGA0 = fic9.f[46];
134  Cis = fic9.f[47];
135  i0 = fic9.f[48];
136  Crc = fic9.f[49];
137  w = fic9.f[50];
138  OMEGAdot = fic9.f[51];
139  idot = fic9.f[53];
140 
141  // - - - Now work on the things that need to be calculated - - -
142 
143  // The system is assumed (legacy navigation message is from GPS)
144  satID.id = static_cast<short>( fic9.f[19] );
145 
146  // The observation ID has a type of navigation, but the
147  // carrier and code types are undefined. They could be
148  // L1/L2 C/A, P, Y,.....
149  obsID.type = ObsID::otNavMsg;
150  obsID.band = ObsID::cbUndefined;
151  obsID.code = ObsID::tcUndefined;

153  // Beginning of Validity
154  // New concept. Admit the following.
155  // (a.) The collection system may not capture the data at earliest transmit.
156  // (b.) The collection system may not capture the three SFs consecutively.
157  // Consider a couple of IS-GPS-200 promises,
158  // (c.) By definition, beginning of validity == beginning of transmission.
159  // (d.) Except for uploads, cutovers will only happen on hour boundaries
160  // (e.) Cutovers can be detected by non-even Toc.
161  // (f.) Even uploads will cutover on a frame (30s) boundary.
162  // Therefore,
163  // 1.) If Toc is NOT even two hour interval, pick lowest HOW time,
164  // round back to even 30s. That's the earliest Xmit time we can prove.
165  // NOTE: For the case where this is the SECOND SF 1/2/3 after an upload,
166  // this may yield a later time as such a SF 1/2/3 will be on a even
167  // hour boundary. Unfortunately, we have no way of knowing whether
168  // this item is first or second after upload without additional information.
169  // 2.) If Toc IS even two hour interval, pick time from SF 1,
170  // round back to nearest EVEN two hour boundary. This assumes collection
171  // SOMETIME in first hour of transmission. Could be more
172  // complete by looking at fit interval and IODC to more accurately
173  // determine earliest transmission time.
174  long longToc = (long) Toc;
175  double XmitSOW = 0.0;
176  if ( (longToc % 7200) != 0) // NOT an even two hour change
177  {
178  long leastHOW = HOWtime[0];
179  if (HOWtime[1]<leastHOW) leastHOW = HOWtime[1];
180  if (HOWtime[2]<leastHOW) leastHOW = HOWtime[2];
181  long Xmit = leastHOW - (leastHOW % 30);
182  XmitSOW = (double) Xmit;
183  }
184  else
185  {
186  long Xmit = HOWtime[0] - HOWtime[0] % 7200;
187  XmitSOW = (double) Xmit;
188  }
189  beginValid = GPSWeekSecond( fullXmitWeekNum, XmitSOW, TimeSystem::GPS );
190 
191  // Determine Transmit Time
192  // Transmit time is the actual time this
193  // SF 1/2/3 sample was collected
194  long leastHOW = HOWtime[0];
195  if (HOWtime[1]<leastHOW) leastHOW = HOWtime[1];
196  if (HOWtime[2]<leastHOW) leastHOW = HOWtime[2];
197  long Xmit = leastHOW - (leastHOW % 30);
198  transmitTime = GPSWeekSecond( fullXmitWeekNum, (double)Xmit, TimeSystem::GPS );
199 
200  // Fully qualified Toe and Toc
201  // As broadcast, Toe and Toc are in GPS SOW and do not include
202  // the GPS week number. OrbElem (rightly) insists on having a
203  // Toe and Toc in CommonTime objects which implies determining
204  // the week number.
205  double timeDiff = Toe - XmitSOW;
206  short epochWeek = fullXmitWeekNum;
207  if (timeDiff < -HALFWEEK) epochWeek++;
208  else if (timeDiff > HALFWEEK) epochWeek--;
209 
210  ctToc = GPSWeekSecond(epochWeek, Toc, TimeSystem::GPS);
211  ctToe = GPSWeekSecond(epochWeek, Toe, TimeSystem::GPS);
212 
213  // End of Validity.
214  // The end of validity is calculated from the fit interval
215  // and the Toe. The fit interval is either trivial
216  // (if fit interval flag==0, fit interval is 4 hours)
217  // or a look-up table based on the IODC.
218  // Round the Toe value to the hour to elminate confusion
219  // due to possible "small offsets" indicating uploads
220  short fitHours = getLegacyFitInterval(IODC, fitint);
221  long ToeOffset = (long) Toe % 3600;
222  double adjToe = Toe; // Default case
223  if (ToeOffset)
224  {
225  adjToe += 3600.0 - (double)ToeOffset; // If offset, then adjust to remove it
226  }
227  long endFitSOW = adjToe + (fitHours/2)*3600;
228  short endFitWk = epochWeek;
229  if (endFitSOW >= FULLWEEK)
230  {
231  endFitSOW -= FULLWEEK;
232  endFitWk++;
233  }
234  endValid = GPSWeekSecond(endFitWk, endFitSOW, TimeSystem::GPS);
235 
236  // Semi-major axis and time-rate-of-change of semi-major axis
237  // Note: Legacy navigation message (SF 1/2/3) used SQRT(A).
238  // The CNAV and CNAV-2 formats use deltaA and Adot. As a
239  // result, OrbElem uses A and Adot and SQRT(A) and deltaA
240  // are converted to A at runtime.
241  A = AHalf * AHalf;
242  Adot = 0.0;
243  // Legacy nav doesn't have Rate of Change to Correction to mean motion,
244  // so set it to zero.
245  dndot = 0.0;
246 
247  // Health
248  // OrbElemFIC9 stores the full 8 bits health from the legacy
249  // navigation message. OrElemn only stores the true/false,
250  // use/don't use based on whether the 8 bit health is 0 or non-zero
251  healthy = (healthy==0);
252

254  // URA Handling
255 
256 
257  // After all this is done, declare that data has been loaded
258  // into this object (so it may be used).
259  dataLoadedFlag = true;
260 
261  return;
