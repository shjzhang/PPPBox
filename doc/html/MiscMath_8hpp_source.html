59  T SimpleLagrangeInterpolation(const std::vector<T>& X, const std::vector<T>& Y,
60  const T x) throw(Exception)
61  {
62  if(Y.size() < X.size()) {
63  GPSTK_THROW(Exception("Input vectors must be of same size"));
64  }
65  size_t i,j;
66  T Yx(0);
67  for(i=0; i<X.size(); i++) {
68  if(x==X[i]) return Y[i];
69 
70  T Li(1);
71  for(j=0; j<X.size(); j++)
72  if(i!=j) Li *= (x-X[j])/(X[i]-X[j]);
73

75  }
76  return Yx;

86  template <class T>
87  T LagrangeInterpolation(const std::vector<T>& X, const std::vector<T>& Y,
88  const T& x, T& err) throw(Exception)
89  {
90  if(Y.size() < X.size() || X.size() < 4) {
91  GPSTK_THROW(Exception("Input vectors must be of same length, at least 4"));
92  }
93 
94  std::size_t i,j,k;
95  T y,del;
96  std::vector<T> D,Q;
97 
98  err = T(0);
99  k = X.size()/2;
100  if(x == X[k]) return Y[k];
101  if(x == X[k-1]) return Y[k-1];
102  if(ABS(x-X[k-1]) < ABS(x-X[k])) k=k-1;
103  for(i=0; i<X.size(); i++) {
104  Q.push_back(Y[i]);
105  D.push_back(Y[i]);
106  }
107  y = Y[k--];
108  for(j=1; j<X.size(); j++) {
109  for(i=0; i<X.size()-j; i++) {
110  del = (Q[i+1]-D[i])/(X[i]-X[i+j]);
111  D[i] = (X[i+j]-x)*del;
112  Q[i] = (X[i]-x)*del;
113  }
114  err = (2*(k+1) < X.size()-j ? Q[k+1] : D[k--]); // NOT 2*k
115  y += err;
116  }
117  return y;
118  } // end T LagrangeInterpolation(vector, vector, const T, T&)

120  // The following is a
121  // Straightforward implementation of Lagrange polynomial and its derivative
122  // { all sums are over index=0,N-1; Xi is short for X[i]; Lp is dL/dx;
123  // y(x) is the function being approximated. }
124  // y(x) = SUM[Li(x)*Yi]
125  // Li(x) = PROD(j!=i)[x-Xj] / PROD(j!=i)[Xi-Xj]
126  // dy(x)/dx = SUM[Lpi(x)*Yi]
127  // Lpi(x) = SUM(k!=i){PROD(j!=i,j!=k)[x-Xj]} / PROD(j!=i)[Xi-Xj]
128  // Define Pi = PROD(j!=i)[x-Xj], Di = PROD(j!=i)[Xi-Xj],
129  // Qij = PROD(k!=i,k!=j)[x-Xk] and Si = SUM(j!=i)Qij.
130  // then Li(x) = Pi/Di, and Lpi(x) = Si/Di.
131  // Qij is symmetric, there are only N(N+1)/2 - N of them, so store them
132  // in a vector of length N(N+1)/2, where Qij==Q[i+j*(j+1)/2] (ignore i=j).

139  template <class T>
140  void LagrangeInterpolation(const std::vector<T>& X, const std::vector<T>& Y,
141  const T& x, T& y, T& dydx) throw(Exception)
142  {
143  if(Y.size() < X.size() || X.size() < 4) {
144  GPSTK_THROW(Exception("Input vectors must be of same length, at least 4"));
145  }
146 
147  std::size_t i,j,k,N=X.size(),M;
148  M = (N*(N+1))/2;
149  std::vector<T> P(N,T(1)),Q(M,T(1)),D(N,T(1));
150  for(i=0; i<N; i++) {
151  for(j=0; j<N; j++) {
152  if(i != j) {
153  P[i] *= x-X[j];
154  D[i] *= X[i]-X[j];
155  if(i < j) {
156  for(k=0; k<N; k++) {
157  if(k == i || k == j) continue;
158  Q[i+(j*(j+1))/2] *= (x-X[k]);
159  }
160  }
161  }
162  }
163  }
164  y = dydx = T(0);
165  for(i=0; i<N; i++) {
166  y += Y[i]*(P[i]/D[i]);
167  T S(0);
168  for(k=0; k<N; k++) if(i != k) {
169  if(k<i) S += Q[k+(i*(i+1))/2]/D[i];
170  else S += Q[i+(k*(k+1))/2]/D[i];
171  }
172  dydx += Y[i]*S;
173  }
174  } // end void LagrangeInterpolation(vector, vector, const T, T&, T&)

178  template <class T>
179  T LagrangeInterpolating2ndDerivative(const std::vector<T>& pos,
180  const std::vector<T>& val,
181  const T desiredPos)
182  {
183  int degree(pos.size());
184  int i,j,m,n;
185 
186  // First, compute interpolation factors
187  typedef std::vector< T > vectorType;
188  std::vector< vectorType > delta(degree, vectorType(degree, 0.0));
189 
190  for(i=0; i < degree; ++i) {
191  for(j=0; j < degree; ++j) {
192  if(j != i) {
193  delta[i][j] = ((desiredPos - pos[j])/(pos[i] - pos[j]));
194  }
195  }
196  }
197 
198  double retVal(0.0);
199  for(i=0; i < degree; ++i) {
200  double sum(0.0);
201 
202  for(m=0; m < degree; ++m) {
203  if(m != i) {
204  double weight1(1.0/(pos[i]-pos[m]));
205  double sum2(0.0);
206 
207  for(j=0; j < degree; ++j) {
208  if((j != i) && (j != m)) {
209  double weight2(1.0/(pos[i]-pos[j]));
210  for(n=0; n < degree; ++n) {
211  if((n != j) && (n != m) && (n != i)) {
212  weight2 *= delta[i][n];
213  }
214  }
215  sum2 += weight2;
216  }
217  }
218  sum += sum2*weight1;
219  }
220  }
221  retVal += val[i] * sum;
222  }
223 
224  return retVal;
225 
226  } // End of 'lagrangeInterpolating2ndDerivative()'

231  template <class T>
232  T RSS (T aa, T bb, T cc)
233  {
234  T a(ABS(aa)), b(ABS(bb)), c(ABS(cc));
235  if(a < b) tswap(a,b);
236  if(a < c) tswap(a,c);
237  if(a == T(0)) return T(0);
238  return a * SQRT(1 + (b/a)*(b/a) + (c/a)*(c/a));
