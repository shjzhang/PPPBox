41 // Generate random numbers uniformly distributed from 0.0 to 1.0. Mbig and
42 // Mseed are large but arbitrary, but Mbig > Mseed. The 55 is not arbitrary.
43 double Rand(long seed)
44 {
45 #define Mbig 1000000000.
46 #define Mseed 161803398.
47 #define imod(x,y) ((x)-((x)/(y))*(y))
48  static short iff=0,inext,inextp;
49  static double Ma[55];
50  double mj,mk;
51  short i,ii,k;
52  if(!iff) {
53  if(seed < 0) seed=-seed;
54  mj = Mseed-seed;
55  mj = fmod(mj,Mbig);
56  Ma[54] = mj;
57  mk = 1.0;
58  for(i=0; i<55; i++) {
59  ii = imod(21*(i+1),55);
60  Ma[ii] = mk;
61  mk = mj - mk;
62  if(mk < 0.0) mk += Mbig;
63  mj = Ma[ii];
64  }
65  for(k=0; k<4; k++) {
66  for(i=0; i<55; i++) {
67  Ma[i] -= Ma[imod(i+30,55)];
68  if(Ma[i] < 0.0) Ma[i] += Mbig;
69  }
70  }
71  inext = -1;
72  inextp = 30;
73  iff=1;
74  }
75  inext++; if(inext == 55) inext=0;
76  inextp++; if(inextp == 55) inextp=0;
77  mj = Ma[inext]-Ma[inextp];
78  if(mj < 0.0) mj += Mbig;
79  Ma[inext] = mj;
80  return mj/Mbig;
81 #undef Mbig
82 #undef Mseed
83 #undef imod
84 }

87 // Generate normally distributed random numbers, zero mean and
88 // sqrt of variance sigma. Uses Box-Muller and Rand() above.
89 double RandNorm(double sigma)
90 {
91 #ifdef RAND_NORM_SAVE
92  static short iset=0;
93  static double saved;
94  double r, v1, v2, fact;
95  if(!iset) {
96  do {
97  v1 = 2.0*Rand(1)-1.0;
98  v2 = 2.0*Rand(1)-1.0;
99  r = v1*v1 + v2*v2;
100  } while( r >= 1.0 || r == 0.0);
101  fact = sigma*sqrt(-2.*log(r)/r);
102  saved = v1*fact;
103  iset = 1;
104  return v2*fact;
105  }
106  iset = 0;
107  return saved;
108 #else
109  double r, v1, v2, fact;
110  do {
111  v1 = 2.0*Rand(1)-1.0;
112  v2 = 2.0*Rand(1)-1.0;
113  r = v1*v1 + v2*v2;
114  } while( r >= 1.0 || r == 0.0);
115  fact = sigma*sqrt(-2.*log(r)/r);
116  return v2*fact;
117 #endif
118 }

121 // Return random integers between low and hi. If you want a different seed,
122 // call Rand(seed) before you call this.
123 int ARand(int low, int hi)
124 {
125  double r=Rand(),d=(double)(hi-low);
126  if(d < 0.0) d = -d;
127  d = r*d;
128  int i=(int)(d+0.5) + low;
129  return i;
130 }

144 // Generate a random walk sequence, given sqrt variance sigma, time step dt
145 // and previous point xlast.
146 double RandomWalk(double dt, double sigma, double xlast)
147 {
148  return xlast+RandNorm(sigma)*dt;
