47  // -------------------------------------------------------------------------
48  // Prepare for the autonomous solution by computing direction cosines,
49  // corrected pseudoranges and satellite system.
50  int PRSolution::PreparePRSolution(const CommonTime& Tr,
51  vector<SatID>& Sats,
52  vector<SatID::SatelliteSystem>& Syss,
53  const vector<double>& Pseudorange,
54  const XvtStore<SatID> *pEph,
55  Matrix<double>& SVP) const
56  throw()
57  {
58  LOG(DEBUG) << "PreparePRSolution at time " << printTime(Tr,timfmt);
59 
60  int j,noeph(0),N,NSVS;
61  size_t i;
62  CommonTime tx;
63  Xvt PVT;
64 
65  // if necessary, define the SystemIDs vector (but NOT the member data one)
66  if(Syss.size() == 0) {
67  for(i=0; i<Sats.size(); i++) {
68  SatID::SatelliteSystem sys=Sats[i].system;
69  if(vectorindex(Syss, sys) == -1) // not found
70  Syss.push_back(sys); // add it
71  }
72  }
73 
74  // must ignore and mark satellites if system is not found in Syss
75  for(N=0,i=0; i<Sats.size(); i++) {
76  if(Sats[i].id <= 0) continue; // already marked
77  if(vectorindex(Syss, Sats[i].system) == -1) {
78  LOG(DEBUG) << " PRSolution ignores satellite (system) "
79  << RinexSatID(Sats[i]) << " at time " << printTime(Tr,timfmt);
80  Sats[i].id = -Sats[i].id; // don't have system - mark it
81  continue;
82  }
83  LOG(DEBUG) << " Count sat " << RinexSatID(Sats[i]);
84  ++N; // count good sat
85  }
86 
87  LOG(DEBUG) << "Sats.size is " << Sats.size();
88  SVP = Matrix<double>(Sats.size(),4,0.0); // define the matrix to return
89  if(N <= 0) return 0; // nothing to do
90  NSVS = 0; // count good sats w/ ephem
91 
92  // loop over all satellites
93  for(i=0; i<Sats.size(); i++) {
94 
95  // skip marked satellites
96  if(Sats[i].id <= 0) {
97  LOG(DEBUG) << " PRSolution ignores marked satellite "
98  << RinexSatID(Sats[i]) << " at time " << printTime(Tr,timfmt);
99  continue;
100  }
101  LOG(DEBUG) << " Process sat " << RinexSatID(Sats[i]);
102 
103  // first estimate of transmit time
104  tx = Tr;
105 
106  // must align time systems.
107  // know system of Tr, and must assume system of pEph(sat) is system(sat).
108  // pEph must do calc in its sys, so must transform Tr to system(sat).
109  // convert time system of tx to that of Sats[i]
110 
111  tx -= Pseudorange[i]/C_MPS;
112  try {
113  LOG(DEBUG) << " go to getXvt with time " << printTime(tx,timfmt);
114  PVT = pEph->getXvt(Sats[i], tx); // get ephemeris range, etc
115  LOG(DEBUG) << " returned from getXvt";
116  }
117  catch(InvalidRequest& e) {
118  LOG(DEBUG) << "Warning - PRSolution ignores satellite (no ephemeris) "
119  << RinexSatID(Sats[i]) << " at time " << printTime(tx,timfmt)
120  << " [" << e.getText() << "]";
121  Sats[i].id = -::abs(Sats[i].id);
122  ++noeph;
123  continue;
124  }
125  catch(Exception& e) {
126  LOG(DEBUG) << "Oops - Exception " << e.getText();
127  }
128 
129  // update transmit time and get ephemeris range again
130  tx -= PVT.clkbias + PVT.relcorr;
131  try {
132  PVT = pEph->getXvt(Sats[i], tx);
133  }
134  catch(InvalidRequest& e) { // unnecessary....you'd think!
135  LOG(DEBUG) << "Warning - PRSolution ignores satellite (no ephemeris 2) "
136  << RinexSatID(Sats[i]) << " at time " << printTime(tx,timfmt)
137  << " [" << e.getText() << "]";
138  Sats[i].id = -::abs(Sats[i].id);
139  ++noeph;
140  continue;
141  }
142 
143  // SVP = {SV position at transmit time}, raw range + clk + rel
144  for(j=0; j<3; j++) SVP(i,j) = PVT.x[j];
145  SVP(i,3) = Pseudorange[i] + C_MPS * (PVT.clkbias + PVT.relcorr);
146 
147  LOG(DEBUG) << "SVP: Sat " << RinexSatID(Sats[i])
148  << " PR " << fixed << setprecision(3) << Pseudorange[i]
149  << " clkbias " << C_MPS*PVT.clkbias
150  << " relcorr " << C_MPS*PVT.relcorr;
151 
152  // count the good satellites
153  NSVS++;
154  }
155 
156  if(noeph == N) return -4; // no ephemeris for any good sat
157 
158  return NSVS;
159 
160  } // end PreparePRPRSolution

163  // -------------------------------------------------------------------------
164  // Compute a straightforward solution using all the unmarked data.
165  // Call PreparePRSolution first.
166  int PRSolution::SimplePRSolution(const CommonTime& T,
167  const vector<SatID>& Sats,
168  const Matrix<double>& SVP,
169  const Matrix<double>& invMC,
170  TropModel *pTropModel,
171  const int& niterLimit,
172  const double& convLimit,
173  const vector<SatID::SatelliteSystem>& Syss,
174  Vector<double>& Resids,
175  Vector<double>& Slopes)
176  throw(Exception)
177  {
178  if(!pTropModel) {
179  Exception e("Undefined tropospheric model");
180  GPSTK_THROW(e);
181  }
182  if(Sats.size() != SVP.rows() ||
183  (invMC.rows() > 0 && invMC.rows() != Sats.size())) {
184  LOG(ERROR) << "Sats has length " << Sats.size();
185  LOG(ERROR) << "SVP has dimension " << SVP.rows() << "x" << SVP.cols();
186  LOG(ERROR) << "invMC has dimension " << invMC.rows() << "x" << invMC.cols();
187  Exception e("Invalid dimensions");
188  GPSTK_THROW(e);
189  }
190  if(Syss.size() == 0) {
191  Exception e("Allowed satellite systems input (Syss) is empty!");
192  GPSTK_THROW(e);
193  }
194 
195  int iret(0),k,n;
196  size_t i, j;
197  double rho,wt,svxyz[3];
198  GPSEllipsoid ellip;
199 
200  Valid = Mixed = false;
201 
202  try {
203  // -----------------------------------------------------------
204  // counts, systems and dimensions
205  vector<SatID::SatelliteSystem> mySyss;
206  {
207  // define the Syss (system IDs) vector, and count good satellites
208  vector<SatID::SatelliteSystem> tempSyss;
209  for(Nsvs=0,i=0; i<Sats.size(); i++) {
210  if(Sats[i].id <= 0) // reject marked sats
211  continue;
212 
213  SatID::SatelliteSystem sys(Sats[i].system); // get this system
214  if(vectorindex(Syss, sys) == -1) // reject disallowed sys
215  continue;
216 
217  Nsvs++; // count it
218  if(vectorindex(tempSyss, sys) == -1) // add unique system
219  tempSyss.push_back(sys);
220  }
221 
222  // must sort as in Syss
223  for(i=0; i<Syss.size(); i++)
224  if(vectorindex(tempSyss, Syss[i]) != -1)
225  mySyss.push_back(Syss[i]);
226  }
227 
228  // dimension of the solution vector (3 pos + 1 clk/sys)
229  const int dim(3 + mySyss.size());
230  if(dim > 4) Mixed = true;
231 
232  // require number of good satellites to be >= number unknowns (no RAIM here)
233  if(Nsvs < dim) return -3;
234 
235  // -----------------------------------------------------------
236  // build the measurement covariance matrix
237  Matrix<double> iMC;
238  if(invMC.rows() > 0) {
239  LOG(DEBUG) << "Build inverse MCov";
240  iMC = Matrix<double>(Nsvs,Nsvs,0.0);
241  for(n=0,i=0; i<Sats.size(); i++) {
242  if(Sats[i].id <= 0) continue;
243  for(k=0,j=0; j<Sats.size(); j++) {
244  if(Sats[j].id <= 0) continue;
245  iMC(n,k) = invMC(i,j);
246  ++k;
247  }
248  ++n;
249  }
250  LOG(DEBUG) << "inv MCov matrix is\n" << fixed << setprecision(4) << iMC;
251  }
252 
253  // -----------------------------------------------------------
254  // define for computation
255  Vector<double> CRange(Nsvs),dX(dim);
256  Matrix<double> P(Nsvs,dim,0.0),PT,G(dim,Nsvs),PG(Nsvs,Nsvs),Rotation;
257  Triple dirCos;
258  Xvt SV,RX;
259 
260  Solution.resize(dim);
261  Covariance.resize(dim,dim);
262  Resids.resize(Nsvs);
263  Slopes.resize(Nsvs);
264  LOG(DEBUG) << " Solution dimension is " << dim << " and Nsvs is " << Nsvs;
265 
266  // prepare for iteration loop
267  // iterate at least twice so that trop model gets evaluated
268  int n_iterate(0), niter_limit(niterLimit < 2 ? 2 : niterLimit);
269  double converge(0.0);
270 
271  // start with solution = apriori
272  Vector<double> APSolution;
273  if(hasMemory) {
274  APSolution = Solution = memory.getAprioriSolution(mySyss);
275  LOG(DEBUG) << " apriori solution (" << Solution.size() << ") is [ "
276  << fixed << setprecision(3) << Solution << " ]";
277  }
278  else {
279  Solution = Vector<double>(dim,0.0);
280  LOG(DEBUG) << " no memory - no apriori solution";
281  }
282 
283  // -----------------------------------------------------------
284  // iteration loop
285  do {
286  TropFlag = false; // true means the trop corr was NOT applied
287 
288  // current estimate of position solution
289  RX.x = Triple(Solution(0),Solution(1),Solution(2));
290 
291  // loop over satellites, computing partials matrix
292  for(n=0,i=0; i<Sats.size(); i++) {
293  // ignore marked satellites
294  if(Sats[i].id <= 0) continue;
295 
296  // ------------ ephemeris
297  // rho is time of flight (sec)
298  if(n_iterate == 0)
299  rho = 0.070; // initial guess: 70ms
300  else
301  rho = RSS(SVP(i,0)-Solution(0),
302  SVP(i,1)-Solution(1), SVP(i,2)-Solution(2))/ellip.c();
303 
304  // correct for earth rotation
305  wt = ellip.angVelocity()*rho; // radians
306  svxyz[0] = ::cos(wt)*SVP(i,0) + ::sin(wt)*SVP(i,1);
307  svxyz[1] = -::sin(wt)*SVP(i,0) + ::cos(wt)*SVP(i,1);
308  svxyz[2] = SVP(i,2);
309 
310  // rho is now geometric range
311  rho = RSS(svxyz[0]-Solution(0),
312  svxyz[1]-Solution(1),
313  svxyz[2]-Solution(2));
314 
315  // direction cosines
316  dirCos[0] = (Solution(0)-svxyz[0])/rho;
317  dirCos[1] = (Solution(1)-svxyz[1])/rho;
318  dirCos[2] = (Solution(2)-svxyz[2])/rho;
319 
320  // ------------ data
321  // corrected pseudorange (m) minus geometric range
322  CRange(n) = SVP(i,3) - rho;
323 
324  // correct for troposphere and PCOs (but not on the first iteration)
325  if(n_iterate > 0) {
326  SV.x = Triple(svxyz[0],svxyz[1],svxyz[2]);
327  Position R,S;
328  R.setECEF(RX.x[0],RX.x[1],RX.x[2]);
329  S.setECEF(SV.x[0],SV.x[1],SV.x[2]);
330 
331  // trop
332  double tc(R.getHeight()); // tc is a dummy here
333  // must test R for reasonableness to avoid corrupting TropModel
334  if(R.elevation(S) < 0.0 || tc > 100000.0 || tc < -1000.0) {
335  tc = 0.0;
336  TropFlag = true; // true means failed to apply trop corr
337  }
338  else
339  tc = pTropModel->correction(R,S,T); // pTropModel not const
340 
341  CRange(n) -= tc;
342  LOG(DEBUG) << "Trop " << i << " " << Sats[i] << " "
343  << fixed << setprecision(3) << tc;
344 
345  } // end if n_iterate > 0
346 
347  // get the index, for this clock, in the solution vector
348  j = 3 + vectorindex(mySyss, Sats[i].system); // Solution ~ X,Y,Z,clks
349 
350  // find the clock for the sat's system
351  const double clk(Solution(j));
352  LOG(DEBUG) << "Clock is (" << j << ") " << clk;
353 
354  // data vector: corrected range residual
355  Resids(n) = CRange(n) - clk;
356 
357  // ------------ least squares
358  // partials matrix
359  P(n,0) = dirCos[0]; // x direction cosine
360  P(n,1) = dirCos[1]; // y direction cosine
361  P(n,2) = dirCos[2]; // z direction cosine
362  P(n,j) = 1.0; // clock
363 
364  // ------------ increment index
365  // n is index and number of good satellites - also used for Slope
366  n++;
367 
368  } // end loop over satellites
369 
370  if(n != Nsvs) {
371  Exception e("Counting error after satellite loop");
372  GPSTK_THROW(e);
373  }
374 
375  LOG(DEBUG) << "Partials (" << P.rows() << "x" << P.cols() << ")\n"
376  << fixed << setprecision(4) << P;
377  LOG(DEBUG) << "Resids (" << Resids.size() << ") "
378  << fixed << setprecision(3) << Resids;
379 
380  // ------------------------------------------------------
381  // compute information matrix (inverse covariance) and generalized inverse
382  PT = transpose(P);
383 
384  // weight matrix = measurement covariance inverse
385  if(invMC.rows() > 0) Covariance = PT * iMC * P;
386  else Covariance = PT * P;
387 
388  // invert using SVD
389  try {
390  Covariance = inverseSVD(Covariance);
391  }
392  catch(SingularMatrixException& sme) { return -2; }
393  LOG(DEBUG) << "InvCov (" << Covariance.rows() << "x" << Covariance.cols()
394  << ")\n" << fixed << setprecision(4) << Covariance;
395 
396  // generalized inverse
397  if(invMC.rows() > 0) G = Covariance * PT * iMC;
398  else G = Covariance * PT;
399 
400  // PG is used for Slope computation
401  PG = P * G;
402  LOG(DEBUG) << "PG (" << PG.rows() << "x" << PG.cols()
403  << ")\n" << fixed << setprecision(4) << PG;
404 
405  n_iterate++; // increment number iterations
406 
407  // ------------------------------------------------------
408  // compute solution
409  dX = G * Resids;
410  LOG(DEBUG) << "Computed dX(" << dX.size() << ")";
411  Solution += dX;
412 
413  // ------------------------------------------------------
414  // test for convergence
415  converge = norm(dX);
416  if(n_iterate > 1 && converge < convLimit) { // success: quit
417  iret = 0;
418  break;
419  }
420  if(n_iterate >= niter_limit || converge > 1.e10) { // failure: quit
421  iret = -1;
422  break;
423  }
424 
425  } while(1); // end iteration loop
426  LOG(DEBUG) << "Out of iteration loop";
427 
428  if(TropFlag) LOG(DEBUG) << "Trop correction not applied at time "
429  << printTime(T,timfmt);
430 
431  // compute slopes and find max member
432  MaxSlope = 0.0;
433  Slopes = 0.0;
434  if(iret == 0) for(j=0,i=0; i<Sats.size(); i++) {
435  if(Sats[i].id <= 0) continue;
436 
437  // NB when one (few) sats have their own clock, PG(j,j) = 1 (nearly 1)
438  // and slope is inf (large)
439  if(::fabs(1.0-PG(j,j)) < 1.e-8) continue;
440 
441  for(int k=0; k<dim; k++) Slopes(j) += G(k,j)*G(k,j); // TD dim=4 here?
442  Slopes(j) = SQRT(Slopes(j)*double(n-dim)/(1.0-PG(j,j)));
443  if(Slopes(j) > MaxSlope) MaxSlope = Slopes(j);
444  j++;
445  }
446 
447  // compute pre-fit residuals
448  if(hasMemory)
449  PreFitResidual = P*(Solution-APSolution) - Resids;
450 
451  // Compute RMS residual (member)
452  RMSResidual = RMS(Resids);
453 
454  // Find the maximum slope (member)
455  //MaxSlope = 0.0;
456  //for(i=0; i<Slopes.size(); i++)
457  // if(Slopes(i) > MaxSlope)
458  // MaxSlope = Slopes[i];
459 
460  // save to member data
461  currTime = T;
462  SatelliteIDs = Sats;
463  SystemIDs = mySyss;
464  invMeasCov = iMC;
465  Partials = P;
466  NIterations = n_iterate;
467  Convergence = converge;
468  Valid = true;
469 
470  return iret;
471 
472  } catch(Exception& e) { GPSTK_RETHROW(e); }
473 
474  } // end PRSolution::SimplePRSolution

477  // -------------------------------------------------------------------------
478  // Compute a solution using RAIM.
479  int PRSolution::RAIMCompute(const CommonTime& Tr,
480  vector<SatID>& Sats,
481  vector<SatID::SatelliteSystem>& Syss,
482  const vector<double>& Pseudorange,
483  const Matrix<double>& invMC,
484  const XvtStore<SatID> *pEph,
485  TropModel *pTropModel)
486  throw(Exception)
487  {
488  try {
489  LOG(DEBUG) << "RAIMCompute at time " << printTime(Tr,gpsfmt);
490 
491  int iret,N;
492  size_t i,j;
493  vector<int> GoodIndexes;
494  // use these to save the 'best' solution within the loop.
495  // BestRMS marks the 'Best' set as unused.
496  bool BestTropFlag(false);
497  int BestNIter(0),BestIret(-5);
498  double BestRMS(-1.0),BestSL(0.0),BestConv(0.0);
499  Vector<double> BestSol(3,0.0),BestPFR;
500  vector<SatID> BestSats,SaveSats;
501  Matrix<double> SVP,BestCov,BestInvMCov,BestPartials;
502  vector<SatID::SatelliteSystem> BestSyss;
503 
504  // initialize
505  Valid = false;
506  currTime = Tr;
507  TropFlag = SlopeFlag = RMSFlag = false;
508 
509  // ----------------------------------------------------------------
510  // fill the SVP matrix, and use it for every solution
511  // NB this routine will: define Syss if it is empty,
512  // reject sat systems not found in Syss, and
513  // reject sats without ephemeris.
514  N = PreparePRSolution(Tr, Sats, Syss, Pseudorange, pEph, SVP);
515 
516  if(LOGlevel >= ConfigureLOG::Level("DEBUG")) {
517  LOG(DEBUG) << "Prepare returns " << N;
518  ostringstream oss;
519  oss << "RAIMCompute: after PrepareAS(): Satellites:";
520  for(i=0; i<Sats.size(); i++) {
521  RinexSatID rs(::abs(Sats[i].id), Sats[i].system);
522  oss << " " << (Sats[i].id < 0 ? "-" : "") << rs;
523  }
524  oss << endl;
525 
526  oss << " SVP matrix("
527  << SVP.rows() << "," << SVP.cols() << ")" << endl;
528  oss << fixed << setw(16) << setprecision(5) << SVP;
529 
530  LOG(DEBUG) << oss.str();
531  }
532 
533  // return is >=0(number of good sats) or -4(no ephemeris)
534  if(N <= 0) return -4;
535 
536  // ----------------------------------------------------------------
537  // Build GoodIndexes based on Sats; save Sats as SaveSats.
538  // Sats is used to mark good sats (id > 0) and those to ignore (id <= 0).
539  // SaveSats saves the original so it can be reused for each RAIM solution.
540  // Let GoodIndexes be all the indexes of Sats that are good:
541  // Sats[GoodIndexes[.]].id > 0 by definition.
542  SaveSats = Sats;
543  for(i=0; i<Sats.size(); i++)
544  if(Sats[i].id > 0)
545  GoodIndexes.push_back(i);
546 
547  // dump good satellites for debug
548  if(LOGlevel >= ConfigureLOG::Level("DEBUG")) {
549  ostringstream oss;
550  oss << " Good satellites (" << N << ") are:";
551  for(i=0; i<GoodIndexes.size(); i++)
552  oss << " " << RinexSatID(Sats[GoodIndexes[i]]);
553  LOG(DEBUG) << oss.str();
554  }
555 
556  // ----------------------------------------------------------------
557  // now compute the solution, first with all the data. If this fails,
558  // RAIM: reject 1 satellite at a time and try again, then 2, etc.
559 
560  // Slopes for each satellite are computed (cf. the RAIM algorithm)
561  Vector<double> Slopes;
562  // Resids stores the post-fit data residuals.
563  Vector<double> Resids;
564 
565  // stage is the number of satellites to reject.
566  int stage(0);
567 
568  do {
569  // compute all the combinations of N satellites taken stage at a time
570  Combinations Combo(N,stage);
571 
572  // compute a solution for each combination of marked satellites
573  do {
574  // Mark the satellites for this combination
575  Sats = SaveSats;
576  for(i=0; i<GoodIndexes.size(); i++)
577  if(Combo.isSelected(i))
578  Sats[GoodIndexes[i]].id = -::abs(Sats[GoodIndexes[i]].id);
579 
580  if(LOGlevel >= ConfigureLOG::Level("DEBUG")) {
581  ostringstream oss;
582  oss << " RAIM: Try the combo ";
583  for(i=0; i<Sats.size(); i++) {
584  RinexSatID rs(::abs(Sats[i].id), Sats[i].system);
585  oss << " " << (Sats[i].id < 0 ? "-" : " ") << rs;
586  }
587  LOG(DEBUG) << oss.str();
588  }
589 
590  // ----------------------------------------------------------------
591  // Compute a solution given the data; ignore ranges for marked
592  // satellites. Fill Vector 'Slopes' with slopes for each unmarked
593  // satellite.
594  // Return 0 ok
595  // -1 failed to converge
596  // -2 singular problem
597  // -3 not enough good data
598  // -4 no ephemeris
599  iret = SimplePRSolution(Tr, Sats, SVP, invMC, pTropModel,
600  MaxNIterations, ConvergenceLimit, Syss, Resids, Slopes);
601 
602  LOG(DEBUG) << " RAIM: SimplePRS returns " << iret;
603  if(iret <= 0 && iret > BestIret) BestIret = iret;
604 
605  // ----------------------------------------------------------------
606  // if error, either quit or continue with next combo (SPS sets Valid F)
607  if(iret < 0) {
608  if(iret == -1) {
609  LOG(DEBUG) << " SPS: Failed to converge - go on";
610  continue;
611  }
612  else if(iret == -2) {
613  LOG(DEBUG) << " SPS: singular - go on";
614  continue;
615  }
616  else if(iret == -3) {
617  LOG(DEBUG) <<" SPS: not enough satellites: quit";
618  break;
619  }
620  else if(iret == -4) {
621  LOG(DEBUG) <<" SPS: no ephemeris: quit";
622  break;
623  }
624  }
625 
626  // ----------------------------------------------------------------
627  // print solution with diagnostic information
628  LOG(DEBUG) << outputString(string("RPS"),iret);
629 
630  // do again for residuals
631  // if memory exists, output residuals
632  //if(hasMemory) LOG(DEBUG) << outputString(string("RAP"), -99,
633  //(Solution-memory.APSolution));
634 
635  // deal with the results of SimplePRSolution()
636  // save 'best' solution for later
637  if(BestRMS < 0.0 || RMSResidual < BestRMS) {
638  BestRMS = RMSResidual;
639  BestSol = Solution;
640  BestSats = SatelliteIDs;
641  BestSyss = SystemIDs;
642  BestSL = MaxSlope;
643  BestConv = Convergence;
644  BestNIter = NIterations;
645  BestCov = Covariance;
646  BestInvMCov = invMeasCov;
647  BestPartials = Partials;
648  BestPFR = PreFitResidual;
649  BestTropFlag = TropFlag;
650  BestIret = iret;
651  }
652 
653  if(stage==0 && RMSResidual < RMSLimit)
654  break;
655 
656  } while(Combo.Next() != -1); // get the next combinations and repeat
657 
658  // end of the stage
659  if(BestRMS > 0.0 && BestRMS < RMSLimit) { // success
660  LOG(DEBUG) << " RAIM: Success in the RAIM loop";
661  iret = 0;
662  break;
663  }
664 
665  // go to next stage
666  stage++;
667 
668  // but not if too many are being rejected
669  if(NSatsReject > -1 && stage > NSatsReject) {
670  LOG(DEBUG) << " RAIM: break before stage " << stage
671  << " due to NSatsReject " << NSatsReject;
672  break;
673  }
674 
675  // already broke out of the combo loop...
676  if(iret == -3 || iret == -4) {
677  LOG(DEBUG) << " RAIM: break before stage " << stage
678  << "; " << (iret==-3 ? "too few sats" : "no ephemeris");
679  break;
680  }
681 
682  LOG(DEBUG) << " RAIM: go to stage " << stage;
683 
684  } while(1); // end loop over stages
685 
686  // ----------------------------------------------------------------
687  // copy out the best solution and return
688  if(iret >= 0) {
689  Sats = SatelliteIDs = BestSats;
690  SystemIDs = BestSyss;
691  Solution = BestSol;
692  Covariance = BestCov;
693  invMeasCov = BestInvMCov;
694  Partials = BestPartials;
695  PreFitResidual = BestPFR;
696  Convergence = BestConv;
697  NIterations = BestNIter;
698  RMSResidual = BestRMS;
699  MaxSlope = BestSL;
700  TropFlag = BestTropFlag;
701  iret = BestIret;
702 
703  if(iret==0) {
704  DOPCompute(); // compute DOPs
705  if(hasMemory) // update memory
706  memory.add(Solution,Covariance,PreFitResidual,Partials,invMeasCov);
707  }
708 
709  // must add zeros to state, covariance and partials if these don't match
710  if(Syss.size() != BestSyss.size()) {
711  N = 3+Syss.size();
712  Solution = Vector<double>(N,0.0);
713  Covariance = Matrix<double>(N,N,0.0);
714  Partials = Matrix<double>(BestPartials.rows(),N,0.0);
715 
716  // build a little map of indexes; note systems are sorted alike
717  vector<int> indexes;
718  for(j=0,i=0; i<Syss.size(); i++) {
719  indexes.push_back(Syss[i] == BestSyss[j] ? j++ : -1);
720  }
721 
722  // copy over position and position,clk parts
723  for(i=0; i<3; i++) {
724  Solution(i) = BestSol(i);
725  for(j=0; j<Partials.rows(); j++) Partials(j,i) = BestPartials(j,i);
726  for(j=0; j<3; j++) Covariance(i,j) = BestCov(i,j);
727  for(j=0; j<indexes.size(); j++) {
728  Covariance(i,3+j)=(indexes[j]==-1 ? 0.:BestCov(i,3+indexes[j]));
729  Covariance(3+j,i)=(indexes[j]==-1 ? 0.:BestCov(3+indexes[j],i));
730  }
731  }
732 
733  // copy the clock part, inserting zeros where needed
734  for(j=0; j<indexes.size(); j++) {
735  int n(indexes[j]);
736  bool ok(n != -1);
737  Solution(3+j) = (ok ? BestSol(3+n) : 0.0);
738  for(i=0; i<Partials.rows(); i++)
739  Partials(i,3+j) = (ok ? BestPartials(i,3+n) : 0.0);
740  for(i=0; i<indexes.size(); i++) {
741  Covariance(3+i,3+j) = (ok ? BestCov(3+i,3+n) : 0.0);
742  Covariance(3+j,3+i) = (ok ? BestCov(3+n,3+i) : 0.0);
743  }
744  }
745  }
746 
747  // compute number of satellites actually used
748  for(Nsvs=0,i=0; i<SatelliteIDs.size(); i++)
749  if(SatelliteIDs[i].id > 0)
750  Nsvs++;
751  }
752 
753  if(iret==0) {
754  if(BestSL > SlopeLimit) { iret = 1; SlopeFlag = true; }
755  if(BestSL > SlopeLimit/2.0 && Nsvs == 5) { iret = 1; SlopeFlag = true; }
756  if(BestRMS >= RMSLimit) { iret = 1; RMSFlag = true; }
757  if(TropFlag) iret = 1;
758  Valid = true;
759  }
760  else if(iret == -1) Valid = false;
761 
762  LOG(DEBUG) << " RAIM exit with ret value " << iret
763  << " and Valid " << (Valid ? "T":"F");
764 
765  return iret;
766  }
767  catch(Exception& e) {
768  GPSTK_RETHROW(e);
769  }
770  } // end PRSolution::RAIMCompute()

773  // -------------------------------------------------------------------------
774  int PRSolution::DOPCompute(void) throw(Exception)
775  {
776  try {
777  Matrix<double> PTP(transpose(Partials)*Partials);
778  Matrix<double> Cov(inverseLUD(PTP));
779  PDOP = SQRT(Cov(0,0)+Cov(1,1)+Cov(2,2));
780  TDOP = 0.0;
781  for(size_t i=3; i<Cov.rows(); i++) TDOP += Cov(i,i);
782  TDOP = SQRT(TDOP);
783  GDOP = RSS(PDOP,TDOP);
784  return 0;
785  }
786  catch(Exception& e) { GPSTK_RETHROW(e); }
787  }
