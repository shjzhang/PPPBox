56  // See notes in the .hpp. This function is designed to be called AFTER all elements
57  // are loaded. It can then make adjustments to time relationships based on
58  // inter-comparisons between sets of elements that cannot be performed until the
59  // ordering has been determined.
60  void GPSEphemerisStore::rationalize(void)
61  {
62  // loop over satellites
63  SatTableMap::iterator it;
64  for (it = satTables.begin(); it != satTables.end(); it++) {
65  TimeOrbitEphTable& table = it->second;
66  TimeOrbitEphTable::iterator ei;
67  TimeOrbitEphTable::iterator eiPrev;
68  bool begin = true;
69  double previousOffset = 0.0;
70  bool previousIsOffset = false;
71  bool currentIsOffset = false;
72 
73  //string tForm = "%02H:%02M:%02S";
74 
75  // Scan the map for this SV looking for uploads. Uploads are identifed by
76  // Toe values that are offset from an even hour.
77  OrbitEph* oePrev = 0;
78  for (ei = table.begin(); ei != table.end(); ei++)
79  {
80  currentIsOffset = false; // start with this assumption
81  OrbitEph* oe = ei->second;
82  long Toe = (long) (static_cast<GPSWeekSecond> (oe->ctToe)).sow;
83  double currentOffset = Toe % 3600;
84 
85  //cout << "Top of For Loop. oe->beginValid = "
86  // << printTime(oe->beginValid,tForm);
87  //cout << ", currentOffset =" << currentOffset << endl;
88 
89  if ( (currentOffset)!=0) {
90  currentIsOffset = true;
91 
92  //cout << "*** Found an offset" << endl;
93  //cout << " currentIsOffest: " << currentIsOffset;
94  //cout << " previousIsOffest: " << previousIsOffset;
95  //cout << " current, previous Offset = " << currentOffset
96  // << ", " << previousOffset << endl;
97 
98  // If this set is offset AND the previous set is offset AND
99  // the two offsets are the same, then this is the SECOND
100  // set of elements in an upload. In that case the OrbitEph
101  // load routines have conservatively set the beginning time
102  // of validity to the transmit time because there was no
103  // way to prove this was the second data set. Since we can
104  // now prove its second by observing the ordering, we can
105  // adjust the beginning of validity as needed.
106  // Since the algorithm is dependent on the message
107  // format, this must be done in OrbitEph.
108  // IMPORTANT NOTE: We also need to adjust the
109  // key in the map, which is based on the beginning
110  // of validity. However, we can't do it in this
111  // loop without destroying the integrity of the
112  // iterator. This is handled later in a second
113  // loop. See notes on the second loop, below.
114  if (previousIsOffset &&
115  currentIsOffset &&
116  currentOffset==previousOffset)
117  {
118  //cout << "*** Adjusting beginValid" << endl;
119  oe->adjustValidity();
120  }
121 
122  // If the previous set is not offset, then
123  // we've found an upload
124  // For that matter, if previous IS offset, but
125  // the current offset is different than the
126  // previous, then it is an upload.
127  if (!previousIsOffset ||
128  (previousIsOffset && (currentOffset!=previousOffset) ) )
129  {
130  // Record the offset for later reference
131  previousOffset = currentOffset;
132 
133  // Adjust the ending time of validity of any elements
134  // broadcast BEFORE the new upload such that they
135  // end at the beginning validity of the upload.
136  // That beginning validity value should already be
137  // set to the transmit time (or earliest transmit
138  // time found) by OrbitEph and GPSEphemerisStore.addOrbitEph( )
139  // This adjustment is consistent with the IS-GPS-XXX
140  // rules that a new upload invalidates previous elements.
141  // Note that this may be necessary for more than one
142  // preceding set of elements.
143  if (!begin)
144  {
145  //cout << "*** Adjusting endValid Times" << endl;
146  TimeOrbitEphTable::iterator ri;
147  // We KNOW it exists in the map
148  ri = table.find(oePrev->beginValid);
149  bool done = false;
150  while (!done)
151  {
152  OrbitEph* oeRev = ri->second;
153  //cout << "Testing Toe of " << printTime(oeRev->ctToe,"%02H:%02M:%02S");
154  //cout << " with endValid of " << printTime(oeRev->endValid,"%02H:%02M:%02S") << endl;
155 
156  // If the current set of elements has an ending
157  // validity prior to the upload, then do NOT
158  // adjust the ending and set done to true.
159  if (oeRev->endValid <= oe->beginValid) done = true;
160 
161  // Otherwise, adjust the ending validity to
162  // match the begin of the upload.
163  else oeRev->endValid = oe->beginValid;
164 
165  // If we've reached the beginning of the map, stop.
166  // Otherwise, decrement and test again.
167  if (ri!=table.begin()) ri--;
168  else done = true;
169  }
170  }
171  }
172  }
173 
174  // Update condition flags for next loop
175  previousIsOffset = currentIsOffset;
176  oePrev = oe; // May need this for next loop.
177  begin = false;
178  //cout << "Bottom of For loop. currentIsOffset: " << currentIsOffset <<
179  // ", previousIsOffset: " << previousIsOffset << endl;
180  } //end inner for-loop
181 
182  // The preceding process has left some elements in a condition
183  // when the beginValid value != the key in the map. This
184  // must be addressed, but the map key is a const (by definition).
185  // We have to search the map for these disagreements. When found,
186  // the offending item need to be copied out of the map, the
187  // existing entry deleted, the item re-entered with the new key.
188  // Since it is unsafe to modify a map while traversing the map,
189  // each time this happens, we have to reset the loop process.
190  //
191  // NOTE: Simplistically, we could restart the loop at the
192  // beginning each time. HOWEVER, we sometimes load a long
193  // period in a map (e.g. a year). At one upload/day, that
194  // would mean ~365 times, each time one day deeper into the map.
195  // As an alternate, we use the variable loopStart to note how
196  // far we scanned prior to finding a problem and manipulating
197  // the map. Then we can restart at that point.
198  bool done = false;
199  CommonTime loopStart = CommonTime::BEGINNING_OF_TIME;
200  while (!done)
201  {
202  ei = table.lower_bound(loopStart);
203  while (ei!=table.end())
204  {
205  OrbitEph* oe = ei->second;
206  if (ei->first!=oe->beginValid)
207  {
208  OrbitEph* oeAdj= oe->clone(); // Adjustment was done in
209  // first loop above.
210  delete ei->second; // oe becomes invalid.
211  table.erase(ei); // Remove the map entry.
212  table[oeAdj->beginValid] = oeAdj->clone(); // Add back to map
213  break; // exit this while loop without finishing
214  }
215  loopStart = ei->first; // Scanned this far successfully, so
216  // save this as a potential restart point.
217  ei++;
218  if (ei==table.end()) done = true; // Successfully completed
219  // the loop w/o finding any
220  // mismatches. We're done.
221  }
222  }
223 
224  // Well, not quite done. We need to update the initial/final
225  // times of the map.
226  const TimeOrbitEphTable::iterator Cei = table.begin( );
227  initialTime = Cei->second->beginValid;
228  const TimeOrbitEphTable::reverse_iterator rCei = table.rbegin();
229  finalTime = rCei->second->endValid;
230 
231  } // end outer for-loop
232  }

235  // Add all ephemerides to an existing list<GPSEphemeris> for given satellite
236  // If sat.id is -1 (the default), all ephemerides are added.
237  // @return the number of ephemerides added.
238  int GPSEphemerisStore::addToList(list<GPSEphemeris>& gpslist, SatID sat) const
239  {
240  // get the list from OrbitEphStore
241  list<OrbitEph*> oelst;
242  OrbitEphStore::addToList(oelst,SatID(-1,SatID::systemGPS));
243 
244  int n(0);
245  list<OrbitEph*>::const_iterator it;
246  for(it = oelst.begin(); it != oelst.end(); ++it) {
247  OrbitEph *ptr = *it;
248  GPSEphemeris *gpsptr = dynamic_cast<GPSEphemeris*>(ptr);
249  GPSEphemeris gpseph(*gpsptr);
250  gpslist.push_back(gpseph);
251  n++;
252  }
253 
254  return n;
255  }
