124  void BrcClockCorrection::loadData(const ObsID obsIDArg, const short PRNIDArg,
125  const short fullweeknum, const long subframe1[10] )
126  throw(InvalidParameter)
127  {
128  double ficked[60];
129 
130  //Load overhead members
131  satSys = "G";
132  obsID = obsIDArg;
133  PRNID = PRNIDArg;
134 
135  //Convert Subframe 1
136  if (!subframeConvert(subframe1, fullweeknum, ficked))
137  {
138  InvalidParameter exc("Subframe 1 not valid.");
139  GPSTK_THROW(exc);
140  }
141  double Txmit = ficked[2]; // Time of week from handover word
142  short weeknum = static_cast<short>( ficked[5] );
143  short accFlag = static_cast<short>( ficked[7] );
144  short health = static_cast<short>( ficked[8] );
145  double TocSOW = ficked[12];
146  af2 = ficked[13];
147  af1 = ficked[14];
148  af0 = ficked[15];
149 
150  double diff = Txmit - TocSOW;
151  if (diff > HALFWEEK) // NOTE: This USED to be in DayTime, but DayTime is going away. Where is it now?
152  weeknum++; // Convert week # of transmission to week # of epoch time when Toc is forward across a week boundary
153  else if (diff < -HALFWEEK)
154  weeknum--; // Convert week # of transmission to week # of epoch time when Toc is back across a week boundary
155 
156  Toc = GPSWeekSecond( weeknum, TocSOW, TimeSystem::GPS );
157  URAoc = accFlag; //Store L1 C/A URA as URAoc
158  URAoc1 = 0;
159  URAoc2 = 0;
160  healthy = false;
161  if (health == 0)
162  healthy = true;
163  dataLoaded = true;
164  return;
165  }

178  double BrcClockCorrection::svClockBias(const CommonTime& t) const
179  throw(gpstk::InvalidRequest)
180  {
181  double dtc,elaptc;
182  elaptc = t - getEpochTime();
183  dtc = af0 + elaptc * ( af1 + elaptc * af2 );

305  static void timeDisplay( ostream & os, const CommonTime& t )
306  {
307  // Convert to CommonTime struct from GPS wk,SOW to M/D/Y, H:M:S.
308  GPSWeekSecond dummyTime;
309  dummyTime = GPSWeekSecond(t);
310  os << setw(4) << dummyTime.week << "(";
311  os << setw(4) << (dummyTime.week & 0x03FF) << ") ";
312  os << setw(6) << setfill(' ') << dummyTime.sow << " ";
313 
314  switch (dummyTime.getDayOfWeek())
315  {
316  case 0: os << "Sun-0"; break;
317  case 1: os << "Mon-1"; break;
318  case 2: os << "Tue-2"; break;
319  case 3: os << "Wed-3"; break;
320  case 4: os << "Thu-4"; break;
321  case 5: os << "Fri-5"; break;
322  case 6: os << "Sat-6"; break;
323  default: break;
324  }
325  os << " " << (static_cast<YDSTime>(t)).printf("%3j %5.0s ")
326  << (static_cast<CivilTime>(t)).printf("%02m/%02d/%04Y %02H:%02M:%02S");
327  }

330  static void shortcut(ostream & os, const long HOW )
331  {
332  short DOW, hour, min, sec;
333  long SOD, SOW;
334  short SOH;
335 
336  SOW = static_cast<long>( HOW );
337  DOW = static_cast<short>( SOW / SEC_PER_DAY );
338  SOD = SOW - static_cast<long>( DOW * SEC_PER_DAY );
339  hour = static_cast<short>( SOD/3600 );
340 
341  SOH = static_cast<short>( SOD - (hour*3600) );
342  min = SOH/60;
343 
344  sec = SOH - min * 60;
345  switch (DOW)
346  {
347  case 0: os << "Sun-0"; break;
348  case 1: os << "Mon-1"; break;
349  case 2: os << "Tue-2"; break;
350  case 3: os << "Wed-3"; break;
351  case 4: os << "Thu-4"; break;
352  case 5: os << "Fri-5"; break;
353  case 6: os << "Sat-6"; break;
354  default: break;
355  }
356 
357  os << ":" << setfill('0')
358  << setw(2) << hour
359  << ":" << setw(2) << min
360  << ":" << setw(2) << sec
361  << setfill(' ');
362  }
