88  // this function writes all valid header records
89  void AntexHeader::WriteHeaderRecords(FFStream& ffs) const
90  throw(FFStreamError, StringException)
91  {
92  AntexStream& strm = dynamic_cast<AntexStream&>(ffs);
93  string line;
94 
95  if((valid & versionValid)|systemValid) {
96  line = rightJustify(asString(version,1), 8);
97  line += string(12,' ');
98  line += system;
99  line = leftJustify(line, 60);
100  line += versionString;
101  strm << leftJustify(line,80) << endl;
102  strm.lineNumber++;
103  }
104  if(valid & pcvTypeValid) {
105  line = pcvType;
106  line += string(19,' ');
107  line += leftJustify(refAntType, 20);
108  line += leftJustify(refAntSerNum, 20);
109  line += pcvTypeString;
110  strm << leftJustify(line,80) << endl;
111  strm.lineNumber++;
112  }
113  if(valid & commentValid) {
114  vector<string>::const_iterator itr = commentList.begin();
115  while(itr != commentList.end()) {
116  line = leftJustify((*itr), 60);
117  line += headerCommentString;
118  strm << leftJustify(line,80) << endl;
119  strm.lineNumber++;
120  itr++;
121  }
122  }
123  if(valid & endValid) {
124  line = string(60, ' ');
125  line += endOfHeaderString;
126  strm << leftJustify(line,80) << endl;
127  strm.lineNumber++;
128  }
129  } // end AntexHeader::WriteHeaderRecords()

131  // this function parses a single header record
132  void AntexHeader::ParseHeaderRecord(string& line)
133  throw(FFStreamError)
134  {
135  string label(line, 60, 20);
136 
137  if(label == versionString) {
138  version = asDouble(line.substr(0,8));
139  system = line[20];
140  if(system != ' ' && system != 'G' &&
141  system != 'R' && system != 'E' && system != 'M')
142  {
143  stringstream os;
144  os >> system;
145  FFStreamError e("Satellite system is invalid: " + os.str());
146  GPSTK_THROW(e);
147  }
148  valid |= versionValid;
149  valid |= systemValid;
150  }
151  else if(label == pcvTypeString) {
152  pcvType = line[0];
153  if(pcvType != 'A' && pcvType != 'R') {
154  stringstream os;
155  os >> pcvType;
156  FFStreamError e("PCV type is invalid: " + os.str());
157  GPSTK_THROW(e);
158  }
159  refAntType = line.substr(20,20);
160  refAntSerNum = line.substr(40,20);
161  valid |= pcvTypeValid;
162  }
163  else if(label == headerCommentString) {
164  commentList.push_back(stripTrailing(line.substr(0,60)));
165  valid |= commentValid;
166  }
167  else if(label == endOfHeaderString) {
168  valid |= endValid;
169  }
170  else {
171  FFStreamError e("Unidentified label: " + label);
172  GPSTK_THROW(e);
173  }
174  } // end of AntexHeader::ParseHeaderRecord(string& line)

177  // This function parses the entire header from the given stream
178  void AntexHeader::reallyGetRecord(FFStream& ffs)
179  throw(std::exception, FFStreamError, StringException)
180  {
181  AntexStream& strm = dynamic_cast<AntexStream&>(ffs);
182 
183  // if already read, just return
184  if (strm.headerRead == true)
185  return;
186 
187  // since we're reading a new header, we need to reinitialize
188  // all our list structures. all the other objects should be ok.
189  // this also applies if we threw an exception the first time we read
190  // the header and are now re-reading it. some of these could be full
191  // and we need to empty them.
192  commentList.clear();
193  version = 1.3;
194  valid = 0;
195 
196  string line;
197  while(!(valid & endValid)) {
198  strm.formattedGetLine(line);
199  stripTrailing(line);
200 
201  if (line.length()==0)
202  continue;
203  else if(line.length()<60 || line.length()>80) {
204  FFStreamError e("Invalid line length");
205  GPSTK_THROW(e);
206  }
207 
208  try { ParseHeaderRecord(line); }
209  catch(FFStreamError& e) { GPSTK_RETHROW(e); }
210 
211  } // end while(not end of header)
212 
213  unsigned long allValid;
214  if (version == 1.3)
215  allValid = allValid13;
216  else {
217  FFStreamError e("Unknown or unsupported Antex version " + asString(version));
218  GPSTK_THROW(e);
219  }
220 
221  if((allValid & valid) != allValid) {
222  FFStreamError e("Incomplete or invalid header");
223  GPSTK_THROW(e);
224  }
225 
226  // If we get here, we should have reached the end of header line
227  strm.header = *this;
228  strm.headerRead = true;
229 
230  } // end of reallyGetRecord()
