49  // indicated time.
50  // @param[in] t the time to look up
51  // @return the position of the Sun at time (as a Triple)

56  throw(InvalidRequest)

78  /* Function to compute Sun position in CIS system (coordinates
79  * in meters)
80  * @param t Epoch
81  */
82  Triple SunPosition::getPositionCIS(const CommonTime& t) const
83  throw(InvalidRequest)
84  {
85 
86  // Test if the time interval is correct
87  if ( (t < SunPosition::initialTime) ||
88  (t > SunPosition::finalTime) )
89  {
90  InvalidRequest ir("Provided epoch is out of bounds.");
91  GPSTK_THROW(ir);
92  }
93 
94  // Compute the years, and fraction of year, pased since J1900.0
95  int y(static_cast<YDSTime>(t).year); // Current year
96  int doy(static_cast<YDSTime>(t).doy); // Day of current year
97  double fd( (static_cast<YDSTime>(t).sod/86400.0 ) ); // Fraction of day
98  int years( (y - 1900) ); // Integer number of years since J1900.0
99  int iy4( ( ((y%4)+4)%4 ) ); // Is it a leap year?
100 
101  // Compute fraction of year
102  double yearfrac = ( ( static_cast<double>(4*(doy-1/(iy4+1))
103  - iy4 - 2) + 4.0 * fd ) / 1461.0 );
104 
105  double time(years+yearfrac);
106 
107  // Compute the geometric mean longitude of the Sun
108  double elm( fmod((4.881628 + TWO_PI*yearfrac +
109  0.0001342*time), TWO_PI) );
110 
111  // Mean longitude of perihelion
112  double gamma(4.90823 + 0.00030005*time);
113 
114  // Mean anomaly
115  double em(elm-gamma);
116 
117  // Mean obliquity
118  double eps0(0.40931975 - 2.27e-6*time);
119 
120  // Eccentricity
121  double e(0.016751 - 4.2e-7*time);
122  double esq(e*e);
123 
124  // True anomaly
125  double v(em + 2.0*e*std::sin(em) + 1.25*esq*std::sin(2.0*em));
126 
127  // True ecliptic longitude
128  double elt(v+gamma);
129 
130  // True distance
131  double r( (1.0 - esq)/(1.0 + e*std::cos(v)) );
132 
133  // Moon's mean longitude
134  double elmm( fmod((4.72 + 83.9971*time),TWO_PI) );
135 
136  // Useful definitions
137  double coselt(std::cos(elt));
138  double sineps(std::sin(eps0));
139  double coseps(std::cos(eps0));
140  double w1(-r*std::sin(elt));
141  double selmm(std::sin(elmm));
142  double celmm(std::cos(elmm));
143 
144  Triple result;
145 
146  // Sun position is the opposite of Earth position
147  result.theArray[0] = (r*coselt+MeanEarthMoonBary*celmm)*AU_CONST;
148  result.theArray[1] = (MeanEarthMoonBary*selmm-w1)*coseps*AU_CONST;
149  result.theArray[2] = (-w1*sineps)*AU_CONST;
150 
151  return result;
152  } // End SunPosition::getPositionCIS()
