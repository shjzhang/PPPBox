123  // This function parses the entire header from the given stream
124  void Rinex3ClockHeader::reallyGetRecord(FFStream& ffs)
125  throw(std::exception, FFStreamError,
126  StringUtils::StringException)
127  {
128 
129  Rinex3ClockStream& strm = dynamic_cast<Rinex3ClockStream&>(ffs);
130 
131  // if already read, just return
132  if (strm.headerRead == true)
133  return;
134 
135  // since we're reading a new header, we need to reinitialize
136  // all our list structures. all the other objects should be ok.
137  // this also applies if we threw an exception the first time we read
138  // the header and are now re-reading it. some of these could be full
139  // and we need to empty them.
140  clear();
141 
142  // one file line
143  string line;
144 
145  while ( !(valid & endValid) )
146  {
147  strm.formattedGetLine(line);
148  StringUtils::stripTrailing(line);
149 
150  if ( line.length() == 0 )
151  {
152  FFStreamError ffse("No data read!");
153  GPSTK_THROW(ffse);
154  }
155  else if ( line.length() < 60 || line.length() > 80 )
156  {
157  FFStreamError ffse("Invalid line length");
158  GPSTK_THROW(ffse);
159  }
160 
161  try
162  {
163  ParseHeaderRecord(line);
164  }
165  catch(FFStreamError& ffse)
166  {
167  GPSTK_RETHROW(ffse);
168  }
169 
170  } // end while(not end of header)
171 
172 
173  // If we get here, we should have reached the end of header line
174  strm.header = *this;
175  strm.headerRead = true;
176 
177  return;
178 
179  } // End of method 'Rinex3ClockHeader::reallyGetRecord(FFStream& ffs)'

183  // this function parses a single header record
184  void Rinex3ClockHeader::ParseHeaderRecord(string& line)
185  throw(FFStreamError)
186  {
187 
188  string label(line, 60, 20);
189 
190  // RINEX VERSION / TYPE
191  if ( label == versionString )
192  {
193 
194  version = asDouble(line.substr(0,9));
195  fileType = strip(line.substr(20, 20));
196 
197  // check version
198  if ( version <= 0.0 ||
199  version > 3.0 )
200  {
201  FFStreamError e( "This isn't an anticipated version number." +
202  asString(version) );
203  GPSTK_THROW(e);
204  }
205 
206  // check type
207  if ( (fileType[0] != 'C') &&
208  (fileType[0] != 'c'))
209  {
210  FFStreamError e( "This isn't a Rinex3 Clock file type." );
211  GPSTK_THROW(e);
212  }
213 
214  // get satellite system
215  string system_str = strip(line.substr(40, 20));
216  try
217  {
218  system.fromString(system_str);
219  }
220  catch (Exception& e)
221  {
222  FFStreamError ffse( "Input satellite system is unsupported: "
223  + system_str + e.getText() );
224  GPSTK_THROW(ffse);
225  }
226 
227  valid |= versionValid;
228 
229  }
230  // PGM / RUN BY / DATE
231  else if ( label == runByString )
232  {
233 
234  fileProgram = strip(line.substr( 0, 20));
235  fileAgency = strip(line.substr(20, 20));
236  date = strip(line.substr(40, 20));
237  isPGM = true;
238 
239  valid |= runByValid;
240 
241  }
242  // COMMENT
243  else if ( label == commentString )
244  {
245 
246  string s = strip(line.substr(0, 60));
247  commentList.push_back(s);
248 
249  valid |= commentValid;
250 
251  }
252  // SYS / # / OBS TYPES
253  else if ( label == numObsString )
254  {
255 
256  numObsTyp = asInt( line.substr(3,3) );
257 
258  // TODO: more work needed
259 
260  valid |= numObsValid;
261 
262  }
263  // TIME SYSTEM ID
264  else if ( label == timeSystemString )
265  {
266 
267  timeSystem = line.substr(3,3);
268  valid |= timeSystemValid;
269 
270  }
271  // LEAP SECONDS
272  else if ( label == leapSecondsString )
273  {
274 
275  leapSeconds = asInt(line.substr(0,6));
276 
277  valid |= leapSecondsValid;
278 
279  }
280  // DCBS APPLIED
281  else if ( label == sysDCBString )
282  {
283 
284  valid |= sysDCBsValid;
285 
286  }
287  // PCVS APPLIED
288  else if ( label == sysPCVString )
289  {
290 
291  valid |= sysPCVsValid;
292 
293  }
294  // # / TYPES OF DATA
295  else if ( label == dataTypesString )
296  {
297 
298  // number of clock data types
299  int nTyp = asInt(line.substr(0,6));
300  // allocate memory
301  dataTypeList.resize(nTyp);
302  // add clock data types
303  for( int iTyp = 0; iTyp < nTyp; iTyp++ )
304  {
305  dataTypeList[iTyp] = strip( line.substr(6*(iTyp+1),6) );
306  }
307 
308  valid |= dataTypesValid;
309 
310  }
311  // STATION NAME / NUM
312  else if ( label == stationNameString )
313  {
314 
315  clk0Name = line.substr(0,4);
316 
317  valid |= stationNameValid;
318 
319  }
320  // STATION CLK REF
321  else if ( label == calibrationClkString )
322  {
323  calName = strip( line.substr(0,60) );
324 
325  valid |= calibrationClkValid;
326 
327  }
328  // ANALYSIS CENTER
329  else if ( label == acNameString )
330  {
331 
332  ac = line.substr(0, 3);
333  acName = strip(line.substr(5,55));
334  isAC = true;
335 
336  valid |= acNameValid;
337 
338  }
339  // # OF CLK REF
340  else if ( label == numRefClkString )
341  {
342 
343  // new reference clock record
344  RefClkRecord ref;
345  // get the number of reference clocks for this record
346  ref.nRef = asInt( line.substr(0,6) );
347  // epoch
348  if( asInt(line.substr(7,4)) )
349  {
350  CommonTime T0 = parseTime( line.substr(7,26) );
351  // only one time
352  if( timeFirst == CommonTime::BEGINNING_OF_TIME )
353  {
354  timeFirst = T0;
355  }
356  // left boundary
357  ref.refWin[0] = T0 - timeFirst;
358  // right boundary
359  T0 = parseTime( line.substr(34,26) );
360  ref.refWin[1] = T0 - timeFirst;
361 
362  // time inconsistency
363  if (T0 < timeFirst)
364  {
365  FFStreamError e( "Wrong epoch of file, expected epoch: " +
366  timeFirst.asString() + " detected epoch: " +
367  T0.asString() );
368  GPSTK_THROW(e);
369  }
370  }
371 
372  // add the ref clk record to the list
373  refClkList.push_back(ref);
374 
375  valid |= numRefClkValid;
376 
377  }
379  else if ( label == analysisClkRefString )
380  {
381 
382  // get the previous reference clock record
383  std::list<RefClkRecord>::iterator iRef = refClkList.end();
384  --iRef;
385 
386  // how many ref clk have been stored
387  size_t nClks = iRef->clk.size();
388 
389  // is there any inconsistency?
390  if ( nClks < iRef->nRef )
391  {
392 
393  RefClk refclk;
394  // reference clock info
395  refclk.name = line.substr(0,4);
396  refclk.sigma = asDouble( strip( line.substr(40,20) ) );
397  refclk.sigma *= 1e6; // ms^2
398  // add into the list
399  iRef->clk.push_back(refclk);
400 
401  }
402  else
403  {
404  FFStreamError e( string("Number of items found in header ") +
405  "is inconsitent to the entry in header" );
406  GPSTK_THROW(e);
407  }
408 
409  valid |= analysisClkRefValid;
410 
411  }
413  else if ( label == numStationsString )
414  {
415 
416  numSta = asInt(line.substr( 0, 6));
417  trfName = strip(line.substr(10, 50));
418 
419  valid |= numStationsValid;
420 
421  }
423  else if ( label == solnStaNameString )
424  {
425 
426  // get 4-character station name
427  string name = line.substr(0,4);
428  // add it into the list
429  clkNameList.push_back( name );
430 
431  // get integer & decimal part of the coordinates
432  int X = asInt( strip(line.substr(25,8)) );
433  int Xf = asInt( strip(line.substr(33,3)) );
434 
435  int Y = asInt( strip(line.substr(37,8)) );
436  int Yf = asInt( strip(line.substr(45,3)) );
437 
438  int Z = asInt( strip(line.substr(49,8)) );
439  int Zf = asInt( strip(line.substr(57,3)) );
440 
441  // geocentric coordinates (be careful to the sign)
442  double x = X>0 ? X*1.0+Xf*1e-3 : X*1.0-Xf*1e-3;
443  double y = Y>0 ? Y*1.0+Yf*1e-3 : Y*1.0-Yf*1e-3;
444  double z = Z>0 ? Z*1.0+Zf*1e-3 : Z*1.0-Zf*1e-3;
445 
446  // check the coordinates
447  double radius = std::sqrt(x*x+y*y+z*z);
448  // add them into the map
449  if (radius >= 5000.0e3 && radius < 12000.0e3)
450  {
451  staCoordList.push_back( Triple(X,Y,Z) );
452  }
453  else
454  {
455  staCoordList.push_back( Triple(0.0,0.0,0.0) );
456  }
457 
458  valid |= solnStaNameValid;
459 
460  }
461  // # OF SOLN SATS
462  else if ( label == numSatsString )
463  {
464 
465  numSVs = asInt(line.substr(0,6));
466 
467  valid |= numSatsValid;
468 
469  }
470  // PRN LIST
471  else if ( label == prnListString )
472  {
473 
474  string s = line.substr(0,60);
475  string word = stripFirstWord(s);
476 
477  while ( !word.empty() )
478  {
479  clkNameList.push_back( word.append(" ") );
480  word = stripFirstWord(s);
481  }
482 
483  valid |= prnListValid;
484 
485  }
486  // END OF HEADER
487  else if ( label == endOfHeader )
488  {
489 
490  valid |= endValid;
491 
492  }
493  else
494  {
495 
496  FFStreamError e("Unidentified label: " + label);
497  GPSTK_THROW(e);
498 
499  }
500 
501  return;
502 
503  } // End of method 'Rinex3ClockHeader::ParseHeaderRecord(string& line)'
