405  RTFileFrame<FileStream, FileData>::waitForData(unsigned wait)
406  {
407  sleep(wait);
408  if(readMode == AppendedData)
409  {
410  // reopen the file and skip to where we left off - openCurrentFile
411  // will overwrite lastPosition...
412  std::streampos cachedPos = lastPosition;
413  if(openCurrentFile())
414  fileStream->seekg(cachedPos);
415  lastPosition = cachedPos;
416  }
417  else
418  {
419  // start from the beginning
420  openCurrentFile();
421  }
422  }

472  template <class FileStream, class FileData>
473  bool
474  RTFileFrame<FileStream, FileData>::getRecord()
475  {
476  if (!isOK)
477  return false;
478 
479  // is the stream still good to read?
480  if (*fileStream >> lastData)
481  {
482  lastPosition = fileStream->tellg();
483  return true;
484  }
485  // the last read failed - try opening the next file until
486  // today's file is reched or else return false
487  else
488  {
489  // read the next file or not?
490  if (getMode == Smart)
491  {
492  if (!endOfDataSet())
493  {
494  // still before today?
495  gpstk::CommonTime today;
496  today.setYDoySod(today.year(), today.DOY(), 0);
497 
498  if (currentTime < today)
499  {
500  openNextDay();
501  return getRecord();
502  }
503  // we've read through today, so just wait until tomorrow's
504  // data comes in - the user needs to call waitForData()
505  // to continue
506  else
507  isOK = false;
508  }
509  else
510  {
511  isOK = false;
512  }
513  } // if (getmode == smart)
514 
515  // in dumb mode, the user needs to call openNextDay or
516  // openCurrentFile to continue.
517  else
518  {
519  isOK = false;
520  }
521  } // if (isOK)
522  return isOK;
523  }
