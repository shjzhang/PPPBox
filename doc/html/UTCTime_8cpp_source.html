73  Epoch UTCTime::asTDB()
74  {
75 
76  Epoch TT = this->asTT();
77 
78  struct MJDTime mjdTT;
79  mjdTT.MJDint = (long)floor(TT.MJD());
80  mjdTT.MJDfr = TT.MJD() - mjdTT.MJDint;
81 
82  double tdbtdt = 0.0;
83  double tdbtdtdot = 0.0;
84  long oldmjd = 0;
85  long l = 0;
86 
87  while ( mjdTT.MJDfr >= 1.0 )
88  {
89  mjdTT.MJDint++ ;
90  mjdTT.MJDfr-- ;
91  }
92  while ( mjdTT.MJDfr < 0.0 )
93  {
94  mjdTT.MJDint-- ;
95  mjdTT.MJDfr++ ;
96  }
97 
98  if ( mjdTT.MJDint != oldmjd )
99  {
100  oldmjd = mjdTT.MJDint ;
101  l = oldmjd + 2400001 ;
102 
103  tdbtdt = ctatv (l, 0.0) ;
104  tdbtdtdot = ctatv (l, 0.5) - ctatv (l, -0.5) ;
105  }
106 
107  double TDB_minus_TT = ( tdbtdt + (mjdTT.MJDfr - 0.5) * tdbtdtdot );
108 
109  Epoch T = TT;
110  T += TDB_minus_TT;
111 
112  return T;
113  }
