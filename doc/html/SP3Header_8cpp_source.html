58  void SP3Header::reallyGetRecord(FFStream& ffs)
59  throw(exception, FFStreamError, StringException)
60  {
61  SP3Stream& strm = dynamic_cast<SP3Stream&>(ffs);
62 
63  string line;
64  strm.formattedGetLine(line);
65  if(debug) std::cout << "SP3 Header Line 1 " << line << std::endl;
66 
67  if (line[0]=='#' && line[1]!='#') // line 1
68  {
69  // version character
70  if(line[1] == 'a') version = SP3a;
71  else if(line[1] == 'b') version = SP3b;
72  else if(line[1] == 'c') version = SP3c;
73  else {
74  FFStreamError e("Unknown version of SP3: " + line.substr(0,3));
75  GPSTK_THROW(e);
76  }
77 
78  // are V records present?
79  if(line[2] == 'V') containsVelocity = true;
80  else containsVelocity = false;
81 
82  // parse the rest of the line
83  int year = asInt(line.substr(3,4));
84  int month = asInt(line.substr(8,2));
85  int dom = asInt(line.substr(11,2));
86  int hour = asInt(line.substr(14,2));
87  int minute = asInt(line.substr(17,2));
88  double second = asInt(line.substr(20,10));
89  try {
90  time = CivilTime(year, month, dom, hour, minute, second);
91  }
92  catch (gpstk::Exception& e) {
93  FFStreamError fe("Invalid time:" + string(1, line[0]));
94  GPSTK_THROW(fe);
95  }
96  numberOfEpochs = asInt(line.substr(32,7));
97  dataUsed = line.substr(40,5);
98  coordSystem = line.substr(46,5);
99  orbitType = line.substr(52,3);
100  agency = line.substr(56,4);
101  }
102  else
103  {
104  FFStreamError e("Unknown label in line 1: " + line.substr(0,2));
105  GPSTK_THROW(e);
106  }
107 
108  strm.formattedGetLine(line);
109  if(debug) std::cout << "SP3 Header Line 2 " << line << std::endl;
110  if (line[0]=='#' && line[1]=='#') // line 2
111  {
112  epochInterval = asDouble(line.substr(24,14));
113  }
114  else
115  {
116  FFStreamError e("Unknown label in line 2: " + line.substr(0,2));
117  GPSTK_THROW(e);
118  }
119 
120  int i, index;
121  int numSVs(0), readSVs(0);
122 
123  // the map stores them sorted, so use svsAsWritten to determine
124  // which SV each accuracy corresponds to.
125  vector<SP3SatID> svsAsWritten;
126  SP3SatID sat;
127 
128  // read in the SV list
129  for(i = 3; i <= 7; i++) // lines 3-7
130  {
131  strm.formattedGetLine(line);
132  if(debug) std::cout << "SP3 Header Line " << i << " " << line << std::endl;
133  if (line[0]=='+')
134  {
135  // get the total number of svs on line 3
136  if (i == 3)
137  {
138  numSVs = asInt(line.substr(4,2));
139  svsAsWritten.resize(numSVs);
140  }
141  for(index = 9; index < 60; index += 3)
142  {
143  if (readSVs < numSVs)
144  {
145  try {
146  sat = SP3SatID(line.substr(index,3));
147  }
148  catch (Exception& e) {
149  FFStreamError ffse(e);
150  GPSTK_THROW(ffse);
151  }
152  svsAsWritten[readSVs] = sat;
153  satList[sat] = 0;
154  readSVs++;
155  }
156  }
157  }
158  else
159  {
160  FFStreamError e("Unknown 1st char in line " + asString(i) + ": "
161  + string(1, line[0]));
162  GPSTK_THROW(e);
163  }
164  }
165 
166  readSVs = 0;
167 
168  // read in the accuracy.
169  for(i = 8; i <= 12; i++) // lines 8-12
170  {
171  strm.formattedGetLine(line);
172  if(debug) std::cout << "SP3 Header Line " << i << " " << line << std::endl;
173  if ((line[0]=='+') && (line[1]=='+'))
174  {
175  for(index = 9; index < 60; index += 3)
176  {
177  if (readSVs < numSVs)
178  {
179  satList[svsAsWritten[readSVs]] = asInt(line.substr(index,3));
180  readSVs++;
181  }
182  }
183  }
184  else
185  {
186  FFStreamError e("Unknown label in line " + asString(i) + ": "
187  + line.substr(0,2));
188  GPSTK_THROW(e);
189  }
190  }
191 
192  strm.formattedGetLine(line);
193  if(debug) std::cout << "SP3 Header Line 13 " << line << std::endl;
194  if (version == SP3c) {
195  if(line[0]=='%' && line[1]=='c') // line 13
196  {
197  // file system
198  system.fromString(line.substr(3,2));
199 
200  // time system
201  string ts = upperCase(line.substr(9,3));
202  timeSystem.fromString(ts);
203  }
204  else
205  {
206  FFStreamError e("Unknown label in line 13: " + line.substr(0,2));
207  GPSTK_THROW(e);
208  }
209  }
210 
211  strm.formattedGetLine(line); // line 14
212  if(debug) std::cout << "SP3 Header Line 14 " << line << std::endl;
213 
214  strm.formattedGetLine(line);
215  if(debug) std::cout << "SP3 Header Line 15 " << line << std::endl;
216  if (version == SP3c) {
217  if (line[0]=='%' && line[1]=='f') // line 15
218  {
219  basePV = asDouble(line.substr(3,10));
220  baseClk = asDouble(line.substr(14,12));
221  }
222  else
223  {
224  FFStreamError e("Unknown label in line 15: " + line.substr(0,2));
225  GPSTK_THROW(e);
226  }
227  }
228 
229  strm.formattedGetLine(line); // line 16
230  if(debug) std::cout << "SP3 Header Line 16 " << line << std::endl;
231 
232  // read in 2 unused %i lines // lines 17,18
233  for(i = 17; i <= 18; i++) {
234  strm.formattedGetLine(line);
235  if(debug) std::cout << "SP3 Header Line " << i << " " << line << std::endl;
236  }
237 
238  // read in 4 comment lines
239  comments.clear();
240  for(i = 19; i <= 22; i++) // lines 19-22
241  {
242  strm.formattedGetLine(line);
243  if(debug) std::cout << "SP3 Header Line " << i << " " << line << std::endl;
244  // strip the first 3 characters
245  line.erase(0, 3);
246  // and add to the comment vector
247  comments.push_back(line);
248  }
249 
250  // save the header, for use later when reading SP3Data records
251  strm.header = *this;
252 
253  } // end SP3Header::reallyGetRecord()

256  void SP3Header::reallyPutRecord(FFStream& ffs) const
257  throw(exception, FFStreamError, StringException)
258  {
259  try {
260  SP3Stream& strm = dynamic_cast<SP3Stream&>(ffs);
261  int i,k;
262  size_t j;
263  string line;
264  SP3SatID SVid;
265  bool isVerA = (version == SP3a);
266  bool isVerC = (version == SP3c);
267 
268  // line 1
269  CivilTime civTime(time);
270  line = "#";
271  line += versionChar();
272  line += (containsVelocity ? "V" : "P");
273  line += civTime.printf("%4Y %2m %2d %2H %2M");
274  line += " " + rightJustify(civTime.printf("%.8f"),11);
275  line += " " + rightJustify(asString(numberOfEpochs),7);
276  line += " " + rightJustify(dataUsed,5);
277  line += " " + rightJustify(coordSystem,5);
278  line += " " + rightJustify(orbitType,3);
279  line += " " + rightJustify(agency,4);
280  strm << line << endl;
281  strm.lineNumber++;
282 
283  // line 2
284  GPSWeekSecond gpsWS(time);
285  line = "##";
286  line += rightJustify(gpsWS.printf("%F"),5);
287  line += rightJustify(gpsWS.printf("%15.8g"),16);
288  line += " " + rightJustify(asString(epochInterval,8),14);
289  line += " " + (static_cast<MJD>(time)).printf("%5.0Q");
290  line += " " + rightJustify(asString(time.getSecondOfDay()/86400.,13),15);
291  strm << line << endl;
292  strm.lineNumber++;
293 
294  // lines 3-7 and 8-12
295  //Map<SV,accuracy flag> (all SVs in data)
296  std::map<SP3SatID, short>::const_iterator it;
297  for(i=3; i<=12; i++) { // loop over the lines
298  if(i==3) line = "+ " + rightJustify(asString(satList.size()),2) + " ";
299  else if(i < 8) line = "+ ";
300  else line = "++ ";
301  k = 0;
302  if(i == 3 || i == 8) // start the iteration
303  it = satList.begin();
304  while(k < 17) { // there are 17 per line
305  if(it != satList.end()) {
306  if(i < 8) { // lines 3-7 - sat id
307  if(!isVerA) {
308  // a satellite in version c -> let j be -1 to mark it
309  SVid = it->first;
310  j = -1;
311  }
312  else j = it->first.id;
313  }
314  else // lines 8-12 - accuracy
315  j = it->second;
316  it++;
317  }
318  else j=0; // no more
319 
320  if(j == -1) // sat version b or c
321  try {
322  line += rightJustify(SP3SatID(SVid).toString(),3);
323  }
324  catch (Exception& e)
325  {
326  FFStreamError ffse(e);
327  GPSTK_THROW(ffse);
328  }
329  else // sat version a, accuracy, or 0
330  line += rightJustify(asString(j),3);
331  k++;
332  }
333  strm << line << endl;
334  strm.lineNumber++;
335  }
336 
337  // line 13
338  string ft("cc");
339  if(!isVerA) {
340  ft[0] = system.systemChar();
341  ft[1] = ' ';
342  }
343  strm << "%c " << ft << " cc"
344  << " " << (isVerA ? "ccc" : timeSystemString())
345  << " ccc cccc cccc cccc cccc ccccc ccccc ccccc ccccc" << endl;
346  strm.lineNumber++;
347 
348  // line 14
349  strm << "%c cc cc ccc ccc cccc cccc cccc cccc ccccc ccccc ccccc ccccc" << endl;
350 
351  // line 15
352  strm << "%f "
353  << (isVerC ? rightJustify(asString(basePV,7),10) : " 0.0000000")
354  << " "
355  << (isVerC ? rightJustify(asString(baseClk,9),12) : " 0.000000000")
356  << " 0.00000000000 0.000000000000000" << endl;
357  strm.lineNumber++;
358 
359  // lines 16-18
360  strm << "%f 0.0000000 0.000000000 0.00000000000 0.000000000000000" << endl;
361  strm.lineNumber++;
362  strm << "%i 0 0 0 0 0 0 0 0 0" << endl;
363  strm.lineNumber++;
364  strm << "%i 0 0 0 0 0 0 0 0 0" << endl;
365  strm.lineNumber++;
366 
367  // lines 19-22
368  //std::vector<std::string> comments; ///< vector of 4 comment lines
369  for(j=0,i=19; i<=22; i++) {
370  line = "/* ";
371  if(j < comments.size()) line += leftJustify(comments[j++],57);
372  else line += string(57,'C');
373  strm << line << endl;
374  strm.lineNumber++;
375  }
376 
377  // save header for use with SP3Data::reallyPut
378  strm.header = *this;
379  }
380  catch(Exception& e) { GPSTK_RETHROW(e); }
381  catch(std::exception& e) { Exception g(e.what()); GPSTK_THROW(g); }
382  }
