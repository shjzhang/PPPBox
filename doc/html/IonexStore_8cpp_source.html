266  Triple IonexStore::getIonexValue( const CommonTime& t,
267  const Position& RX,
268  int strategy ) const
269  throw(InvalidRequest)
270  {
271 
272  // Here we store the necessary IONEX-extracted values
273  // (i.e, TEC, RMS, ionosphere height)
274  Triple tecval(0.0,0.0,0.0);
275 
276  // current time check
277  if (t < getInitialTime())
278  {
279  InvalidRequest e("Inadequate data before requested time");
280  GPSTK_THROW(e);
281  }
282 
283  if (t > getFinalTime() )
284  {
285  InvalidRequest e("Inadequate data after requested time");
286  GPSTK_THROW(e);
287  }
288 
289  // this never should happen but just in case
290  Position pos(RX);
291  if ( pos.getSystemName() != "Geocentric" )
292  {
293 
294  InvalidRequest e("Position object is not in GEOCENTRIC coordinates");
295 
296  GPSTK_THROW(e);
297 
298  }
299 
300  //let's define the number of maps to be considered
301  int nmap;
302  if (strategy == 1) nmap = 1;
303  else if (strategy == 2) nmap = 2;
304  else if (strategy == 3) nmap = 2;
305  else if (strategy == 4) nmap = 1;
306  else
307  {
308  InvalidRequest e("Invalid interpolation stategy");
309  GPSTK_THROW(e);
310  }
311 
312  // let's look for valid Ionex maps
313  CommonTime T[2];
314  // iterator
315  IonexMap::const_iterator itm = inxMaps.find(t);
316  try
317  {
318 
319  if( itm != inxMaps.end() ) // exact match of t
320  {
321 
322  // get the current map
323  itm = inxMaps.lower_bound(t);
324  // store current and next epoch
325  T[0] = itm->first;
326  T[1] = (++itm)->first;
327 
328  }
329  else // t is between two maps
330  {
331 
332  // get the next valid map
333  itm = inxMaps.lower_bound(t);
334  // store the next and previous epoch
335  T[1] = itm->first;
336  T[0] = (--itm)->first;
337 
338  } // end of 'if( itm != inxMaps.end() ) ... else ... ''
339 
340  }
341  catch (...)
342  {
343  InvalidRequest e("IonexStore::getIonexValue() ... Invalid time!");
344  GPSTK_THROW(e);
345  }
346 
347 
348  // factors (As in Eq.(3), pag.2 of the manual)
349  double f[2];
350  f[0] = (T[1]-t ) / (T[1]-T[0]);
351  f[1] = (t -T[0]) / (T[1]-T[0]);
352 
353  // if only one map, then we have to use the neareast
354  if( nmap == 1 )
355  {
356 
357  // closer to the next map
358  if( f[1] > f[0] )
359  {
360  T[0] = T[1];
361  }
362 
363  // than the factor is unit
364  f[0] = 1.0;
365 
366  } // if( nmap == 1 )
367 
368  // loop over the number of maps considered
369  for(int imap = 0; imap < nmap; imap++)
370  {
371 
372  // now let's determine if we keep fixed position or
373  // take into account the rotation around the Sun
374  Position pos;
375  if (strategy == 1 || strategy == 2) // fixed position
376  {
377 
378  pos = RX;
379 
380  }
381  else // rotate the position
382  {
383 
384  // seconds of time to degree (360.0 / 86400.0)
385  double sec2deg( 4.16666666666667e-3 );
386 
387  // count the rotation
388  pos = RX;
389  pos.theArray[1] = pos.theArray[1] + ( t - T[imap] ) * sec2deg;
390 
391  } // End of 'if (strategy == 1 || strategy == 2)...'
392 
393 
394  // iterator to the current map
395  itm = inxMaps.find(T[imap]);
396 
397  // map to hold the IONEX types for the current map
398  IonexValTypeMap ivtm = (*itm).second;
399 
400  // object to hold the corresponding data to the current map
401  IonexData iod;
402 
403  // Compute TEC value
404  if ( ivtm.find(IonexData::TEC) != ivtm.end() )
405  {
406 
407  iod = ivtm[IonexData::TEC];
408  tecval[0] = tecval[0] + f[imap]*iod.getValue(pos);
409 
410  }
411 
412  // Compute RMS value
413  if ( ivtm.find(IonexData::RMS) != ivtm.end() )
414  {
415 
416  iod = ivtm[IonexData::RMS];
417  tecval[1] = tecval[1] + f[imap]*iod.getValue(pos);
418 
419  }
420 
421  } // End of 'for(int imap = 0; imap < nmap; imap++)...'
422 
423 
424  // ionosphere height in meters
425  tecval[2] = RX.theArray[2];
426 
427  return tecval;
428 
429  } // End of method 'IonexStore::getIonexValue()'

543  double IonexStore::iono_mapping_function( const double& elevation,
544  const std::string& ionoMapType ) const
545  {
546 
547  // map
548  double imap(1.0);
549  // Earth's radius in KM
550  double Re = 6371.0;
551  // zenith angle
552  double z0 = 90.0 - elevation;
553 
554  if( ionoMapType == "SLM" )
555  {
556 
557  // As explained in: Hofmann-Wellenhof et al. (2004) - GPS Theory and
558  // practice, 5th edition, SpringerWienNewYork, Chapter 6.3, pg. 102
559 
560  // ionosphere height in KM
561  double ionoHeight = 450.0;
562  // zenith angle of the ionospheric pierce point (IPP)
563  double sinzipp = Re / (Re + ionoHeight) * std::sin(z0*DEG_TO_RAD);
564  double zipprad = std::asin(sinzipp);
565 
566  imap = 1.0/std::cos(zipprad);
567 
568  }
569  else if( ionoMapType == "MSLM" )
570  {
571  // maximum zenith distance is 80 degrees
572  if( z0 <= 80.0 )
573  {
574  // ionosphere height in KM
575  double ionoHeight = 506.7;
576  double alfa = 0.9782;
577  // zenith angle of the ionospheric pierce point (IPP)
578  double sinzipp = Re / (Re + ionoHeight)
579  * std::sin(alfa * z0 * DEG_TO_RAD);
580  double zipprad = std::asin(sinzipp);
581 
582  imap = 1.0/std::cos(zipprad);
583  }
584 
585  }
586  else if( ionoMapType == "ESM" )
587  {
588  //TODO
589  }
590  else // that means ionoMapType == "NONE"
591  {
592  // TODO
593  }
594 
595 
596  return imap;
597 
598  } // End of method 'IonexStore::iono_mapping_function()'

611  double IonexStore::findDCB( const SatID sat,
612  const CommonTime& time ) const
613  throw(InvalidRequest)
614  {
615  // current time check. This is passed even if there are gaps
616  if ( time < getInitialTime() )
617  {
618  InvalidRequest e("Inadequate data before requested time");
619  GPSTK_THROW(e);
620  }
621 
622  if ( time > getFinalTime() )
623  {
624  InvalidRequest e("Inadequate data after requested time");
625  GPSTK_THROW(e);
626  }
627 
628  double dt(0.0);
629  IonexDCBMap::const_iterator itm = inxDCBMap.begin();
630 
631 
632  // looping through the map
633  while ( itm != inxDCBMap.end() )
634  {
635 
636  // let's get the relative reference
637  dt = time - itm->first;
638 
639  // this means we don't have maps for this day and there is a gap
640  if( dt < 0.0 )
641  {
642 
643  InvalidRequest e( "Inadequate data after requested time: " +
644  time.asString() );
645  GPSTK_THROW(e);
646 
647  } // works fine for consecutive files, otherwise last epoch is thrown
648  else
649  {
650 
651  if( dt < 86400.0 )
652  {
653 
654  IonexHeader::SatDCBMap satdcb( itm->second );
655  IonexHeader::SatDCBMap::const_iterator iprn( satdcb.find(sat) );
656 
657  // if satellite is not found, throw
658  if ( iprn == satdcb.end() )
659  {
660 
661  InvalidRequest e( "There is no DCB value for satellite " +
662  asString(sat) );
663  GPSTK_THROW(e);
664 
665  }
666  else // ... otherwise, fetch the value
667  {
668 
669  return iprn->second.bias;
670 
671  } // End of 'if ( iprn == satdcb.end() ) ... else ...'
672 
673  }
674  else // If everything is fine, then move forward in the map
675  {
676 
677  itm++;
678 
679  } // End of 'if( dt < 86400.0 ) ... else ...'
680 
681  } // End of 'if( dt < 0.0 ) ... else ...'
682 
683  } // End of 'while ( itm != inxDCBMap.end() )'
684 
685 
686  // You should never get here, but just in case
687  return 0.0;
688 
689  } // End of method 'IonexStore::findDCB()'
