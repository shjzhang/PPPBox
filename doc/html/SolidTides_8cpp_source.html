46  /* Returns the effect of solid Earth tides (meters) at the given
47  * position and epoch, in the Up-East-Down (UEN) reference frame.
48  *
49  * @param[in] t Epoch to look up
50  * @param[in] p Position of interest
51  *
52  * @return a Triple with the solid tidal effect, in meters and in
53  * the UED reference frame.
54  *
55  * @throw InvalidRequest If the request can not be completed for any
56  * reason, this is thrown. The text may have additional information
57  * as to why the request failed.
58  */
59  Triple SolidTides::getSolidTide(const CommonTime& t,
60  const Position& p) const
61  throw(InvalidRequest)
62  {
63 
64  // We will store here the results
65  Triple res;
66 
67  // Objects to compute Sun and Moon positions
68  SunPosition sunPosition;
69  MoonPosition moonPosition;
70 
71 
72  try
73  {
74 
75  // Variables to hold Sun and Moon positions
76  Triple sunPos(sunPosition.getPosition(t));
77  Triple moonPos(moonPosition.getPosition(t));
78 
79 
80  // Compute the factors for the Sun
81  double rpRs( p.X()*sunPos.theArray[0] +
82  p.Y()*sunPos.theArray[1] +
83  p.Z()*sunPos.theArray[2]);
84 
85  double Rs2(sunPos.theArray[0]*sunPos.theArray[0] +
86  sunPos.theArray[1]*sunPos.theArray[1] +
87  sunPos.theArray[2]*sunPos.theArray[2]);
88 
89  double rp2( p.X()*p.X() + p.Y()*p.Y() + p.Z()*p.Z() );
90 
91  double xy2p( p.X()*p.X() + p.Y()*p.Y() );
92  double sqxy2p( std::sqrt(xy2p) );
93 
94  double sqRs2(std::sqrt(Rs2));
95 
96  double fac_s( 3.0*MU_SUN*rp2/(sqRs2*sqRs2*sqRs2*sqRs2*sqRs2) );
97 
98  double g1sun( fac_s*(rpRs*rpRs/2.0 - rp2*Rs2/6.0) );
99 
100  double g2sun( fac_s * rpRs * (sunPos.theArray[1]*p.X() -
101  sunPos.theArray[0]*p.Y()) * std::sqrt(rp2)/sqxy2p );
102 
103  double g3sun( fac_s * rpRs * ( sqxy2p* sunPos.theArray[2] -
104  p.Z()/sqxy2p * (p.X()*sunPos.theArray[0] +
105  p.Y()*sunPos.theArray[1]) ) );
106 
107 
108  // Compute the factors for the Moon
109  double rpRm( p.X()*moonPos.theArray[0] +
110  p.Y()*moonPos.theArray[1] +
111  p.Z()*moonPos.theArray[2]);
112 
113  double Rm2(moonPos.theArray[0]*moonPos.theArray[0] +
114  moonPos.theArray[1]*moonPos.theArray[1] +
115  moonPos.theArray[2]*moonPos.theArray[2]);
116 
117  double sqRm2(std::sqrt(Rm2));
118 
119  double fac_m( 3.0*MU_MOON*rp2/(sqRm2*sqRm2*sqRm2*sqRm2*sqRm2) );
120 
121  double g1moon( fac_m*(rpRm*rpRm/2.0 - rp2*Rm2/6.0) );
122 
123  double g2moon( fac_m * rpRm * (moonPos.theArray[1]*p.X() -
124  moonPos.theArray[0]*p.Y()) * std::sqrt(rp2)/sqxy2p );
125 
126  double g3moon( fac_m * rpRm * ( sqxy2p* moonPos.theArray[2] -
127  p.Z()/sqxy2p * (p.X()*moonPos.theArray[0] +
128  p.Y()*moonPos.theArray[1]) ) );
129 
130  // Effects due to the Sun
131  double delta_sun1(H_LOVE*g1sun);
132  double delta_sun2(L_LOVE*g2sun);
133  double delta_sun3(L_LOVE*g3sun);
134 
135  // Effects due to the Moon
136  double delta_moon1(H_LOVE*g1moon);
137  double delta_moon2(L_LOVE*g2moon);
138  double delta_moon3(L_LOVE*g3moon);
139 
140  // Combined effect
141  res.theArray[0] = delta_sun1 + delta_moon1;
142  res.theArray[1] = delta_sun2 + delta_moon2;
143  res.theArray[2] = delta_sun3 + delta_moon3;
144 
145  } // End of try block
146  catch(InvalidRequest& ir)
147  {
148  GPSTK_RETHROW(ir);
149  }
150 
151  return res;
152 
153  } // End SolidTides::getSolidTide
