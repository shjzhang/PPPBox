310  int expandTimeCorrMap(void)
311  {
312  int n(0);
313  std::map<std::string, TimeSystemCorrection>::iterator it,jt;
314 
315  // GLGP : GLO to GPS
316  if(mapTimeCorr.find(std::string("GLGP")) == mapTimeCorr.end()) {
317  it = mapTimeCorr.find(std::string("GPUT"));
318  jt = mapTimeCorr.find(std::string("GLUT"));
319  if(it != mapTimeCorr.end() && jt != mapTimeCorr.end()
320  && mapTimeCorr.find(std::string("GLGP")) == mapTimeCorr.end())
321  {
322  TimeSystemCorrection tc("GLGP");
323  tc.A0 = jt->second.A0 - it->second.A0;
324  tc.A1 = jt->second.A1 - it->second.A1; // probably zeros
325  tc.refYr = it->second.refYr;
326  tc.refMon = it->second.refMon;
327  tc.refDay = it->second.refDay;
328  tc.refWeek = jt->second.refWeek;
329  tc.refSOW = jt->second.refSOW;
330  tc.geoProvider = jt->second.geoProvider; // blank
331  tc.geoUTCid = 0; // NA
332  mapTimeCorr[tc.asString4()] = tc;
333  n++;
334  }
335  }
336 
337  // GPGA : GPS to GAL
338  if(mapTimeCorr.find(std::string("GPGA")) == mapTimeCorr.end()) {
339  it = mapTimeCorr.find(std::string("GAUT"));
340  jt = mapTimeCorr.find(std::string("GPUT"));
341  if(it != mapTimeCorr.end() && jt != mapTimeCorr.end()
342  && mapTimeCorr.find(std::string("GPGA")) == mapTimeCorr.end())
343  {
344  TimeSystemCorrection tc("GPGA");
345  tc.A0 = jt->second.A0 - it->second.A0;
346  tc.A1 = jt->second.A1 - it->second.A1;
347  tc.refYr = it->second.refYr; // take ref time from GAL
348  tc.refMon = it->second.refMon;
349  tc.refDay = it->second.refDay;
350  tc.refWeek = it->second.refWeek;
351  tc.refSOW = it->second.refSOW;
352  tc.geoProvider = it->second.geoProvider; // blank
353  tc.geoUTCid = 0; // NA
354  mapTimeCorr[tc.asString4()] = tc;
355  n++;
356  }
357  }
358 
359  // BDGP : BDS to GPS
360  if(mapTimeCorr.find(std::string("BDGP")) == mapTimeCorr.end()) {
361  it = mapTimeCorr.find(std::string("GPUT"));
362  jt = mapTimeCorr.find(std::string("BDUT"));
363  if(it != mapTimeCorr.end() && jt != mapTimeCorr.end()
364  && mapTimeCorr.find(std::string("BDGP")) == mapTimeCorr.end())
365  {
366  TimeSystemCorrection tc("BDGP");
367  tc.A0 = jt->second.A0 - it->second.A0;
368  tc.A1 = jt->second.A1 - it->second.A1;
369  tc.refYr = it->second.refYr;
370  tc.refMon = it->second.refMon;
371  tc.refDay = it->second.refDay;
372  BDSWeekSecond bws(jt->second.refWeek,jt->second.refSOW);
373  GPSWeekSecond gws(bws);
374  tc.refWeek = gws.week;
375  tc.refSOW = gws.sow;
376  tc.geoProvider = jt->second.geoProvider; // blank
377  tc.geoUTCid = 7;
378  mapTimeCorr[tc.asString4()] = tc;
379  n++;
380  }
381  }
382 
383  return n;
384  }
