104  void Add(const T& x, const T& wt_in=T())
105  {
106  T wt(ABS(wt_in));
107  if(wt != T()) { weighted=true; }
108 
109  if(n == 0) {
110  min = max = ave = x;
111  var = T();
112  W = T();
113  }
114  else {
115  if(x < min) min=x;
116  if(x > max) max=x;
117  }
118 
119  if(weighted) {
120  if(W+wt > T(1.e-10)) // if W+wt=0, nothing yet has had non-zero weight
121  ave += (x-ave)*(wt/(W+wt));
122  if(n > 0 && W > 1.e-10)
123  var = (W/(W+wt))*var + (x-ave)*(x-ave)*(wt/W);
124  W += wt;
125  }
126  else {
127  ave += (x-ave)/T(n+1);
128  if(n > 0)
129  var = T(n)*var/T(n+1) + (x-ave)*(x-ave)/T(n);
130  }
131 
132  n++;
133  }

153  void Subtract(const T x, const T wt_in=T())
154  {
155  if(n == 0) return;
156  if(weighted) {
157  if(W > T(1.e-10)) {
158  T wt(ABS(wt_in));
159  if(W-wt > T(1.e-10))
160  var = (var - (wt/(W-wt))*(x-ave)*(x-ave)) * (W/(W-wt));
161  else
162  var = T();
163  ave = (ave - wt*x/W)*W/(W-wt);
164  W -= wt;
165  }
166  else { ave = var = W = T(); }
167  }
168  else {
169  if(n > 1)

171  else
172  var = T();
173  ave = T(n)*(ave - x/T(n))/(T(n)-T(1));
174  }
175  n--;

300  void Add(const T& X, const T& Y)
301  {
302  if(n == 0) {
303  sumx = sumy = sumx2 = sumy2 = sumxy = T();
304  xmin = xmax = X;
305  ymin = ymax = Y;
306  scalex = scaley = T(1);
307  }
308  if(scalex==T(1) && X!=T()) scalex=ABS(X);
309  if(scaley==T(1) && Y!=T()) scaley=ABS(Y);
310  sumx += X/scalex;
311  sumy += Y/scaley;
312  sumx2 += (X/scalex)*(X/scalex);
313  sumy2 += (Y/scaley)*(Y/scaley);
314  sumxy += (X/scalex)*(Y/scaley);
315  if(X < xmin) xmin=X;
316  if(X > xmax) xmax=X;
317  if(Y < ymin) ymin=Y;
318  if(Y > ymax) ymax=Y;
319  n++;
320  }

502  inline T median(const Vector<T>& v)
503  {
504  if(v.size()==0) return T();
505  if(v.size()==1) return v(0);
506  if(v.size()==2) return (v(0)+v(1))/T(2);
507  // insert sort
508  int i,j;
509  T x;
510  Vector<T> w(v);
511  for(i=0; i<v.size(); i++) {
512  x = w[i] = v(i);
513  j = i-1;
514  while(j>=0 && x<w[j]) {
515  w[j+1] = w[j];
516  j--;
517  }
518  w[j+1] = x;
519  }
520  if(v.size() % 2)
521  x=w[(v.size()+1)/2-1];
522  else
523  x=(w[v.size()/2-1]+w[v.size()/2])/T(2);
524 
525  return x;
526  } // end median(Vector)
