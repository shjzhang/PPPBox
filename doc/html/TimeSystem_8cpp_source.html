91  // NB. The table 'leaps' must be modified when a new leap second is announced.
92  // Return the number of leap seconds between UTC and TAI, that is the
93  // difference in time scales UTC-TAI at an epoch defined by (year, month, day).
94  // NB. Input day in a floating quantity and thus any epoch may be represented;
95  // this is relevant the period 1960 to 1972, when UTC-TAI was not integral.
96  // NB. GPS = TAI - 19sec and so GPS-UTC = getLeapSeconds()-19.
97  double TimeSystem::getLeapSeconds(const int& year,
98  const int& month,
99  const double& day)
100  {
101  // Leap second data --------------------------------------------------------
102  // number of changes before leap seconds (1960-1971) - this should never change.
103  static const int NPRE=14;
104 
105  // epoch year, epoch month(1-12), delta t(sec), rate (sec/day) for [1960,1972).
106  static const struct {
107  int year, month;
108  double delt, rate;
109  } preleap[NPRE] = {
110  { 1960, 1, 1.4178180, 0.0012960 },
111  { 1961, 1, 1.4228180, 0.0012960 },
112  { 1961, 8, 1.3728180, 0.0012960 },
113  { 1962, 1, 1.8458580, 0.0011232 },
114  { 1963, 11, 1.9458580, 0.0011232 },
115  { 1964, 1, 3.2401300, 0.0012960 },
116  { 1964, 4, 3.3401300, 0.0012960 },
117  { 1964, 9, 3.4401300, 0.0012960 },
118  { 1965, 1, 3.5401300, 0.0012960 },
119  { 1965, 3, 3.6401300, 0.0012960 },
120  { 1965, 7, 3.7401300, 0.0012960 },
121  { 1965, 9, 3.8401300, 0.0012960 },
122  { 1966, 1, 4.3131700, 0.0025920 },
123  { 1968, 2, 4.2131700, 0.0025920 }
124  };
125 
126  // Leap seconds history
127  // ***** This table must be updated for new leap seconds **************
128  static const struct {
129  int year, month, nleap;
130  } leaps[] = {
131  { 1972, 1, 10 },
132  { 1972, 7, 11 },
133  { 1973, 1, 12 },
134  { 1974, 1, 13 },
135  { 1975, 1, 14 },
136  { 1976, 1, 15 },
137  { 1977, 1, 16 },
138  { 1978, 1, 17 },
139  { 1979, 1, 18 },
140  { 1980, 1, 19 },
141  { 1981, 7, 20 },
142  { 1982, 7, 21 },
143  { 1983, 7, 22 },
144  { 1985, 7, 23 },
145  { 1988, 1, 24 },
146  { 1990, 1, 25 },
147  { 1991, 1, 26 },
148  { 1992, 7, 27 },
149  { 1993, 7, 28 },
150  { 1994, 7, 29 },
151  { 1996, 1, 30 },
152  { 1997, 7, 31 },
153  { 1999, 1, 32 },
154  { 2006, 1, 33 },
155  { 2009, 1, 34 },
156  { 2012, 7, 35 }, // leave the last comma!
157  // add new entry here, of the form:
158  // { year, month(1-12), leap_sec }, // leave the last comma!
159  };
160 
161  // the number of leaps (do not change this)
162  static const int NLEAPS = sizeof(leaps)/sizeof(leaps[0]);
163 
164  // last year in leaps
165  //static const int MAXYEAR = leaps[NLEAPS-1].year;
166 
167  // END static data -----------------------------------------------------
168 
169  // search for the input year, month
170  if(year < 1960) // pre-1960 no deltas
171  ;
172  else if(month < 1 || month > 12) // blunder, should never happen - throw?
173  ;
174  else if(year < 1972) { // [1960-1972) pre-leap
175  for(int i=NPRE-1; i>=0; i--) {
176  if(preleap[i].year > year ||
177  (preleap[i].year == year && preleap[i].month > month)) continue;
178 
179  // found last record with < rec.year >= year and rec.month >= month
180  // watch out - cannot use CommonTime here
181  int iday(day);
182  double dday(iday-int(day));
183  if(iday == 0) { iday = 1; dday = 1.0-dday; }
184  long JD0 = convertCalendarToJD(year,month,iday);
185  long JD = convertCalendarToJD(preleap[i].year,preleap[i].month,1);
186  return (preleap[i].delt + (double(JD0-JD)+dday)*preleap[i].rate);
187  }
188  }
189  else { // [1972- leap seconds
190  for(int i=NLEAPS-1; i>=0; i--) {
191  if(leaps[i].year > year ||
192  (leaps[i].year == year && leaps[i].month > month)) continue;
193  return double(leaps[i].nleap);
194  }
195  }
196 
197  return 0.0;
198  }

200  // Compute the conversion (in seconds) from one time system (inTS) to another
201  // (outTS), given the year and month of the time to be converted.
202  // Result is to be added to the first time (inTS) to yield the converted (outTS),
203  // that is t(outTS) = t(inTS) + correction(inTS,outTS).
204  // NB. the caller must not forget to change to outTS after adding this correction.
205  // @param TimeSystem inTS, input system
206  // @param TimeSystem outTS, output system
207  // @param int year, year of the time to be converted.
208  // @param int month, month (1-12) of the time to be converted.
209  // @return double dt, correction (sec) to be added to t(in) to yield t(out).
210  // @throw if input system(s) are invalid or Unknown.
211  double TimeSystem::Correction(const TimeSystem& inTS, const TimeSystem& outTS,
212  const int& year, const int& month, const double& day)
213  {
214  double dt(0.0);
215 
216  // identity
217  if(inTS == outTS)
218  return dt;
219 
220  // cannot convert unknowns
221  if(inTS == Unknown || outTS == Unknown) {
222  Exception e("Cannot compute correction for TimeSystem::Unknown");
223  GPSTK_THROW(e);
224  }
225 
226  // compute TT-TDB here; ref Astronomical Almanac B7
227  double TDBmTT(0.0);
228  if(inTS == TDB || outTS == TDB) {
229  int iday = int(day);
230  long jday = convertCalendarToJD(year, month, iday) ;
231  double frac(day-iday);
232  double TJ2000(jday-2451545.5+frac); // t-J2000
233  // 0.0001657 sec * sin(357.53 + 0.98560028 * TJ2000 deg)
234  frac = ::fmod(0.017201969994578 * TJ2000, 6.2831853071796);
235  TDBmTT = 0.0001657 * ::sin(6.240075674 + frac);
236  // 0.000022 sec * sin(246.11 + 0.90251792 * TJ2000 deg)
237  frac = ::fmod(0.015751909262251 * TJ2000, 6.2831853071796);
238  TDBmTT += 0.000022 * ::sin(4.295429822 + frac);
239  }
240 
241  // -----------------------------------------------------------
242  // conversions: first convert inTS->TAI ...
243  // TAI = GPS + 19s
244  // TAI = UTC + getLeapSeconds()
245  // TAI = TT - 32.184s
246  if(inTS == GPS || // GPS -> TAI
247  inTS == GAL) // GAL -> TAI
248  dt = 19.;
249  else if(inTS == UTC | // UTC -> TAI
250  inTS == BDT | // BDT -> TAI // TD is this right?
251  inTS == GLO) // GLO -> TAI
252  dt = getLeapSeconds(year, month, day);
253  //else if(inTS == BDT) // BDT -> TAI // RINEX 3.02 seems to say this
254  // dt = 34.;
255  else if(inTS == TAI) // TAI
256  ;
257  else if(inTS == TT) // TT -> TAI
258  dt = -32.184;
259  else if(inTS == TDB) // TDB -> TAI
260  dt = -32.184 + TDBmTT;
261  else { // other
262  Exception e("Invalid input TimeSystem " + inTS.asString());
263  GPSTK_THROW(e);
264  }
265 
266  // -----------------------------------------------------------
267  // ... then convert TAI->outTS
268  // GPS = TAI - 19s
269  // UTC = TAI - getLeapSeconds()
270  // TT = TAI + 32.184s
271  if(outTS == GPS || // TAI -> GPS
272  outTS == GAL) // TAI -> GAL
273  dt -= 19.;
274  else if(outTS == UTC | // TAI -> UTC
275  outTS == BDT | // TAI -> BDT
276  outTS == GLO) // TAI -> GLO
277  dt -= getLeapSeconds(year, month, day);
278  //else if(outTS == BDT) // TAI -> BDT
279  // dt -= 34.;
280  else if(outTS == TAI) // TAI
281  ;
282  else if(outTS == TT) // TAI -> TT
283  dt += 32.184;
284  else if(outTS == TDB) // TAI -> TDB
285  dt += 32.184 - TDBmTT;
286  else if(outTS == GAL) // TD
287  dt = 0.0;
288  else { // other
289  Exception e("Invalid output TimeSystem " + outTS.asString());
290  GPSTK_THROW(e);
291  }
292 
293  return dt;
294  }
