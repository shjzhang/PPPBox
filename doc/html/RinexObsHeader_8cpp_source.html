153  void RinexObsHeader::reallyPutRecord(FFStream& ffs) const
154  throw(std::exception, FFStreamError, StringException)
155  {
156  RinexObsStream& strm = dynamic_cast<RinexObsStream&>(ffs);
157 
158  strm.header = *this;
159 
160  unsigned long allValid;
161  if (version == 2.0) allValid = allValid20;
162  else if (version == 2.1) allValid = allValid21;
163  else if (version == 2.11) allValid = allValid211;
164  else
165  {
166  FFStreamError err("Unknown RINEX version: " + asString(version,2));
167  err.addText("Make sure to set the version correctly.");
168  GPSTK_THROW(err);
169  }
170 
171  if ((valid & allValid) != allValid)
172  {
173  FFStreamError err("Incomplete or invalid header.");
174  err.addText("Make sure you set all header valid bits for all of the available data.");
175  GPSTK_THROW(err);
176  }
177 
178  try
179  {
180  WriteHeaderRecords(strm);
181  }
182  catch(FFStreamError& e)
183  {
184  GPSTK_RETHROW(e);
185  }
186  catch(StringException& e)
187  {
188  GPSTK_RETHROW(e);
189  }
190 
191  } // end RinexObsHeader::reallyPutRecord

194  // this function computes the number of valid header records which WriteHeaderRecords will write
195  int RinexObsHeader::NumberHeaderRecordsToBeWritten(void) const throw()
196  {
197  int n=0;
198  if(valid & RinexObsHeader::versionValid) n++;
199  if(valid & RinexObsHeader::runByValid) n++;
200  if(valid & RinexObsHeader::markerNameValid) n++;
201  if(valid & RinexObsHeader::observerValid) n++;
202  if(valid & RinexObsHeader::receiverValid) n++;
203  if(valid & RinexObsHeader::antennaTypeValid) n++;
204  if(valid & RinexObsHeader::antennaPositionValid) n++;
205  if(valid & RinexObsHeader::antennaOffsetValid) n++;
206  if(valid & RinexObsHeader::waveFactValid) {
207  n++;
208  if(extraWaveFactList.size()) n += 1 + (extraWaveFactList.size()-1)/7;
209  }
210  if(valid & RinexObsHeader::obsTypeValid) n += 1 + (obsTypeList.size()-1)/9;
211  if(valid & RinexObsHeader::intervalValid) n++;
212  if(valid & RinexObsHeader::firstTimeValid) n++;
213  if(valid & RinexObsHeader::lastTimeValid) n++;
214  if(valid & RinexObsHeader::markerNumberValid) n++;
215  if(valid & RinexObsHeader::receiverOffsetValid) n++;
216  if(valid & RinexObsHeader::leapSecondsValid) n++;
217  if(valid & RinexObsHeader::commentValid) n += commentList.size();
218  if(valid & RinexObsHeader::numSatsValid) n++;
219  if(valid & RinexObsHeader::prnObsValid)
220  n += numObsForSat.size() * (1+numObsForSat.begin()->second.size()/9);
221  if(valid & RinexObsHeader::endValid) n++;
222  return n;

225  // this function writes all valid header records
226  void RinexObsHeader::WriteHeaderRecords(FFStream& ffs) const
227  throw(FFStreamError, StringException)
228  {
229  RinexObsStream& strm = dynamic_cast<RinexObsStream&>(ffs);
230  string line;
231  if (valid & versionValid)
232  {
233  line = rightJustify(asString(version,2), 9);
234  line += string(11, ' ');
235  if ((fileType[0] != 'O') && (fileType[0] != 'o'))
236  {
237  FFStreamError err("This isn't a Rinex Observation file: " +
238  fileType.substr(0,1));
239  GPSTK_THROW(err);
240  }
241 
242  if (system.system == RinexSatID::systemUnknown)
243  {
244  FFStreamError err("Invalid satellite system");
245  GPSTK_THROW(err);
246  }
247 
248  line += leftJustify(string("Observation"), 20);
249  std::string str;
250  str = system.systemChar();
251  str = str + " (" + system.systemString() + ")";
252  line += leftJustify(str, 20);
253  line += versionString;
254  strm << line << endl;
255  strm.lineNumber++;
256  }
257  if (valid & runByValid)
258  {
259  line = leftJustify(fileProgram,20);
260  line += leftJustify(fileAgency,20);
261  CommonTime dt;
262  SystemTime sysTime;
263  string dat = (static_cast<CivilTime>(sysTime)).printf("%04Y%02m%02d %02H%02M%02S %P");
264  line += leftJustify(dat, 20);
265  line += runByString;
266  strm << line << endl;
267  strm.lineNumber++;
268  }
269  if (valid & markerNameValid)
270  {
271  line = leftJustify(markerName, 60);
272  line += markerNameString;
273  strm << line << endl;
274  strm.lineNumber++;
275  }
276  if (valid & observerValid)
277  {
278  line = leftJustify(observer, 20);
279  line += leftJustify(agency, 40);
280  line += observerString;
281  strm << line << endl;
282  strm.lineNumber++;
283  }
284  if (valid & receiverValid)
285  {
286  line = leftJustify(recNo, 20);
287  line += leftJustify(recType, 20);
288  line += leftJustify(recVers, 20);
289  line += receiverString;
290  strm << line << endl;
291  strm.lineNumber++;
292  }
293  if (valid & antennaTypeValid)
294  {
295  line = leftJustify(antNo, 20);
296  line += leftJustify(antType, 20);
297  line += string(20, ' ');
298  line += antennaTypeString;
299  strm << line << endl;
300  strm.lineNumber++;
301  }
302  if (valid & antennaPositionValid)
303  {
304  line = rightJustify(asString(antennaPosition[0], 4), 14);
305  line += rightJustify(asString(antennaPosition[1], 4), 14);
306  line += rightJustify(asString(antennaPosition[2], 4), 14);
307  line += string(18, ' ');
308  line += antennaPositionString;
309  strm << line << endl;
310  strm.lineNumber++;
311  }
312  if (valid & antennaOffsetValid)
313  {
314  line = rightJustify(asString(antennaOffset[0], 4), 14);
315  line += rightJustify(asString(antennaOffset[1], 4), 14);
316  line += rightJustify(asString(antennaOffset[2], 4), 14);
317  line += string(18, ' ');
318  line += antennaOffsetString;
319  strm << line << endl;
320  strm.lineNumber++;
321  }
322  if (valid & waveFactValid)
323  {
324  line = rightJustify(asString<short>(wavelengthFactor[0]),6);
325  line += rightJustify(asString<short>(wavelengthFactor[1]),6);
326  line += string(48, ' ');
327  line += waveFactString;
328  strm << line << endl;
329  strm.lineNumber++;
330 
331  // handle continuation lines
332  if (!extraWaveFactList.empty())
333  {
334  vector<ExtraWaveFact>::const_iterator itr = extraWaveFactList.begin();
335 
336  while (itr != extraWaveFactList.end())
337  {
338  const int maxSatsPerLine = 7;
339  short satsWritten = 0, satsLeft = (*itr).satList.size(), satsThisLine;
340  vector<SatID>::const_iterator vecItr = (*itr).satList.begin();
341 
342  while ((vecItr != (*itr).satList.end())) {
343  if(satsWritten == 0) {
344  line = rightJustify(asString<short>((*itr).wavelengthFactor[0]),6);
345  line += rightJustify(asString<short>((*itr).wavelengthFactor[1]),6);
346  satsThisLine = (satsLeft > maxSatsPerLine ? maxSatsPerLine : satsLeft);
347  line += rightJustify(asString<short>(satsThisLine),6);
348  }
349  try {
350  line += string(3, ' ') + RinexSatID(*vecItr).toString();
351  }
352  catch (Exception& e) {
353  FFStreamError ffse(e);
354  GPSTK_THROW(ffse);
355  }
356  satsWritten++;
357  satsLeft--;
358  if(satsWritten==maxSatsPerLine || satsLeft==0) { // output a complete line
359  line += string(60 - line.size(), ' ');
360  line += waveFactString;
361  strm << line << endl;
362  strm.lineNumber++;
363  satsWritten = 0;
364  }
365  vecItr++;
366  }
367  itr++;
368  }
369  }
370  }
371  if (valid & obsTypeValid)
372  {
373  const int maxObsPerLine = 9;
374  int obsWritten = 0;
375  line = ""; // make sure the line contents are reset.
376 
377  vector<RinexObsType>::const_iterator itr = obsTypeList.begin();
378 
379  while (itr != obsTypeList.end())
380  {
381  // the first line needs to have the # of obs
382  if (obsWritten == 0)
383  line = rightJustify(asString(obsTypeList.size()), 6);
384  // if you hit 9, write out the line and start a new one
385  else if ((obsWritten % maxObsPerLine) == 0)
386  {
387  line += numObsString;
388  strm << line << endl;
389  strm.lineNumber++;
390  line = string(6, ' ');
391  }
392  line += rightJustify(convertObsType(*itr), 6);
393  obsWritten++;
394  itr++;
395  }
396  line += string(60 - line.size(), ' ');
397  line += numObsString;
398  strm << line << endl;
399  strm.lineNumber++;
400  }
401  if (valid & intervalValid)
402  {
403  line = rightJustify(asString(interval, 3), 10);
404  line += string(50, ' ');
405  line += intervalString;
406  strm << line << endl;
407  strm.lineNumber++;
408  }
409  if (valid & firstTimeValid)
410  {
411  line = writeTime(firstObs);
412  line += string(48-line.size(),' ');
413  if(firstSystem.system == RinexSatID::systemGPS) line += "GPS";
414  if(firstSystem.system == RinexSatID::systemGlonass) line += "GLO";
415  if(firstSystem.system == RinexSatID::systemGalileo) line += "GAL";
416  line += string(60 - line.size(), ' ');
417  line += firstTimeString;
418  strm << line << endl;
419  strm.lineNumber++;
420  }
421  if (valid & lastTimeValid)
422  {
423  line = writeTime(lastObs);
424  line += string(48-line.size(),' ');
425  if(lastSystem.system == RinexSatID::systemGPS) line += "GPS";
426  if(lastSystem.system == RinexSatID::systemGlonass) line += "GLO";
427  if(lastSystem.system == RinexSatID::systemGalileo) line += "GAL";
428  line += string(60 - line.size(), ' ');
429  line += lastTimeString;
430  strm << line << endl;
431  strm.lineNumber++;
432  }
433  if (valid & markerNumberValid)
434  {
435  line = leftJustify(markerNumber, 20);
436  line += string(40, ' ');
437  line += markerNumberString;
438  strm << line << endl;
439  strm.lineNumber++;
440  }
441  if (valid & receiverOffsetValid)
442  {
443  line = rightJustify(asString(receiverOffset),6);
444  line += string(54, ' ');
445  line += receiverOffsetString;
446  strm << line << endl;
447  strm.lineNumber++;
448  }
449  if (valid & leapSecondsValid)
450  {
451  line = rightJustify(asString(leapSeconds),6);
452  line += string(54, ' ');
453  line += leapSecondsString;
454  strm << line << endl;
455  strm.lineNumber++;
456  }
457  if (valid & commentValid)
458  {
459  vector<string>::const_iterator itr = commentList.begin();
460  while (itr != commentList.end())
461  {
462  line = leftJustify((*itr), 60);
463  line += commentString;
464  strm << line << endl;
465  strm.lineNumber++;
466  itr++;
467  }
468  }
469  if (valid & numSatsValid)
470  {
471  line = rightJustify(asString(numSVs), 6);
472  line += string(54, ' ');
473  line += numSatsString;
474  strm << line << endl;
475  strm.lineNumber++;
476  }
477  if (valid & prnObsValid)
478  {
479  const int maxObsPerLine = 9;
480  map<SatID, vector<int> >::const_iterator itr = numObsForSat.begin();
481  while (itr != numObsForSat.end())
482  {
483  int numObsWritten = 0;
484 
485  vector<int>::const_iterator vecItr = (*itr).second.begin();
486  while (vecItr != (*itr).second.end())
487  {
488  if (numObsWritten == 0)
489  {
490  try {
491  RinexSatID prn((*itr).first);
492  line = string(3, ' ') + prn.toString();
493  }
494  catch (Exception& e) {
495  FFStreamError ffse(e);
496  GPSTK_RETHROW(ffse);
497  }
498  }
499  else if ((numObsWritten % maxObsPerLine) == 0)
500  {
501  line += prnObsString;
502  strm << line << endl;
503  strm.lineNumber++;
504  line = string(6, ' ');
505  }
506  line += rightJustify(asString(*vecItr), 6);
507  ++vecItr;
508  ++numObsWritten;
509  }
510  line += string(60 - line.size(), ' ');
511  line += prnObsString;
512  strm << line << endl;
513  strm.lineNumber++;
514  itr++;
515  }
516  }
517  if (valid & endValid)
518  {
519  line = string(60, ' ');
520  line += endOfHeader;
521  strm << line << endl;
522  strm.lineNumber++;
523  }
524  } // end RinexObsHeader::WriteHeaderRecords()

527  // this function parses a single header record
528  void RinexObsHeader::ParseHeaderRecord(string& line)
529  throw(FFStreamError)
530  {
531  string label(line, 60, 20);
532 
533  if (label == versionString)
534  {
535  version = asDouble(line.substr(0,20));
536 // cout << "R2ObsHeader:ParseHeaderRecord:version = " << version << endl;
537  fileType = strip(line.substr(20, 20));
538  if ( (fileType[0] != 'O') &&
539  (fileType[0] != 'o'))
540  {
541  FFStreamError e("This isn't a Rinex Obs file");
542  GPSTK_THROW(e);
543  }
544  string system_str = strip(line.substr(40, 20));
545  try {
546  system.fromString(system_str);
547  }
548  catch (Exception& e)
549  {
550  FFStreamError ffse("Input satellite system is unsupported: " + system_str);
551  GPSTK_THROW(ffse);
552  }
553  valid |= versionValid;
554  }
555  else if (label == runByString )
556  {
557  fileProgram = strip(line.substr(0, 20));
558  fileAgency = strip(line.substr(20, 20));
559  date = strip(line.substr(40, 20));
560  valid |= runByValid;
561  }
562  else if (label == commentString)
563  {
564  string s = strip(line.substr(0, 60));
565  commentList.push_back(s);
566  valid |= commentValid;
567  }
568  else if (label == markerNameString)
569  {
570  // shjzhang
571  markerName = strip(line.substr(0,4));
572  valid |= markerNameValid;
573  }
574  else if (label == markerNumberString)
575  {
576  markerNumber = strip(line.substr(0,20));
577  valid |= markerNumberValid;
578  }
579  else if (label == observerString)
580  {
581  observer = strip(line.substr(0,20));
582  agency = strip(line.substr(20,40));
583  valid |= observerValid;
584  }
585  else if (label == receiverString)
586  {
587  recNo = strip(line.substr(0, 20));
588  recType = strip(line.substr(20,20));
589  recVers = strip(line.substr(40,20));
590  valid |= receiverValid;
591  }

773  // This function parses the entire header from the given stream
774  void RinexObsHeader::reallyGetRecord(FFStream& ffs)
775  throw(std::exception, FFStreamError,
776  gpstk::StringUtils::StringException)
777  {
778  RinexObsStream& strm = dynamic_cast<RinexObsStream&>(ffs);
779 
780  // if already read, just return
781  if (strm.headerRead == true)
782  return;
783 
784  // since we're reading a new header, we need to reinitialize
785  // all our list structures. all the other objects should be ok.
786  // this also applies if we threw an exception the first time we read
787  // the header and are now re-reading it. some of these could be full
788  // and we need to empty them.
789  commentList.clear();
790  wavelengthFactor[0] = wavelengthFactor[1] = 1;
791  extraWaveFactList.clear();
792  obsTypeList.clear();
793  numObsForSat.clear();
794  valid = 0;
795  numObs = 0;
796  lastPRN.id = -1;
797 
798  string line;
799 
800  while (!(valid & endValid))
801  {
802  strm.formattedGetLine(line);
803  StringUtils::stripTrailing(line);
804 
805  if (line.length()==0)
806  {
807  FFStreamError e("No data read");
808  GPSTK_THROW(e);
809  }
810  else if (line.length()<60 || line.length()>80)
811  {
812  FFStreamError e("Invalid line length");
813  GPSTK_THROW(e);
814  }
815 
816  try
817  {
818  ParseHeaderRecord(line);
819  }
820  catch(FFStreamError& e)
821  {
822  GPSTK_RETHROW(e);
823  }
824 
825  } // end while(not end of header)
826 
827  unsigned long allValid;
828  if (version == 2.0) allValid = allValid20;
829  else if (version == 2.1) allValid = allValid21;
830  else if (version == 2.11) allValid = allValid211;
831  else
832  {
833  FFStreamError e("Unknown or unsupported RINEX version " +
834  asString(version));
835  GPSTK_THROW(e);
836  }
837 
838  if ( (allValid & valid) != allValid)
839  {
840  FFStreamError e("Incomplete or invalid header");
841  GPSTK_THROW(e);
842  }
843 
844  // If we get here, we should have reached the end of header line
845  strm.header = *this;
846  strm.headerRead = true;
847 
848  } // end of reallyGetRecord()
