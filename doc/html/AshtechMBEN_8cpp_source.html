52  void AshtechMBEN::reallyGetRecord(FFStream& ffs)
53  throw(std::exception, FFStreamError, EndOfFile)
54  {
55  AshtechStream& stream=dynamic_cast<AshtechStream&>(ffs);
56 
57  // make sure the object is reset before starting the search
58  clear(fmtbit | lenbit | crcbit);
59  string& rawData = stream.rawData;
60 
61  // If this object doesn't have an id set yet, assume that the streams
62  // most recent read id is what we need to be
63  if (id == "" && rawData.size()>=11 &&
64  rawData.substr(0,7) == preamble &&
65  rawData[10]==',')
66  id = rawData.substr(7,3);
67 
68  // If that didn't work, or this is object is not of the right type,
69  // then give up.
70  if (id == "" || !checkId(id))
71  return;
72 
73  readBody(stream);
74  }

78  void AshtechMBEN::decode(const std::string& data)
79  throw(std::exception, FFStreamError)
80  {
81  using gpstk::BinUtils::decodeVar;
82 
83  string str(data);
84 
85  uint8_t csum=0;
86  if (str.length() == 108 || str.length()==52)
87  {
88  ascii=false;
89  header = str.substr(0,11); str.erase(0,11);
90 
91  seq = decodeVar<uint16_t>(str);
92  left = decodeVar<uint8_t>(str);
93  svprn = decodeVar<uint8_t>(str);
94  el = decodeVar<uint8_t>(str);
95  az = decodeVar<uint8_t>(str);
96  chid = decodeVar<uint8_t>(str);
97 
98  ca.decodeBIN(str);
99 
100  if (id == mpcId)
101  {
102  p1.decodeBIN(str);
103  p2.decodeBIN(str);
104  }
105 
106  checksum = decodeVar<uint8_t>(str);
107 
108  clear();
109 
110  int end = data.size() - 3;
111  for (int i=11; i<end; i++)
112  csum ^= data[i];
113  }
114  else
115  {
116  ascii=true;
117  header = str.substr(0,11); str.erase(0,11);
118  stringstream iss(str);
119  char c;
120  iss >> seq >> c
121  >> left >> c
122  >> svprn >> c
123  >> el >> c
124  >> az >> c
125  >> chid >> c;
126 
127  ca.decodeASCII(iss);
128 
129  if (id == mpcId)
130  {
131  p1.decodeASCII(iss);
132  p2.decodeASCII(iss);
133  }
134 
135  iss >> checksum;
136 
137  if (iss)
138  clear();
139 
140  int end=data.rfind(',');
141  for (int i=11; i<=end; i++)
142  csum ^= data[i];
143  }
144 
145 
146  if (csum != checksum)
147  {
148  setstate(crcbit);
149  if (debugLevel)
150  cout << "checksum error, computed:" << hex << (uint16_t) csum
151  << " received:" << checksum << dec << endl;
152  }
153 
154  if (seq>36000)
155  setstate(fmtbit);
156  }

183  void AshtechMBEN::code_block::decodeBIN(string& str)
184  throw(std::exception, FFStreamError)
185  {
186  using gpstk::BinUtils::decodeVar;
187  uint32_t smo;
188  warning = decodeVar<uint8_t>(str);
189  goodbad = decodeVar<uint8_t>(str);
190  polarity_known = decodeVar<uint8_t>(str);
191  ireg = decodeVar<uint8_t>(str);
192  qa_phase = decodeVar<uint8_t>(str);
193  full_phase = decodeVar<double>(str);
194  raw_range = decodeVar<double>(str);
195  doppler = decodeVar<int32_t>(str);
196  smo = decodeVar<uint32_t>(str);
197 
198  doppler *= 1e-4;
199  smoothing = (smo & 0x800000 ? -1e-3 : 1e-3) * (smo & 0x7fffff);
200  smooth_cnt = (smo >> 24) & 0xff;
201  }

224  float AshtechMBEN::code_block::snr(float chipRate) const throw()
225  {
226  const float n = 20000; // number of samples in 1 ms
227  // magnitude of the carrier estimate; should be 4.14 for 2 bit
228  // quantization, 2.18 for 1 bit.
229  const float m = 2.18f;
230  const float bw = 0.9 * chipRate; // equivalent noise bandwidth (Hz)
231 
232  const float d = PI/(n*n*m*m*4.0);
233  float snr=0;
234  // Note that ireg is as output from the MBEN which is 0-255, not the 0-99
235  // as displayed on the front panel.
236  if (ireg)
237  {
238  snr = exp(((float)ireg)/25.0);
239  snr = snr*snr*bw*d;
240  snr = 10 * log10(snr);
241  }
242 
243  return snr;
244  }
