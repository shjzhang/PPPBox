48 // compute the relativity correction
49 double gpstk::Xvt::computeRelativityCorrection(void)
50 {
51  relcorr = -2.0*( (x[0]/C_MPS)*(v[0]/C_MPS)
52  +(x[1]/C_MPS)*(v[1]/C_MPS)
53  +(x[2]/C_MPS)*(v[2]/C_MPS) );
54  return relcorr;
55 }

57 // Function to find the range and position from a ground
58 // location, rxPos, to the spacecraft position (*this).x
59 // Use the pseudorange corrected for SV clock effects to get a
60 // rough time of flight (dt). Account directly for Earth
61 // rotation, then compute a rough receiver bias by differencing
62 // the initial time of flight with the new estimate. Then
63 // correct the rotation by a small amount.
64 double gpstk::Xvt::preciseRho(const Triple& rxPos,
65  const EllipsoidModel& ellips,
66  double correction) const
67  throw()
68 {
69  // Compute initial time of flight estimate using the
70  // geometric range at transmit time. This fails to account
71  // for the rotation of the earth, but should be good to
72  // within about 40 m
73  double sr1 = rxPos.slantRange(x);
74  double dt = sr1 / ellips.c();
75 
76  // compute rotation angle in the time of signal transit
77  double rotation_angle = -ellips.angVelocity() * dt;
78 
79  // rotate original GS coordinates to new values to correct for
80  // rotation of ECEF frame
81  // Ref: Satellite Geodesy, Gunter Seeber, 1993, pg 291 and the
82  // ICD-GPS-200 sheet 102 May 1993 version
83  // xnew[0]=xg[0]*cos(rotation_angle)-xg[1]*sin(rotation_angle);
84  // xnew[1]=xg[1]*cos(rotation_angle)+xg[0]*sin(rotation_angle);
85  // xnew[2]=xg[2];
86  // since cosine and sine are small, approximate by the first
87  // order terms in an expansion.
88  gpstk::Triple xnew;
89  for (int i = 0; i < 2; i++)
90  {
91  xnew[0] = x[0] - x[1] * rotation_angle;
92  xnew[1] = x[1] + x[0] * rotation_angle;
93  xnew[2] = x[2];
94 
95  // Compute geometric slant range from ground station to
96  // the rotated new coord's
97  sr1 = rxPos.slantRange(xnew);
98 
99  // Recompute the time of flight (dt) based on PR, with the
100  // time of flight based on geometric range. Note that
101  // this is a really unneeded, in that the change in PR is
102  // < 40 m, hence the change in tof is < 20 ns
103  dt = sr1 / ellips.c();
104 
105  // Compute new rotation in this time
106  rotation_angle = -ellips.angVelocity() * dt;
107  }
108 
109  // Account for SV clock drift, relativity and user input correction
110  double rho = sr1 - (clkbias + relcorr) * ellips.c() - correction;
111 
112  return rho;
113 
114 } // end of preciseRho()
