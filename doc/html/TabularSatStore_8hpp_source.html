366  virtual bool getNonCenteredTableInterval(const SatID& sat,
367  const CommonTime& ttag,
368  const int& nhalf,
369  typename DataTable::const_iterator& it1,
370  typename DataTable::const_iterator& it2,
371  bool exactReturn=true)
372  const throw(InvalidRequest)
373  {
374  try {
375  // find the DataTable for this sat
376  typename std::map<SatID, DataTable>::const_iterator satit;
377  satit = tables.find(sat);
378  if(satit == tables.end()) {
379  InvalidRequest ir("Satellite " + gpstk::StringUtils::asString(sat)
380  + " not found.");
381  GPSTK_THROW(ir);
382  }
383 
384  // this is the data table for the sat
385  const DataTable& dtable(satit->second);
386  static const char *fmt=" at time %4Y/%02m/%02d %2H:%02M:%02S";
387 
388  // find the timetag in this table
389  // NB. throw here if time systems do not match and are not "Any"
390  it1 = dtable.find(ttag);
391  // is it an exact match?
392  bool exactMatch(it1 != dtable.end());
393 
394  // user must decide whether to return with exact value; e.g. without
395  // velocity data, user needs the interval to compute v from x data
396  if(exactMatch && exactReturn) return true;
397 
398  // lower_bound points to the first element with key >= ttag
399  it1 = it2 = dtable.lower_bound(ttag);
400 
401  // Should we allow to predict data?
402  if(it1 == dtable.end()) {
403  InvalidRequest ir("No data for the requested time of satellite "
404  + gpstk::StringUtils::asString(sat) + printTime(ttag,fmt));
405  GPSTK_THROW(ir);
406  }
407 
408  if(int(dtable.size()) < 2*nhalf) {
409  InvalidRequest ir("Inadequate data for satellite "
410  + gpstk::StringUtils::asString(sat) + printTime(ttag,fmt));
411  GPSTK_THROW(ir);
412  }
413 
414  // check for gap only when exactMatch==false
415  if(!exactMatch && checkDataGap) {
416  typename DataTable::const_iterator lit,rit;
417  lit = rit = it2;
418 
419  if(rit!=dtable.begin()) {
420  lit--;
421  if((rit->first-lit->first) > gapInterval) {
422  InvalidRequest ir(
423  "Gap at interpolation time for satellite "
424  + gpstk::StringUtils::asString(sat) + printTime(ttag,fmt));
425  GPSTK_THROW(ir);
426  }
427  }
428  else{
429  if((rit->first - ttag) > gapInterval) {
430  InvalidRequest ir(
431  "Gap may produce bad interpolation precision for satellite "
432  + gpstk::StringUtils::asString(sat) + printTime(ttag,fmt));
433  GPSTK_THROW(ir);
434  }
435  }
436  }
437 
438  // try to get the beginning of the window
439  int i, ileft(nhalf);
440  for(i = 0; i < nhalf; i++) {
441  if(it1 == dtable.begin()) break;
442  it1--;
443  ileft--;
444  }
445 
446  // try to get the ending of the window
447  int iright = nhalf - 1 + ileft;
448  for(i = 0; i < (nhalf-1+ileft); i++) {
449  it2++;
450  if(it2 == dtable.end()) {
451  it2--;
452  break;
453  }
454  iright--;
455  }
456 
457  // adjust the beginning of the window again
458  int ileft2 = iright;
459  for(i = 0; i < iright; i++) {
460  if(it1 == dtable.begin()) break;
461  it1--;
462  ileft2--;
463  }
464 
465  // We have checked that there are enough data, so ileft2 should
466  // always be zero.
467  if(ileft2 > 0) {
468  InvalidRequest ir("This should never happen");
469  GPSTK_THROW(ir);
470  }
471 
472  // check that the interval is not too large
473  if(checkInterval && (it2->first - it1->first) > maxInterval) {
474  InvalidRequest ir("Interpolation interval too large for satellite "
475  + gpstk::StringUtils::asString(sat) + printTime(ttag,fmt));
476  GPSTK_THROW(ir);
477  }
478 
479  return exactMatch;
480  }
481  catch(InvalidRequest& ir) { GPSTK_RETHROW(ir); }
482  }

749  double nomTimeStep(const SatID& sat) const throw()
750  {
751  // get the table for this sat
752  typename SatTable::const_iterator it(tables.find(sat));
753 
754  // not found or empty
755  if(it == tables.end() || it->second.size() == 0) return 0.0;
756 
757  // save the most frequent N step sizes
758  static const int N=3;
759  int i,ndt[N]={0,0,0};
760  double dt[N],del;
761 
762  // loop over the table
763  typename DataTable::const_iterator jt(it->second.begin());
764  CommonTime prevT(jt->first);
765  ++jt;
766  while(jt != it->second.end()) {
767  del = jt->first - prevT;
768  if(del > 1.0e-8) for(i=0; i<N; i++) {
769  if(ndt[i] == 0) { dt[i] = del; ndt[i] = 1; break; }
770  if(fabs(del-dt[i]) < 1.0e-8) { ndt[i]++; break; }
771  }
772  prevT = jt->first;
773  ++jt;
774  }
775 
776  // find the most frequent interval
777  del = dt[0];
778  for(i=1; i<N; i++) if(ndt[i] > ndt[0]) {
779  del = dt[i];
780  ndt[0] = ndt[i];
781  }
782 
783  return del;
784  }
