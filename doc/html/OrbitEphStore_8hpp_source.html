152  if(satTables.find(sat) == satTables.end())
153  return CommonTime::END_OF_TIME;
154 
155  CommonTime limit(CommonTime::END_OF_TIME);
156  const TimeOrbitEphTable& table = getTimeOrbitEphMap(sat);
157  TimeOrbitEphTable::const_iterator it;
158  for (it = table.begin(); it != table.end(); ++it) {
159  CommonTime ct(it->first);
160  ct.setTimeSystem(timeSystem);
161  if(ct < limit) limit = ct;
162  }
163  return limit;
164  }
