79  int EOPPrediction::loadFile(string filename)
80  throw(FileMissingException)
81  {
82  bool ok;
83  int n;
84  string line,word;
85  ifstream inpf(filename.c_str());
86  if(!inpf) {
87  FileMissingException fme("Could not open EOPP file " + filename);
88  GPSTK_THROW(fme);

91  ok = true;
92  n = 0; // n is line number
93  while(!inpf.eof() && inpf.good()) {
94  getline(inpf,line);
95  StringUtils::stripTrailing(line,'\r');
96  if(inpf.bad()) break;
97  if(line.size() > 80) { ok=false; break; }
98  switch(n) {
99  case 0:
100  if(line.size() < 76) { ok=false; break; }
101  word = line.substr( 0,10); ta = StringUtils::asDouble(word);
102  word = line.substr(10,10); A = StringUtils::asDouble(word);
103  word = line.substr(20,10); B = StringUtils::asDouble(word);
104  word = line.substr(30,10); C1 = StringUtils::asDouble(word);
105  word = line.substr(40,10); C2 = StringUtils::asDouble(word);
106  word = line.substr(50,10); D1 = StringUtils::asDouble(word);
107  word = line.substr(60,10); D2 = StringUtils::asDouble(word);
108  word = line.substr(70, 6); P1 = StringUtils::asDouble(word);
109  break;
110  case 1:
111  if(line.size() < 78) { ok=false; break; }
112  word = line.substr( 0, 6); P2 = StringUtils::asDouble(word);
113  word = line.substr( 6,10); E = StringUtils::asDouble(word);
114  word = line.substr(16,10); F = StringUtils::asDouble(word);
115  word = line.substr(26,10); G1 = StringUtils::asDouble(word);
116  word = line.substr(36,10); G2 = StringUtils::asDouble(word);
117  word = line.substr(46,10); H1 = StringUtils::asDouble(word);
118  word = line.substr(56,10); H2 = StringUtils::asDouble(word);
119  word = line.substr(66, 6); Q1 = StringUtils::asDouble(word);
120  word = line.substr(72, 6); Q2 = StringUtils::asDouble(word);
121  break;
122  case 2:
123  if(line.size() < 70) { ok=false; break; }
124  word = line.substr( 0,10); tb = StringUtils::asDouble(word);
125  word = line.substr(10,10); I = StringUtils::asDouble(word);
126  word = line.substr(20,10); J = StringUtils::asDouble(word);
127  word = line.substr(30,10); K1 = StringUtils::asDouble(word);
128  word = line.substr(40,10); K2 = StringUtils::asDouble(word);
129  word = line.substr(50,10); K3 = StringUtils::asDouble(word);
130  word = line.substr(60,10); K4 = StringUtils::asDouble(word);
131  break;
132  case 3:
133  if(line.size() < 76) { ok=false; break; }
134  word = line.substr( 0,10); L1 = StringUtils::asDouble(word);
135  word = line.substr(10,10); L2 = StringUtils::asDouble(word);
136  word = line.substr(20,10); L3 = StringUtils::asDouble(word);
137  word = line.substr(30,10); L4 = StringUtils::asDouble(word);
138  word = line.substr(40, 9); R1 = StringUtils::asDouble(word);
139  word = line.substr(49, 9); R2 = StringUtils::asDouble(word);
140  word = line.substr(58, 9); R3 = StringUtils::asDouble(word);
141  word = line.substr(67, 9); R4 = StringUtils::asDouble(word);
142  break;
143  case 4:
144  if(line.size() < 16) { ok=false; break; }
145  word = line.substr( 0, 4); TAIUTC = StringUtils::asInt(word);
146  word = line.substr( 4, 5); SerialNo = StringUtils::asInt(word);
147  // this actually integer : mjd of begin valid period
148  word = line.substr( 9, 7); tv = StringUtils::asDouble(word);
149  Info = line.substr(16,19);
150  break;
151  } // end switch on n=line number
152  if(!ok) break;
153  n++;
154  };
155  inpf.close();
156  if(!ok) {
157  FileMissingException fme("EOPP File " + filename
158  + " is corrupted or wrong format");
159  GPSTK_THROW(fme);
160  }
161  if(inpf.bad()) return -1;
162  return 0;
163  }

166  // generate serial number (NGA files are named EOPP<sn>.txt) from epoch
167  // SN = Year (1 digit) + week of year
168  int EOPPrediction::getSerialNumber(CommonTime& t)
169  throw(Exception)
170  {
171  int w2 = static_cast<GPSWeekSecond>(t).week-1; // the previous week
172  if(w2 < 0) {
173  using namespace StringUtils;
174  Exception dte("Invalid week in EOPP file: "
175  + asString<short>(w2));
176  GPSTK_THROW(dte);
177  }
178 
179  int yr,w1;
180  try {
181  CommonTime ht;
182  ht=GPSWeekSecond(w2,475200.0); // Friday (noon) of previous week
183  yr = static_cast<YDSTime>(ht).year; // save the year for later
184  ht=CivilTime(yr,1,1,0,0,0.0); // first day of that year
185  w1 = static_cast<GPSWeekSecond>(ht).week;
186  if(static_cast<GPSWeekSecond>(ht).getDayOfWeek() == 6) w1++; // GPS week of first Friday in the year
187  yr = yr % 10; // last digit of the year
188  }
189  catch(Exception& dte) {
190  GPSTK_RETHROW(dte);
191  }
192  return (100*yr + w2-w1+1); // SN = Year (1 digit) + week of year
193  }

207  //---------------------------------------------------------------------------------
208  // 2 2
209  // xp(t)= A + B(t-ta) + SUM(Cj sin[2pi(t-ta)/Pj]) + SUM(Dj cos[2pi(t-ta)/Pj])
210  // j=1 j=1
211  //
212  // 2 2
213  // yp(t)= E + F(t-ta) + SUM(Gk sin[2pi(t-ta)/Qk]) + SUM(Hk cos[2pi(t-ta)/Qk])
214  // k=1 k=1
215  //
216  // 4 4
217  // UT1-UTC(t)= I+J(t-tb) + SUM(Km sin[2pi(t-tb)/Rm]) + SUM(Lm cos[2pi(t-tb)/Rm])
218  // m=1 m=1
219  //---------------------------------------------------------------------------------
220  EarthOrientation EOPPrediction::computeEOP(CommonTime& ep) const
221  throw()
222  {
223  double t,dt,arg;
224  EarthOrientation eo;
225 
226  t = static_cast<MJD>(ep).mjd + static_cast<YDSTime>(ep).sod/86400.0;
227  //if(t < tv || t > tv+7) // TD warn - outside valid range
228  //
229  dt = t - ta;
230  arg = TWO_PI*dt;
231  eo.xp = A + B*dt + C1*sin(arg/P1) + D1*cos(arg/P1)
232  + C2*sin(arg/P2) + D2*cos(arg/P2);
233  eo.yp = E + F*dt + G1*sin(arg/Q1) + H1*cos(arg/Q1)
234  + G2*sin(arg/Q2) + H2*cos(arg/Q2);
235 
236  dt = t - tb;
237  arg = TWO_PI*dt;
238  eo.UT1mUTC = I + J*dt
239  + K1*sin(arg/R1) + L1*cos(arg/R1)
240  + K2*sin(arg/R2) + L2*cos(arg/R2)
241  + K3*sin(arg/R3) + L3*cos(arg/R3)
242  + K4*sin(arg/R4) + L4*cos(arg/R4);
243 
244  return eo;
245  }

438  //---------------------------------------------------------------------------------
439  // Edit the store by deleting all entries before(after)
440  // the given min(max) MJDs. If mjdmin is later than mjdmax,
441  // the two times are switched.
442  // @param mjdmin integer MJD desired earliest store time.
443  // @param mjdmax integer MJD desired latest store time.
444  void EOPStore::edit(int mjdmin, int mjdmax)
445  throw()
446  {
447  if(mjdmin > mjdmax) {
448  int m=mjdmin;
449  mjdmin = mjdmax;
450  mjdmax = m;
451  }
452 
453  if(mjdmin > endMJD) return;
454  if(mjdmax < begMJD) return;
455 
456  map<int,EarthOrientation>::iterator it;
457  it = mapMJD_EOP.lower_bound(mjdmin);
458  if(it != mapMJD_EOP.begin())
459  mapMJD_EOP.erase(mapMJD_EOP.begin(), it);
460 
461  it = mapMJD_EOP.upper_bound(mjdmax);
462  if(it != mapMJD_EOP.end())
463  mapMJD_EOP.erase(it, mapMJD_EOP.end());
464 
465  it = mapMJD_EOP.begin();
466  if(it == mapMJD_EOP.end())
467  begMJD = -1;
468  else
469  begMJD = it->first;
470 
471  it = mapMJD_EOP.end();
472  if(--it == mapMJD_EOP.end())
473  endMJD = -1;
474  else
475  endMJD = it->first;
476  }

508  //---------------------------------------------------------------------------------
509  // Get the EOP at the given epoch and return it.
510  // @param t CommonTime at which to compute the EOPs.
511  // @return EarthOrientation EOPs at time t.
512  // @throw InvalidRequest if the (int) MJDs on either side of t
513  // cannot be found in the map.
514  EarthOrientation EOPStore::getEOP(CommonTime& t) const
515  throw(InvalidRequest)
516  {
517  // find the EOs before and after epoch
518  int loMJD = int(static_cast<MJD>(t).mjd);
519  int hiMJD = loMJD + 1;
520  // find these EOPs
521  map<int,EarthOrientation>::const_iterator hit,lit;
522  lit = mapMJD_EOP.find(loMJD);
523  hit = mapMJD_EOP.find(hiMJD);
524  if(lit == mapMJD_EOP.end() || hit == mapMJD_EOP.end()) {
525  InvalidRequest ire(string("Time tag (MJD=")
526  + (lit == mapMJD_EOP.end() ?
527  StringUtils::asString(loMJD) : StringUtils::asString(hiMJD))
528  + string(") not found within the EOP store - EOPP files are out-of-date"));
529  GPSTK_THROW(ire);
530  }
531  // linearly interpolate to get EOP at the desired time
532  EarthOrientation eo;
533  double dt=static_cast<MJD>(t).mjd-double(loMJD);
534  eo.xp = (1.0-dt) * lit->second.xp + dt * hit->second.xp;
535  eo.yp = (1.0-dt) * lit->second.yp + dt * hit->second.yp;
536  eo.UT1mUTC = (1.0-dt) * lit->second.UT1mUTC + dt * hit->second.UT1mUTC;
537 
538  return eo;
539  }
