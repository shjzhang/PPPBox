99  template <class BaseClass>
100  Matrix(const ConstMatrixBase<T, BaseClass>& mat, size_t topRow,
101  size_t topCol, size_t numRows, size_t numCols)
102  : v((size_t)0), r(0), c(0), s(0)
103  {
104  // sanity checks...
105  if ( (topCol > mat.cols()) ||
106  (topRow > mat.rows()) ||
107  ((topRow + numRows) > mat.rows()) ||
108  ((topCol + numCols) > mat.cols()) )
109  {
110  MatrixException e("Invalid dimensions or size for Matrix(MatrixBase)");
111  GPSTK_THROW(e);
112  }
113 
114  // seems ok - make the valarray and copy column by column
115  r = numRows;
116  c = numCols;
117  s = r * c;
118  v.resize(r * c);
119  size_t i, j;
120  for(i = 0; i < r; i++)
121  for(j = 0; j < c; j++)
122  (*this)(i,j) = mat(topRow + i, topCol + j);
123  }
