79  // Singular Value Decomposition
80  template <class BaseClass>
81  bool operator() (const ConstMatrixBase<T, BaseClass>& mat)
82  throw (MatrixException)
83  {
84  const T eps=T(8)*std::numeric_limits<T>::epsilon();
85  bool flip=false;
86  U = mat;
87  if(mat.rows() > mat.cols()) {
88  flip = true;
89  U = transpose(mat);
90  }
91 
92  size_t n(U.cols()), m(U.rows());
93  size_t i, j, k, l, nm, jj;
94  T anorm(0), scale(0), g(0), s, f, h, c, x, y, z;
95 
96  V = Matrix<T>(n, n, T(0));
97  S = Vector<T>(n, T(1));
98  Vector<T> B(n, T(1));
99 
100  for (i = 0; i < n; i++) {
101  l = i + 1;
102  B[i] = scale * g;
103  g = s = scale = T(0);
104  if (i < m) {
105  for(k = i; k < m; k++) scale += ABS(U(k, i));
106  if (scale) {
107  for(k = i; k < m; k++) {
108  U(k, i) /= scale;
109  s += U(k, i) * U(k, i);
110  }
111  f = U(i, i);
112  g = -SIGN(SQRT(s),f);
113  h = f * g - s;
114  U(i,i) = f - g;
115  for(j = l; j < n; j++) {
116  for(s = T(0), k = i; k < m; k++) s += U(k, i) * U(k, j);
117  f = s / h;
118  for(k = i; k < m; k++) U(k, j) += f * U(k, i);
119  }
120  for(k = i; k < m; k++) U(k, i) *= scale;
121  } // if (scale)
122  } // if (i < m)
123  S[i] = scale * g;
124  g = s = scale = T(0);
125  if ( (i < m) && (i != n-1) ) {
126  for(k = l; k < n; k++) scale += ABS(U(i, k));
127  if (scale) {
128  for(k = l; k < n; k++) {
129  U(i, k) /= scale;
130  s += U(i, k) * U(i, k);
131  }
132  f = U(i, l);
133  g = -SIGN(SQRT(s),f);
134  h = f * g - s;
135  U(i, l) = f - g;
136  for(k = l; k < n; k++) B[k] = U(i, k) / h;
137  for(j = l; j < m; j++) {
138  for(s = T(0), k = l; k < n; k++) s += U(j, k) * U(i, k);
139  for(k = l; k < n; k++) U(j, k) += s * B[k];
140  }
141  for(k = l; k < n; k++) U(i, k) *= scale;
142  }
143  }
144  if(ABS(S[i])+ABS(B[i]) > anorm) anorm=ABS(S[i])+ABS(B[i]);
145  }
146  for(i = n - 1; ; i--) {
147  if (i < n - 1) {
148  if (g) {
149  for(j = l; j < n; j++) V(j, i) = (U(i, j) / U(i, l)) / g;
150  for(j = l; j < n; j++) {
151  for(s = T(0), k = l; k < n; k++) s += U(i, k) * V(k, j);
152  for(k = l; k < n; k++) V(k, j) += s * V(k, i);
153  }
154  }
155  for(j = l; j < n; j++) V(j, i) = V(i, j) = T(0);
156  }
157  V(i,i) =T(1);
158  g = B[i];
159  l = i;
160  if(i==0) break;
161  }
162  for(i = ( (m-1 < n-1) ? m-1 : n-1); ; i--) {
163  l = i+1;
164  g = S[i];
165  for(j=l; j<n; j++) U(i, j) = T(0);
166  if (g) {
167  g = T(1) / g;
168  for(j = l; j < n; j++) {
169  for(s = T(0), k = l; k < m; k++) s += U(k,i) * U(k,j);
170  f = (s / U(i,i)) * g;
171  for(k=i; k<m; k++) U(k,j) += f * U(k,i);
172  }
173  for(j = i; j < m; j++) U(j,i) *= g;
174  }
175  else {
176  for(j=i; j<m; j++) U(j,i) = T(0);
177  }
178  ++U(i,i);
179  if(i==0) break;
180  }
181  for(k = n - 1; ; k--) {
182  size_t its;
183  for(its = 1; its <= iterationMax; its++) {
184  bool flag = true;
185  for(l = k; ; l--) {
186  nm = l - 1;
187  if (ABS(B[l])/MAX(ABS(B[l]),anorm) < eps) {
188  flag = false;
189  break;
190  }
191  if (l == 0) { // should never happen
192  MatrixException e("SVD algorithm has nm==-1");
193  GPSTK_THROW(e);
194  }
195  if (ABS(S[nm])/MAX(ABS(S[nm]),anorm) < eps) break;
196  if(l == 0) break; // since l is unsigned...
197  }
198  if (flag) {
199  c = T(0);
200  s = T(1);
201  for(i = l; i <= k; i++) {
202  f = s * B[i];
203  B[i] = c * B[i];
204  if (ABS(f)/MAX(ABS(f),anorm) < eps) break;
205  g = S[i];
206  h = RSS(f,g);
207  S[i] = h;
208  h = T(1) / h;
209  c = g * h;
210  s = -f * h;
211  for(j = 0; j < m; j++) {
212  y = U(j, nm);
213  z = U(j,i);
214  U(j, nm) = y * c + z * s;
215  U(j,i) = z * c - y * s;
216  }
217  }
218  }
219  z = S[k];
220  if (l == k) {
221  if (z < T(0)) {
222  S[k] = -z;
223  for(j = 0; j < n; j++) V(j,k) = -V(j,k);
224  }
225  break;
226  }
227 
228  if (its == iterationMax) {
229  MatrixException e("SVD algorithm did not converge");
230  GPSTK_THROW(e);
231  }
232  x = S[l];
233  if(k == 0) { // should never happen
234  MatrixException e("SVD algorithm has k==0");
235  GPSTK_THROW(e);
236  }
237  nm = k - 1;
238  y = S[nm];
239  g = B[nm];
240  h = B[k];
241  f = ( (y-z) * (y+z) + (g-h) * (g+h)) / (T(2) * h * y);
242  g = RSS(f,T(1));
243  f = ( (x-z) * (x+z) + h * ((y/(f + SIGN(g,f))) - h)) / x;
244  c = s = 1.0;
245  for(j = l; j <= nm; j++) {
246  i = j + 1;
247  g = B[i];
248  y = S[i];
249  h = s * g;
250  g = c * g;
251  z = RSS(f, h);
252  B[j] = z;
253  c = f / z;
254  s = h / z;
255  f = x * c + g * s;
256  g = g * c - x * s;
257  h = y * s;
258  y *= c;
259  for(jj = 0; jj < n; jj++) {
260  x = V(jj, j);
261  z = V(jj, i);
262  V(jj, j) = x * c + z * s;
263  V(jj, i) = z * c - x * s;
264  }
265  z = RSS(f, h);
266  S[j] = z;
267  if (z) {
268  z = T(1) / z;
269  c = f * z;
270  s = h * z;
271  }
272  f = c * g + s * y;
273  x = c * y - s * g;
274  for(jj = 0; jj < m; jj++) {
275  y = U(jj, j);
276  z = U(jj, i);
277  U(jj, j) = y * c + z * s;
278  U(jj, i) = z * c - y * s;
279  }
280  }
281  B[l] = T(0);
282  B[k] = f;
283  S[k] = x;
284  }
285  if(k==0) break; // since k is unsigned...
286  }
287  // if U is not square - last n-m columns of U are zero - remove
288  if(U.cols() > U.rows()) {
289  for(i=1; i<S.size(); i++) { // sort in descending order
290  T sv=S[i],svj;
291  int kk = i-1; // not size_t ~ unsigned
292  while(kk >= 0) {
293  svj = S[kk];
294  if(sv < svj) break;
295  S[kk+1] = svj;
296  // exchange columns kk and kk+1 in U and V
297  U.swapCols(kk,kk+1);
298  V.swapCols(kk,kk+1);
299  kk = kk - 1;
300  }
301  S[kk+1] = sv;
302  }
303  Matrix<T> Temp(U);
304  U = Matrix<T>(Temp,0,0,Temp.rows(),Temp.rows());
305  S.resize(Temp.rows());
306  }
307 
308  if(flip) {
309  Matrix<T> Temp(U);
310  U = V;
311  V = Temp;
312  }
313 
314  return true;
315 
316  } // end SVD::operator() - the SVD algorithm

318  // Backsubstitution using SVD.
319  // Solve A*x=b for vector x where A [mxn] has been SVD'ed and is given by
320  // U,W,V (*this); that is A[mxn] = U[mxm]*W[mxn]*VT[nxn]. b has dimension m,
321  // x dimension n. Singular values are NOT edited, except that if s.v. == 0,
322  // 1/0 is replaced by 0. Result is returned as b.
323  template <class BaseClass>
324  void backSub(RefVectorBase<T, BaseClass>& b) const
325  throw(MatrixException)
326  {
327  if(b.size() != U.rows())
328  {
329  MatrixException e("SVD::BackSub called with unequal dimensions");
330  GPSTK_THROW(e);
331  }
332 
333  size_t i, n=V.cols(), m=U.rows();
334  Matrix<T> W(n,m,T(0)); // build the 'inverse singular values' matrix
335  for(i=0; i<S.size(); i++) W(i,i)=(S(i)==T(0)?T(0):T(1)/S(i));
336  Vector<T> Y;
337  Y = V*W*transpose(U)*b;
338  //b = Y;

340  // (op= not inherited); use assignFrom instead

397 // Performs the lower/upper triangular decomposition of a matrix PA = LU.
398 // The results are put into the matricies L, U, and P (pivot), and sign
399 // (representing even (positive) or odd (negative) row swaps.
400  template <class T>
401  class LUDecomp
402  {
403  public:
404  LUDecomp() {} // why is there no constructor from ConstMatrixBase?
405 
407  template <class BaseClass>
408  void operator() (const ConstMatrixBase<T, BaseClass>& m)
409  throw (MatrixException)
410  {
411  if(!m.isSquare() || m.rows()<=1) {
412  MatrixException e("LUDecomp requires a square, non-trivial matrix");
413  GPSTK_THROW(e);
414  }
415 
416  size_t N=m.rows(),i,j,k,imax;
417  T big,t,d;
418  Vector<T> V(N,T(0));
419 
420  LU = m;
421  Pivot = Vector<int>(N);
422  parity = 1;
423 
424  for(i=0; i<N; i++) { // get scale of each row
425  big = T(0);
426  for(j=0; j<N; j++) {
427  t = ABS(LU(i,j));
428  if(t > big) big=t;
429  }
430  if(big <= T(0)) { // m is singular
431  //LU *= T(0);
432  SingularMatrixException e("singular matrix!");
433  GPSTK_THROW(e);
434  }
435  V(i) = T(1)/big;
436  }
437 
438  for(j=0; j<N; j++) { // loop over columns
439  for(i=0; i<j; i++) {
440  t = LU(i,j);
441  for(k=0; k<i; k++) t -= LU(i,k)*LU(k,j);
442  LU(i,j) = t;
443  }
444  big = T(0); // find largest pivot
445  for(i=j; i<N; i++) {
446  t = LU(i,j);
447  for(k=0; k<j; k++) t -= LU(i,k)*LU(k,j);
448  LU(i,j) = t;
449  d = V(i)*ABS(t);
450  if(d >= big) {
451  big = d;
452  imax = i;
453  }
454  }
455  if(j != imax) {
456  LU.swapRows(imax,j);
457  V(imax) = V(j);
458  parity = -parity;
459  }
460  Pivot(j) = imax;
461 
462  t = LU(j,j);
463  if(t == 0.0) { // m is singular
464  //LU *= T(0);
465  SingularMatrixException e("singular matrix!");
466  GPSTK_THROW(e);
467  }
468  if(j != N-1) {
469  d = T(1)/t;
470  for(i=j+1; i<N; i++) LU(i,j) *= d;
471  }
472  }
473  } // end LUDecomp()

475  // Compute inverse(m)*v, where *this is LUD(m), via back substitution
476  // Solution overwrites input Vector v
477  template <class BaseClass2>
478  void backSub(RefVectorBase<T, BaseClass2>& v) const
479  throw (MatrixException)
480  {
481  if(LU.rows() != v.size()) {
482  MatrixException e("Vector size does not match dimension of LUDecomp");
483  GPSTK_THROW(e);
484  }
485 
486  bool first=true;
487  size_t N=LU.rows(),i,j,ii(0);
488  T sum;
489 
490  // un-pivot
491  for(i=0; i<N; i++) {
492  sum = v(Pivot(i));
493  v(Pivot(i)) = v(i);
494  if(first && sum != T(0)) {
495  ii = i;
496  first = false;
497  }
498  else for(j=ii; j<i; j++) sum -= LU(i,j)*v(j);
499  v(i) = sum;
500  }
501  // back substitution
502  for(i=N-1; ; i--) {
503  sum = v(i);
504  for(j=i+1; j<N; j++) sum -= LU(i,j)*v(j);
505  v(i) = sum / LU(i,i);
506  if(i == 0) break; // b/c i is unsigned
507  }
508  } // end LUD::backSub

530  // Compute Cholesky decomposition (triangular square root) of the given matrix,
531  // which must be square and positive definite. That is, if M is square and positive
532  // definite, then M = C * transpose(C) where C is the Cholesky decomposition. This
533  // routine computes both the upper (U) and lower (L) triangular C's;
534  // thus M = L * transpose(L) = U * transpose(U).
535  //
536  // NB Note that M = U*UT NOT UT*U; if M = U'T*U' is desired, use U'=transpose(L);
537  // then note the similarity to LU decomposition: M = L*U' = L*LT = U'T*U'.
538  //
539  // Positive definite (PD) implies positive eigenvalues, and the reverse.
540  //
541  // Note that the result is not unique, as any orthogonal transformation R of UT,
542  // including a change in sign, will preserve the result m = U*UT = U*RT*R*UT.
543  //
544  // @code
545  // Matrix<double> M(i,i); // square and PD
547  // Ch(M);
548  // cout << Ch.U << endl << Ch.L << endl;
549  // // note that M = Ch.L * transpose(Ch.L);
550  // // note that M = Ch.U * transpose(Ch.U);
551  // // and that if Matrix<double> Up(transpose(Ch.L)); then
552  // // M = transpose(Up) * Up;
553  // @endcode
554  template <class T>
555  class Cholesky
556  {
557  public:
558  Cholesky() {}
559 
561  template <class BaseClass>
562  void operator() (const ConstMatrixBase<T, BaseClass>& m)
563  throw (MatrixException)
564  {
565  if(!m.isSquare()) {
566  MatrixException e("Cholesky requires a square matrix");
567  GPSTK_THROW(e);
568  }
569 
570  size_t N=m.rows(),i,j,k;
571  double d;
572  Matrix<T> P(m);
573  U = Matrix<T>(m.rows(),m.cols(),T(0));
574 
575  for(j=N-1; ; j--) {
576  if(P(j,j) <= T(0)) {
577  MatrixException e("Cholesky fails - eigenvalue <= 0");
578  GPSTK_THROW(e);
579  }
580  U(j,j) = SQRT(P(j,j));
581  d = T(1)/U(j,j);
582  if(j > 0) {
583  for(k=0; k<j; k++) U(k,j)=d*P(k,j);
584  for(k=0; k<j; k++)
585  for(i=0; i<=k; i++)
586  P(i,k) -= U(k,j)*U(i,j);
587  }
588  if(j==0) break; // since j is unsigned
589  }
590 
591  P = m;
592  L = Matrix<T>(m.rows(),m.cols(),T(0));
593  for(j=0; j<=N-1; j++) {
594  if(P(j,j) <= T(0)) {
595  MatrixException e("Cholesky fails - eigenvalue <= 0");
596  GPSTK_THROW(e);
597  }
598  L(j,j) = SQRT(P(j,j));
599  d = T(1)/L(j,j);
600  if(j < N-1) {
601  for(k=j+1; k<N; k++) L(k,j)=d*P(k,j);
602  for(k=j+1; k<N; k++) {
603  for(i=k; i<N; i++) {
604  P(i,k) -= L(i,j)*L(k,j);
605  }
606  }
607  }
608  }
609 
610  } // end Cholesky::operator()

612  // Use backsubstition to solve the equation A*x=b where *this Cholesky
613  // has been applied to A, i.e. A = L*transpose(L). The algorithm is in
614  // two steps: since A*x=L*LT*x=b, first solve L*y=b for y, then solve
615  // LT*x=y for x. x is returned as b.
616  template <class BaseClass2>
617  void backSub(RefVectorBase<T, BaseClass2>& b) const
618  throw (MatrixException)
619  {
620  if (L.rows() != b.size())
621  {
622  MatrixException e("Vector size does not match dimension of Cholesky");
623  GPSTK_THROW(e);
624  }
625  size_t i,j,N=L.rows();
626 
627  Vector<T> y(b.size());
628  y(0) = b(0)/L(0,0);
629  for(i=1; i<N; i++) {
630  y(i) = b(i);
631  for(j=0; j<i; j++) y(i)-=L(i,j)*y(j);
632  y(i) /= L(i,i);
633  }
634  // b is now x
635  b(N-1) = y(N-1)/L(N-1,N-1);
636  for(i=N-1; ; i--) {
637  b(i) = y(i);
638  for(j=i+1; j<N; j++) b(i)-=L(j,i)*b(j);
639  b(i) /= L(i,i);
640  if(i==0) break;
641  }
642 
643  } // end Cholesky::backSub

650  // Compute the Cholesky decomposition using the Cholesky-Crout algorithm,
651  // which is very fast; if A is the given matrix we will get L, where A = L*LT.
652  // A must be symetric and positive definite. This is the usual case when
653  // A comes from applying a Least Mean-Square (LMS) or Weighted Least
654  // Mean-Square (WLMS) method.
655  //
656  // NB. U here is NOT the same as U in class Cholesky (but L is the same);
657  // here m = transpose(U)*U, but there m = U * transpose(U); see doc for Cholesky.
658  template <class T>
659  class CholeskyCrout : public Cholesky<T>
660  {
661  public:
662  template <class BaseClass>
663  void operator() (const ConstMatrixBase<T, BaseClass>& m) throw(MatrixException)
664  {
665  if(!m.isSquare()) {
666  MatrixException e("CholeskyCrout requires a square matrix");
667  GPSTK_THROW(e);
668  }
669 
670  int N = m.rows(), i, j, k;
671  double sum;
672  (*this).L = Matrix<T>(N,N, 0.0);
673 
674  for(j=0; j<N; j++) {
675  sum = m(j,j);
676  for(k=0; k<j; k++ ) sum -= (*this).L(j,k)*(*this).L(j,k);
677  if(sum > 0.0) {
678  (*this).L(j,j) = SQRT(sum);
679  } else {
680  MatrixException e("CholeskyCrout fails - eigenvalue <= 0");
681  GPSTK_THROW(e);
682  }
683 
684  for(i=j+1; i<N; i++){
685  sum = m(i,j);
686  for(k=0; k<j; k++) sum -= (*this).L(i,k)*(*this).L(j,k);
687  (*this).L(i,j) = sum/(*this).L(j,j);
688  }
689  }
690 
691  (*this).U = transpose((*this).L);
692  }

697  // The Householder transformation is simply an orthogonal transformation
698  // designed to make the elements below the diagonal zero. It applies to any
699  // matrix.
700  template <class T>
701  class Householder
702  {
703  public:
704  Householder() {}
705 
706  // Explicitly perform the transformation, one column at a time, without
707  // actually constructing the transformation matrix. Let y be column k of the
708  // input matrix. y can be zeroed below the diagonal as follows:
709  // let sum=sign(y(k))*sqrt(y*y), and define vector u(k)=y(k)+sum,
710  // u(j)=y(j) (j.gt.k). This defines the transformation matrix as (1-bu*u),
711  // with b=2/u*u=1/sum*u(k). Redefine y(k)=u(k) and apply the transformation to
712  // elements of the input matrix below and to the right of the (k,k) element.
713  // This algorithm for each column k=0,n-1 in turn is equivalent to a single
714  // orthogonal transformation which triangularizes the matrix.
715  template <class BaseClass>
716  inline void operator() (const ConstMatrixBase<T, BaseClass>& m)
717  throw (MatrixException)
718  {
719  A = m;
720  size_t i,j,k;
721  Vector<T> v(A.rows());
722  T sum,alpha;
723 
724  // loop over cols
725  const T EPS(1.e-200);
726  for(j=0; (j<A.cols()-1 && j<A.rows()-1); j++) {
727  sum = T(0);
728  // loop over rows at diagonal and below
729  for(i=j; i<A.rows(); i++) {
730  v(i) = A(i,j);
731  A(i,j) = T(0); // this is optional - below the diag is trash
732  sum += v(i)*v(i);
733  }
734 
735  if(sum < EPS) continue; // already zero below diagonal
736 
737  sum = SQRT(sum);
738  if(v(j) > T(0)) sum = -sum;
739  A(j,j) = sum;
740  v(j) = v(j) - sum;
741  sum = 1/(sum*v(j));
742 
743  // loop over columns beyond j
744  for(k=j+1; k<A.cols(); k++) {
745  alpha = T(0);
746  // loop over rows at and below j
747  for(i=j; i<A.rows(); i++) {
748  alpha += A(i,k)*v(i);
749  }
750  alpha *= sum;
751  // modify column k at and below j
752  for(i=j; i<A.rows(); i++) {
753  A(i,k) += alpha*v(i);
754  }
755  } // end loop over cols > j
756 
757  } // end loop over cols
758 
759  } // end Householder::operator()
760 
762  Matrix<T> A;
763 
764  }; // end class Householder
