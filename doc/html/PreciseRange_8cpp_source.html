58  double PreciseRange::ComputeAtTransmitTime(const CommonTime& nomRecTime,
59  const double pr,
60  const Position& Receiver,
61  const SatID sat,
62  const AntexData& antenna,
63  const SolarSystem& SSEph,
64  const EarthOrientation& EO,
65  const XvtStore<SatID>& Eph,
66  const bool isCOM)
67  throw(Exception)
68  {
69  try {
70  int i;
71  Position Rx(Receiver);
72  GPSEllipsoid ell;
73  Xvt svPosVel;
74 
75  // nominal transmit time
76  transmit = nomRecTime; // receive time on receiver's clock
77  transmit -= pr/ell.c(); // correct for measured time of flight and Rx clock
78 
79  // get the satellite position at the nominal time, computing and
80  // correcting for the satellite clock bias and other delays
81  try {
82  svPosVel = Eph.getXvt(sat,transmit);
83  SatR.setECEF(svPosVel.x[0],svPosVel.x[1],svPosVel.x[2]);
84  }
85  // this should be a 'no ephemeris' exception
86  catch(InvalidRequest& e) { GPSTK_RETHROW(e); }
87 
88  // update the transmit time for sat clk bias + relativity
89  transmit -= svPosVel.clkbias + svPosVel.relcorr;
90 
91  // Sagnac effect
92  // ref. Ashby and Spilker, GPS: Theory and Application, 1996 Vol 1, pg 673.
93  // this is w(Earth) * (SatR cross Rx).Z() / c*c in seconds
94  // beware numerical error by differencing very large to get very small
95  Sagnac = ( (SatR.X()/ell.c()) * (Rx.Y()/ell.c())
96  - (SatR.Y()/ell.c()) * (Rx.X()/ell.c()) ) * ell.angVelocity();
97  transmit -= Sagnac;
98 
99  // compute other delays -- very small
100  // 2GM/c^2 = 0.00887005608 m^3/s^2 * s^2/m^2 = m
101  double rx = Rx.radius();
102  double rs = SatR.radius();
103  double dr = range(SatR,Rx);
104  relativity2 = -0.00887005608 * ::log((rx+rs+dr)/(rx+rs-dr));
105  transmit -= relativity2 / ell.c();
106 
107  // iterate satellite position
108  try {
109  svPosVel = Eph.getXvt(sat,transmit);
110  // Do NOT replace these with Xvt
111  SatR.setECEF(svPosVel.x[0],svPosVel.x[1],svPosVel.x[2]);
112  SatV.setECEF(svPosVel.v[0],svPosVel.v[1],svPosVel.v[2]);
113  }
114  catch(InvalidRequest& e) { GPSTK_RETHROW(e); }
115 
116  // ----------------------------------------------------------
118  //relativity = RelativityCorrection(SatR,SatV) * ell.c();
119 
120  // save relativity (m), clock bias (m) and drift (m/s)
121  relativity = svPosVel.relcorr * ell.c();
122  satclkbias = svPosVel.clkbias * ell.c();
123  satclkdrift = svPosVel.clkdrift * ell.c();
124 
125  // correct for Earth rotation
126  double sxyz[3], wt;
127  rawrange = range(SatR,Rx);
128  wt = ell.angVelocity() * rawrange/ell.c();
129  sxyz[0] = ::cos(wt)*SatR.X() + ::sin(wt)*SatR.Y();
130  sxyz[1] = -::sin(wt)*SatR.X() + ::cos(wt)*SatR.Y();
131  sxyz[2] = SatR.Z();
132  SatR.setECEF(sxyz);
133  sxyz[0] = ::cos(wt)*SatV.X() + ::sin(wt)*SatV.Y();
134  sxyz[1] = -::sin(wt)*SatV.X() + ::cos(wt)*SatV.Y();
135  sxyz[2] = SatV.Z();
136  SatV.setECEF(sxyz);
137 
138  // geometric range, again
139  rawrange = range(SatR,Rx);
140 
141  // Compute line of sight, satellite to receiver
142  Triple S2R(Rx.X()-SatR.X(),Rx.Y()-SatR.Y(),Rx.Z()-SatR.Z());
143  S2R = S2R.unitVector();
144 
145  // ----------------------------------------------------------
146  // satellite antenna pco and pcv
147  if(isCOM && antenna.isValid()) {
148  double sf;
149  // rotation matrix from satellite attitude: Rot*[XYZ]=[body frame]
150  Matrix<double> Rotation;
151  if(SSEph.JPLNumber() > -1) {
152  // use full JPL ephemeris
153  Rotation = SatelliteAttitude(transmit, SatR, SSEph, EO, sf);
154  }
155  else {
156  // use SolarPosition
157  Rotation = SatelliteAttitude(transmit, SatR, sf);
158  }
159 
160  // phase center offset vector in body frame (at L1)
161  Triple pco = antenna.getPhaseCenterOffset(1);
162  Vector<double> PCO(3);
163  for(i=0; i<3; i++) PCO(i) = pco[i]/1000.0; // body frame, mm -> m
164 
165  // PCO vector (from COM to PC) in ECEF XYZ frame, m
166  SatPCOXYZ = transpose(Rotation) * PCO;
167 
168  Triple pcoxyz(SatPCOXYZ(0),SatPCOXYZ(1),SatPCOXYZ(2));
169  satLOSPCO = pcoxyz.dot(S2R); // meters
170 
171  // phase center variation NB. this should be subtracted from rawrange
172  // get the body frame azimuth and nadir angles
173  double nadir,az;
174  SatelliteNadirAzimuthAngles(SatR, Rx, Rotation, nadir, az);
175  satLOSPCV = antenna.getPhaseCenterVariation(1, az, nadir) / 1000.; // meters
176 
177  }
178  else {
179  satLOSPCO = satLOSPCV = 0.0;
180  SatPCOXYZ=Vector<double>(3,0.0);
181  }
182 
183  // ----------------------------------------------------------
184  // other quanitites
185  // direction cosines
186  // cosines[0] = (Rx.X()-SatR.X())/rawrange;
187  // cosines[1] = (Rx.Y()-SatR.Y())/rawrange;
188  // cosines[2] = (Rx.Z()-SatR.Z())/rawrange;
189  for(i=0; i<3; i++) cosines[i] = -S2R[i]; // receiver to satellite
190 
191  // elevation and azimuth
192  elevation = Rx.elevation(SatR);
193  azimuth = Rx.azimuth(SatR);
194  elevationGeodetic = Rx.elevationGeodetic(SatR);
195  azimuthGeodetic = Rx.azimuthGeodetic(SatR);
196 
197  // return corrected ephemeris range
198  return (rawrange-satclkbias-relativity-relativity2-satLOSPCO+satLOSPCV);
199 
200  } // end try
201  catch(gpstk::Exception& e) { GPSTK_RETHROW(e); }
202 
203  } // end PreciseRange::ComputeAtTransmitTime
